36  // better-than-nothing emulation of C99 snprintf(),

40  inline
41  int
42  arma_snprintf(char* out, size_t size, const char* fmt, ...)
43  {
44  size_t i;
45 
46  for(i=0; i<size; ++i)
47  {
48  out[i] = fmt[i];
49  if(fmt[i] == char(0))
50  break;
51  }
52 
53  if(size > 0)
54  out[size-1] = char(0);
55 
56  return int(i);
57  }
