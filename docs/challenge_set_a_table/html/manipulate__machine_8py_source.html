35  def resolve(self):
36  """ Resolves
37 
38  :return: entity in the <area_description> of the <surface_designator> that is closest to the robot
39  """
40  # Get the surface as an entity
41  surface = self._surface_designator.resolve()
42  if surface is None:
43  rospy.logerror("Cannot resolve surface designator")
44  return None
45 
46  # Get all entities and check which ones are on the table
47  all_entities = self._robot.ed.get_entities()
48  entities = []
49  for e in all_entities:
50  point = VectorStamped.from_framestamped(e.pose)
51  if surface.in_volume(point=point, volume_id=self.area_description):
52  entities.append(e)
53 
54  # Remove all entities that are too large or too small
55  entities = [e for e in entities if (e.shape.z_max - e.shape.z_min) > MIN_GRAB_OBJECT_HEIGHT]
56  entities = [e for e in entities if (e.shape.y_max - e.shape.y_min) < MAX_GRAB_OBJECT_WIDTH]
57  entities = [e for e in entities if (e.shape.x_max - e.shape.x_min) < MAX_GRAB_OBJECT_WIDTH]
58 
59  # Check if there are any
60  if not entities:
61  return None
62 
63  # Sort the entities and return the closest one
64  base_pose = self._robot.base.get_location().frame
65  entities = sorted(entities, key=lambda e: e.distance_to_2d(base_pose.p))
66  return entities[0]
