17 def color_map(N=256, normalized=False):
18  """
19  Generate an RGB color map of N different colors
20 
21  :param N : int amount of colors to generate
22  :param normalized: bool indicating range of each channel: float32 in [0, 1] or int in [0, 255]
23  :return: a numpy.array of shape (N, 3) with a row for each color and each row is [R,G,B]
24  """
25 
26  def bitget(byteval, idx):
27  return ((byteval & (1 << idx)) != 0)
28 
29  dtype = 'float32' if normalized else 'uint8'
30  cmap = np.zeros((N, 3), dtype=dtype)
31  for i in range(N):
32  r = g = b = 0
33  c = i + 1 # skip the first color (black)
34  for j in range(8):
35  r |= bitget(c, 0) << 7 - j
36  g |= bitget(c, 1) << 7 - j
37  b |= bitget(c, 2) << 7 - j
38  c >>= 3
39 
40  cmap[i] = np.array([r, g, b])
41 
42  cmap = cmap / 255 if normalized else cmap
43  return cmap
