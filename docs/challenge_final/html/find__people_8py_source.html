30 def color_map(N=256, normalized=False):
31  """
32  Generate an RGB color map of N different colors
33 
34  :param N : int amount of colors to generate
35  :param normalized: bool indicating range of each channel: float32 in [0, 1] or int in [0, 255]
36  :return: a numpy.array of shape (N, 3) with a row for each color and each row is [R,G,B]
37  """
38 
39  def bitget(byteval, idx):
40  return ((byteval & (1 << idx)) != 0)
41 
42  dtype = 'float32' if normalized else 'uint8'
43  cmap = np.zeros((N, 3), dtype=dtype)
44  for i in range(N):
45  r = g = b = 0
46  c = i + 1 # skip the first color (black)
47  for j in range(8):
48  r |= bitget(c, 0) << 7 - j
49  g |= bitget(c, 1) << 7 - j
50  b |= bitget(c, 2) << 7 - j
51  c >>= 3
52 
53  cmap[i] = np.array([r, g, b])
54 
55  cmap = cmap / 255 if normalized else cmap
56  return cmap
