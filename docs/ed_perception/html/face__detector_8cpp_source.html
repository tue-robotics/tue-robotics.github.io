114 void FaceDetector::classify(const ed::Entity& e, const std::string& property, const ed::perception::CategoricalDistribution& prior,
115  ed::perception::ClassificationOutput& output) const
116 {
117  if (property != "type" && property != "name")
118  return;
119 
120  // If we already know that this is not going to be a human, skip face detection altogether
121  double prior_human;
122  if (prior.getScore("human", prior_human) && prior_human == 0)
123  return;
124 
125  // ---------- Prepare measurement ----------
126 
127  // Get the best measurement from the entity
128  ed::MeasurementConstPtr msr = e.lastMeasurement();
129 
130  if (!msr)
131  return;
132 
133  // get color image
134  const cv::Mat& color_image = msr->image()->getRGBImage();
135 
136  // get depth image
137  const cv::Mat& depth_image = msr->image()->getDepthImage();
138 
139  // Mask color image
140  cv::Rect rgb_roi;
141  cv::Mat color_image_masked = ed::perception::maskImage(color_image, msr->imageMask(), rgb_roi);
142 
143  // ---------- Detect faces ----------
144 
145  std::vector<cv::Rect> faces_front;
146  std::vector<cv::Rect> faces_profile;
147 
148  // Detect faces in the measurment and assert the results
149  if (DetectFaces(color_image_masked(rgb_roi), faces_front, faces_profile))
150  {
151  // write face information to config if a frontal face was found
152  int face_counter = 0;
153  if (faces_front.size() > 0)
154  {
155  output.data.writeArray("faces_front");
156  writeFaceDetectionResult(*msr, rgb_roi, faces_front, face_counter, output.data);
157  output.data.endArray();
158  }
159 
160  // write face information to config if a profile face was found
161  if (faces_profile.size() > 0)
162  {
163  output.data.writeArray("faces_profile");
164  writeFaceDetectionResult(*msr, rgb_roi, faces_profile, face_counter, output.data);
165  output.data.endArray();
166  }
167 
168  if (property == "type")
169  {
170  output.likelihood.setScore("human", 1);
171  }
172  else if (property == "name")
173  {
174  if (!faces_front.empty())
175  {
176  recognizeFace(color_image_masked(rgb_roi), faces_front[0], output);
177  }
178  }
179  }
180 }

196 bool FaceDetector::DetectFaces(const cv::Mat& cropped_img,
197  std::vector<cv::Rect>& faces_front,
198  std::vector<cv::Rect>& faces_profile) const{
199 
200  cv::Mat cascade_img;
201  std::vector<cv::Rect>::iterator face_it;
202 
203  // create a copy of the image
204  cropped_img.copyTo(cascade_img);
205 
206  // increase contrast of the image
207  normalize(cascade_img, cascade_img, 0, 255, cv::NORM_MINMAX, CV_8UC1);
208 
209 
210 
211  // detect frontal faces
212  classifier_front_.detectMultiScale(cascade_img,
213  faces_front,
214  classifier_front_scale_factor_,
215  classifier_front_min_neighbours_,
216  0|CV_HAAR_SCALE_IMAGE,
217  classif_front_min_size_);
218 
219  // discard face if its not close to the top of the region, false positive
220  face_it = faces_front.begin();
221  for ( ; face_it != faces_front.end(); ) {
222  // allowed area is the full width and three times the size of the detected face
223  cv::Rect allowed_area (0, 0, cropped_img.cols, face_it->height * 3);
224 
225  // test if the rectangles intersect
226  if ( !(allowed_area & *face_it).area()) {
227  face_it = faces_front.erase(face_it);
228  }else
229  ++face_it;
230  }
231 
232  // only search profile faces if the frontal face detection failed
233  if (faces_front.empty())
234  {
235  classifier_profile_.detectMultiScale(cascade_img,
236  faces_profile,
237  classifier_profile_scale_factor_,
238  classifier_profile_min_neighbours_,
239  0|CV_HAAR_SCALE_IMAGE,
240  classif_profile_min_size_);
241 
242  // discard face if its not close to the top of the region, false positive
243  face_it = faces_profile.begin();
244  for ( ; face_it != faces_profile.end(); ) {
245  // allowed area is the full width and three times the size of the detected face
246  cv::Rect allowed_area (0, 0, cropped_img.cols, face_it->height * 3);
247 
248  // test if the rectangles intersect
249  if ( !(allowed_area & *face_it).area()) {
250  face_it = faces_profile.erase(face_it);
251  }else
252  ++face_it;
253  }
254  }
255 
256 
257  // if debug mode is active and faces were found
258  if (debug_mode_)
259  {
260  cv::Mat debugImg(cropped_img);
261 
262  for (uint j = 0; j < faces_front.size(); j++)
263  cv::rectangle(debugImg, faces_front[j], cv::Scalar(0, 255, 0), 2, CV_AA);
264 
265  for (uint j = 0; j < faces_profile.size(); j++)
266  cv::rectangle(debugImg, faces_profile[j], cv::Scalar(0, 0, 255), 2, CV_AA);
267 
268 
269  cv::imwrite(debug_folder_ + ed::Entity::generateID().c_str() + "_face_detector.png", debugImg);
270  cv::imshow("Face Detector Output", debugImg);
271  }
272 
273  // return true if a face was found
274  return (!faces_front.empty() || !faces_profile.empty());
275 }

297 void FaceDetector::writeFaceDetectionResult(const ed::Measurement& msr, const cv::Rect& rgb_roi,
298  const std::vector<cv::Rect>& rgb_face_rois,
299  int& face_counter, tue::Configuration& result) const
300 {
301  // get color image
302  const cv::Mat& color_image = msr.image()->getRGBImage();
303 
304  // get color image
305  const cv::Mat& depth_image = msr.image()->getDepthImage();
306 
307  // Calculate size factor between depth and rgb images
308  double f_depth_rgb = (double)depth_image.cols / color_image.cols;
309 
310  // Create depth view
311  rgbd::View depth_view(*msr.image(), depth_image.cols);
312 
313  for (uint j = 0; j < rgb_face_rois.size(); j++)
314  {
315  cv::Rect rgb_face_roi = rgb_face_rois[j];
316  rgb_face_roi.x += rgb_roi.x;
317  rgb_face_roi.y += rgb_roi.y;
318 
319  if (debug_mode_)
320  ed::perception::saveDebugImage("face_detector-rgb", color_image(rgb_face_roi));
321 
322  result.addArrayItem();
323 
324  result.setValue("index", face_counter);
325 
326  // add 2D location of the face
327  result.setValue("x", rgb_face_roi.x);
328  result.setValue("y", rgb_face_roi.y);
329  result.setValue("width", rgb_face_roi.width);
330  result.setValue("height", rgb_face_roi.height);
331 
332  // Compute face roi for depth image
333  cv::Rect depth_face_roi(f_depth_rgb * rgb_face_roi.x, f_depth_rgb * rgb_face_roi.y,
334  f_depth_rgb * rgb_face_roi.width, f_depth_rgb * rgb_face_roi.height);
335 
336  if (debug_mode_)
337  ed::perception::saveDebugImage("face_detector-depth", depth_image(depth_face_roi));
338 
339  cv::Mat face_area = depth_image(depth_face_roi);
340  float avg_depth = ed::perception::getMedianDepth(face_area);
341 
342  if (avg_depth > 0)
343  {
344  // calculate the center point of the face
345  cv::Point2i p_2d(depth_face_roi.x + depth_face_roi.width/2,
346  depth_face_roi.y + depth_face_roi.height/2);
347 
348  geo::Vector3 projection = depth_view.getRasterizer().project2Dto3D(p_2d.x, p_2d.y) * avg_depth;
349  geo::Vector3 point_map = msr.sensorPose() * projection;
350 
351  // add 3D location of the face
352  result.setValue("map_x", point_map.x);
353  result.setValue("map_y", point_map.y);
354  result.setValue("map_z", point_map.z);
355 
356  }
357  else
358  {
359  std::cout << "[ED FACE DETECTOR] Could not calculate face's average depth. Map coordinates might be incorrect!" << std::endl;
360  }
361 
362  result.endArrayItem();
363  face_counter++;
364  }
365 }
