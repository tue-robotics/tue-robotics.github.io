27 void decomposePose(const geo::Pose3D& pose, geo::Pose3D& pose_xya, geo::Pose3D& pose_zrp)
28 {
29  tf::Matrix3x3 m;
30  geo::convert(pose.R, m);
31 
32  double roll, pitch, yaw;
33  m.getRPY(roll, pitch, yaw);
34 
35  pose_xya.R.setRPY(0, 0, yaw);
36  pose_xya.t = geo::Vec3(pose.t.x, pose.t.y, 0);
37 
38  pose_zrp = pose_xya.inverse() * pose;
39 }
