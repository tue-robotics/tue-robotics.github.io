209 void TypeAggregator::determineType(std::vector<Feature>& features, std::string& type, float& score) const{
210 
211  bool face = false;
212  bool multiple_faces = false;
213  bool human_shape = false;
214  bool small_size = false;
215  bool medium_size = false;
216  bool large_size = false;
217 
218 
219  // get booleans from features
220  for (std::vector<Feature>::iterator feat_it = features.begin() ; feat_it != features.end(); ++feat_it){
221  face = ((feat_it->name.compare("face") == 0 && feat_it->score == 1) || face == true);
222  multiple_faces = ((feat_it->name.compare("multiple_faces") == 0 && feat_it->score == 1) || multiple_faces == true);
223  human_shape = ((feat_it->name.compare("human_shape") == 0 && feat_it->score == 1) || human_shape == true);
224 
225  small_size = ((feat_it->name.compare("small_size") == 0 && feat_it->score == 1) || small_size == true);
226  medium_size = ((feat_it->name.compare("medium_size") == 0 && feat_it->score == 1) || medium_size == true);
227  large_size = ((feat_it->name.compare("large_size") == 0 && feat_it->score == 1) || large_size == true);
228  }
229 
230  // if it has a face or human shape, and its not small, it must be a human
231  if ((face || (human_shape && !multiple_faces)) && (medium_size || large_size)){
232  type = "human";
233  score = 1;
234  }else if (multiple_faces && (medium_size || large_size)){
235  type = "crowd";
236  score = 1;
237  // if it has no face, no human shape, and its not small, then its not an ordinary object
238  }else if(!human_shape && !face && !multiple_faces && (medium_size || large_size)){
239  type = "unknown";
240  score = 0;
241  }
242 }
