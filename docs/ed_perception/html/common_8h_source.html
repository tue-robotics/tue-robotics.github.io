63 inline int cluster_points(ANNpointArray points, int points_count,
64  std::vector<int>& membership, float r_max =
65  600.0f, float r_min = 200.0f, float A = 800.0f, float K = 0.02f)
66 {
67  // adaptive radius calculation
68  // (see paper Fast and Robust Object Detection in Household Environments
69  // Using Vocabulary Trees with SIFT Descriptors by Pangercic and Haltakov)
70  double radius = (1 - logsig((points_count - A) * K)) * (r_max - r_min) + r_min;
71  //radius = (1-logsig((points_count-800.0)/50.0))*300 + 300;
72  //printf("\n\n NEW RADIUS = %f\n\n", radius);
73  ANNidxArray nnIdx = new ANNidx[points_count];
74  // ANNdistArray dists = new ANNdist[points_count];
75  ANNkd_tree* kdTree = new ANNkd_tree(points, points_count, 2);
76  membership.assign(points_count, -1);
77  int last_unassigned_id = 0;
78  int current_cluster = 0;
79 
80  while (last_unassigned_id < points_count)
81  {
82  std::vector<int> points_stack;
83  points_stack.push_back(last_unassigned_id);
84  while (points_stack.size() > 0)
85  {
86  //printf("STACK SIZE = %d\n", points_stack.size());
87  int current_point_id = points_stack.back();
88  points_stack.pop_back();
89  membership[current_point_id] = current_cluster;
90  int points_found = kdTree->annkFRSearch(points[current_point_id],
91  radius, points_count, nnIdx);
92  //printf("Neighbours found = %d\n", points_found);
93  int newPointsCount = 0;
94  for (int i = 0; i < points_found; ++i)
95  if (membership[nnIdx[i]] == -1)
96  ++newPointsCount;
97 
98  if (newPointsCount > 3)
99  {
100  for (int i = 0; i < points_found; ++i)
101  if (membership[nnIdx[i]] == -1)
102  points_stack.push_back(nnIdx[i]);
103  }
104  }
105 
106  ++current_cluster;
107  ++last_unassigned_id;
108  while (last_unassigned_id < points_count
109  && membership[last_unassigned_id] != -1)
110  ++last_unassigned_id;
111  }
112 
113  delete[] nnIdx;
114  //delete [] dists;
115  delete kdTree;
116  annClose();
117  return current_cluster;
118 }
