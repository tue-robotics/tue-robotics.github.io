132 void HumanContourMatcher::process(const ed::perception::WorkerInput& input, ed::perception::WorkerOutput& output) const
133 {
134  ed::ErrorContext errc("Processing entity in HumanContourMatcher");
135 
136  // if initialization failed, return
137  if (!init_success_)
138  return;
139 
140  const ed::EntityConstPtr& e = input.entity;
141  tue::Configuration& result = output.data;
142 
143  float classification_error = 0;
144  float classification_deviation = 0;
145  std::string classification_stance;
146  bool is_human = false;
147 
148  // ---------- Prepare measurement ----------
149 
150  // Get the best measurement from the entity
151  ed::MeasurementConstPtr msr = e->lastMeasurement();
152  if (!msr)
153  return;
154 
155  int min_x, max_x, min_y, max_y;
156 
157  // create a view
158 // rgbd::View view(*msr->image(), msr->image()->getRGBImage().cols);
159 
160  // get color image
161  const cv::Mat& color_image = msr->image()->getRGBImage();
162 
163  // get color image
164  const cv::Mat& depth_image = msr->image()->getDepthImage();
165 
166  // crop it to match the view
167 // cv::Mat cropped_image(color_image(cv::Rect(0,0,view.getWidth(), view.getHeight())));
168 
169  // initialize bounding box points
170  max_x = 0;
171  max_y = 0;
172  min_x = depth_image.cols;
173  min_y = depth_image.rows;
174 
175  cv::Mat depth_mask = cv::Mat::zeros(depth_image.rows, depth_image.cols, CV_8UC1);
176  // Iterate over all points in the mask
177  for(ed::ImageMask::const_iterator it = msr->imageMask().begin(depth_image.cols); it != msr->imageMask().end(); ++it)
178  {
179  // mask's (x, y) coordinate in the depth image
180  const cv::Point2i p_2d(it());
181 
182  // paint a mask
183  depth_mask.at<unsigned char>(p_2d) = 255;
184 
185  // update the boundary coordinates
186  if (min_x > p_2d.x) min_x = p_2d.x;
187  if (max_x < p_2d.x) max_x = p_2d.x;
188  if (min_y > p_2d.y) min_y = p_2d.y;
189  if (max_y < p_2d.y) max_y = p_2d.y;
190  }
191 
192  cv::Rect bouding_box (min_x, min_y, max_x - min_x, max_y - min_y);
193 
194  // create a copy of the depth image region of interest, masked
195  cv::Mat masked_depth_image(depth_image);
196  masked_depth_image.copyTo(masked_depth_image, depth_mask);
197  masked_depth_image = masked_depth_image(bouding_box);
198 
199  // get entity average depth
200  float avg_depth = ed::perception::getAverageDepth(masked_depth_image);
201 
202  // call classifier
203  is_human = human_classifier_.Classify(depth_image, color_image, depth_mask, avg_depth, classification_error, classification_deviation, classification_stance);
204 
205 
206  // ----------------------- Assert results -----------------------
207 
208  // create group if it doesnt exist
209  if (!result.readGroup("perception_result", tue::OPTIONAL))
210  {
211  result.writeGroup("perception_result");
212  }
213 
214  result.writeGroup("human_contour_matcher");
215 
216  result.setValue("label", "human_shape");
217 
218  // only assert something if error is above 0, otherwise matching could not be performed
219  if (classification_error > 0){
220  result.setValue("stance", classification_stance);
221  result.setValue("error", classification_error);
222  result.setValue("deviation", classification_deviation);
223  }
224 
225  if(is_human){
226  // classified as human
227  result.setValue("score", type_positive_score_);
228 
229 // output.type_update.setScore("crowd", type_positive_score_);
230  output.type_update.setScore("human", type_positive_score_);
231 
232  }else if (!is_human && classification_error > 0){
233  // not classified as human but matching was possible
234  result.setValue("score", type_negative_score_);
235 
236 // output.type_update.setScore("crowd", type_negative_score_);
237  output.type_update.setScore("human", type_negative_score_);
238 
239  }else if (!is_human && classification_error == 0){
240  // not classified as human and matching was not possible
241  result.setValue("score", 0);
242 
243 // output.type_update.setScore("crowd", type_negative_score_);
244  output.type_update.setScore("human", type_negative_score_);
245  }
246 
247 
248  output.type_update.setUnknownScore(type_unknown_score_);
249 
250  result.endGroup(); // close human_contour_matcher group
251  result.endGroup(); // close perception_result group
252 }
