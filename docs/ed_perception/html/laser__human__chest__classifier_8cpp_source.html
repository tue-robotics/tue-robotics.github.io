126  // ---------- Prepare measurement ----------

130 // double t_start = ros::Time::now().toSec();

141  for (int i = 0; i < num_readings; i++) {

144  else ranges[i] = msg->ranges[i];
145  angles[i] = msg->angle_min + i * msg->angle_increment;
146 
147  }

150 // ROS_DEBUG("Looping over all segments takes %f [ms]", 1000*(ros::Time::now().toSec()-t0));

155  people_detector_.segmentScan(people_detector_threshold_, num_readings, angles, ranges, list_segments);

173  {
174

177 
178  // CHECK 1: number of laser points
179  if ( s->num() < min_n_points_ )
180  {
181  s->type = -1;
182  index_laser_point += s->num();
183  }
184  else
185  {
186  // Calculate size of the segment (exclude possible shadow points)
187  int i_min = index_laser_point;
188  int i_max = index_laser_point + s->num();
189  double x_min = ranges[i_min]*cos(angles[i_min]);
190  double x_max = ranges[i_max]*cos(angles[i_max]);
191  double y_min = ranges[i_min]*sin(angles[i_min]);
192  double y_max = ranges[i_max]*sin(angles[i_max]);
193  double size_segment = sqrt((x_max-x_min)*(x_max-x_min) + (y_max-y_min)*(y_max-y_min));
194 
195  // Determine position of the segment
196  double x_pos = 0, y_pos = 0;
197  for (int out = 0; out < s->num(); out++) {
198  if (out == static_cast<int>(s->num()/2)) {

200  y_pos = ranges[index_laser_point]*sin(angles[index_laser_point]);
201  }
202  ++index_laser_point;
203  }
204 
205  // CHECK 2: size of the segment
206  if (size_segment > min_segment_size_){
207 
208  // Classify current segment

213  {
