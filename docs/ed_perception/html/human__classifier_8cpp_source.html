243 bool HumanClassifier::PerfectMatch(const cv::Mat& mask,
244  cv::Mat &map_dt,
245  std::vector<cv::Point> template_pts,
246  cv::Point3i& best_pos,
247  cv::Point3i& start_pos,
248  float& best_error,
249  float& variance,
250  float& deviation) const{
251 
252  float err; // error value
253  int occupancy_best;
254  int occupancy;
255  int start_col;
256  cv::Mat grad_x; // sobel gradient in X
257  cv::Mat grad_y; // sobel gradient in Y
258  cv::Vec3f curr_grad; // gradient value
259  cv::Vec3f pos; // position to test
260  cv::Vec3f prev_grad; // old gradient value
261  cv::Vec3f step_width; // (jump distance in X, jump distance in Y)
262  cv::Vec3f upper_limits; // limits for the position tests
263  cv::Vec3f lower_limits; // limits for the position tests
264  std::vector<float> err_values;
265  std::vector<float> curr_err_values;
266  cv::Rect templtBox;
267  cv::Rect colCounter;
268  cv::Scalar meanErr;
269  cv::Scalar devErr;
270 
271  templtBox = boundingRect(template_pts);
272 
273  // discard regions that are too small for the given template
274  if (map_dt.rows - 2*border_size_ < templtBox.height || map_dt.cols - 2*border_size_ < templtBox.width) {
275  best_pos = cv::Point3i(0,0,0);
276  start_pos = cv::Point3i(0,0,0);
277  best_error = 0;
278  variance = 0;
279  deviation = 0;
280  return false;
281  }
282 
283  // Initializations
284  prev_grad = cv::Vec3f(0.0, 0.0, 0.0);
285  step_width = cv::Vec3f(1.0, 1.0, 0.05);
286  upper_limits = cv::Vec3f(map_dt.cols,map_dt.rows, M_PI);
287  lower_limits = cv::Vec3f(0, 0, -M_PI);
288  err = best_error = std::numeric_limits<float>::max(); // initial error
289  occupancy_best = 0;
290  start_col = 0;
291  colCounter = cv::Rect(0,0, mask.cols / num_slices_matching_, mask.rows);
292  variance = 0;
293 
294  // calculate best start position, divide the regions in vertical sections and choose the most occupied
295  for (int i = 0; i < num_slices_matching_; i++) {
296  // count number of non-black pixels of the current region
297  occupancy = cv::countNonZero( mask(cv::Rect((mask.cols / num_slices_matching_) * i, 0, (mask.cols / num_slices_matching_), mask.rows)));
298 
299  if (occupancy > occupancy_best) {
300  occupancy_best = occupancy;
301  // zone = midle of the zone - offset of half the template
302  start_col = ClipInt(floor(((mask.cols / num_slices_matching_) * i) - (templtBox.width/2)), 0, mask.cols);
303  }
304  }
305 
306  // set intial position, add borderSize or remove it for original location
307  pos = cv::Vec3f(start_col + border_size_, border_size_, 0.0);
308  start_pos = cv::Point3i(start_col + border_size_, border_size_, 0.0);
309 
310  // Gradient in X
311  Sobel(map_dt, grad_x, CV_32FC1, 1, 0, 3, 1, 0, cv::BORDER_DEFAULT);
312  // Gradient in Y
313  Sobel(map_dt, grad_y, CV_32FC1, 0, 1, 3, 1, 0, cv::BORDER_DEFAULT);
314 
315 
316  //---------------------------------------------------------
317 
318 
319  // RPROP algorithm
320  for (int i = 0; i < match_iterations_; i++) {
321  curr_err_values.clear();
322  err = ErrorFunction(grad_x, grad_y, map_dt, template_pts, pos, curr_grad, curr_err_values);
323 
324  // paint the path taken by the template in the map, point(0,0) should be close to "white"
325  if (debug_mode_){
326  map_dt.at<float>(cv::Point(pos[0], pos[1])) = map_dt.at<float>(0,0);
327  }
328 
329  // sometimes it happened, now it appears to be fixed, but just in case
330  if (err < 0)
331  std::cout << "[" << module_name_ << "] " << "Incorrect value for error (" << err << ")\n" << std::endl;
332 
333  // update best error
334  if (err < best_error && err > 0) {
335  err_values.clear();
336 
337  // backup error values for the best match, to calculate mean and deviation later
338  for (uint x = 0; x < curr_err_values.size(); x++)
339  err_values.push_back(curr_err_values[x]);
340 
341  // backup best error
342  best_error = err;
343 
344  // update the pos to go from mapDT location to mask location
345  best_pos = cv::Point3i(floor(pos[0]), floor(pos[1]), floor((pos[2]*180.0)/M_PI));
346  }
347 
348  // move template in x, y, and teta (rotation)
349  for (uint j = 0; j < 3; j++) {
350  // make updates for each parameter
351  if (curr_grad[j] == 0)
352  prev_grad[j] = 0;
353  else {
354  // Increment adjustment
355  if (curr_grad[j] * prev_grad[j] > 0)
356  step_width[j] *= 1.2;
357  else if (curr_grad[j] * prev_grad[j] < 0)
358  step_width[j] *= 0.5;
359 
360  // backup last gradient result
361  prev_grad[j] = curr_grad[j];
362 
363  // Adjustment of the parameters if it doesn't move the template outside the image
364  if (curr_grad[j] > 0 && (pos[j] - step_width[j]) > lower_limits[j])
365  pos[j] -= step_width[j];
366  else if (curr_grad[j] < 0 && (pos[j] + step_width[j]) < upper_limits[j])
367  pos[j] += step_width[j];
368  }
369  }
370 
371  // fitting is finished when the templates barelly moves in any direction
372  if (step_width[0] < 0.3 && step_width[1] < 0.3 && step_width[2] < 0.2){
373  break;
374  }
375  }
376 
377  // get mean error and deviation
378  meanStdDev(err_values, meanErr, devErr);
379 
380  // calculate error variance
381  for (uint i = 0; i < err_values.size(); ++i){
382  variance += pow(err_values[i] - meanErr[0], 2);
383  }
384 
385  variance /= err_values.size();
386  deviation = devErr[0];
387  best_error = meanErr[0];
388 
389  return true;
390 }

393 float HumanClassifier::ErrorFunction(const cv::Mat& grad_x,
394  const cv::Mat& grad_y,
395  const cv::Mat& distance_transf,
396  std::vector<cv::Point>& template_pts,
397  cv::Vec3f& anchor,
398  cv::Vec3f& grad,
399  std::vector<float>& err_values) const{
400 
401  cv::Point pos;
402  float err = 0;
403  float gx;
404  float gy;
405  float d;
406  std::vector<cv::Point> pos_tested;
407  float anchor_sin = sin(anchor[2]);
408  float anchor_cos = cos(anchor[2]);
409 
410  // reset the gradient value
411  grad = cv::Vec3f(0.0, 0.0, 0.0);
412 
413  // test all template points
414  for (uint i = 0; i < template_pts.size(); i++){
415 
416  // set testing position, with transformation (rotation)
417  pos.x = ClipInt(floor(anchor[0] + (template_pts[i].x * anchor_cos - template_pts[i].y * anchor_sin) ),
418  0, distance_transf.cols-1);
419 
420  pos.y = ClipInt(floor(anchor[1] + (template_pts[i].x * anchor_sin + template_pts[i].y * anchor_cos) ),
421  0 , distance_transf.rows-1);
422 
423  // avoid points that have already been tested
424  if (find(pos_tested.begin(), pos_tested.end(), pos) != pos_tested.end()) {
425  //ROS_INFO("Already tested (%d, %d)", pos.x, pos.y);
426  } else {
427  gx = grad_x.at<float>(pos);
428  gy = grad_y.at<float>(pos);
429  d = distance_transf.at<float>(pos) * distance_transf.at<float>(pos);
430 
431  grad[0] += gx; // gradient in X
432  grad[1] += gy; // gradient in Y
433  grad[2] += gx * (-template_pts[i].x * anchor_sin - template_pts[i].y * anchor_cos)
434  + gy * (template_pts[i].x * anchor_cos + template_pts[i].y * anchor_sin);
435 
436  err += d; // error for this position, or "distance"
437 
438  err_values.push_back(d);
439  pos_tested.push_back(pos);
440  }
441 
442  // a warning just in case..
443  if (pos.x > distance_transf.cols || pos.y > distance_transf.rows || pos.x < 0 || pos.y < 0)
444  std::cout << "[" << module_name_ << "] " << "Position outside of the map ("<< pos.x << ", " << pos.y <<")" << std::endl;
445  }
446 
447  return err;
448 }

554 float HumanClassifier::ResizeTemplate(std::vector<cv::Point> template_src,
555  float depth,
556  float alpha,
557  TemplateType type,
558  std::vector<cv::Point>& template_dst) const{
559 
560  float scaleFactor;
561 
562  if (depth <= 0)
563  std::cout << "[" << module_name_ << "] " << "Incorrect depth value = "<< depth << std::endl;
564 
565  // scale_factor = -15,763 X + 106,33 (linear)
566  // scale_factor = -52.6 * log(X) + 114.06 (logarithmic)
567  // scale_factor = 308 * X^(-1.018)
568  // scaleFactor = (308 * pow(depth, -1.018)) + alpha;
569 
570  // use different scaling functions for different template types
571  if (type == FaceFront)
