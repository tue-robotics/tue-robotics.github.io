40 void ColorMatcher::classify(const ed::Entity& e, const std::string& property,
41  const ed::perception::CategoricalDistribution& prior,
42  ed::perception::ClassificationOutput& output) const
43 {
44  tue::Configuration& result = output.data;
45 
46  ColorHistogram color_histogram;
47  calculateHistogram(e, color_histogram);
48 
49  if (property == "color")
50  {
51  for(unsigned int i = 0; i < ColorNameTable::NUM_COLORS; ++i)
52  output.likelihood.setScore(color_table_.intToColorName(i), color_histogram[i]);
53  output.likelihood.setUnknownScore(0);
54  return;
55  }
56 
57  for(std::map<std::string, ColorModel>::const_iterator it = models_.begin(); it != models_.end(); ++it)
58  {
59  const std::string& model_name = it->first;
60  const ColorModel& model = it->second;
61 
62  output.likelihood.setScore(model_name, 1);
63  for(unsigned int i = 0; i < ColorNameTable::NUM_COLORS; ++i)
64  {
65  float v = color_histogram[i];
66  if (v < model.min[i] - color_margin_ || v > model.max[i] + color_margin_)
67  output.likelihood.setScore(model_name, 0);
68  }
69  }
70 
71  // Represent data (for debugging)
72  result.writeArray("colors");
73  for(unsigned int i = 0; i < ColorNameTable::NUM_COLORS; ++i)
74  {
75  result.addArrayItem();
76  result.setValue("color", ColorNameTable::intToColorName(i));
77  result.setValue("value", color_histogram[i]);
78  result.endArrayItem();
79  }

81 }

85 void ColorMatcher::addTrainingInstance(const ed::Entity& e, const std::string& property, const std::string& value)
86 {
87  if (property != "type")
88  return;
89 
90  ColorHistogram color_histogram;
91  calculateHistogram(e, color_histogram);
92 
93  std::map<std::string, ColorModel>::iterator it = models_.find(value);
94  if (it == models_.end())
95  {
96  // No other training instances for this type
97  ColorModel& model = models_[value];
98  model.min = color_histogram;
99  model.max = color_histogram;
100  }
101  else
102  {
103  ColorModel& model = it->second;
104 
105  // Determine maximum for each histogram bin (i.e., determine maximum for each color)
106  for(unsigned int i = 0; i < ColorNameTable::NUM_COLORS; ++i)
107  {
108  model.min[i] = std::min(model.min[i], color_histogram[i]);
109  model.max[i] = std::max(model.max[i], color_histogram[i]);
110  }
111  }

191 void ColorMatcher::calculateHistogram(const ed::Entity& e, ColorHistogram& histogram) const
192 {
193  ed::MeasurementConstPtr msr = e.lastMeasurement();
194  if (!msr)
195  return;
196 
197  histogram.resize(ColorNameTable::NUM_COLORS);
198  histogram.assign(ColorNameTable::NUM_COLORS, 0);
199 
200  // get color image
201  const cv::Mat& img = msr->image()->getRGBImage();
202 
203  int pixel_count = 0;
204 
205  for(ed::ImageMask::const_iterator it = msr->imageMask().begin(img.cols); it != msr->imageMask().end(); ++it)
206  {
207  ++pixel_count;
208  const cv::Point2i p(it());
209 
210  // Calculate prob distribution
211  const cv::Vec3b& bgr = img.at<cv::Vec3b>(p);
212 
213  const float* probs = color_table_.rgbToDistribution(bgr[2], bgr[1], bgr[0]);
214 
215  for(unsigned int i = 0; i < ColorNameTable::NUM_COLORS; ++i)
216  histogram[i] += probs[i];
217  }
218 
219  // normalize histogram
220  for(unsigned int i = 0; i < ColorNameTable::NUM_COLORS; ++i)
221  histogram[i] /= pixel_count;
222 }
