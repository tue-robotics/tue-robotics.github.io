187  DWAPlanner::DWAPlanner(std::string name, base_local_planner::LocalPlannerUtil *planner_util) :
188  planner_util_(planner_util),
189  occ_vel_costs_(planner_util->getCostmap()),
190  plan_costs_(planner_util->getCostmap()),
191  goal_costs_(planner_util->getCostmap()),
192  obstacle_costs_(planner_util->getCostmap()),
193  vis_(planner_util->getCostmap(), goal_costs_, plan_costs_, planner_util->getGlobalFrame())
194  {
195  // set up all the cost functions that will be applied in order
196  // (any function returning negative values will abort scoring, so the order can improve performance)
197  std::vector<base_local_planner::TrajectoryCostFunction*> critics;
198  critics.push_back(&goal_costs_);
199  critics.push_back(&occ_vel_costs_);
200  critics.push_back(&plan_costs_);
201  critics.push_back(&alignment_costs_);
202  critics.push_back(&cmd_vel_costs_);
203  critics.push_back(&obstacle_costs_);
204 
205  // trajectory generators
206  std::vector<base_local_planner::TrajectorySampleGenerator*> generator_list;
207  generator_list.push_back(&generator_);
208 
209  scored_sampling_planner_ = base_local_planner::SimpleScoredSamplingPlanner(generator_list, critics);
210  }

212  void DWAPlanner::updatePlanAndLocalCosts(
213  const geometry_msgs::PoseStamped& robot_pose,
214  const std::vector<geometry_msgs::PoseStamped>& local_plan,
215  double lookahead,
216  const std::vector<geometry_msgs::Point>& footprint_spec) {
217 
218  obstacle_costs_.setFootprint(footprint_spec);
219 
220  // Determine the errors
221  double yaw_error = base_local_planner::getGoalOrientationAngleDifference(robot_pose, tf2::getYaw(local_plan.front().pose.orientation));
222  double plan_distance = base_local_planner::getGoalPositionDistance(robot_pose, local_plan.front().pose.position.x, local_plan.front().pose.position.y);
223  double goal_distance = base_local_planner::getGoalPositionDistance(robot_pose, local_plan.back().pose.position.x, local_plan.back().pose.position.y);
224 
225  // Determine state of the controller
226  LocalPlannerState state = determineState(robot_pose, yaw_error, plan_distance, goal_distance);
227 
228  // Update the cost functions depending on the state we are in
229  switch (state)
230  {
231  case NotMoving:
232  // Almost similar to Align (see below). Difference in the desired orientation
233  alignment_costs_.setScale(align_align_scale_);
234  plan_costs_.setScale(align_plan_scale_);
235  goal_costs_.setScale(align_goal_scale_);
236 
237  alignment_costs_.setDesiredOrientation(tf2::getYaw(local_plan.back().pose.orientation));
238 
239  cmd_vel_costs_.setCoefficients(align_cmd_px_, align_cmd_nx_, align_cmd_py_, align_cmd_ny_, align_cmd_pth_, align_cmd_nth_);
240 
241  obstacle_costs_.setScale(align_obstacle_scale_);
242 
243  break;
244 
245  case Align:
246  alignment_costs_.setScale(align_align_scale_);
247  plan_costs_.setScale(align_plan_scale_);
248  goal_costs_.setScale(align_goal_scale_);
249 
250  alignment_costs_.setDesiredOrientation(tf2::getYaw(local_plan.front().pose.orientation));
251 
252  cmd_vel_costs_.setCoefficients(align_cmd_px_, align_cmd_nx_, align_cmd_py_, align_cmd_ny_, align_cmd_pth_, align_cmd_nth_);
253 
254  obstacle_costs_.setScale(align_obstacle_scale_);
255 
256  break;
257 
258  case Default:
259  alignment_costs_.setScale(default_align_scale_);
260  plan_costs_.setScale(default_plan_scale_);
261  goal_costs_.setScale(default_goal_scale_);
262 
263  alignment_costs_.setDesiredOrientation(tf2::getYaw(local_plan.front().pose.orientation));
264 
265  cmd_vel_costs_.setCoefficients(default_cmd_px_, default_cmd_nx_, default_cmd_py_, default_cmd_ny_, default_cmd_pth_, default_cmd_nth_);
266 
267  obstacle_costs_.setScale(default_obstacle_scale_);
268 
269  break;
270 
271  case Arrive:
272  alignment_costs_.setScale(arrive_align_scale_);
273  plan_costs_.setScale(arrive_plan_scale_);
274  goal_costs_.setScale(arrive_goal_scale_);
275 
276  alignment_costs_.setDesiredOrientation(tf2::getYaw(local_plan.back().pose.orientation));
277 
278  cmd_vel_costs_.setCoefficients(arrive_cmd_px_, arrive_cmd_nx_, arrive_cmd_py_, arrive_cmd_ny_, arrive_cmd_pth_, arrive_cmd_nth_);
279 
280  obstacle_costs_.setScale(arrive_obstacle_scale_);
281 
282  break;
283  }
284 
285  // Optimization data (Set local plan)
286  std::vector<geometry_msgs::PoseStamped> local_plan_from_lookahead;
287  base_local_planner::planFromLookahead(local_plan, lookahead, local_plan_from_lookahead);
288 
289  goal_costs_.setTargetPoses(local_plan_from_lookahead);
290  plan_costs_.setTargetPoses(local_plan);
291 
292  // Update footprint if changed
293  occ_vel_costs_.setFootprint(footprint_spec);
294  }

300  base_local_planner::Trajectory DWAPlanner::findBestPath(
301  const geometry_msgs::PoseStamped& global_pose,
302  const geometry_msgs::PoseStamped& global_vel,
303  const geometry_msgs::PoseStamped& goal_pose) {
304 
305  //make sure that our configuration doesn't change mid-run
306  boost::mutex::scoped_lock l(configuration_mutex_);
307 
308  Eigen::Vector3f pos(global_pose.pose.position.x, global_pose.pose.position.y, tf2::getYaw(global_pose.pose.orientation));
309  Eigen::Vector3f vel(global_vel.pose.position.x, global_vel.pose.position.y, tf2::getYaw(global_vel.pose.orientation));
310  Eigen::Vector3f goal(goal_pose.pose.position.x, goal_pose.pose.position.y, tf2::getYaw(goal_pose.pose.orientation));
311  base_local_planner::LocalPlannerLimits limits = planner_util_->getCurrentLimits();
312 
313  // prepare cost functions and generators for this run
314  generator_.initialise(pos,
315  vel,
316  goal,
317  &limits,
318  vsamples_);
319 
320  base_local_planner::Trajectory result_traj;
321  result_traj.cost_ = -7;
322  // find best trajectory by sampling and scoring the samples
323  std::vector<base_local_planner::Trajectory> all_explored;
324  scored_sampling_planner_.findBestTrajectory(result_traj, &all_explored);
325 
326  vis_.publishDesiredOrientation(alignment_costs_.getDesiredOrientation(), global_pose);
327  vis_.publishCostGrid();
328  vis_.publishTrajectoryCloud(all_explored);
329 
330  return result_traj;
331  }

333  LocalPlannerState DWAPlanner::determineState(const geometry_msgs::PoseStamped& robot_pose, double yaw_error, double /*plan_distance*/, double goal_distance) {
334  static LocalPlannerState prev_state = None;
335  LocalPlannerState state = Default;
336 
337  // todo: optional path_distance state
338 
339  // ToDo: can we make this more generic???
340  if (!isMoving(robot_pose)) // ToDo: prevent switching behavior???
341  state = NotMoving;
342  else if (goal_distance < switch_goal_distance_)
343  state = Arrive;
344  else if (fabs(yaw_error) > switch_yaw_error_ || ( prev_state == Align && fabs(yaw_error) > (switch_yaw_error_/2) ) )
345  state = Align;
346  else
347  state = Default;
348 
349  // To print state
350  if (prev_state != state)
351  {
352  ROS_INFO_STREAM("State = " << StateName[state]);
353  prev_state = state;
354  }
355 
356  return state;
357  }

359  bool DWAPlanner::isMoving(const geometry_msgs::PoseStamped& robot_pose) {
360  static double x_saved = 0.0;
361  static double y_saved = 0.0;
362  static bool prev_is_moving = true;
363  static ros::Time stamp_not_moving = ros::Time::now(); // Time stamp to indicate when the robot went to 'not moving'
364 
365  double x = robot_pose.pose.position.x;
366  double y = robot_pose.pose.position.y;
367 
368  double dist_sq = (x_saved - x) * (x_saved - x) + (y_saved - y) * (y_saved - y);
369 
370  double time_since_move = (ros::Time::now() - stamp_last_motion_).toSec();
371  double time_since_stop = (ros::Time::now() - stamp_not_moving).toSec();
372 
373  bool is_moving = false;
374 
375  // Check whether we're actually moving
376  if (dist_sq > not_moving_distance_) {
377  resetMotionStamp();
378  x_saved = x;
379  y_saved = y;
380  is_moving = true;
381  }
382 
383  if (!prev_is_moving && time_since_stop < not_moving_minimal_duration_) {
384  // Configure the minimal 'notMoving' duration to avoid switching behavior
385  return false;
386  }
387  else if (is_moving) {
388  prev_is_moving = true;
389  return true;
390  }
391  else if ( time_since_move > not_moving_time_window_) {
392  if (prev_is_moving)
393  {
394  ROS_WARN("Robot has not moved significantly for more than %f seconds", not_moving_time_window_);
395  stamp_not_moving = ros::Time::now();
396  prev_is_moving = false;
397  }
398  return false;
399  }
400  else {
401  return true;
402  }
403  }
