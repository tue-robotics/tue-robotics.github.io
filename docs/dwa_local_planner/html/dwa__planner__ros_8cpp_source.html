57  void setCmdVel(const base_local_planner::Trajectory& traj,
58  const geometry_msgs::PoseStamped& robot_vel,
59  const base_local_planner::LocalPlannerLimits& limits,
60  double sim_period,
61  geometry_msgs::Twist& cmd_vel)
62  {
63  Eigen::Vector3f vel(robot_vel.pose.position.x, robot_vel.pose.position.y, tf2::getYaw(robot_vel.pose.orientation));
64 
65  double vx = traj.xv_;
66  double vy = traj.yv_;
67  double vth = traj.thetav_;
68 
69  if (traj.cost_ < 0)
70  {
71  double v0 = fabs(vel[0]);
72  double v1 = fabs(vel[1]);
73  double v0frac = (v0/(v0+v1));
74  double v1frac = (v1/(v0+v1));
75 
76  if (vel[0] > 0) vx = std::max(0.0, vel[0] - v0frac * limits.deacc_lim_trans * sim_period);
77  if (vel[0] < 0) vx = std::min(0.0, vel[0] + v0frac * limits.deacc_lim_trans * sim_period);
78 
79  if (vel[1] > 0) vy = std::max(0.0, vel[1] - v1frac * limits.deacc_lim_trans * sim_period);
80  if (vel[1] < 0) vy = std::min(0.0, vel[1] + v1frac * limits.deacc_lim_trans * sim_period);
81 
82  if (vel[2] > 0) vth = std::max(0.0, vel[2] - limits.acc_lim_theta * sim_period);
83  if (vel[2] < 0) vth = std::min(0.0, vel[2] + limits.acc_lim_theta * sim_period);
84 
85  ROS_WARN_STREAM_DELAYED_THROTTLE(10, "DWA Planner discarded all trajectories. Enable ROS_DEBUG for more information.");
86  ROS_DEBUG_STREAM("DWA PLANNER DISCARDED ALL TRAJECTORIES, DEACCELERATING WITH MAX; COST: " << traj.cost_ << "\n"
87  << " vx: " << vel[0] << " --> " << vx << "-- frac: " << v0frac << "\n"
88  << " vy: " << vel[1] << " --> " << vy << "-- frac: " << v1frac << "\n"
89  << " vth: " << vel[2] << " --> " << vth << "\n"
90  << " deacc_lim_trans: " << limits.deacc_lim_trans);
91  }
92 
93  cmd_vel.linear.x = vx;
94  cmd_vel.linear.y = vy;
95  cmd_vel.angular.z = vth;
96  }

98  void DWAPlannerROS::reconfigureCB(DWAPlannerConfig &config, uint32_t level) {
99  if (setup_ && config.restore_defaults) {
100  config = default_config_;
101  config.restore_defaults = false;
102  }
103  if ( ! setup_) {
104  default_config_ = config;
105  setup_ = true;
106  }
107 
108  // update generic local planner params
109  base_local_planner::LocalPlannerLimits limits;
110  limits.max_vel_trans = config.max_vel_trans;
111  limits.min_vel_trans = config.min_vel_trans;
112  limits.max_vel_x = config.max_vel_x;
113  limits.min_vel_x = config.min_vel_x;
114  limits.max_vel_y = config.max_vel_y;
115  limits.min_vel_y = config.min_vel_y;
116  limits.max_vel_theta = config.max_vel_theta;
117  limits.min_vel_theta = config.min_vel_theta;
118  limits.acc_lim_x = config.acc_lim_x;
119  limits.acc_lim_y = config.acc_lim_y;
120  limits.acc_lim_theta = config.acc_lim_theta;
121  limits.acc_lim_trans = config.acc_lim_trans;
122  limits.deacc_lim_trans = config.deacc_lim_trans;
123  limits.xy_goal_tolerance = config.xy_goal_tolerance;
124  limits.yaw_goal_tolerance = config.yaw_goal_tolerance;
125  limits.prune_plan = config.prune_plan;
126  limits.trans_stopped_vel = config.trans_stopped_vel;
127  limits.theta_stopped_vel = config.theta_stopped_vel;
128  limits.min_lookahead_distance = 1;
129  planner_util_.reconfigureCB(limits, config.restore_defaults);
130 
131  // update dwa specific configuration
132  dp_->reconfigure(config);
133  }

140  void DWAPlannerROS::initialize(
141  std::string name,
142  tf2_ros::Buffer* tf,
143  costmap_2d::Costmap2DROS* costmap_ros) {
144  if (! isInitialized()) {
145 
146  ros::NodeHandle private_nh("~/" + name);
147  l_plan_pub_ = private_nh.advertise<nav_msgs::Path>("local_plan", 1);
148  l_traj_pub_ = private_nh.advertise<nav_msgs::Path>("local_traj", 1);
149  costmap_ros_ = costmap_ros;
150 
151  // make sure to update the costmap we'll use for this cycle
152  costmap_2d::Costmap2D* costmap = costmap_ros_->getCostmap();
153 
154  planner_util_.initialize(tf, costmap, costmap_ros_->getGlobalFrameID());
155 
156  //create the actual planner that we'll use.. it'll configure itself from the parameter server
157  dp_ = boost::shared_ptr<DWAPlanner>(new DWAPlanner(name, &planner_util_));
158 
159  initialized_ = true;
160 
161  // Warn about deprecated parameters -- remove this block in N-turtle
162  nav_core::warnRenamedParameter(private_nh, "max_vel_trans", "max_trans_vel");
163  nav_core::warnRenamedParameter(private_nh, "min_vel_trans", "min_trans_vel");
164  nav_core::warnRenamedParameter(private_nh, "max_vel_theta", "max_rot_vel");
165  nav_core::warnRenamedParameter(private_nh, "min_vel_theta", "min_rot_vel");
166  nav_core::warnRenamedParameter(private_nh, "acc_lim_trans", "acc_limit_trans");
167  nav_core::warnRenamedParameter(private_nh, "theta_stopped_vel", "rot_stopped_vel");
168 
169  dsrv_ = new dynamic_reconfigure::Server<DWAPlannerConfig>(private_nh);
170  dynamic_reconfigure::Server<DWAPlannerConfig>::CallbackType cb = [this](auto& config, auto level){ reconfigureCB(config, level); };
171  dsrv_->setCallback(cb);
172  }
173  else{
174  ROS_WARN("This planner has already been initialized, doing nothing.");
175  }

189  bool DWAPlannerROS::isGoalReached() {
190  if (! isInitialized()) {
191  ROS_ERROR("This planner has not been initialized, please call initialize() before using this planner");
192  return false;
193  }
194  geometry_msgs::PoseStamped robot_pose, robot_vel;
195  if (!getRobotState(robot_pose, robot_vel))
196  return false;
197 
198  base_local_planner::LocalPlannerLimits l = planner_util_.getCurrentLimits();
199  geometry_msgs::PoseStamped goal_pose;
200  if (!planner_util_.getGoal(goal_pose))
201  return false;
202 
203  double xy_to_goal = base_local_planner::getGoalPositionDistance(robot_pose, goal_pose.pose.position.x, goal_pose.pose.position.y);
204  double angle_to_goal = base_local_planner::getGoalOrientationAngleDifference(robot_pose, tf2::getYaw(goal_pose.pose.orientation));
205  bool stopped = base_local_planner::stopped(robot_vel, l.theta_stopped_vel, l.trans_stopped_vel);
206 
207  return (xy_to_goal <= l.xy_goal_tolerance && fabs(angle_to_goal) <= l.yaw_goal_tolerance && stopped);
208  }

215  bool DWAPlannerROS::computeVelocityCommands(geometry_msgs::Twist& cmd_vel) {
216  std::vector<geometry_msgs::PoseStamped> local_plan;
217  geometry_msgs::PoseStamped robot_pose, robot_vel, goal_pose;
218 
219  base_local_planner::Trajectory traj;
220  if (getRobotStateAndLocalPlan(robot_pose, robot_vel, local_plan))
221  {
222  // Calculate lookahead
223  double trans_vel = hypot(robot_vel.pose.position.x, robot_vel.pose.position.y);
224  double lookahead = dp_->getSimTime() * (trans_vel + planner_util_.getCurrentLimits().acc_lim_trans * dp_->getSimPeriod());
225  lookahead = std::max(planner_util_.getCurrentLimits().min_lookahead_distance, lookahead);
226 
227  // Publish the local plan
228  base_local_planner::publishPlan(local_plan, l_plan_pub_);
229 
230  // Get the local goal
231  goal_pose = local_plan.back();
232 
233  // update plan in dwa planner to calculate cost grid
234  dp_->updatePlanAndLocalCosts(robot_pose, local_plan, lookahead, costmap_ros_->getRobotFootprint());
235 
236  // call with updated footprint
237  traj = dp_->findBestPath(robot_pose, robot_vel, goal_pose);
238  }
239 
240  // Set the command velocity
241  setCmdVel(traj, robot_vel, planner_util_.getCurrentLimits(), dp_->getSimPeriod(), cmd_vel);
242 
243  // Publish the local plan
244  publishTrajectory(traj);
245 
246  return traj.cost_ >= 0;
247  }

281  void DWAPlannerROS::publishTrajectory(const base_local_planner::Trajectory& traj) {
282  std::vector<geometry_msgs::PoseStamped> local_plan;
283  if (traj.cost_ >= 0) {
284  // Fill out the local plan
285  for (unsigned int i = 0; i < traj.getPointsSize(); ++i) {
286  double p_x, p_y, p_th;
287  traj.getPoint(i, p_x, p_y, p_th);
288  geometry_msgs::PoseStamped pose;
289  pose.pose.position.x = p_x;
290  pose.pose.position.y = p_y;
291  tf2::Quaternion q;
292  q.setRPY(0, 0, p_th);
293  pose.pose.orientation = tf2::toMsg(q);
294  pose.header.stamp = ros::Time::now();
295  pose.header.frame_id = costmap_ros_->getGlobalFrameID();
296  local_plan.push_back(pose);
297  }
298  }
299  base_local_planner::publishPlan(local_plan, l_traj_pub_);
300  }
