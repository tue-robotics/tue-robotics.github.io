193 
194  if (goal_handles_.size() > 0)
195  {
196  // Take the highest priority goal
197  HeadReferenceActionServer::GoalHandle gh = goal_handles_.front();
198  goal = *gh.getGoal();
199 
200  if (goal.goal_type == head_ref_msgs::HeadReferenceGoal::LOOKAT || goal.goal_type == head_ref_msgs::HeadReferenceGoal::LOOKAT_AND_FREEZE) {
201  // Update pan and tilt

204  if (!targetToPanTilt(goal.target_point, pan_goal, tilt_goal))
205  {

207  goal_handles_.erase (goal_handles_.begin(), goal_handles_.begin()+1);
208  return;
209  }
210  else
211  {
212  goal.pan = limitReferences(pan_joint_props_, pan_goal * pan_joint_props_.direction);
213  goal.tilt = limitReferences(tilt_joint_props_, tilt_goal * tilt_joint_props_.direction);
214  }
215  publishMarker(goal.target_point);
216  }
217  else if (goal.goal_type == head_ref_msgs::HeadReferenceGoal::PAN_TILT)
218  {
219  goal.pan = limitReferences(pan_joint_props_, goal.pan * pan_joint_props_.direction);
220  goal.tilt = limitReferences(tilt_joint_props_, goal.tilt * tilt_joint_props_.direction);
221  }
222 
223  // Check whether we are there
224  head_ref_msgs::HeadReferenceFeedback fb;
225  double pan_error = goal.pan - current_pan_;
226  double tilt_error = goal.tilt - current_tilt_;
227  if (fabs(pan_error) < goal_error_tolerance_ && fabs(tilt_error) < goal_error_tolerance_)
228  {
229  fb.at_setpoint = true;
230  gh.publishFeedback(fb);
231  return;
232  }
233  else

235  ROS_DEBUG_STREAM_THROTTLE(1.0, "Pan error: " << pan_error << ", tilt error: " << tilt_error);
236  fb.at_setpoint = false;
237  gh.publishFeedback(fb);
238  }
239  }
240  else
241  {
242  goal.goal_type = head_ref_msgs::HeadReferenceGoal::PAN_TILT;
243  goal.pan = default_pan_;
244  goal.tilt = default_tilt_;
245  }
246 
247  ROS_DEBUG_THROTTLE(1.0, "Current head goal (pan/tilt): %.3f,%.3f",goal.pan,goal.tilt);
248 
249  // ROBOCUP HACK
250  if (goal.goal_type == head_ref_msgs::HeadReferenceGoal::LOOKAT_AND_FREEZE)
251  {
252  goal = lookat_and_freeze_goal_;
253  goal.goal_type = head_ref_msgs::HeadReferenceGoal::PAN_TILT;
254  goal.pan = limitReferences(pan_joint_props_, goal.pan * pan_joint_props_.direction);
255  goal.tilt = limitReferences(tilt_joint_props_, goal.tilt * tilt_joint_props_.direction);

306 bool HeadReference::targetToPanTilt(const geometry_msgs::PointStamped& target, double& pan, double& tilt)
307 {
308  tf2::Stamped<tf2::Vector3> target_MOUNT;
309  try
310  {
311  tf_buffer_.transform(target, target_MOUNT, frame_mount_);
312  }
313  catch(tf2::TransformException& ex)
314  {
315  ROS_ERROR("%s", ex.what());
316  return false;
317  }
318 
319  tf2::Stamped<tf2::Vector3> target_NECK;
320  try
321  {
322  tf_buffer_.transform(target, target_NECK, frame_neck_);
323  }
324  catch(tf2::TransformException& ex)
325  {
326  ROS_ERROR("%s", ex.what());
327  return false;
328  }
329 
330  double mount_to_neck;
331  try
332  {
333  geometry_msgs::TransformStamped transform;
334  transform = tf_buffer_.lookupTransform(frame_mount_, frame_neck_, ros::Time());
335  mount_to_neck = transform.transform.translation.x;
336  }
337  catch(tf2::TransformException& ex)
338  {
339  ROS_ERROR("%s", ex.what());
340  return false;
341  }
342 
343  double neck_to_head_vert;
344  try
345  {
346  geometry_msgs::TransformStamped transform;
347  transform = tf_buffer_.lookupTransform(frame_neck_, frame_head_, ros::Time());
348  neck_to_head_vert = transform.transform.translation.z;
349  }
350  catch(tf2::TransformException& ex)
351  {
352  ROS_ERROR("%s", ex.what());
353  return false;
354  }
355 
356  pan = -atan2(target_MOUNT.getY(), target_MOUNT.getZ());
357 
358  double neck_to_target = target_NECK.length();
359 
360  double target_to_neck_vert = target_MOUNT.getX() - mount_to_neck;
361 
362  double head_mount_to_target_flat = sqrt(target_MOUNT.getY() * target_MOUNT.getY() + target_MOUNT.getZ() * target_MOUNT.getZ());
363 
364  double tilt_basic = -atan(target_to_neck_vert / head_mount_to_target_flat);
365  double tilt_camera_offset_correction = asin(neck_to_head_vert / neck_to_target);
366 
367  tilt = tilt_basic + tilt_camera_offset_correction;
368 
369  return true;
370 }

392 bool HeadReference::getJointInfo(const urdf::Model &model, JointProps &props)
393 {
394  // Get joint info from urdf model
395  urdf::JointConstSharedPtr joint_ptr = model.getJoint(props.name);
396  if (joint_ptr == nullptr)
397  {
398  ROS_ERROR_STREAM("Could not get joint '" << props.name << "' from the urdf model");
399  return false;
400  }
401 
402  // Get joint limits from joint info
403  props.lower = joint_ptr->limits->lower;
404  props.upper = joint_ptr->limits->upper;
405 
406  // Get direction from joint info
407  if (joint_ptr->axis.x > 0.99 || joint_ptr->axis.y > 0.99 || joint_ptr->axis.z > 0.99)
408  props.direction = 1.0;
409  else if (joint_ptr->axis.x < -0.99 || joint_ptr->axis.y < -0.99 || joint_ptr->axis.z < -0.99)
410  props.direction = -1.0;
411  else
412  {
413  ROS_ERROR_STREAM("Don't know how to define the direction of joint " << props.name
414  << " (this is not supported), "
415  << "axis of rotation should coincide with one of the principle axes (x, y, z, -x, -y, -z) of the joint");
416  return false;
417  }
418 
419  ROS_INFO_STREAM("Limits for " << props.name << ": "
420  << props.lower << ", " << props.upper
421  << ", direction: " << props.direction);
422 
423  // All is well
424  return true;
425 }
