21 void decomposePose(const geo::Pose3D& pose, geo::Pose3D& pose_xya, geo::Pose3D& pose_zrp)
22 {
23  tf::Matrix3x3 m;
24  geo::convert(pose.R, m);

26  double roll, pitch, yaw;
27  m.getRPY(roll, pitch, yaw);
28 
29  pose_xya.R.setRPY(0, 0, yaw);
30  pose_xya.t = geo::Vec3(pose.t.x, pose.t.y, 0);
31
