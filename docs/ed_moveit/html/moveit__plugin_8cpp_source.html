62 {
63  ROS_INFO("[ED MOVEIT] Generating moveit planning scene");
64  moveit_msgs::PlanningSceneWorld msg;
65  for(ed::WorldModel::const_iterator it = world_model_->begin(); it != world_model_->end(); ++it)
66  {
67  const ed::EntityConstPtr& e = *it;
68 
69  if (!e->has_pose() || !e->shape() || e->existenceProbability() < 0.95 || e->hasFlag("self") || e->id() == "floor")
70  continue;
71 
72  const geo::Mesh mesh = e->shape()->getMesh();
73 
74  shape_msgs::Mesh mesh_msg;
75  geo::convert(mesh, mesh_msg);
76 
77  moveit_msgs::CollisionObject object_msg;
78  object_msg.meshes.push_back(mesh_msg);
79 
80  //Pose is in /map frame. When publishing in own frame, pose can be zero.
81  geometry_msgs::Pose pose_msg;
82  geo::convert(e->pose(), pose_msg);
83  object_msg.mesh_poses.push_back(pose_msg);
84 
85  object_msg.operation = moveit_msgs::CollisionObject::ADD;
86  object_msg.id = e->id().str();
87  object_msg.header.frame_id = "/map";
88  object_msg.header.stamp = ros::Time::now();

90  }
91
