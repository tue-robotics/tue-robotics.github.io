36  def resolve(self):
37  """ Resolves
38 
39  :return: entity in the <area_description> of the <surface_designator> that is closest to the robot
40  """
41  # Get the surface as an entity
42  surface = self._surface_designator.resolve()
43  if surface is None:
44  rospy.logerror("Cannot resolve surface designator")
45  return None
46 
47  # Get all entities and check which ones are on the table
48  all_entities = self._robot.ed.get_entities()
49  entities = []
50 
51  rospy.loginfo("Gathering all entities in {vol} of surface {ent}".format(vol=self._area_description, ent=surface))
52  # import ipdb; ipdb.set_trace()
53  for e in all_entities:
54  point = robot_skills.util.kdl_conversions.VectorStamped(frame_id=e.frame_id, vector=e._pose.p)
55  if surface.in_volume(point=point, volume_id=self._area_description):
56  entities.append(e)
57  rospy.loginfo("{l} entities in {vol} of surface {ent}".format(l=len(entities), vol=self._area_description, ent=surface.id))
58 
59  # Remove all entities that are too large or too small
60  entities = [e for e in entities if (e.shape.z_max - e.shape.z_min) > MIN_GRAB_OBJECT_HEIGHT]
61  entities = [e for e in entities if (e.shape.y_max - e.shape.y_min) < MAX_GRAB_OBJECT_WIDTH]
62  entities = [e for e in entities if (e.shape.x_max - e.shape.x_min) < MAX_GRAB_OBJECT_WIDTH]
63  rospy.loginfo("Keeping {l} entities in {vol} of surface {ent} that are not too big".format(l=len(entities), vol=self._area_description, ent=surface.id))
64 
65  # Check if there are any
66  if not entities:
67  return None
68 
69  # Sort the entities and return the closest one
70  base_pose = self._robot.base.get_location().frame
71  entities = sorted(entities, key=lambda e: e.distance_to_2d(base_pose.p))
72  return entities[0]
