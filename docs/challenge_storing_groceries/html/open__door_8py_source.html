22 def _clamp(abs_value, value):
23  """
24  Clamps the value to be between -abs_value and +abs_value
25 
26  :param abs_value: limit of the value in both positive and negative direction
27  :param value: value to be clamped...
28  :return: The -abs_value when value is smaller than -abs_value OR +abs_value when value is more than +abs_value
29  """
30  return max(-abs_value, min(abs_value, value))
