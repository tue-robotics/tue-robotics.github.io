79 bool OccupancyGridPublisher::getMapData(const ed::WorldModel& world, std::vector<ed::EntityConstPtr>& entities_to_be_projected)
80 {
81  // default size
82  geo::Vector3 min(-min_map_size_x_/2,-min_map_size_y_/2,0);
83  geo::Vector3 max(min_map_size_x_/2,min_map_size_y_/2,0);
84 
85  for(ed::WorldModel::const_iterator it = world.begin(); it != world.end(); ++it)
86  {
87  const ed::EntityConstPtr& e = *it;
88  geo::ShapeConstPtr shape = e->collision();
89 
90  if (!e->has_pose() || !shape || e->existenceProbability() < 0.95 || e->hasFlag("self"))
91  continue;
92 
94  entities_to_be_projected.push_back(e);
95 
97  const std::vector<geo::Vector3>& vertices = shape->getMesh().getPoints();
98  for(std::vector<geo::Vector3>::const_iterator it2 = vertices.begin(); it2 != vertices.end(); ++it2)
99  {
100  geo::Vector3 p1w = e->pose() * (*it2);
101 
102  // Filter the ground
103  if (p1w.getZ() > min_z_)
104  {
105  min.x = std::min(p1w.x, min.x);
106  max.x = std::max(p1w.x, max.x);
107 
108  min.y = std::min(p1w.y, min.y);
109  max.y = std::max(p1w.y, max.y);
110  }
111  }
112  }
113 
114  // Bounds fix
115  min.x-=1.0;
116  min.y-=1.0;
117 
118  max.x+=1.0;
119  max.y+=1.0;
120 
122  map_.setOrigin(min);
123 
124  map_.setSizeAndClear((max.x - min.x) / map_.resolution(), (max.y - min.y) / map_.resolution());
125 
126  return (map_.width_in_cells() > 0 && map_.height_in_cells() > 0 && map_.width_in_cells() * map_.height_in_cells() < 100000000);
