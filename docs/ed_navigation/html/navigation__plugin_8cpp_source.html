29 {
30  if (chull.points.size() < 3)
31  {
32  ROS_ERROR("Convex hull has to consist of at least three points !!");
33  return;
34  }
35 
36  // To make sure we don't get e powers in the string
37  constraint << std::fixed << std::setprecision(6);
38 
39  constraint << "(";
40 
41  double dx,dy,xi,yi,xs,ys,length;
42  for (unsigned int i = 0; i < chull.points.size(); ++i)
43  {
44  if (i > 0)
45  constraint << " and ";
46 
47  xi = chull.points[i].x;
48  yi = chull.points[i].y;
49 
50  dx = ((i + 1 < chull.points.size()) ? chull.points[i+1].x : chull.points[0].x) - xi;
51  dy = ((i + 1 < chull.points.size()) ? chull.points[i+1].y : chull.points[0].y) - yi;
52  length = sqrt(dx * dx + dy * dy);
53 
54  xs = xi + (dy/length)*offset;
55  ys = yi - (dx/length)*offset;
56 
57  constraint << "-(x-" << xs << ")*" << dy << "+(y-" << ys << ")*" << dx << " > 0";
58  }
59 
60  constraint << ")";
61 }
