104  virtual inline unsigned char computeCost(double distance) const
105  {
106  unsigned char cost = 0;
107  if (distance == 0)
108  cost = target_cell_value_;
109  else if (distance * resolution_ <= dilation_radius_)
110  cost = dilation_cell_value_;
111  else
112  {
113  // make sure cost falls off by Euclidean distance
114  double euclidean_distance = distance * resolution_;
115  double factor = exp(-1.0 * weight_ * (euclidean_distance - dilation_radius_));
116  cost = (unsigned char)((dilation_cell_value_ - 1) * factor);
117  }
118  return cost;
119  }
