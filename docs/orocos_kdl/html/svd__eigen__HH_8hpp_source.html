36  double at,bt,ct;
37  at = fabs(a);
38  bt = fabs(b);
39  if (at > bt ) {
40  ct=bt/at;
41  return at*sqrt(1.0+ct*ct);
42  } else {
43  if (bt==0)
44  return 0.0;
45  else {
46  ct=at/bt;
47  return bt*sqrt(1.0+ct*ct);
48  }
