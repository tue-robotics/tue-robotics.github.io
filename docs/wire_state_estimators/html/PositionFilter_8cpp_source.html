68 void PositionFilter::propagate(const mhf::Time& time) {
69  if (t_last_propagation_ == 0) {
70  t_last_propagation_ = time;
71  return;
72  }
73 
74  mhf::Duration dt = time - t_last_propagation_;
75  t_last_propagation_ = time;
76 
77  assert(dt >= 0);
78 
79  if (!kalman_filter_) {
80  return;
81  }
82 
83  if ((time - t_last_update_) > kalman_timeout_ && kalman_timeout_ > 0) {
84  //if ((time - t_last_update_) > 10.0) {
85  if (!fixed_pdf_) {
86  int dimensions = kalman_filter_->getGaussian().getMean().n_rows;
87  pbl::Matrix cov = arma::eye(dimensions, dimensions) * fixed_pdf_cov_;
88  fixed_pdf_ = new pbl::Gaussian(kalman_filter_->getGaussian().getMean(), cov);
89  } else {
90  fixed_pdf_->setMean(kalman_filter_->getGaussian().getMean());
91  }
92 
93  // kalman_filter_ has been checked before, so it isn't a nullptr
94  delete kalman_filter_;
95  kalman_filter_ = 0;
96  return;
97  }
98 
99  // TODO: fix the kalman filter update (we shouldn't need a loop here...)
100  mhf::Duration small_dt = 0.05;
101  if (dt < small_dt) {
102  kalman_filter_->propagate(dt);
103  } else {
104  double total_dt = 0;
105  for(; total_dt < dt; total_dt += small_dt) {
106  kalman_filter_->propagate(small_dt);
107  }
108  if (total_dt < dt) {
109  kalman_filter_->propagate(dt - total_dt);
110  }
111  }
112 }

114 void PositionFilter::update(const pbl::PDF& z, const mhf::Time& time) {
115  t_last_update_ = time;
116 
117  if (z.type() == pbl::PDF::GAUSSIAN) {
118  const pbl::Gaussian* G = pbl::PDFtoGaussian(z);
119 
120  if (!kalman_filter_) {
121  kalman_filter_ = new KalmanFilter(z.dimensions());
122  kalman_filter_->setMaxAcceleration(max_acceleration_);
123  kalman_filter_->init(*G);
124  } else {
125  kalman_filter_->update(*G);
126  }
127  } else {
128  std::cout << "PositionFilter can only be updated with Gaussians, not with " << z.type() << std::endl;
129  }
130 }
