52 void OrientationFilter::processPath(const geometry_msgs::PoseStamped& start,
53  std::vector<geometry_msgs::PoseStamped>& path)
54 {
55  int n = path.size();
56  if (n == 0) return;
57  switch(omode_) {
58  case FORWARD:
59  for(int i=0;i<n-1;i++){
60  setAngleBasedOnPositionDerivative(path, i);
61  }
62  break;
63  case BACKWARD:
64  for(int i=0;i<n-1;i++){
65  setAngleBasedOnPositionDerivative(path, i);
66  set_angle(&path[i], angles::normalize_angle(tf2::getYaw(path[i].pose.orientation) + M_PI));
67  }
68  break;
69  case LEFTWARD:
70  for(int i=0;i<n-1;i++){
71  setAngleBasedOnPositionDerivative(path, i);
72  set_angle(&path[i], angles::normalize_angle(tf2::getYaw(path[i].pose.orientation) - M_PI_2));
73  }
74  break;
75  case RIGHTWARD:
76  for(int i=0;i<n-1;i++){
77  setAngleBasedOnPositionDerivative(path, i);
78  set_angle(&path[i], angles::normalize_angle(tf2::getYaw(path[i].pose.orientation) + M_PI_2));
79  }
80  break;
81  case INTERPOLATE:
82  path[0].pose.orientation = start.pose.orientation;
83  interpolate(path, 0, n-1);
84  break;
85  case FORWARDTHENINTERPOLATE:
86  for(int i=0;i<n-1;i++){
87  setAngleBasedOnPositionDerivative(path, i);
88  }
89 
90  int i=n-3;
91  const double last = tf2::getYaw(path[i].pose.orientation);
92  while( i>0 ){
93  const double new_angle = tf2::getYaw(path[i-1].pose.orientation);
94  double diff = fabs(angles::shortest_angular_distance(new_angle, last));
95  if( diff>0.35)
96  break;
97  else
98  i--;
99  }
100 
101  path[0].pose.orientation = start.pose.orientation;
102  interpolate(path, i, n-1);
103  break;
104  }
105 }

107 void OrientationFilter::setAngleBasedOnPositionDerivative(std::vector<geometry_msgs::PoseStamped>& path, int index)
108 {
109  int index0 = std::max(0, index - window_size_);
110  int index1 = std::min((int)path.size() - 1, index + window_size_);
111 
112  double x0 = path[index0].pose.position.x,
113  y0 = path[index0].pose.position.y,
114  x1 = path[index1].pose.position.x,
115  y1 = path[index1].pose.position.y;
116 
117  double angle = atan2(y1-y0,x1-x0);
118  set_angle(&path[index], angle);
119 }

121 void OrientationFilter::interpolate(std::vector<geometry_msgs::PoseStamped>& path,
122  int start_index, int end_index)
123 {
124  const double start_yaw = tf2::getYaw(path[start_index].pose.orientation),
125  end_yaw = tf2::getYaw(path[end_index ].pose.orientation);
126  double diff = angles::shortest_angular_distance(start_yaw, end_yaw);
127  double increment = diff/(end_index-start_index);
128  for(int i=start_index; i<=end_index; i++){
129  double angle = start_yaw + increment * i;

131  }
