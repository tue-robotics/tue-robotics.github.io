241 double ReferenceInterpolator::calculateTimeNeeded(double x0, double v0, double x1, double v1)
242 {
243  // Check velocity limits
244  if (v0 < -max_vel_ || v0 > max_vel_ || v1 < -max_vel_ || v1 > max_vel_)
245  return -1;
246 
247  // Calculate distance to travel
248  double X = x1 - x0;
249 
250  // Precalculate (we need it later)
251  double v_diff_sq = v0 * v0 + v1 * v1;
252 
253  // Calculate the center velocity 'vc'. This is a peak velocity that makes sure we get
254  // to the goal position as soons as possible. It is either larger than both v0 and v1,
255  // or smaller, as illustrated in the following velocity profiles:
256 
257  //
258  // * vc * v1
259  // / \ /
260  // / \ v0 /
261  // / \ * /
262  // / * v1 or \ /
263  // / \ /
264  // * v0 * vc
265  //
266  // ----------------> t ----------------> t
267 
268  // We now if it is higher or lower by comparing the distance we need to travel (X) by the
269  // distance we traveled if we would directly change from velocity v0 to v1 (Y).
270  // If X is lower than Y, it means we have to decelerate first, then accelerate to v1,
271  // and vc is below v0 and v1. Otherwise 'vc' is above v0 and v1.
272 
273  // Calculate the distance traveled if we would directly change from velocity v0 to v1
274  double Y = (std::abs(v1 - v0) / max_acc_) * (v0 + v1) / 2;
275 
276  double vc;
277  if (X < Y)
278  // vc is below v0 and v1
279  vc = -sqrt(v_diff_sq / 2 - max_acc_ * X);
280  else
281  // vc is above v0 and v1
282  vc = sqrt(v_diff_sq / 2 + max_acc_ * X);
283 
284  // If vc is larger than max_vel, it means we would exceed our maximum velocity. Instead
285  // we should accelerate to the maximum velocity as fast as possible, then maintain it for
286  // some time, and decelerate.
287  if (vc > max_vel_)
288  {
289  double time = (max_vel_ - v0) / max_acc_ + (max_vel_ - v1) / max_acc_;
290  double X_covered = (2 * max_vel_ * max_vel_ - v_diff_sq) / (2 * max_acc_);
291  double X_rest = X - X_covered;
292  double t_rest = X_rest / max_vel_;
293  return time + t_rest;
294  }
295  // A similar thing holds for when vc is negative and smaller than -max_vel
296  else if (vc < -max_vel_)
297  {
298  double time = (v0 + max_vel_) / max_acc_ + (v1 + max_vel_) / max_acc_;
299  double X_covered = (v_diff_sq - 2 * max_vel_ * max_vel_) / (2 * max_acc_);
300  double X_rest = X_covered - X;
301  double t_rest = X_rest / max_vel_;
302  return time + t_rest;
303  }
304  // Otherwise we can just calculate how long it will take to change velocity from v0 to vc,
305  // and from vc to v1.
306  else
307  {
308  return std::abs(vc - v0) / max_acc_ + std::abs(vc - v1) / max_acc_;
309  }
310 }
