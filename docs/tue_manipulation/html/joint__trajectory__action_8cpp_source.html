121  void goalCB(GoalHandle gh)
122  {
123  current_point = 0;
124  now = ros::Time::now();
125 
126  // Cancels the currently active goal.
127  if (has_active_goal_)
128  {
129  // Stops the controller.
130 
131  // Torso
132  sensor_msgs::JointState torso_msg;
133  for (uint i = 0; i < nr_torso_joints_; i++) {
134  torso_msg.name.push_back(joint_names_[i]);
135  torso_msg.position.push_back(cur_pos_[joint_names_[i]]);
136  }
137  torso_pub.publish(torso_msg);
138 
139  // arm
140  sensor_msgs::JointState arm_msg;
141  for (uint i = 0; i < 7; i++) {
142  arm_msg.name.push_back(joint_names_[i+nr_torso_joints_]);
143  arm_msg.position.push_back(cur_pos_[joint_names_[i+nr_torso_joints_]]);
144  }
145  pub.publish(arm_msg);
146 
147  // Marks the current goal as canceled.
148  active_goal_.setCanceled();
149  has_active_goal_ = false;
150  ROS_WARN("Canceling previous goal");
151  }
152 
153  number_of_goal_joints_ = gh.getGoal()->trajectory.joint_names.size();
154 
155  // Check feasibility of arm joint goals
156  for (uint i = 0; i < number_of_goal_joints_; i++) {
157  std::string joint_name = gh.getGoal()->trajectory.joint_names[i];
158  for (uint j = 0; j < gh.getGoal()->trajectory.points.size(); j++) {
159  double ref = gh.getGoal()->trajectory.points[j].positions[i];
160  if (ref < joint_min_constraints_[joint_name] || ref > joint_max_constraints_[joint_name]) {
161  ROS_WARN("Reference for joint %s is %f but should be between %f and %f.",joint_name.c_str(),ref,joint_min_constraints_[joint_name],joint_max_constraints_[joint_name]);
162  gh.setRejected();
163  has_active_goal_=false;
164  return;
165  }
166  }
167  }
168 
170  gh.setAccepted();
171  active_goal_ = gh;
172  has_active_goal_ = true;
173 
174  // Start by assuming hardware works
175  arm_status = 2;
176  torso_status = 2;
177 
178  }

262  void controllerCB() {
263 
264  int converged_joints=0;
265  float abs_error=0.0;
266 
267  // Check hardware status
268  // ToDo: are we happy with this?
269  if ( arm_status == 2 && torso_status == 2 ) {
270  //ROS_INFO("Hardware status OK");
271  } else if ( arm_status == 4 || torso_status == 4 ) {
272  ROS_WARN("Arm (%u) or torso (%u) is in error, joint trajectory goal cannot be reached, aborting", arm_status, torso_status);
273  active_goal_.setAborted();
274  has_active_goal_=false;
275  return;
276  } else if ( arm_status == 0 || torso_status == 0 ) {
277  ROS_WARN("Arm (%u) or torso (%u) is stale, joint trajectory goal cannot be reached, aborting", arm_status, torso_status);
278  active_goal_.setAborted();
279  has_active_goal_=false;
280  return;
281  } else if ( arm_status == 3 || torso_status == 3 ) {
282  ROS_WARN("Arm (%u) or torso (%u) is still homing, joint trajectory goal may not be reached", arm_status, torso_status);
283  } else if ( arm_status == 1 || torso_status == 1 ) {
284  ROS_WARN("Arm (%u) or torso (%u) is in idle, joint trajectory goal cannot be reached, aborting", arm_status, torso_status);
285  active_goal_.setAborted();
286  has_active_goal_=false;
287  return;
288  }
289 
290  // Check if the time constraint is not violated
291  if(ros::Time::now().toSec() > goal_time_constraint_ + now.toSec())
292  {
293  ROS_WARN("Aborting because the time constraint was violated");
294  active_goal_.setAborted();
295  has_active_goal_=false;
296  return;
297  }
298 
301 
302  for (unsigned int i = 0; i < active_goal_.getGoal()->trajectory.joint_names.size(); i++) {
303  std::string joint_name = active_goal_.getGoal()->trajectory.joint_names[i];
304 
305  // Compute absolute error
306  double ref_pos = active_goal_.getGoal()->trajectory.points[current_point].positions[i];
307  ref_pos_[joint_name] = ref_pos; // Required to push reference
308  double cur_pos = cur_pos_[joint_name];
309  abs_error = fabs(ref_pos - cur_pos);
310  //ROS_DEBUG("%s: r: %f\t q: %f\t e: %f",joint_name.c_str(), ref_pos_[joint_name], cur_pos_[joint_name], abs_error);
311 
312  // Check trajectory constraint
313  if(abs_error > trajectory_constraints_[joint_name]) {
314  ROS_WARN("Aborting because the trajectory constraint of %s (%f) was violated (%f)", joint_name.c_str(), trajectory_constraints_[joint_name], abs_error);
315  active_goal_.setAborted();
316  has_active_goal_=false;
317  return;
318  }
319 
320  // Check if this joint has converged
321  if(current_point < ((int)active_goal_.getGoal()->trajectory.points.size()-1))
322  {
323  if(abs_error < intermediate_goal_constraints_[joint_name])
324  {
325  //ROS_DEBUG("intermediate goal constraints for %s converged", joint_name.c_str());
326  converged_joints += 1;
327  }
328  }
329  else
330  {
331  if(abs_error < final_goal_constraints_[joint_name])
332  {
333  //ROS_DEBUG("final goal constraints for %s converged", joint_name.c_str());
334  converged_joints += 1;
335  }
336  }
337 
338  }
339 
340  // Joint trajectory action should work for both seven (old situation, only arm) and eight (new situation, incl torso) joints
341 
342  // Only publish torso if requested
343  if( number_of_goal_joints_ > 7) {
344  sensor_msgs::JointState torso_msg;
345  for (unsigned int i = 0; i < nr_torso_joints_; i++) {
346  torso_msg.name.push_back(joint_names_[i]);
347  torso_msg.position.push_back(ref_pos_[joint_names_[i]]);
348  }
349  torso_pub.publish(torso_msg);
350  }
351 
352  // Always publish arm msg
353  sensor_msgs::JointState arm_msg;
354  for (unsigned int i = 0; i < 7; i++) {
355  arm_msg.name.push_back(joint_names_[i+nr_torso_joints_]);
356  arm_msg.position.push_back(ref_pos_[joint_names_[i+nr_torso_joints_]]);
357  }
358  pub.publish(arm_msg);
359 
360  //if(converged_joints==(int)number_of_goal_joints_)
361  if (converged_joints==(int)number_of_goal_joints_)
362  {
363  now = ros::Time::now();
364  current_point = current_point + 1;
365  ROS_INFO("every joint has converged, go to the next point");
366  }
367 
368  if(current_point==(int)active_goal_.getGoal()->trajectory.points.size())
369  {
370  ROS_INFO("active goal succeeded");
371  active_goal_.setSucceeded();
372  has_active_goal_ = false;
373  }
374 
375  ROS_DEBUG("Converged joints = %i of %i, current_point = %i of %i", converged_joints, (int)number_of_goal_joints_, current_point, (int)active_goal_.getGoal()->trajectory.points.size());
376  }
