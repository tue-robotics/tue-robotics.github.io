67  int ConstrainedChainIkSolverVel_pinv::CartToJnt(const JntArray& q_in, const Twist& v_in, JntArray& qdot_out)
68  {
69  //Let the ChainJntToJacSolver calculate the jacobian "jac" for
70  //the current joint positions "q_in"
71  jnt2jac.JntToJac(q_in, jac);
72 
73  // Apply constraints (would JntToJac be a better place to do this immediately?)
74  // ToDo: don't hardcode
75  // ToDo: use polynomials
76  // Compare this with the function under 'apply constraints' in the iksolverpos
77  double dcdq = 3*1.3365*q_in(0)*q_in(0) - 2*1.9862*q_in(0) + 2.6392;
78  // Loop over columns
79  for (unsigned int col=0; col<jac_reduced.columns(); col++) {
80  for (unsigned int row=0; row<jac_reduced.rows(); row++) {
81  // ToDo: don't hardcode
82  if (col == 0) {
83  jac_reduced(row, col) = jac(row, col) + dcdq*jac(row, col+1);
84  } else {
85  jac_reduced(row, col) = jac(row, col+1);
86  }
87  }
88  }
89 
90  //Do a singular value decomposition of "jac" with maximum
91  //iterations "maxiter", put the results in "U", "S" and "V"
92  //jac = U*S*Vt
93  int ret = svd.calculate(jac_reduced, U, S, V, maxiter);
94 
95  double sum;
96  unsigned int i, j;
97 
98  // We have to calculate qdot_out = jac_pinv*v_in
99  // Using the svd decomposition this becomes(jac_pinv=V*S_pinv*Ut):
100  // qdot_out = V*S_pinv*Ut*v_in
101  //first we calculate Ut*v_in
102  for (i=0; i<jac_reduced.columns(); i++) {
103  sum = 0.0;
104  for (j=0; j<jac_reduced.rows(); j++) {
105  sum+= U[j](i)*v_in(j);
106  }
107  //If the singular value is too small (<eps), don't invert it but
108  //set the inverted singular value to zero (truncated svd)
109  tmp(i) = sum*(fabs(S(i))<eps?0.0:1.0/S(i));
110  }
111  //tmp is now: tmp=S_pinv*Ut*v_in, we still have to premultiply
112  //it with V to get qdot_out
113  for (i=0; i<jac_reduced.columns(); i++) {
114  sum = 0.0;
115  for (j=0; j<jac_reduced.columns(); j++) {
116  sum+=V[i](j)*tmp(j);
117  }
118  //Put the result in qdot_out
119  qdot_out_reduced(i)=sum;
120  }
121  //return the return value of the svd decomposition
122  // Expand solution
123  qdot_out.data.setZero();
124  for (unsigned int k = 0; k < qdot_out_reduced.rows(); k++) {
125  // ToDo: don't hardcode bookkeeping
126  if (k == 0) {
127  qdot_out(k) = qdot_out_reduced(k);
128  } else {
129  qdot_out(k+1) = qdot_out_reduced(k);
130  }
131  }
132  return ret;
133  }
