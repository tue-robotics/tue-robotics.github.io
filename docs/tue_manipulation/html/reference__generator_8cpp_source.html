45 // Interpolate using Hermite curve
46 void interpolateCubic(trajectory_msgs::JointTrajectoryPoint& p_out,
47  const trajectory_msgs::JointTrajectoryPoint& p0,
48  const trajectory_msgs::JointTrajectoryPoint& p1,
49  double t_abs)
50 {
51  double T = (p1.time_from_start - p0.time_from_start).toSec();
52  double t = t_abs - p0.time_from_start.toSec();
53  unsigned int njoints = p0.positions.size();
54 
55  p_out.positions.resize(njoints);
56  p_out.velocities.resize(njoints);
57  p_out.accelerations.resize(njoints);
58 
59  // Transform time to [0, 1]
60  double f = t / T;
61 
62  // Pre-calculate some things
63  double f2 = f * f;
64  double f3 = f * f2;
65 
66  // Interpolate for every joint
67  for(unsigned int k = 0; k < njoints; ++k)
68  {
69  p_out.positions[k] = (2 * f3 - 3 * f2 + 1) * p0.positions[k]
70  + (f3 - 2 * f2 + f) * (p0.velocities[k] * T)
71  + (-2 * f3 + 3 * f2) * p1.positions[k]
72  + (f3 - f2) * (p1.velocities[k] * T);
73 
74  p_out.velocities[k] = (6 * f2 - 6 * f) * p0.positions[k] / T
75  + (3 * f2 - 4 * f + 1) * p0.velocities[k]
76  + (-6 * f2 + 6 * f) * p1.positions[k] / T
77  + (3 * f2 - 2 * f) * p1.velocities[k];
78 
79  p_out.accelerations[k] = ((12 * f - 6) * p0.positions[k] / T
80  + (6 * f - 4) * p0.velocities[k]
81  + (-12 * f + 6) * p1.positions[k] / T
82  + (6 * f - 2) * p1.velocities[k]) / T;
83  }
84 
85  p_out.time_from_start = ros::Duration(t_abs);
86 }

386 bool ReferenceGenerator::calculatePositionReferencesInternal(JointGoal& goal, double dt)
387 {
388  time_ += dt;
389  goal.time_since_start += dt;
390 
391  // - - - - - - - - - - - - - - - - - - - - - - - - - - - -- - - - - - -
392  // Check if the sub goal is reached
393 
394  bool sub_goal_reached = false;
395  if (goal.sub_goal_idx < 0)
396  {
397  sub_goal_reached = true;
398  }
399  else
400  {
401  const trajectory_msgs::JointTrajectoryPoint& sub_goal = goal.goal_msg.trajectory.points[goal.sub_goal_idx];
402 
403  if (goal.use_cubic_interpolation)
404  {
405  sub_goal_reached = (goal.time_since_start >= sub_goal.time_from_start.toSec());
406  }
407  else
408  {
409  sub_goal_reached = true;
410  for(unsigned int i = 0; i < goal.num_goal_joints; ++i)
411  {
412  unsigned int joint_idx = goal.joint_index_mapping[i];
413 
414  if (!joint_info_[joint_idx].interpolator.done())
415  {
416  sub_goal_reached = false;
417  break;
418  }
419  }
420  }
421  }
422 
423  // - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
424  // If so, go to next unreached sub goal
425 
426  size_t MAX_ITERS = 100;
427  size_t ITER = 0;
428 
429  if (sub_goal_reached)
430  {
431  // If it has been reached, go to the next one
432  ++goal.sub_goal_idx;
433 
434  // Check if this was the last trajectory point. If so, this goal is finished!
435  if (goal.sub_goal_idx >= static_cast<long>(goal.goal_msg.trajectory.points.size()))
436  {
437  for(unsigned int i = 0; i < goal.num_goal_joints; ++i)
438  joint_info_[goal.joint_index_mapping[i]].goal_id.clear();
439 
440 // std::cout << "Goal reached in " << goal.time_since_start << " seconds" << std::endl;
441 
442  goal.status = JOINT_GOAL_SUCCEEDED;
443  return true;
444  }
445 
446  const trajectory_msgs::JointTrajectoryPoint& sub_goal = goal.goal_msg.trajectory.points[goal.sub_goal_idx];
447 
448  // If the velocities for the next and previous point are defined and the time needed in between is more than 0,
449  // use cubic interpolation
450  if (goal.sub_goal_idx > 0
451  && sub_goal.velocities.size() == goal.num_goal_joints
452  && goal.goal_msg.trajectory.points[goal.sub_goal_idx - 1].velocities.size() == goal.num_goal_joints
453  && (sub_goal.time_from_start - goal.goal_msg.trajectory.points[goal.sub_goal_idx - 1].time_from_start).toSec() > 0)
454  {
455  goal.use_cubic_interpolation = true;
456  goal.time_since_start = goal.goal_msg.trajectory.points[goal.sub_goal_idx - 1].time_from_start.toSec();
457  }
458  else
459  {
460  goal.use_cubic_interpolation = false;
461 
462  // Let's do some smoothing! We don't want to decelerate to 0 for each sub goal. However, we did not receive any
463  // intermediate velocities or timestamps in the given goal, so we have to do some calculation of our own.
464 
465  // First determine for each joint the maximum velocity we are allowed to have when reaching the next sub goal,
466  // such that we can still fully brake to 0 velocity in the goal after that.
467 
468  std::vector<double> sub_goal_velocities(goal.num_goal_joints, 0);
469  if (sub_goal.velocities.size() == goal.num_goal_joints)
470  {
471  sub_goal_velocities = sub_goal.velocities;
472  }
473  else if (goal.sub_goal_idx + 1 < static_cast<long>(goal.goal_msg.trajectory.points.size()))
474  {
475  for(unsigned int i = 0; i < goal.num_goal_joints; ++i)
476  {
477  const JointInfo& js = joint_info_[goal.joint_index_mapping[i]];
478 
479  double v0 = js.velocity(); // Current velocity
480  double x0 = js.position(); // Current position
481  double x1 = sub_goal.positions[i]; // Next sub goal position
482  double x2 = goal.goal_msg.trajectory.points[goal.sub_goal_idx + 1].positions[i]; // Sub goal position after that
483 
484  // Check if x0, x1 and x2 are going in the same direction. If not, the velocity has to be 0 in x1 because
485  // we have to change direction there.
486  if ((x0 < x1) == (x1 < x2))
487  {
488  // Calculate the maximum velocity we can reach from x0 to x1
489  double v_max_01 = sqrt(2 * js.max_acc * std::abs(x1 - x0) + v0 * v0);
490 
491  // Calculate the maximum velocity we are allowed to have such that we can still reach x2 with 0 velocity
492  double v_max_12 = sqrt(2 * js.max_acc * std::abs(x2 - x1));
493 
494  sub_goal_velocities[i] = std::min(js.max_vel, std::min(v_max_01, v_max_12));
495 
496  if (x2 < x1)
497  sub_goal_velocities[i] = -sub_goal_velocities[i];
498  }
499  }
500  }
501 
502  // Now given our current joint positions and velocities, the sub goal positions and the sub goal velocities we
503  // just calculated, calculate the time needed for each joint to reach the sub goal position and velocity, and
504  // remember the longest time.
505 
506  double time = 0;
507  for(unsigned int i = 0; i < goal.num_goal_joints; ++i)
508  {
509  JointInfo& js = joint_info_[goal.joint_index_mapping[i]];
510  time = std::max<double>(time, js.interpolator.calculateTimeNeeded(sub_goal.positions[i], sub_goal_velocities[i]));
511  }
512 
513  // Now just give each individual joint the calculated sub goal velocity and position and goal time calculated above.
514  // There is one problem: it might be the case that the max time calculated is too long for the joint to reach the
515  // given sub goal position and velocity. For example, it might have to brake to take more time, but then not have enough
516  // position margin left to accerelate to the sub goal velocity (maybe a bit hard to grasp, but think about it for a
517  // while...). Therefore we need to check for each joint if the calculated sub goal and time is still feasible. If not,
518  // we do a dirty trick: we lower the sub goal velocity a bit, see if that alters the max time, and try again. We do this
519  // in an iterative fashion until all joint goals are reachable.
520  // Don't worry: in most cases repetition is not needed and if it is, the number of iterations will be quite low.
521  while (true)
522  {
523  ++ITER;
524  if (ITER > MAX_ITERS)
525  {
526  break;
527  }
528 
529  bool all_goals_ok = true;
530 
531  for(unsigned int i = 0; i < goal.num_goal_joints && all_goals_ok; ++i)
532  {
533  JointInfo& js = joint_info_[goal.joint_index_mapping[i]];
534 
535  while(!js.interpolator.setGoal(sub_goal.positions[i], sub_goal_velocities[i], time))
536  {
537  ++ITER;
538 
539  if (ITER > MAX_ITERS)
540  {
541  break;
542  }
543 
544  // Whoops, we can't reach the goal in the time given! Let's lower the sub goal velocity and try again
545  sub_goal_velocities[i] *= 0.9;
546 
547  // Before trying again, we check if the time needed has changed.
548  double new_joint_time = js.interpolator.calculateTimeNeeded(sub_goal.positions[i], sub_goal_velocities[i]);
549 
550  if (new_joint_time > time)
551  {
552  // If now the joint needs longer than the current max time, we have to recalculate the max time
553  // and repeat the process for all joints
554  time = std::max(new_joint_time, time);
555  all_goals_ok = false;
556  break;
557  }
558  }
559  }
560 
561  if (all_goals_ok)
562  break;
563  }
564  }
565  }
566 
567  // - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
568  // Determine references using interpolation
569 
570  const trajectory_msgs::JointTrajectoryPoint& sub_goal = goal.goal_msg.trajectory.points[goal.sub_goal_idx];
571 
572  if (goal.use_cubic_interpolation)
573  {
574  const trajectory_msgs::JointTrajectoryPoint& prev_sub_goal = goal.goal_msg.trajectory.points[goal.sub_goal_idx - 1];
575 
576  trajectory_msgs::JointTrajectoryPoint p_interpolated;
577  interpolateCubic(p_interpolated, prev_sub_goal, sub_goal, goal.time_since_start);
578 
579  // Update the state of the interpolators (otherwise we'll have a problem if we switch to non-cubic interpolation)
580  for(unsigned int i = 0; i < goal.num_goal_joints; ++i)
581  {
582  unsigned int joint_idx = goal.joint_index_mapping[i];
583  joint_info_[joint_idx].interpolator.setState(p_interpolated.positions[i], p_interpolated.velocities[i],
584  p_interpolated.accelerations[i]);
585  }
586  }
587  else
588  {
589  for (unsigned int i = 0; i < goal.num_goal_joints; ++i)
590  {
591  unsigned int joint_idx = goal.joint_index_mapping[i];
592  JointInfo& j = joint_info_[joint_idx];
593  j.interpolator.update(dt);
594  }
595  }
596 
597  if (visualize_)
598  {
599  for(unsigned int i = 0; i < goal.num_goal_joints; ++i)
600  {
601  unsigned int joint_idx = goal.joint_index_mapping[i];
602  graph_vis_pos_.addPoint(0, joint_idx, time_, joint_info_[joint_idx].position());
603  graph_vis_vel_.addPoint(0, joint_idx, time_, joint_info_[joint_idx].velocity());
604  graph_vis_acc_.addPoint(0, joint_idx, time_, joint_info_[joint_idx].acceleration());
605  }
606  }
607 
608  return (ITER < MAX_ITERS);
609 }
