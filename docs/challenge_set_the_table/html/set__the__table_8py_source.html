26 @cb_interface(outcomes=['we_have_it_all', 'keep_going'], input_keys=['item_picked'])
27 def check_if_we_have_it_all(user_data, robot):
28  global items_picked

31  iteration += 1

33  return 'we_have_it_all'
34

47 
48  return 'keep_going' if missing_items else 'we_have_it_all'
