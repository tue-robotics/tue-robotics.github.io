61 inline void stopTimer(string msg, int ID = 0, double factor = 1) {
62  timespec t_end;
63  clock_gettime(CLOCK_PROCESS_CPUTIME_ID, &t_end);
64 
65  timespec& t_start = TIMERS.back();
66  TIMER_LOG << msg << ": " << factor * ((t_end.tv_sec - t_start.tv_sec) + double(t_end.tv_nsec - t_start.tv_nsec) / 1e9) << " secs." << endl;
67  TIMERS.pop_back();
68 }
