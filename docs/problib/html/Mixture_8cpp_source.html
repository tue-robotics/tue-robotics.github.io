87 double Mixture::getLikelihood(const PDF& pdf) const {
88  assert_msg(ptr_, "Mixture does not contain components.");
89  assert(ptr_->num_components_ > 0);
90  assert(ptr_->weights_total_ == 1);

92  double likelihood = 0;

151
