21 {
22  Vec2 diff = p2 - p1;
23  double line_length_sq = diff.length2();
24 
25  // Get rid of null cases
26  if ((p1.x == 0 && p1.y == 0) || (p2.x == 0 && p2.y == 0) || line_length_sq == 0)
27  return;
28 
29 
30  if (lrf_->range_max_ > 0)
31  {
32  // Calculate distance to the line
33 
34  double t = p1.dot(diff) / -line_length_sq;
35 
36  double distance_sq;
37 
38  if (t < 0)
39  distance_sq = p1.length2();
40  else if (t > 1)
41  distance_sq = p2.length2();
42  else
43  distance_sq = (p1 + t * diff).length2();
44 
45  // If too far, skip
46  if (distance_sq > lrf_->range_max_ * lrf_->range_max_)
47  return;
48  }
49 
50  // Get the angle / beam indices based on the slope
51  int i_p1 = lrf_->getAngleUpperIndex(p1.x, p1.y);
52  int i_p2 = lrf_->getAngleUpperIndex(p2.x, p2.y);
53 
54  // Get the minimum and maximum
55  int i_min = std::min(i_p1, i_p2);
56  int i_max = std::max(i_p1, i_p2);
57 
58  // We need to differentiate between two cases:
59  // - from min to max is less than half a circle
60  // - from min to max is more than half a circle (the line can be 'occluded' by the blind spot of the sensor)
61 
62  // In the latter case, we may need to render two parts
63 
64  int i_min1, i_max1, i_min2, i_max2;
65  if (i_max - i_min < lrf_->i_half_circle_)
66  {
67  // Back-face culling: if the normal is pointing outwards, ommit this line
68  if (i_p1 > i_p2)
69  return;
70 
71  // Both points in the blind spot (i's are both larger number of beams), so don't render a line
72  if (i_min >= lrf_->num_beams_)
73  return;
74 
75  // The line is fully in view, so only need to render one part
76  i_min1 = i_min;
77  i_max1 = std::min(lrf_->num_beams_, i_max);
78 
79  // No second part
80  i_min2 = 0;
81  i_max2 = 0;
82 
83  min_i = std::min(min_i, (int)i_min);
84  max_i = std::max(max_i, (int)i_max);
85  }
86  else
87  {
88  // Back-face culling: if the normal is pointing outwards, ommit this line
89  if (i_p2 > i_p1)
90  return;
91 
92  // We may need to draw two parts, because the line can be 'occluded' by the blind spot of the sensor
93  i_min1 = i_max;
94  i_max1 = lrf_->num_beams_;
95 
96  i_min2 = 0;
97  i_max2 = std::min(lrf_->num_beams_, i_min);
98 
99  min_i = 0;
100  max_i = lrf_->num_beams_;
101  }
102 
103  // d = (q1 - ray_start) x s / (r x s)
104  // = (q1 x s) / (r x s)
105  Vec2 s = p2 - p1;
106 
107  // For all beam regions found above (1 or 2 regions), calculate the intersection
108  // of each beam with the line
109 
110  // Draw part 1
111  for(int i = i_min1; i < i_max1; ++i)
112  {
113  const Vector3& r = lrf_->ray_dirs_[i];
114  double d = (p1.x * s.y - p1.y * s.x) / (r.x * s.y - r.y * s.x);
115  if (d > 0)
116  renderPoint(i, d);
117  }
118 
119  // Draw part 2
120  for(int i = i_min2; i < i_max2; ++i)
121  {
122  const Vector3& r = lrf_->ray_dirs_[i];
123  double d = (p1.x * s.y - p1.y * s.x) / (r.x * s.y - r.y * s.x);

125  renderPoint(i, d);

141  res.min_i = 0;
142  res.max_i = ray_dirs_.size();
143 
144  if (res.ranges.size() != ray_dirs_.size()) {
145  res.ranges.resize(ray_dirs_.size(), 0);
146  }
147 
148  res.lrf_ = this;
149 
150  const geo::Pose3D& pose = opt.getPose();
151 
152  double max_radius = opt.getMesh().getMaxRadius();
153  if (max_radius > 0)
154  {
155  // If object is too far above or below the laser plane, do not render
156  if (std::abs(pose.getOrigin().getZ()) > max_radius)
157  return;
158 
159  double dist_sq = pose.getOrigin().length2();
160 
161  if (dist_sq > max_radius * max_radius) {
162  // If nearest object point is certainly further away than max_range, do not render
163  if (sqrt(dist_sq) - max_radius > range_max_)
164  return;
165  }
166  }
167 
168  const std::vector<TriangleI>& triangles = opt.getMesh().getTriangleIs();
169  const std::vector<Vector3>& points = opt.getMesh().getPoints();
170 
171  // transform Z-coordinates of all vertices
172  std::vector<double> zs_t(points.size());
173  Vector3 Rz = pose.getBasis().getRow(2);
174  double z_offset = pose.getOrigin().getZ();
175  for(unsigned int i = 0; i < points.size(); ++i)
176  zs_t[i] = Rz.dot(points[i]) + z_offset;
177 
178  Vector3 Rx = pose.getBasis().getRow(0);
179  Vector3 Ry = pose.getBasis().getRow(1);
180 
181  // Iterate over all triangles
182  for(std::vector<TriangleI>::const_iterator it_tri = triangles.begin(); it_tri != triangles.end(); ++it_tri)
183  {
184  double z1 = zs_t[it_tri->i1_];
185  double z2 = zs_t[it_tri->i2_];
186  double z3 = zs_t[it_tri->i3_];
187 
188  bool p1_under_plane = z1 < 0;
189  bool p2_under_plane = z2 < 0;
190  bool p3_under_plane = z3 < 0;
191 
192  // Check if not all points of the triangle are on the same side of the plane
193  if (p1_under_plane != p2_under_plane || p2_under_plane != p3_under_plane)
194  {
195  // Transform the vertices to the sensor frame
196  Vec2 p1_3d(Rx.dot(points[it_tri->i1_]) + pose.t.x, Ry.dot(points[it_tri->i1_]) + pose.t.y);
197  Vec2 p2_3d(Rx.dot(points[it_tri->i2_]) + pose.t.x, Ry.dot(points[it_tri->i2_]) + pose.t.y);
198  Vec2 p3_3d(Rx.dot(points[it_tri->i3_]) + pose.t.x, Ry.dot(points[it_tri->i3_]) + pose.t.y);
199 
200  // Calculate the distances of the vertices to the plane
201  double z1_abs = std::abs(z1);
202  double z2_abs = std::abs(z2);
203  double z3_abs = std::abs(z3);
204 
205  // Calculate the intersections of the triangle edges with the plane,
206  // respecting the orientation of the triangle (normal is towards or away from sensor)
207  // such that later on we can do back-face culling.
208 
209  Vec2 q1, q2;
210  if (p2_under_plane == p3_under_plane) {
211  if (p2_under_plane)
212  {
213  q2 = (p1_3d * z2_abs + p2_3d * z1_abs) / (z1_abs + z2_abs);
214  q1 = (p1_3d * z3_abs + p3_3d * z1_abs) / (z1_abs + z3_abs);
215  }
216  else
217  {
218  q1 = (p1_3d * z2_abs + p2_3d * z1_abs) / (z1_abs + z2_abs);
219  q2 = (p1_3d * z3_abs + p3_3d * z1_abs) / (z1_abs + z3_abs);
220  }
221  } else if (p1_under_plane == p3_under_plane) {
222  if (p1_under_plane)
223  {
224  q1 = (p2_3d * z1_abs + p1_3d * z2_abs) / (z2_abs + z1_abs);
225  q2 = (p2_3d * z3_abs + p3_3d * z2_abs) / (z2_abs + z3_abs);
226  }
227  else
228  {
229  q1 = (p2_3d * z3_abs + p3_3d * z2_abs) / (z2_abs + z3_abs);
230  q2 = (p2_3d * z1_abs + p1_3d * z2_abs) / (z2_abs + z1_abs);
231  }
232  }
233 
234  if (p1_under_plane == p2_under_plane) {
235  if (p1_under_plane)
236  {
237  q1 = (p3_3d * z2_abs + p2_3d * z3_abs) / (z3_abs + z2_abs);
238  q2 = (p3_3d * z1_abs + p1_3d * z3_abs) / (z3_abs + z1_abs);
239  }
240  else
241  {
242  q1 = (p3_3d * z1_abs + p1_3d * z3_abs) / (z3_abs + z1_abs);
243  q2 = (p3_3d * z2_abs + p2_3d * z3_abs) / (z3_abs + z2_abs);
244  }
245  }

248  res.renderLine(geo::Vec2(q1.x, q1.y), geo::Vec2(q2.x, q2.y));

303 void LaserRangeFinder::calculateRays() {
304  ray_dirs_.clear();
305  angles_.clear();
306 
307  // Pre-calculate the unit direction vectors of all the rays
308  double a_incr = getAngleIncrement();
309  double a = a_min_;
310  for(int i = 0; i < num_beams_; ++i) {
311  Vector3 dir = polarTo2D(a, 1);
312  ray_dirs_.push_back(dir);
313  angles_.push_back(a);
314  a += a_incr;
315  }
316 
317  // Create a look-up table which translates slope to ray index. This way,
318  // we can later on calculate the approximate polar coordinates a point on the
319  // laser plane without using atan.
320 
321  // Determine the needed resolution of the look-up table (TODO: make less ad-hoc)
322  slope_factor_ = 1 / tan(getAngleIncrement()) * 10;
323  int N = (int)slope_factor_ + 1;
324 
325  for(unsigned int i = 0; i < 8; ++i)
326  slope_to_index_[i].resize(N);
327 
328  // We divide the unit circle into 8 parts, split by 3 criteria:
329  // abs(x) < abs(y)
330  // x < 0
331  // y < 0
332 
333  // This way we can always calculate a slope between 0 and 1 (by dividing the
334  // smallest coordinate by the biggest).
335  for(int j = 0; j < 8; ++j)
336  {
337  slope_to_index_[j].resize(N);
338  for(int k = 0; k < N; ++k)
339  {
340  // Calculate the slope corresponding to this table entry
341  double slope = (double)k / slope_factor_;
342 
343  // Calculate a virtual (x, y) cartesian point which corresponds to this table entry
344  // based on the slope. Determine the part of the unit circle in which (x, y) lies
345  // base on j
346  double x, y;
347  if (j & 1)
348  {
349  // x_abs > y_abs
350  x = 1;
351  y = slope + 1e-16; // Necessary because of 0-check in getAngle
352  }
353  else
354  {
355  // x_abs < y_abs
356  x = slope + 1e-16; // Necessary because of 0-check in getAngle
357  y = 1;
358  }
359 
360  if (!(j & 2))
361  {
362  // x < 0
363  x = -x;
364  }
365 
366  if (!(j & 4))
367  {
368  // y < 0
369  y = -y;
370  }
371 
372  // Calculate the angle of the virtual point (x, y)
373  double a = getAngle(x, y);
374  double a_temp = a - a_min_;
375  if (a_temp < 0)
376  a_temp += 2 * M_PI;
377 
378  // Calculate the ray index based on the angle
379  slope_to_index_[j][k] = a_temp / (a_max_ - a_min_) * num_beams_ + 1;
380  }
381  }
382 
383  i_half_circle_ = M_PI / getAngleIncrement();
384 }

407 int LaserRangeFinder::getAngleUpperIndex(double x, double y) const
408 {
409  // Calculate the ray index corresponding to the cartesian point (x, y)
410  // We use the look-up table to avoid calculating the atan of (x / y)
411 
412  double x_abs = std::abs(x);
413  double y_abs = std::abs(y);
414 
415  // Calculcate slope
416  double slope;
417  if (x_abs < y_abs)
418  slope = x_abs / y_abs;
419  else
420  slope = y_abs / x_abs;
421 
422  int k = slope_factor_ * slope;
423 
424  // Calculate region number (0 - 7)
425  int j = (x_abs < y_abs ? 0 : 1) + (x < 0 ? 0 : 2) + (y < 0 ? 0 : 4);
426 
427  return slope_to_index_[j][k];
428 }
