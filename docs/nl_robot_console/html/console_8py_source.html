28 def recurse_replace_in_dict(d, mapping):
29  """
30  Recursively replace values in a dictionary according to some mapping, if the mapping has an entry
31  :param d: a python dictionary in which some items must be replaced
32  :param mapping: a python dictionary
33  :return: the dictionary with some values replaces according to the given mapping
34 
35  >>> mapping = {'A': 'apple', 'B':'banana'}
36  >>> flat = {'a': 'A', 'b': 'B'}
37  >>> recurse_replace_in_dict(flat, mapping)
38  {'a': 'apple', 'b': 'banana'}
39 
40  >>> flat = {'a': 'A', 'b': {'c':'courgette'}}
41  >>> recurse_replace_in_dict(flat, mapping)
42  {'a': 'apple', 'b': {'c': 'courgette'}}
43 
44  >>> deep_nested = {'a': 'A', 'b': {'c':{'a': 'A', 'b': 'B'}}}
45  >>> recurse_replace_in_dict(deep_nested, mapping)
46  {'a': 'apple', 'b': {'c': {'a': 'apple', 'b': 'banana'}}}
47  """
48  for key in d.keys():
49  # Looks magic, but the .get-method of a dict checks if the given entry is in the dict and if it is not, use a default.
50  # In this case, if there is an entry for d[key], use it but otherwise default to current value of d[key]
51  if isinstance(d[key], dict):
52  d[key] = recurse_replace_in_dict(d[key], mapping)
53  else:
54  d[key] = mapping.get(d[key], d[key])
55  return d
