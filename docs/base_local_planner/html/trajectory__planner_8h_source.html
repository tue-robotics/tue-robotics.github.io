357  //compute velocity based on acceleration
366  inline double computeNewVelocity(double vg, double vi, double a_max, double dt){
367  if((vg - vi) >= 0) {
368  return std::min(vg, vi + a_max * dt);
369  }
370  return std::max(vg, vi - a_max * dt);
371  }
