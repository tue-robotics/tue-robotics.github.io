74 double ObstacleCostFunction::scoreTrajectory(Trajectory &traj) {
75  double cost = 0;
76  double scale = getScalingFactor(traj, scaling_speed_, max_trans_vel_, max_scaling_factor_);
77  double px, py, pth;
78  if (footprint_spec_.size() == 0) {
79  // Bug, should never happen
80  ROS_ERROR("Footprint spec is empty, maybe missing call to setFootprint?");
81  return -9;
82  }
83 
84  for (unsigned int i = 0; i < traj.getPointsSize(); ++i) {
85  traj.getPoint(i, px, py, pth);
86  double f_cost = footprintCost(px, py, pth,
87  scale, footprint_spec_,
88  costmap_, world_model_);
89 
90  if(f_cost < 0){
91  return f_cost;
92  }
93 
94  // Count cell costs instead of footprint costs for optimization
95  unsigned int cell_x, cell_y;
96  costmap_->worldToMap(px, py, cell_x, cell_y);
97  double c_cost = double(costmap_->getCost(cell_x, cell_y));
98 
99  if(sum_scores_)
100  cost += c_cost;
101  else
102  cost = std::max(cost, c_cost);
103  }
104  return cost;
105 }

107 double ObstacleCostFunction::getScalingFactor(Trajectory &traj, double scaling_speed, double max_trans_vel, double max_scaling_factor) {
108  double vmag = hypot(traj.xv_, traj.yv_);
109 
110  //if we're over a certain speed threshold, we'll scale the robot's
111  //footprint to make it either slow down or stay further from walls
112  double scale = 1.0;
113  if (vmag > scaling_speed) {
114  //scale up to the max scaling factor linearly... this could be changed later
115  double ratio = (vmag - scaling_speed) / (max_trans_vel - scaling_speed);
116  scale = max_scaling_factor * ratio + 1.0;
117  }
118  return scale;
119 }

121 double ObstacleCostFunction::footprintCost (
122  const double& x,
123  const double& y,
124  const double& th,
125  double scale,
126  std::vector<geometry_msgs::Point> footprint_spec,
127  costmap_2d::Costmap2D* costmap,
128  base_local_planner::WorldModel* world_model) {
129 
130  std::vector<geometry_msgs::Point> scaled_footprint;
131  for(unsigned int i = 0; i < footprint_spec.size(); ++i) {
132  geometry_msgs::Point new_pt;
133  new_pt.x = scale * footprint_spec[i].x;
134  new_pt.y = scale * footprint_spec[i].y;
135  scaled_footprint.push_back(new_pt);
136  }
137 
138  //check if the footprint is legal
139  // TODO: Cache inscribed radius
140  double footprint_cost = world_model->footprintCost(x, y, th, scaled_footprint);
141 
142  if (footprint_cost < 0) {
143  return -6.0;
144  }
145  unsigned int cell_x, cell_y;
146 
147  //we won't allow trajectories that go off the map... shouldn't happen that often anyways
148  if ( ! costmap->worldToMap(x, y, cell_x, cell_y)) {
149  return -7.0;
150  }
151 
152  double occ_cost = std::max(std::max(0.0, footprint_cost), double(costmap->getCost(cell_x, cell_y)));
153 
154  return occ_cost;
155 }
