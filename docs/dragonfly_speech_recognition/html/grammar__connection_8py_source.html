52 class ConnectionGrammar(Grammar):
53  """
54  Grammar class for maintaining a COM connection well
55  within a given context. This is useful for controlling
56  applications through COM while they are in the
57  foreground. This grammar class will take care of
58  dispatching the correct COM interface when the
59  application comes to the foreground, and releasing it
60  when the application is no longer there.
61 
62  * ``name`` -- name of this grammar.
63  * ``description`` -- description for this grammar.
64  * ``context`` -- context within which to maintain
65  the COM connection.
66  * ``app_name`` -- COM name to dispatch.
67  """
68 
69  def __init__(self, name, description=None, context=None, app_name=None):
70  assert isinstance(app_name, string_types) or app_name is None
71  self._app_name = app_name
72  self._application = None
73  Grammar.__init__(self, name=name, description=description,
74  context=context)

101  def _process_begin(self, executable, title, handle):
102  # If not connected yet, retry. If the connection fails after
103  # single attempt, give up.
104  if not self._application:
105  if not self.connect():
106  return False

113  def connect(self):
114  if not self._app_name:
115  return True
116  try:
117  self._application = Dispatch(self._app_name)
118  except com_error as e:
119  if self._log_begin:
120  self._log_begin.warning("Grammar %s: failed to connect to "
121  "%r: %s.", self, self._app_name, e)
122  return False
123  else:
124  [r.activate() for r in self._rules if not r.active]
125  return True
126 
127  def disconnect(self):
128  self._application = None
