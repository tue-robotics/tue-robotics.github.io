53 class FlagContainer(object):
54  """ Container for a predefined set of boolean flags. """
55 
56  flag_names = ()
57 
58  def __init__(self, *names):
59  self._flags_true = set()
60  for name in names:
61  setattr(self, name, True)
62 
63  def flags_string(self):
64  flags_true_names = []
65  for flag in self.flag_names:
66  if flag in self._flags_true:
67  flags_true_names.append(flag)
68  return u", ".join(flags_true_names)
69 
70  def __unicode__(self):
71  return u"%s(%s)" % (self.__class__.__name__, self.flags_string())
72 
73  def __repr__(self):
74  if PY2:
75  return self.__unicode__().encode(getpreferredencoding())
76  else:
77  return self.__unicode__()
78 
79  def __getattr__(self, name):
80  if name not in self.flag_names:
81  raise AttributeError("Invalid flag name: %r" % (name,))
82  return name in self._flags_true
83 
84  # pylint: disable=inconsistent-return-statements
85  def __setattr__(self, name, value):
86  if name.startswith("_"):
87  return object.__setattr__(self, name, value)
88  if name not in self.flag_names:
89  raise AttributeError("Invalid flag name: received %r, expected"
90  " one of %r" % (name, self.flag_names))
91  if value:
92  self._flags_true.add(name)
93  else:
94  self._flags_true.discard(name)
95 
96  def clone(self):
97  clone = self.__class__()
98  for name in self._flags_true:
99  setattr(clone, name, True)
100  return clone

266  if info is None:

277  def parse_input(self, input):
278  if not isinstance(input, string_types):
279  raise TypeError("input must be a string, not {0!r}"
280  .format(input))
281 
282  if isinstance(input, binary_type):
283  # DNS and Natlink provide recognized words as "Windows-1252"
284  # encoded strings. Here we convert them to Unicode for internal
285  # processing.
286  input = input.decode("windows-1252")
287 
288  # The written and spoken forms of a word are separated by a "\"
289  # character.
290  index = input.rfind("\\")
291  if index == -1:
292  # Input doesn't contain a backslash, so written and spoken forms
293  # are the same as the input.
294  written = input
295  spoken = input
296  else:
297  # Input contains one or more backslashes, so written and spoken
298  # forms are separated by the last backslash.
299  written = input[:index]
300  spoken = input[index+1:]
301 
302  info = natlink.getWordInfo(input.encode("windows-1252"))
303  word_flags = self.create_word_flags(info)
304 
305  word = Word(written, spoken, word_flags)
306  self._log.debug("Parsed input %r -> %s", input, word)
307 # print ("Parsed input {0!r} -> {1}".format(input, word))
308  return word

370  def create_word_flags(self, property):
371  if not property:
372  # None indicates the word is not in DNS' vocabulary.
373  flags = WordFlags()
374  elif property in self.property_map:
375  flags = self.property_map[property].clone()
376  elif property.startswith("left-"):
377  flags = self.property_map["left-*"].clone()
378  elif property.startswith("right-"):
379  flags = self.property_map["right-*"].clone()
380  else:
381  self._log.warning("Unknown word property: %r", property)
382  flags = WordFlags()

385  def parse_input(self, input):
386  if not isinstance(input, string_types):
387  raise TypeError("input must be a string, not {0!r}"
388  .format(input))
389 
390  if isinstance(input, binary_type):
391  # DNS and Natlink provide recognized words as "Windows-1252"
392  # encoded strings. Here we convert them to Unicode for internal
393  # processing.
394  input = input.decode("windows-1252")
395 
396  parts = input.split("\\")
397  if len(parts) == 1:
398  # Word doesn't have "written\property\spoken" form, so
399  # written and spoken forms are equal to input and there are
400  # no formatting flags.
401  written = input
402  spoken = input
403  property = None
404  elif len(parts) == 2:
405  if parts[1] == "letter":
406  # Format: "X \ letter"
407  written = spoken = parts[0]
408  property = "letter"
409  else:
410  # Format: "written \ spoken"
411  written, spoken = parts
412  property = None
413  elif len(parts) == 3:
414  # Format: "written \ property \ spoken"
415  written, property, spoken = parts
416  else: # len(parts) > 3:
417  # Format: "wri\tt\en \ property \ spoken"
418  written = "\\".join(parts[:-2])
419  property = parts[-2]
420  spoken = parts[-1]
421 
422  word_flags = self.create_word_flags(property)
423 
424  word = Word(written, spoken, word_flags)
425  self._log.debug("Parsed input %r -> %s", input, word)
426 # print ("Parsed input {0!r} -> {1}".format(input, word))
427  return word

432 class WordParserFactory(object):
433  """
434  Detects appropriate WordParser class and creates objects
435 
436  This class calls natlink.getWordInfo() for special words and
437  interprets the result to determine whether words should be parsed
438  using natlink.getWordInfo(), e.g. for DNS v10 and lower, or using
439  "in-line word properties", e.g. for DNS v11 and higher.
440 
441  This class performs detection only once per instance and caches
442  the detection result for reuse.
443 
444  """
445 
446  _log = logging.getLogger("dictation.word_parser_factory")
447 
448  # The following dictionary determines which words are used for which
449  # language to detect whether natlink.getWordInfo() returns useful info.
450  words_with_info = {
451  "en": ".\\dot",
452  "nl": ".\\dot",
453  }
454 
455  def __init__(self):
456  self.parser_class = None
457 
458  def detect_parser_class(self):
459  engine = dragonfly.engines.get_engine()
460  word = self.words_with_info.get(engine.language, ".\\dot")
461  if PY2:
462  info = natlink.getWordInfo(word.encode("windows-1252"))
463  else:
464  info = natlink.getWordInfo(word)
465  if info is None:
466  parser_class = WordParserDns11
467  else:
468  parser_class = WordParserDns10
469 
470  parser_class_string = parser_class.__name__
471  if info is None: info_string = "None"
472  else: info_string = "0x" + format(info, "08x")
473  self._log.info("Selected word parser class %s because"
474  " natlink.getWordInfo(%r) returned %s",
475  parser_class_string, word, info_string)
476 
477  return parser_class

481  if not self.parser_class:

506  def format_dictation(self, input_words):
507  if isinstance(input_words, string_types):
508  raise ValueError("Argument input_words must be a sequence of"
509  " words, but received a single string: {0!r}"

511

513  for input_word in input_words:
514  word = self.parser.parse_input(input_word)
515  formatted_words.append(self.apply_formatting(word))
516  new_state = self.update_state(word)
517  self._log.debug("Processing %s, formatted output: %r,"
518  " %s -> %s",
519  word, formatted_words[-1], self.state,
520  new_state)
521  self.state = new_state
522  return u"".join(formatted_words)

524  def apply_formatting(self, word):
525  state = self.state
526 
527  # Determine prefix.
528  if word.flags.no_format: prefix = ""
529  elif state.no_space_mode: prefix = ""
530  elif word.flags.no_space_before: prefix = ""
531  elif state.no_space_before: prefix = ""
532  elif state.no_space_between and word.flags.no_space_between:
533  prefix = ""
534  elif state.two_spaces_before:
535  if self.two_spaces_after_period: prefix = " "
536  else: prefix = " "
537  else: prefix = " "
538 
539  # Determine formatted written form.
540  if word.flags.no_format: written = word.written
541  elif state.cap_mode and not word.flags.no_title_cap:
542  written = self.capitalize_all(word.written)
543  elif state.upper_mode: written = self.upper_all(word.written)
544  elif state.lower_mode: written = self.lower_all(word.written)
545  elif state.cap_next: written = self.capitalize_first(word.written)
546  elif state.cap_next_force: written = self.capitalize_first(word.written)
547  elif state.upper_next: written = self.upper_first(word.written)
548  elif state.lower_next: written = self.lower_first(word.written)
549  else: written = word.written
550 
551  # Remove first period character if needed.
552  if (state.prev_ended_in_period
553  and word.flags.not_after_period
554  and written.startswith(".")):
555  written = written[1:]
556 
557  # Determine suffix.
558  if word.flags.newline_after: suffix = "\n"
559  elif word.flags.two_newlines_after: suffix = "\n\n"
560  elif word.flags.spacebar: suffix = " "
561  else: suffix = ""
562 
563  return prefix + written + suffix
564 
565  def update_state(self, word_object):
566  word = word_object.flags
567  prev = self.state
568  state = StateFlags()
569 
570  # Update capitalization state.
571  state.cap_next_force = word.cap_next_force or (word.no_cap_reset and prev.cap_next_force)
572  state.cap_next = word.cap_next or word.cap_mode or (word.no_cap_reset and prev.cap_next)
573  state.upper_next = word.upper_next or (word.no_cap_reset and prev.upper_next)
574  state.lower_next = word.lower_next or (word.no_cap_reset and prev.lower_next)
575  state.cap_mode = word.cap_mode or (prev.cap_mode and not word.reset_cap)
576  state.upper_mode = word.upper_mode or (prev.upper_mode and not word.reset_cap)
577  state.lower_mode = word.lower_mode or (prev.lower_mode and not word.reset_cap)
578 
579  # Update spacing state.
580  state.no_space_before = word.no_space_after or (prev.no_space_before and word.no_space_reset and word.no_format)
581  state.two_spaces_before = word.two_spaces_after or (prev.two_spaces_before and word.no_space_reset and word.no_format)
582  state.no_space_between = word.no_space_between
583  state.no_space_mode = word.no_space_mode or (prev.no_space_mode and not word.reset_no_space)
584 
585  # Record whether this word ended in a period.
586  state.prev_ended_in_period = word_object.written.endswith(".")
587 
588  # Return newly created state.
589  return state
