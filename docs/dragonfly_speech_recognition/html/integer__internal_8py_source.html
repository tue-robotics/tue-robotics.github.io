109  def build_element(self, min, max):
110 
111  # Sanity check.
112  if min >= max: return None
113 
114  # Calculate ranges of multipliers and remainders.
115  first_multiplier = min / self._factor
116  last_multiplier = (max - 1) / self._factor + 1
117  first_remainder = min % self._factor
118  last_remainder = max % self._factor
119  if last_remainder == 0:
120  last_remainder = self._factor
121 
122  # Handle special case of only one possible multiplier value.
123  if first_multiplier == last_multiplier - 1:
124  return self._build_range(first_multiplier, last_multiplier,
125  first_remainder, last_remainder)
126 
127  children = []
128 
129  # Build partial range for first multiplier value, if necessary.
130  if first_remainder > 0:
131  c = self._build_range(first_multiplier, first_multiplier + 1,
132  first_remainder, self._factor)
133  if c: children.append(c)
134  first_multiplier += 1
135 
136  # Build partial range for last multiplier value, if necessary.
137  if last_remainder > 0:
138  c = self._build_range(last_multiplier - 1, last_multiplier,
139  0, last_remainder)
140  if c: children.append(c)
141  last_multiplier -= 1
142 
143  # Build range for multiplier values which have the full
144  # range of remainder values.
145  if first_multiplier < last_multiplier:
146  c = self._build_range(first_multiplier, last_multiplier,
147  0, self._factor)
148  if c: children.append(c)
149 
150  # Wrap up result as is appropriate.
151  if len(children) == 0: return None
152  elif len(children) == 1: return children[0]
153  else: return Alternative(children)
