163  def _build_element(self, min, max, memo):
164 
165  # Sanity check.
166  if min >= max: return None
167 
168  # Calculate ranges of multipliers and remainders.
169  first_multiplier = int(min / self._factor)
170  last_multiplier = int((max - 1) / self._factor + 1)
171  first_remainder = min % self._factor
172  last_remainder = max % self._factor
173  if last_remainder == 0:
174  last_remainder = self._factor
175 
176  # Handle special case of only one possible multiplier value.
177  if first_multiplier == last_multiplier - 1:
178  return self._build_range(first_multiplier, last_multiplier,
179  first_remainder, last_remainder, memo)
180 
181  children = []
182 
183  # Build partial range for first multiplier value, if necessary.
184  if first_remainder > 0:
185  c = self._build_range(first_multiplier, first_multiplier + 1,
186  first_remainder, self._factor, memo)
187  if c: children.append(c)
188  first_multiplier += 1
189 
190  # Build partial range for last multiplier value, if necessary.
191  if last_remainder > 0:
192  c = self._build_range(last_multiplier - 1, last_multiplier,
193  0, last_remainder, memo)
194  if c: children.append(c)
195  last_multiplier -= 1
196 
197  # Build range for multiplier values which have the full
198  # range of remainder values.
199  if first_multiplier < last_multiplier:
200  c = self._build_range(first_multiplier, last_multiplier,
201  0, self._factor, memo)
202  if c: children.append(c)
203 
204  # Build modified path elements, if necessary.
205  if self._modifier_function is not None:
206  c = self._build_modified_paths(children, memo)
207  if c:
208  # Don't use other children if in modifier mode REPLACE.
209  mode = self._modifier_mode
210  if mode == ModifiedPathsCollection.MODE_REPLACE:
211  del children[:]
212 
213  # Add the ModifiedPathsCollection element.
214  children.append(c)
215 
216  # Wrap up result as is appropriate.
217  if len(children) == 0: return None
218  elif len(children) == 1: return children[0]
219  else: return Alternative(children)

302 class ModifiedPathsCollection(Alternative):
303 
304  # Collect and use paths to *augment* recognition of the given element
305  # (default). Discards nil value or unchanged strings returned by the
306  # modifier function.
307  MODE_AUGMENT = 1
308 
309  # Collect and use paths to *replace* recognition of the given element.
310  # Discards nil value strings returned by the modifier function. Keeps
311  # unchanged path strings.
312  MODE_REPLACE = 2
313 
314  def __init__(self, element, modifier_function, modifier_mode,
315  name=None):
316 
317  if not 0 < modifier_mode <= 2:
318  raise ValueError("Invalid modifier mode: %d" % modifier_mode)
319 
320  # Generate element tree recognition paths and use them to create
321  # elements.
322  children = []
323  specs = set()
324  for path in self._generate_paths(element):
325  # Get a flat list of each word in this path.
326  all_words = []
327  for words in path:
328  all_words.extend(words)
329 
330  # Get a spec using the modifier function.
331  text = " ".join(all_words)
332  spec = modifier_function(text)
333 
334  # Handle the result based on the specified modifier mode.
335  if modifier_mode is self.MODE_AUGMENT:
336  if not spec or text == spec:
337  continue
338  elif modifier_mode is self.MODE_REPLACE:
339  if not spec:
340  continue
341 
342  # Always skip duplicate specs.
343  if spec in specs:
344  continue
345 
346  # Initialize a new ModifiedPath element, passing the original
347  # element and words for decode-time calculation of the integer
348  # value.
349  specs.add(spec)
350  children.append(ModifiedPath(spec, all_words, element))
351 
352  # Initialize super class.
353  Alternative.__init__(self, children=children, name=name)
354 
355  def _generate_sequence_paths(self, element, child_no):
356  if child_no > len(element.children) - 1:
357  yield []
358  return
359 
360  child = element.children[child_no]
361  next_child = child_no + 1
362  for head in self._generate_paths(child):
363  if None in head:
364  continue
365  for tail in self._generate_sequence_paths(element, next_child):
366  yield head + tail
367 
368  def _generate_paths(self, element):
369  if isinstance(element, Literal):
370  yield [element.words]
371 
372  elif isinstance(element, ListRef):
373  if len(element.list) == 0:
374  # Impossible path.
375  yield [None]
376  else:
377  # Generate paths from the list words.
378  for word in element.list.get_list_items():
379  yield [word.split()]
380 
381  elif isinstance(element, Sequence):
382  # Generate each complete path of the sequence.
383  for path in self._generate_sequence_paths(element, 0):
384  if None not in path:
385  yield path
386 
387  elif isinstance(element, Alternative):
388  # Generate the paths of each alternative.
389  for child in element.children:
390  for path in self._generate_paths(child):
391  if None not in path:
392  yield path
393 
394  elif isinstance(element, Optional):
395  # Generate the possible paths.
396  # Optional element is not spoken.
397  yield []
398 
399  # Optional element is spoken.
400  for path in self._generate_paths(element.children[0]):
401  if None not in path:
402  yield path
403 
404  elif isinstance(element, Impossible):
405  # Impossible path.
406  yield [None]
407 
408  elif isinstance(element, RuleRef):
409  # Generate the possible paths from the referenced rule's element
410  # tree.
411  for path in self._generate_paths(element.rule.element):
412  if None not in path:
413  yield path
414 
415  else:
416  yield []

419 class ModifiedPath(Compound):
420 
421  def __init__(self, spec, words, orig_root_element, name=None):
422  self._words = words
423  self._orig_root_element = orig_root_element
424  Compound.__init__(self, spec, name=name)
425 
426  def value(self, node):
427  """
428  The *value* of a :class:`ModifiedPath` is the *value*
429  obtained by decoding the original words.
430 
431  """
432  import dragonfly.grammar.state as state_
433  words_rules = tuple((word, 0) for word in self._words)
434  s = state_.State(words_rules, [], node.engine)
435  s.initialize_decoding()
436  for _ in self._orig_root_element.decode(s):
437  if s.finished():
438  root = s.build_parse_tree()
439  return root.value()
440  self._log_decode.error("CompoundWord %s: failed to decode original"
441  " words %r.", self, " ".join(self._words))
442  return None
