21 """
22  This module is a simple example of Dragonfly use.
23 
24  It shows how to use Dragonfly's Function action to call Python functions
25  with the values associated with spoken Choice, Dictation and IntegerRef
26  elements. This module can be activated in the same way as other
27  Natlink macros by placing it in the "NatLink/MacroSystem" folder.

105 # Create a mapping rule class which maps things you can say to actions.
106 #
107 # Note the relationship between the *mapping* and *extras* keyword
108 # arguments. The extras is a list of Dragonfly elements which are
109 # available to be used in the specs of the mapping. In this example
110 # the IntegerRef("n") extra makes it possible to use "<n>" within a
111 # mapping spec and pass an integer "n" to the associated action's
112 # function.
113 
114 class FunctionExamplesRule(MappingRule):
115  mapping = {
116  # Call the complete() function.
117  "function complete": Function(complete),
118 
119  # Type capitalized text.
120  "[dictation] capitalize <text>": Function(cap_text),
121 
122  # Say numbers.
123  "say number <n>": Function(say_number),
124  "say double number <n>": Function(lambda n: say_number(n * 2)),
125 
126  # Control the volume.
127  "volume <volume_mode> [<n>]": Function(volume_control),
128 
129  # Retrieves address and country of <any location>.
130  # 1. Say "get Yellowstone National Park".
131  "get <location>": Function(get_location),
132  # The address and country of <any location> can be retrieved
133  # afterwards.
134  # 2. Say "what is the country".
135  # 3. Say "what is the address".
136  "what is [the] <location_info>": Function(get_last_location_info),
137 
138  # Print secret fruits and animals mapped to spelling words.
139  # The values of 'stuff' and 'things' will also be printed.
140  "secret <fruit> [<animals>]": Function(busy_func, stuff="shhhh...",
141  things="more"),
142  }
143 
144  extras = [
145  IntegerRef("n", 1, 50),
146  Dictation("text"),
147  Dictation("location"),
148 
149  Choice("volume_mode", {
150  "mute": "mute",
151  "up": "up",
152  "down": "down",
153  }),
154 
155  Choice("fruit", {
156  "alpha": "apple",
157  "bravo": "blueberry",
158  "charlie": "cherry",
159  }),
160 
161  Choice("animals", {
162  "delta": "dogs",
163  "echo": "emus",
164  "foxtrot": "foxes",
165  }),
166 
167  Choice("location_info", {
168  "address": "address",
169  "country": "country",
170  }),
171  ]
172 
173  defaults = {
174  "n": 1,
175  "animals": "",
176  "text": ""
177  }
