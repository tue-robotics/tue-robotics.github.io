108  def get_properties(self):
109  """
110  Method to get the properties of a macOS window.
111 
112  :rtype: dict
113  :returns: window properties
114  """
115  script = '''
116  tell application "System Events" to tell application process id "%s"
117  try
118  get properties of window 1
119  on error errmess
120  log errmess
121  end try
122  end tell
123  ''' % self._id
124  properties = applescript.AppleScript(script).run()
125  if not properties:
126  return {}
127 
128  result = {}
129  encoding = locale.getpreferredencoding()
130  for key, value in properties.items():
131  key = key.code
132  if isinstance(key, binary_type):
133  key = key.decode(encoding)
134  if isinstance(value, applescript.AEType):
135  value = value.code
136  if isinstance(value, binary_type):
137  value = value.decode(encoding)
138  result[key] = value
139  return result
