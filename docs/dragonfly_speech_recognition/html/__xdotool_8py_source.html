34 
35 def _run_xdotool_command(arguments):
36  arguments = [str(arg) for arg in arguments]
37  full_command = ["xdotool"] + arguments
38  kwargs = dict(stdout=PIPE, stderr=PIPE)
39 
40  # Fork the process with setsid() if on a POSIX system.
41  if os.name == 'posix':
42  kwargs.update(dict(preexec_fn=os.setsid))
43 
44  # Execute the command with Popen, logging an error and re-raising
45  # the exception on failure.
46  try:
47  p = Popen(full_command, **kwargs)
48  stdout, stderr = p.communicate()
49  except OSError as exception:
50  raise exception
51 
52  # Decode output if it is binary.
53  encoding = locale.getpreferredencoding()
54  if isinstance(stdout, binary_type):
55  stdout = stdout.decode(encoding)
56  if isinstance(stderr, binary_type):
57  stderr = stderr.decode(encoding)
58 
59  # Print error messages to stderr. Filter BadWindow messages.
60  stderr = stderr.rstrip()
61  if stderr:
62  print(stderr, file=sys.stderr)
63 
64  # Return the process output and return code.
65  return stdout.rstrip(), p.returncode

71 def get_cursor_position():
72  # Get the position of the cursor on the screen using xdotool.
73  arguments = ["getmouselocation", "--shell"]
74  stdout, returncode = _run_xdotool_command(arguments)
75  result = None
76  if returncode == 0:
77  lines = stdout.split("\n")
78  assert lines[0].startswith("X=")
79  assert lines[1].startswith("Y=")
80  x = int(lines[0][2:])
81  y = int(lines[1][2:])
82  result = (x, y)
83  return result
