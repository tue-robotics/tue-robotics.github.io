231  def vad_collector(self, start_window_ms=150, start_padding_ms=100,
232  end_window_ms=150, end_padding_ms=None, complex_end_window_ms=None,
233  ratio=0.8, blocks=None, nowait=False, audio_auto_reconnect=False,
234  ):
235  """Generator/coroutine that yields series of consecutive audio blocks comprising each phrase, separated by yielding a single None.
236  Determines voice activity by ratio of blocks in window_ms. Uses a buffer to include window_ms prior to being triggered.
237  Example: (block, ..., block, None, block, ..., block, None, ...)
238  |----phrase-----| |----phrase-----|
239  """
240  assert end_padding_ms == None, "end_padding_ms not supported yet"
241  num_start_window_blocks = max(1, int(start_window_ms // self.BLOCK_DURATION_MS))
242  num_start_padding_blocks = max(0, int((start_padding_ms or 0) // self.BLOCK_DURATION_MS))
243  num_end_window_blocks = max(1, int(end_window_ms // self.BLOCK_DURATION_MS))
244  num_complex_end_window_blocks = max(1, int((complex_end_window_ms or end_window_ms) // self.BLOCK_DURATION_MS))
245  num_end_padding_blocks = max(0, int((end_padding_ms or 0) // self.BLOCK_DURATION_MS))
246  _log.debug("%s: vad_collector: num_start_window_blocks=%s num_end_window_blocks=%s num_complex_end_window_blocks=%s",
247  self, num_start_window_blocks, num_end_window_blocks, num_complex_end_window_blocks)
248  audio_reconnect_threshold_blocks = 5
249  audio_reconnect_threshold_time = 50 * self.BLOCK_DURATION_MS / 1000
250 
251  ring_buffer = collections.deque(maxlen=max(
252  (num_start_window_blocks + num_start_padding_blocks),
253  (num_end_window_blocks + num_end_padding_blocks),
254  (num_complex_end_window_blocks + num_end_padding_blocks),
255  ))
256  ring_buffer_recent_slice = lambda num_blocks: itertools.islice(ring_buffer, max(0, (len(ring_buffer) - num_blocks)), None)
257 
258  triggered = False
259  in_complex_phrase = False
260  num_empty_blocks = 0
261  last_good_block_time = time.time()
262 
263  if blocks is None: blocks = self.iter(nowait=nowait)
264  for block in blocks:
265  if block is False or block is None:
266  # Bad/empty block
267  num_empty_blocks += 1
268  if audio_auto_reconnect and (num_empty_blocks >= audio_reconnect_threshold_blocks) and (time.time() - last_good_block_time >= audio_reconnect_threshold_time):
269  _log.warning("%s: no good block received recently, so reconnecting audio", self)
270  self.reconnect()
271  num_empty_blocks = 0
272  last_good_block_time = time.time()
273  in_complex_phrase = yield False
274 
275  else:
276  # Good block
277  num_empty_blocks = 0
278  last_good_block_time = time.time()
279  is_speech = self.vad.is_speech(block, self.SAMPLE_RATE)
280 
281  if not triggered:
282  # Between phrases
283  ring_buffer.append((block, is_speech))
284  num_voiced = len([1 for (_, speech) in ring_buffer_recent_slice(num_start_window_blocks) if speech])
285  if num_voiced >= (num_start_window_blocks * ratio):
286  # Start of phrase
287  triggered = True
288  for block, _ in ring_buffer_recent_slice(num_start_padding_blocks + num_start_window_blocks):
289  # print('|' if is_speech else '.', end='')
290  # print('|' if in_complex_phrase else '.', end='')
291  in_complex_phrase = yield block
292  # print('#', end='')
293  ring_buffer.clear()
294 
295  else:
296  # Ongoing phrase
297  in_complex_phrase = yield block
298  # print('|' if is_speech else '.', end='')
299  # print('|' if in_complex_phrase else '.', end='')
300  ring_buffer.append((block, is_speech))
301  num_unvoiced = len([1 for (_, speech) in ring_buffer_recent_slice(num_end_window_blocks) if not speech])
302  num_complex_unvoiced = len([1 for (_, speech) in ring_buffer_recent_slice(num_complex_end_window_blocks) if not speech])
303  if (not in_complex_phrase and num_unvoiced >= (num_end_window_blocks * ratio)) or \
304  (in_complex_phrase and num_complex_unvoiced >= (num_complex_end_window_blocks * ratio)):
305  # End of phrase
306  triggered = False
307  in_complex_phrase = yield None
308  # print('*')
309  ring_buffer.clear()
310 
311  if triggered:
312  # We were in a phrase, so we must terminate it (this may be abrupt!)
313  yield None

328 class AudioStore(object):
329  """
330  Stores the current audio data being recognized, which is cleared upon calling `finalize()`.
331  Also, optionally stores the last `maxlen` recognitions as `AudioStoreEntry` objects,
332  indexed in reverse order (0 is most recent), and advanced upon calling `finalize()`.
333  Note: `finalize()` should be called after the recognition has been parsed and its actions executed.
334 
335  Constructor arguments:
336  - *maxlen* (*int*, default *None*): if set, the number of previous recognitions to temporarily store.
337  - *save_dir* (*str*, default *None*): if set, the directory to save the `retain.tsv` file and optionally wav files.
338  - *save_metadata* (*bool*, default *None*): whether to automatically save the recognition metadata.
339  - *save_audio* (*bool*, default *None*): whether to automatically save the recognition audio data (in addition to just the recognition metadata).
340  - *retain_approval_func* (*Callable*, default *None*): if set, will be called with the `AudioStoreEntry` object about to be saved,
341  and should return `bool` whether to actually save. Example: `retain_approval_func=lambda entry: bool(entry.grammar_name != 'noisegrammar')`
342  """

444  @classmethod
445  def read_file(cls, filename, realtime=False):
446  """ Yields raw audio blocks from wav file, terminated by a None element. """
447  if not os.path.isfile(filename):
448  raise IOError("'%s' is not a file. Please use a different file path.")
449 
450  with contextlib.closing(wave.open(filename, 'rb')) as file:
451  # Validate the wave file's header
452  if file.getnchannels() != MicAudio.CHANNELS:
453  raise ValueError("WAV file '%s' should use %d channel(s), not %d!"
454  % (filename, MicAudio.CHANNELS, file.getnchannels()))
455  elif file.getsampwidth() != MicAudio.SAMPLE_WIDTH:
456  raise ValueError("WAV file '%s' should use sample width %d, not "
457  "%d!" % (filename, MicAudio.SAMPLE_WIDTH, file.getsampwidth()))
458  elif file.getframerate() != MicAudio.SAMPLE_RATE:
459  raise ValueError("WAV file '%s' should use sample rate %d, not "
460  "%d!" % (filename, MicAudio.SAMPLE_RATE, file.getframerate()))
461 
462  next_time = time.time()
463  for _ in range(0, int(file.getnframes() / MicAudio.BLOCK_SIZE_SAMPLES) + 1):
464  data = file.readframes(MicAudio.BLOCK_SIZE_SAMPLES)
465  if not data:
466  break
467  if realtime:
468  time_behind = next_time - time.time()
469  if time_behind > 0:
470  time.sleep(time_behind)
471  next_time += float(MicAudio.BLOCK_SIZE_SAMPLES) / MicAudio.SAMPLE_RATE
472  yield data
473  yield None
