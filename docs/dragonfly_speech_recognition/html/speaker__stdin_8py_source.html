47  def speak(self, text):
48  """ Speak the given *text* using text-to-speech. """
49  if len(self._read_stdin_command) == 0:
50  raise NotImplementedError("Virtual method not implemented.")
51 
52  # Encode input, if necessary.
53  encoding = getpreferredencoding()
54  if isinstance(text, text_type):
55  text = text.encode(encoding)
56 
57  # Speak *text* using the specified command.
58  p = Popen(self._read_stdin_command, stdout=PIPE, stdin=PIPE,
59  stderr=PIPE)
60  stdout, stderr = p.communicate(input=text)
61  returncode = p.wait()
62 
63  # Decode output if it is binary.
64  if isinstance(stdout, binary_type): stdout = stdout.decode(encoding)
65  if isinstance(stderr, binary_type): stderr = stderr.decode(encoding)
66  if stdout: print(stdout)
67  if stderr: print(stderr, file=sys.stderr)
68 
69  # Handle non-zero return codes.
70  if returncode != 0:
71  raise RuntimeError("%s exited with non-zero return code %d"
72  % (self.name, p.returncode))
