107  # Test format_hdrop with existing file paths.
108  with NamedTemporaryFile() as f1, NamedTemporaryFile() as f2:
109  # Strings are converted to tuples containing file paths. Test
110  # using Unicode and binary strings.
111  # First, test using one existing file.
112  for test_string in [u"%s\0\0", u"%s\n\n", u"%s\0", u"%s\n"]:
113  test_string = test_string % f1.name
114  f1name = u"%s" % f1.name
115  self.assertEqual(convert(format_hdrop, test_string),
116  (f1name,))
117  test_string = test_string.encode()
118  f1name = f1name.encode()
119  self.assertEqual(convert(format_hdrop, test_string),
120  (f1name,))
121 
122  # Then, test using two.
123  for test_string in ["%s\0%s\0\0", "%s\n%s\0", "%s\n%s\n"]:
124  test_string = test_string % (f1.name, f2.name)
125  f1name, f2name = u"%s" % f1.name, u"%s" % f2.name
126  self.assertEqual(convert(format_hdrop, test_string),
127  (f1name, f2name))
128  test_string = test_string.encode()
129  f1name, f2name = f1name.encode(), f2name.encode()
130  self.assertEqual(convert(format_hdrop, test_string),
131  (f1name, f2name))

134  # converted to tuples.
