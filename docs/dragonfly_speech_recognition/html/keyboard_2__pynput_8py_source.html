44 class SafeKeyCode(object):
45  """
46  Class to safely get key codes from pynput.
47  """
48 
49  def __getattr__(self, name):
50  # Get the key code from pynput, returning KeyCode(vk=-1, char=name)
51  # if the key name isn't present. Keys are undefined on some
52  # platforms, e.g. "pause" on Darwin.
53  return getattr(Key, name, KeyCode(vk=-1, char=name))

228 class PynputKeyboard(BaseKeyboard):
229  """Static class wrapper around pynput.keyboard."""
230 
231  _controller = None
232 
233  @classmethod
234  def send_keyboard_events(cls, events):
235  """
236  Send a sequence of keyboard events.
237 
238  Positional arguments:
239  events -- a sequence of tuples of the form
240  (keycode, down, timeout), where
241  keycode (str|KeyCode): pynput key code.
242  down (boolean): True means the key will be pressed down,
243  False means the key will be released.
244  timeout (int): number of seconds to sleep after
245  the keyboard event.
246 
247  """
248  # Initialize the pynput keyboard controller, if necessary.
249  if cls._controller is None:
250  cls._controller = Controller()
251 
252  # Send keyboard events.
253  cls._log.debug("Keyboard.send_keyboard_events %r", events)
254  for event in events:
255  (key, down, timeout) = event
256 
257  # Raise an error if the key is unsupported. 'key' can also be a
258  # string, e.g. "a", "b", "/", etc, but we don't check if those
259  # are valid.
260  if isinstance(key, KeyCode) and key.vk == -1:
261  raise ValueError("Unsupported key: %r" % key.char)
262 
263  # Press/release the key, catching any errors.
264  try:
265  cls._controller.touch(key, down)
266  except Exception as e:
267  cls._log.exception("Failed to type key code %s: %s",
268  key, e)
269 
270  # Sleep after the keyboard event if necessary.
271  if timeout:
272  time.sleep(timeout)
273

275  def get_typeable(cls, char, is_text=False):
276  return PynputTypeable(char, is_text=is_text)
