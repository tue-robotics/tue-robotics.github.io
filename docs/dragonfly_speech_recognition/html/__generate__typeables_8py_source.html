218 print("# ---- Code for typeables.py")
219 for group_name, indent_level, group_map in key_map:
220  indent = indent_level * " "
221  print("%s# %s" % (indent , group_name))
222  for key_type, key_value, key_names in group_map:
223  for key_name in key_names.split():
224  if key_type == lookup:
225  print('%s_add_typeable(name="%s", char=\'%s\')'
226  % (indent, key_name, key_value))
227  elif key_type == vkey:
228  value_code = ("Typeable(code=%s, name=%r)"
229  % (key_value, key_name))
230  print(('%s"%s": %s%s,'
231  % (indent, key_name, " " * (16-len(key_name)),
232  value_code)))
233  else:
234  raise Exception("Invalid key type: {0!r} (for {1!r} {2!r})"
235  .format(key_type, key_value, key_names))
236  print("")

238 print("# ---- Code for documentation in action_key.py")
239 # Exclude reserved characters in the generated documentation.
240 excluded = "-,:/"
241 for group_name, _, group_map in key_map:
242  parts = [group_name + ":"]
243  for key_type, key_value, key_names in group_map:
244  key_name_parts = []
245  for key_name in key_names.split():
246  if key_name not in excluded:
247  key_name = key_name.decode("string_escape")
248  key_name_parts.append("``" + key_name + "``")
249  parts.append(" or ".join(key_name_parts) + ",")
250  parts[-1] = parts[-1][:-1] # Remove trailing comma.
251  line = " -"
252  for part in parts:
