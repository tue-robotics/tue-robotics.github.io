155  def _get_window_module(self):
156  # Get this window's process ID.
157  pid = c_ulong()
158  windll.user32.GetWindowThreadProcessId(self._handle, pointer(pid))
159 
160  # Get the process handle of this window's process ID.
161  # Access permission flags:
162  # 0x0410 = PROCESS_QUERY_INFORMATION | PROCESS_VM_READ
163  handle = windll.kernel32.OpenProcess(0x0410, 0, pid)
164 
165  # Retrieve and return the process's executable path.
166  try:
167  # Try to use the QueryForProcessImageNameW function
168  # available since Windows Vista.
169  buffer_len = c_ulong(256)
170  buffer = (c_wchar * buffer_len.value)()
171  windll.kernel32.QueryFullProcessImageNameW(handle, 0,
172  pointer(buffer),
173  pointer(buffer_len))
174  buffer = buffer[:]
175  buffer = buffer[:buffer.index("\0")]
176  except Exception:
177  # If the function above failed, fall back to the older
178  # GetModuleFileNameEx function, available since windows XP.
179  # Note that this fallback function seems to fail when
180  # this process is 32 bit Python and handle refers to a
181  # 64-bit process.
182  buffer_len = 256
183  buffer = (c_wchar * buffer_len)()
184  windll.psapi.GetModuleFileNameExW(handle, 0, pointer(buffer),
185  buffer_len)
186  buffer = buffer[:]
187  buffer = buffer[:buffer.index("\0")]
188  finally:
189  windll.kernel32.CloseHandle(handle)
190 
191  return str(buffer)
