21 """
22 Loader of language-dependent objects
23 ============================================================================
24 
25 This file implements the loader for language-dependent objects within
26 the dragonfly.language sub-package. At runtime it attempts to figure
27 out what the current user's language is, and then loads the appropriate
28 objects.
29 
30 Examples of language-dependent objects within Dragonfly are the Integer,
31 Digits, and Number classes.
32 
33 """

52  # Mapping of language tags to module names. Each of the module names
53  # should implement the appropriate classes for the given
54  # speaker language.
55 
56  language_map = {
57  "ar": "dragonfly.language.ar",
58  "de": "dragonfly.language.de",
59  "en": "dragonfly.language.en",
60  "id": "dragonfly.language.id",

66  def __init__(self, language=None):
67  self._language = language
68  self._module = None
69 
70  def __getattr__(self, name):
71  return self.get_attribute(name)
72 
73  def get_attribute(self, name):
74  # Make sure the appropriate language-specific module has been
75  # loaded and is available.
76  if not self._module:
77  self._load_module()
78 
79  # Attempt to retrieve the requested attribute from the
80  # language-specific module.
81  try:
82  return getattr(self._module, name)
83  except AttributeError:
84  # Raise an error unless getting ShortIntegerContent.
85  if name != "ShortIntegerContent":
86  raise DragonflyError("Language %r does not implement %r."
87  % (self._language, name))
88  else:
89  self._log.warning("Language %r does not implement %r, "
90  "falling back on %r."
91  %(self._language, name, "IntegerContent"))
92  return getattr(self._module, "IntegerContent")
93 
94  def _load_module(self):
95  if not self._language:
96  self._language = self._get_engine_language()
97  self._module = self._load_module_for_language(self._language)
98 
99  def _get_engine_language(self):
100  try:
101  from dragonfly.engines import get_engine
102  language = get_engine().language
103  except Exception as e:
104  self._log.exception("Failed to retrieve speaker language: %s" % e)
105  raise
106  return language
107 
108  def _load_module_for_language(self, language):
109  # Lookup the module name for the given language.
110  try:
111  module_name = self.language_map[language]
112  except KeyError:
113  raise DragonflyError("Attempt to load unknown language: %r"
114  % language)
115 
116  # Attempt to import the language-dependent module.
117  try:
118  top_module = __import__(module_name)
119  except ImportError as e:
120  self._log.exception("Failed to load module %r for language %r"
121  % (module_name, language))
122  raise
123  try:
124  module = sys.modules[module_name]
125  except KeyError:
126  self._log.exception("Failed to load module %r for language %r"
127  % (module_name, language))
128  raise
129 
130  return module
