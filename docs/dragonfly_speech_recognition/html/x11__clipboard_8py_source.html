111  def copy_from_system(self, formats=None, clear=False):
112  # pylint: disable=too-many-branches
113  # Determine which formats to retrieve.
114  supported_formats = (self.format_text, self.format_unicode,
115  self.format_hdrop, self.format_x_primary,
116  self.format_x_secondary)
117  caller_specified_formats = bool(formats)
118  if not formats:
119  formats = supported_formats
120  elif isinstance(formats, integer_types):
121  formats = (formats,)
122 
123  # Verify that the given formats are valid.
124  for format in formats:
125  if not isinstance(format, integer_types):
126  raise TypeError("Invalid clipboard format: %r"
127  % format)
128 
129  # Retrieve the system clipboard content.
130  try:
131  clipboard_text = self._get_x_selection(self.format_x_clipboard)
132  except TypeError:
133  clipboard_text = u""
134  contents = {}
135 
136  # Populate the clipboard contents dictionary and raise errors for
137  # unavailable formats.
138  for format in formats:
139  err = False
140  try:
141  # Retrieve and use other X selections as text, if required.
142  if format in (self.format_x_primary,
143  self.format_x_secondary):
144  text = self._get_x_selection(format)
145  else:
146  text = clipboard_text
147 
148  # Attempt to convert and set text for this clipboard format.
149  content = self.convert_format_content(format, text)
150  contents[format] = content
151  except (ValueError, TypeError):
152  err = True
153 
154  # Do not raise errors for formats that the caller did not
155  # specify.
156  if format in supported_formats and not caller_specified_formats:
157  err = False
158 
159  # Always raise an error if the format is not supported.
160  if format not in supported_formats:
161  err = True
162 
163  # Raise an error if a specified clipboard format was not
164  # available.
165  if err:
166  format_repr = self.format_names.get(format, format)
167  message = ("Specified clipboard format %r is not "
168  "available." % format_repr)
169  raise TypeError(message)
170  self._contents = contents
171 
172  # Then clear the system clipboard, if requested.
173  if clear:
174  self.clear_clipboard()

176  def copy_to_system(self, clear=True):
177  # Clear the system clipboard, if requested.
178  if clear:
179  self.clear_clipboard()
180 
181  # Transfer content to the X selections.
182  # Text content is put on the clipboard selection, if necessary.
183  text = self.get_text()
184  if text is not None:
185  self._set_x_selection(self.format_x_clipboard, text)
186 
187  # CF_HDROP content is put on the clipboard selection, if necessary.
188  # This might not work properly.
189  elif self.has_format(self.format_hdrop):
190  file_paths = self.get_format(self.format_hdrop)
191  content = u"\n".join(file_paths) + u"\n"
192  self._set_x_selection(self.format_x_clipboard, content)
193 
194  # Set the primary selection content, if necessary.
195  if self.has_format(self.format_x_primary):
196  content = self.get_format(self.format_x_primary)
197  self._set_x_selection(self.format_x_primary, content)
198 
199  # Set the secondary selection content, if necessary.
200  if self.has_format(self.format_x_secondary):
201  content = self.get_format(self.format_x_secondary)
202  self._set_x_selection(self.format_x_secondary, content)

208 class XselClipboard(BaseX11Clipboard):
209  """
210  Class for interacting with X selections (clipboards) using xsel.
211 
212  This is Dragonfly's default clipboard class on X11/Linux.
213 
214  """
215 
216  @classmethod
217  def _run_command(cls, command, arguments, input_data=None):
218  """
219  Run a command with arguments and return the result.
220  """
221  arguments = [str(arg) for arg in arguments]
222  full_command = [command] + arguments
223  full_readable_command = ' '.join(full_command)
224  cls._log.debug(full_readable_command)
225  try:
226  # Execute the child process, passing input data if necessary.
227  encoding = locale.getpreferredencoding()
228  popen_kwargs = dict(stdout=PIPE, stderr=PIPE)
229  comm_kwargs = {}
230  if input_data is not None:
231  popen_kwargs["stdin"] = PIPE
232  if not isinstance(input_data, binary_type):
233  input_data = input_data.encode(encoding)
234  comm_kwargs["input"] = input_data
235  p = Popen(full_command, **popen_kwargs)
236  stdout, stderr = p.communicate(**comm_kwargs)
237 
238  # Decode output if it is binary.
239  if isinstance(stdout, binary_type):
240  stdout = stdout.decode(encoding)
241  if isinstance(stderr, binary_type):
242  stderr = stderr.decode(encoding)
243 
244  # Print error messages to stderr.
245  if stderr:
246  print(stderr, file=sys.stderr)
247 
248  # Return the process output and return code.
249  return stdout, p.returncode
250  except OSError as e:
251  cls._log.error("Failed to execute command '%s': %s. Is "
252  "%s installed?",
253  full_readable_command, e, command)
254  raise e

256  @classmethod
257  def _get_x_selection(cls, format):
258  if format == cls.format_x_primary:
259  arguments = ["--primary", "-o"]
260  elif format == cls.format_x_secondary:
261  arguments = ["--secondary", "-o"]
262  elif format == cls.format_x_clipboard:
263  arguments = ["--clipboard", "-o"]
264  else:
265  raise ValueError("Invalid X selection: %r" % format)
266 
267  stdout, return_code = cls._run_command("xsel", arguments)
268  if return_code != 0 or not stdout:
269  format_repr = cls.format_names.get(format, format)
270  message = ("Specified X selection %r is not available."
271  % format_repr)
272  raise TypeError(message)
273 
274  return stdout

276  @classmethod
277  def _set_x_selection(cls, format, content):
278  if format == cls.format_x_primary:
279  arguments = ["--primary", "-i"]
280  elif format == cls.format_x_secondary:
281  arguments = ["--secondary", "-i"]
282  elif format == cls.format_x_clipboard:
283  arguments = ["--clipboard", "-i"]
284  else:
285  raise ValueError("Invalid X selection: %r" % format)
286  _, return_code = cls._run_command("xsel", arguments, content)
287  if return_code != 0:
288  format_repr = cls.format_names.get(format, format)
289  message = ("Specified X selection %r could not be set."
290  % format_repr)
