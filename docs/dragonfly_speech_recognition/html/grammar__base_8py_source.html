39 class Grammar(object):
40  """
41  Grammar class for managing a set of rules.
42 
43  This base grammar class takes care of the communication
44  between Dragonfly's object model and the backend speech
45  recognition engine. This includes compiling rules and
46  elements, loading them, activating and deactivating
47  them, and unloading them. It may, depending on the
48  engine, also include receiving recognition results and
49  dispatching them to the appropriate rule.
50 
51  - *name* -- name of this grammar
52  - *description* (str, default: None) --
53  description for this grammar
54  - *context* (Context, default: None) --
55  context within which to be active. If *None*, the grammar will
56  always be active.
57 
58  """

198  def add_rule(self, rule):
199  """
200  Add a rule to this grammar.
201 
202  The following rules apply when adding rules into grammars:
203 
204  #. Rules **cannot** be added to grammars that are currently loaded.
205  #. Two or more rules with the same name are **not** allowed.
206 
207  .. warning::
208 
209  Note that while adding the same ``Rule`` object to more than one
210  grammar is allowed, it is **not** recommended! This is because
211  the context and active/enabled states of these rules will not
212  function correctly if used. It is better to use *separate*
213  ``Rule`` instances for each grammar instead.
214 
215  :param rule: Dragonfly rule
216  :type rule: Rule
217  """
218  self._log_load.debug("Grammar %s: adding rule %s.",
219  self._name, rule.name)
220 
221  # Check for correct type and duplicate rules or rule names.
222  if self._loaded:
223  raise GrammarError("Cannot add rule while loaded.")
224  elif not isinstance(rule, Rule):
225  raise GrammarError("Invalid rule object: %s" % rule)
226  elif rule in self._rules:
227  return
228  elif rule.imported:
229  return
230  elif [True for r in self._rules if r.name == rule.name]:
231  raise GrammarError("Two rules with the same name '%s' not"
232  " allowed." % rule.name)
233  elif rule.grammar is not None and rule.exported:
234  self._log_load.warning("Exported rule %s is already in grammar "
235  "%s, adding it to grammar %s is not "
236  "recommended.", rule.name,
237  rule.grammar.name, self._name)
238 
239  # Append the rule to this grammar object's internal list.
240  self._rules.append(rule)
241  rule.grammar = self

468  def get_complexity_string(self):
469  """
470  Build and return a human-readable text giving insight into the
471  complexity of this grammar.
472 
473  """
474  rules_all = self.rules
475  rules_top = [r for r in self.rules if r.exported]
476  rules_imp = [r for r in self.rules if r.imported]
477  elements = []
478  for rule in rules_all:
479  elements.extend(self._get_element_list(rule))
480  text = ("Grammar: %3d (%3d, %3d) rules, %4d elements (%3d avg) %s"
481  % (
482  len(rules_all), len(rules_top), len(rules_imp),
483  len(elements), len(elements) / len(rules_all),
484  self,
485  )
486  )
487  for rule in rules_all:
488  elements = self._get_element_list(rule)
489  text += "\n Rule: %4d %s" % (len(elements), rule)
490  return text

505  def process_begin(self, executable, title, handle):
506  """
507  Start of phrase callback.
508 
509  *Usually derived grammar classes override
510  ``Grammar._process_begin`` instead of this method, because
511  this method merely wraps that method adding context matching.*
512 
513  This method is called when the speech recognition
514  engine detects that the user has begun to speak a
515  phrase.
516 
517  Arguments:
518  - *executable* -- the full path to the module whose
519  window is currently in the foreground.
520  - *title* -- window title of the foreground window.
521  - *handle* -- window handle to the foreground window.
522 
523  """
524  # pylint: disable=expression-not-assigned
525 
526  self._log_begin.debug("Grammar %s: detected beginning of "
527  "utterance.", self._name)
528  self._log_begin.debug("Grammar %s: executable '%s', title '%s'.",
529  self._name, executable, title)
530 
531  if not self._enabled:
532  # Grammar is disabled, so deactivate all active rules.
533  [r.deactivate() for r in self._rules if r.active]
534 
535  elif not self._context \
536  or self._context.matches(executable, title, handle):
537  # Grammar is within context.
538  if not self._in_context:
539  self._in_context = True
540  self.enter_context()
541  self._process_begin(executable, title, handle)
542  for r in self._rules:
543  if r.exported and hasattr(r, "process_begin"):
544  r.process_begin(executable, title, handle)
545 
546  else:
547  # Grammar's context doesn't match, deactivate active rules.
548  if self._in_context:
549  self._in_context = False
550  self.exit_context()
551  [r.deactivate() for r in self._rules if r.active]
552 
553  self._log_begin.debug("Grammar %s: active rules: %s.",
554  self._name,
555  [r.name for r in self._rules if r.active])
