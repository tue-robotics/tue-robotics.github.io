77  @classmethod
78  def get_matching_windows(cls, executable=None, title=None):
79  """
80  Find windows with a matching executable or title.
81 
82  Window searches are case-insensitive.
83 
84  If neither parameter is be specified, then it is effectively the
85  same as calling :meth:`get_all_windows`.
86 
87  :param executable: -- part of the filename of the application's
88  executable to which the target window belongs; not case
89  sensitive.
90  :param title: -- part of the title of the target window; not case
91  sensitive.
92  :type executable: str
93  :type title: str
94  :rtype: list
95  """
96  # Make window searches case-insensitive.
97  if executable:
98  executable = executable.lower()
99  if title:
100  title = title.lower()
101 
102  matching = []
103  for window in cls.get_all_windows():
104  if executable:
105  if window.executable.lower().find(executable) == -1:
106  continue
107  if title:
108  if window.title.lower().find(title) == -1:
109  continue
110  matching.append(window)
111  return matching

274  def get_containing_monitor(self):
275  """

277 
278  This checks which monitor contains the center of the window.

282  """
