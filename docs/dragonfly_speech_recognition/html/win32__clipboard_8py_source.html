47 class win32_clipboard_ctx(object):
48  """
49  Python context manager for safely opening the Windows clipboard by
50  polling for access, timing out after the specified number of seconds.
51 
52  Arguments:
53  - *timeout* (float, default: 0.5) -- timeout in seconds.
54  - *step* (float, default: 0.001) -- number of seconds between each
55  attempt to open the clipboard.
56 
57  Notes:
58  - The polling is necessary because the clipboard is a shared resource
59  which may be in use by another process.
60  - Nested usage will not close the clipboard early.
61 
62  Use with a Python 'with' block::
63 
64  with win32_clipboard_ctx():
65  # Do clipboard operation(s).
66  win32clipboard.EmptyClipboard()
67 
68  """
69 
70  _ctx_data = threading.local()
71  _log = logging.getLogger("clipboard")
72 
73  def __init__(self, timeout=0.5, step=0.001):
74  self._timeout = float(timeout)
75  self._step = float(step)
76 
77  def __enter__(self):
78  timeout = time.time() + self._timeout
79  success = False
80  while time.time() < timeout:
81  # Attempt to open the clipboard, catching Windows errors.
82  try:
83  win32clipboard.OpenClipboard()
84  success = True
85  break
86  except pywintypes.error:
87  # Failure. Try again after *step* seconds.
88  time.sleep(self._step)
89 
90  # Try opening the clipboard one more time if it still isn't open.
91  # If this fails, then an error will be raised this time.
92  if not success:
93  win32clipboard.OpenClipboard()
94 
95  # The clipboard is open now. Increment our thread-local counter to
96  # keep track of nested usage.
97  ctx_data = self._ctx_data
98  ctx_data.count = getattr(ctx_data, "count", 0) + 1
99 
100  def __exit__(self, exc_type, exc_val, exc_tb):
101  # The clipboard is (presumably) still open. Decrement our thread-
102  # local counter and close the clipboard if it is at 0 again.
103  ctx_data = self._ctx_data
104  ctx_data.count -= 1
105  if ctx_data.count == 0:
106  win32clipboard.CloseClipboard()

163  @classmethod
164  def _wait_for_change(cls, timeout, step, formats, initial_clipboard,
165  seq_no):
166  # This method determines if the system clipboard has changed by
167  # repeatedly checking the current sequence number. Contents are
168  # compared if specific clipboard formats are given.
169  timeout = time.time() + float(timeout)
170  step = float(step)
171  if isinstance(formats, integer_types):
172  formats = (formats,)
173  elif formats:
174  for format in formats:
175  if not isinstance(format, integer_types):
176  raise TypeError("Invalid clipboard format: %r"
177  % format)
178  clipboard2 = cls() if formats or initial_clipboard else None
179  result = False
180  while time.time() < timeout:
181  seq_no_change = (win32clipboard.GetClipboardSequenceNumber()
182  != seq_no)
183  if seq_no_change and initial_clipboard:
184  # Check if the content of any relevant format has changed.
185  clipboard2.copy_from_system()
186  formats_to_compare = formats if formats else "all"
187  result = cls._clipboard_formats_changed(formats_to_compare,
188  initial_clipboard,
189  clipboard2)
190 
191  # Reset the sequence number if the clipboard change is not
192  # related.
193  if not result:
194  seq_no = win32clipboard.GetClipboardSequenceNumber()
195  elif seq_no_change:
196  result = True
197 
198  if result:
199  break
200 
201  # Failure. Try again after *step* seconds.
202  time.sleep(step)
203  return result

234  def _update_format_names(self, formats):
235  # Retrieve and store the names of any registered clipboard formats
236  # in format_names, if necessary.
237  for format in formats:
238  if format in self.format_names:
239  continue
240 
241  # Registered clipboard formats are identified by values in
242  # the range 0xC000 through 0xFFFF.
243  if 0xC000 <= format <= 0xFFFF:
244  format_name = win32clipboard.GetClipboardFormatName(format)
245  self.format_names[format] = format_name

247  def copy_from_system(self, formats=None, clear=False):
248  with win32_clipboard_ctx():
249  # Determine which formats to retrieve.
250  caller_specified_formats = bool(formats)
251  if not formats:
252  format = 0
253  formats = []
254  while 1:
255  format = win32clipboard.EnumClipboardFormats(format)
256  if not format:
257  break
258  formats.append(format)
259  elif isinstance(formats, integer_types):
260  formats = (formats,)
261 
262  # Verify that the given formats are valid.
263  for format in formats:
264  if not isinstance(format, integer_types):
265  raise TypeError("Invalid clipboard format: %r"
266  % format)
267 
268  # Update clipboard format names.
269  self._update_format_names(formats)
270 
271  # Retrieve Windows system clipboard content.
272  contents = {}
273  for format in formats:
274  # Allow GetClipboardData() to raise Windows API errors if
275  # the caller specified one or more formats. If the
276  # function raises an error at this point, it is likely the
277  # format is not retrievable and the caller will want to
278  # know, since they explicitly asked for this format.
279  if caller_specified_formats:
280  content = win32clipboard.GetClipboardData(format)
281  contents[format] = content
282  continue
283 
284  # Otherwise, catch and log Windows API errors. We assume
285  # here that the caller is not much interested in
286  # irretrievable formats.
287  try:
288  content = win32clipboard.GetClipboardData(format)
289  contents[format] = content
290  except pywintypes.error as err:
291  format_repr = self.format_names.get(format, format)
292  self._log.warning("Could not retrieve clipboard data "
293  "for format %r: %s", format_repr, err)
294 
295  self._contents = contents
296 
297  # Clear the system clipboard, if requested, and close it.
298  if clear:
299  win32clipboard.EmptyClipboard()
