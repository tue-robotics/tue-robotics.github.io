22 RunCommand action
23 ============================================================================
24 
25 The :class:`RunCommand` action takes a command-line program to run including
26 any required arguments. On execution, the program will be started as a
27 subprocess.
28 
29 Processing will occur asynchronously by default. Commands running
30 asynchronously should not normally prevent the Python process from exiting.
31 
32 It may sometimes be necessary to use a list for the action's *command*
33 argument instead of a string. This is because some command-line shells may
34 not work 100% correctly with Python's built-in :meth:`shlex.split` function.
35 
36 This action should work on Windows and other platforms.
37 
38 Example using the ping command::
39 
40  from dragonfly import RunCommand
41 
42  # Ping localhost for 4 seconds.
43  RunCommand('ping -w 4 localhost').execute()
44 
45 
46 Example using a command list instead of a string::
47 
48  from dragonfly import RunCommand

51  RunCommand(['ping', '-w', '4', 'localhost']).execute()

152  def __init__(self, command=None, process_command=None,
153  synchronous=False, hide_window=True):
154  """
155  Constructor arguments:
156  - *command* (str or list) -- the command to run when this
157  action is executed. It will be parsed by :meth:`shlex.split`
158  if it is a string and passed directly to ``subprocess.Popen``
159  if it is a list. Command arguments can be included.
160  - *process_command* (callable) -- optional callable to invoke
161  with the :class:`Popen` object after successfully starting
162  the subprocess. Using this argument overrides the
163  :meth:`process_command` method.
164  - *synchronous* (bool, default *False*) -- whether to wait
165  until :meth:`process_command` has finished executing before
166  continuing.
167  - *hide_window* (bool, default *True*) -- whether to hide the
168  application window. Set to *False* if using this action with
169  GUI programs. This argument only applies to Windows. It has
170  no effect on other platforms.
171 
172  """
173  ActionBase.__init__(self)
174  self._proc = None
175 
176  # Complex handling of arguments because of clashing use of the names
177  # at the class level: property & class-value.

181  if not (self.command and isinstance(self.command, command_types)):
182  raise TypeError("command must be a non-empty string or list, "
183  "not %s" % self.command)
184  if synchronous is not False:
185  self.synchronous = synchronous

208  # pylint: disable=no-self-use
209  def process_command(self, proc):
210  """
211  Method to override for custom handling of the command's
212  :class:`Popen` object.
213 
214  By default this method prints lines from the subprocess until it
215  exits.
216  """
217  encoding = locale.getpreferredencoding()
218  for line in iter(proc.stdout.readline, b''):
219  if isinstance(line, binary_type):
220  line = line.decode(encoding)
221 
222  print(line, end='')

224  def _execute(self, data=None):
225  self._log.info("Executing: %s", self.command)
226 
227  # Suppress showing the new CMD.exe window on Windows.
228  startupinfo = None
229  if os.name == 'nt' and self._hide_window:
230  startupinfo = subprocess.STARTUPINFO()
231  startupinfo.dwFlags |= subprocess.STARTF_USESHOWWINDOW
232 
233  # Pre-process self.command before passing it to subprocess.Popen.
234  command = self.command
235  if isinstance(command, string_types):
236  # Split command strings using shlex before passing it to Popen.
237  # Use POSIX mode only if on a POSIX platform.
238  command = shlex.split(command, posix=os.name == "posix")
239  try:
240  self._proc = subprocess.Popen(command,
241  stdout=subprocess.PIPE,
242  stderr=subprocess.STDOUT,
243  stdin=subprocess.PIPE,
244  startupinfo=startupinfo)
245  except Exception as e:
246  self._log.exception("Exception from starting subprocess %s: "
247  "%s", self._str, e)
248  return False
249 
250  # Call process_command either synchronously or asynchronously.
251  def call():
252  try:
253  if self._process_command:
254  process_func = self._process_command
255  else:
256  process_func = self.process_command
257  process_func(self._proc)
258  return_code = self._proc.wait()
259  if return_code != 0:
260  self._log.error("Command %s failed with return code "
261  "%d", self._str, return_code)
262  return False
263  return True
264  except Exception as e:
265  self._log.exception("Exception processing command %s: %s",
266  self._str, e)
267  return False
268  finally:
269  self._proc = None
270 
271  if self.synchronous:
272  return call()
273 
274  # Execute in a new daemonized thread so that the command cannot
275  # stop the SR engine from exiting.
276  thread = threading.Thread(target=call)
277  thread.setDaemon(True)
278  thread.start()
279  return True
