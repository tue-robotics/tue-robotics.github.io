76  def _process_observer_callbacks(self, cb_name, required_names,
77  **kwargs):
78  for observer in self._observers:

83  # If the callback function takes keyword arguments, only send
84  # those that it accepts. Always pass required names.
85  func_kwargs = kwargs

89  if not kwargs_names:

91  if k in arg_names or k in required_names}
