22  This module is a simple example of Dragonfly use.
23 
24  It shows how to use Dragonfly's Grammar, AppContext, and MappingRule
25  classes. This module can be activated in the same way as other

28 """

30 from dragonfly import (Grammar, AppContext, MappingRule, Dictation,
31  Key, Text)
32 
33 
34 #---------------------------------------------------------------------------
35 # Create this module's grammar and the context under which it'll be active.
36 
37 grammar_context = AppContext(executable="notepad")
38 grammar = Grammar("notepad_example", context=grammar_context)

40 
41 #---------------------------------------------------------------------------
42 # Create a mapping rule which maps things you can say to actions.
43 #
44 # Note the relationship between the *mapping* and *extras* keyword
45 # arguments. The extras is a list of Dragonfly elements which are
46 # available to be used in the specs of the mapping. In this example
47 # the Dictation("text")* extra makes it possible to use "<text>"
48 # within a mapping spec and "%(text)s" within the associated action.
49 
50 example_rule = MappingRule(
51  name="example", # The name of the rule.
52  mapping={ # The mapping dict: spec -> action.
53  "save [file]": Key("c-s"),
54  "save [file] as": Key("a-f, a"),
55  "save [file] as <text>": Key("a-f, a/20") + Text("%(text)s"),
56  "find <text>": Key("c-f/20") + Text("%(text)s\n"),
57  },
58  extras=[ # Special elements in the specs of the mapping.
59  Dictation("text"),
60  ],
61  )
62 
63 # Add the action rule to the grammar instance.
64 grammar.add_rule(example_rule)
65 
66

69 
70 grammar.load()

74  global grammar
