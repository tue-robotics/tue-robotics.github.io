42 def is_engine_available(**kwargs):
43  """
44  Check whether Natlink is available.
45 
46  :param \\**kwargs: optional keyword arguments passed through to the
47  engine for engine-specific configuration.
48  """
49  # pylint: disable=too-many-return-statements
50  global _engine
51  if _engine:
52  return True
53 
54  platform_name = platform.system()
55  if platform_name != 'Windows':
56  _log.warning("%s is not supported by the Natlink engine backend",
57  platform_name)
58  return False
59 
60  if struct.calcsize("P") == 8: # 64-bit
61  _log.warning("The python environment is 64-bit. Natlink requires a "
62  "32-bit python environment")
63  return False
64 
65  # Attempt to import natlink.
66  try:
67  import natlink
68  except ImportError as e:
69  # Add Natlink's default 'core' directory path to sys.path if
70  # necessary.
71  coredir_path = r'C:\\NatLink\\NatLink\\MacroSystem\\core'
72  pyd_filename = 'natlink.pyd'
73  pyd_path = os.path.join(coredir_path, pyd_filename)
74  import_failure = True
75  if os.path.isdir(coredir_path):
76  if not os.path.isfile(pyd_path):
77  _log.warning("Requested engine 'natlink' is not available: "
78  "The %r file is missing from Natlink's core "
79  "directory", pyd_filename)
80  return False
81 
82  # Add the core directory to the path and try importing again.
83  sys.path.append(coredir_path)
84  try:
85  import natlink
86  import_failure = False
87  except ImportError:
88  pass
89  if import_failure:
90  _log.warning("Requested engine 'natlink' is not available: "
91  "Natlink is not installed: %s", e)
92  return False
93  except Exception as e:
94  _log.exception("Exception during import of natlink package: "
95  "%s", e)
96  return False
97 
98  try:
99  if natlink.isNatSpeakRunning():
100  return True
101  else:
102  _log.warning("Requested engine 'natlink' is not available: "
103  "Dragon NaturallySpeaking is not running")
104  return False
105  except Exception as e:
106  _log.exception("Exception during natlink.isNatSpeakRunning(): "
107  "%s", e)
108  return False
