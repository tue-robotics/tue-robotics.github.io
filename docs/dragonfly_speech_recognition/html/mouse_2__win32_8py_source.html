107  def execute(self, window):
108  # Ensure that the primary mouse button is the *left* button before
109  # sending events.
110  primary_changed = windll.user32.SwapMouseButton(0)
111  try:
112  # Prepare and send the mouse events.
113  zero = pointer(c_ulong(0))
114  inputs = [MouseInput(0, 0, flag[1], flag[0], 0, zero)
115  for flag in self._flags]
116  array = make_input_array(inputs)
117  send_input_array(array)
118  finally:
119  # Swap the primary mouse button back if it was previously set
120  # to *right*.
121  if primary_changed:
122  windll.user32.SwapMouseButton(1)
