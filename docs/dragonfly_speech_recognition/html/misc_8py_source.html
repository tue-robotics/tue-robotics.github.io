34 def get_decoder_config_object():
35  """ Get the default Sphinx decoder object for the engine. """
36  decoder_config = DefaultConfig()
37 
38  # Silence the decoder output by default.
39  decoder_config.set_string("-logfn", os.devnull)
40 
41  # Set voice activity detection configuration options for the decoder.
42  # These values can be changed if noise in the background triggers speech
43  # start and/or false recognitions (e.g. of short words) frequently.
44  # Descriptions for VAD config options were retrieved from:
45  # https://cmusphinx.github.io/doc/sphinxbase/fe_8h_source.html
46 
47  # Number of silence frames to keep after from speech to silence.
48  decoder_config.set_int("-vad_postspeech", 30)
49 
50  # Number of speech frames to keep before silence to speech.
51  decoder_config.set_int("-vad_prespeech", 20)
52 
53  # Number of speech frames to trigger vad from silence to speech.
54  decoder_config.set_int("-vad_startspeech", 10)
55 
56  # Threshold for decision between noise and silence frames. Log-ratio
57  # between signal level and noise level.
58  decoder_config.set_float("-vad_threshold", 3.0)
59  return decoder_config
