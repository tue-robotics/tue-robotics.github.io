38 class Paste(DynStrActionBase):
39 
40  """
41  Paste-from-clipboard action.
42 
43  Constructor arguments:
44  - *contents* (*str* | *dict*) -- contents to paste. This may be a
45  simple string to paste, a dynamic action *spec* or a dictionary
46  of clipboard format ints to contents (typically strings).
47  - *format* (*int*, clipboard format integer) --
48  clipboard format. This argument is ignored if *contents* is a
49  dictionary.
50  - *paste* (instance derived from *ActionBase*) --
51  paste action
52  - *static* (boolean) --
53  flag indicating whether the
54  specification contains dynamic elements
55 
56  This action inserts the given *contents* into the system
57  clipboard, and then performs the *paste* action to paste it into
58  the foreground application. By default, the *paste* action is the
59  :kbd:`Ctrl-v` keystroke or :kbd:`Super-v` on a mac. The default
60  clipboard format used by this action is the *Unicode* text format.
61 
62  """

71  if not format:

73  if paste is None:
74  # Pass use_hardware=True to guarantee that Ctrl+V is always

76  paste = Key(self._default_paste_spec, use_hardware=True)

93  def _execute_events(self, events):
94  original = Clipboard()
95  try:

98  self._log.warning("Failed to store original clipboard contents:"
