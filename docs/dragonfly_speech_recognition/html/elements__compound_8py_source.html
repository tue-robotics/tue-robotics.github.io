229  def __init__(self, spec, extras=None, actions=None, name=None,
230  value=None, value_func=None, elements=None, default=None):
231  self._spec = spec
232  self._value = value
233  self._value_func = value_func
234 
235  if extras is None: extras = {}
236  if actions is None: actions = {}
237  if elements is None: elements = {}
238 
239  # Convert extras argument from sequence to mapping.
240  if isinstance(extras, (tuple, list)):
241  mapping = {}
242  for element in extras:
243  if not isinstance(element, elements_.ElementBase):
244  self._log.error("Invalid extras item: %s" % element)
245  raise TypeError("Invalid extras item: %s" % element)
246  if not element.name:
247  self._log.error("Extras item does not have a name: %s" % element)
248  raise TypeError("Extras item does not have a name: %s" % element)
249  if element.name in mapping:
250  self._log.warning("Multiple extras items with the same name: %s" % element)
251  mapping[element.name] = element
252  extras = mapping
253  elif not isinstance(extras, dict):
254  self._log.error("Invalid extras argument: %s" % extras)
255  raise TypeError("Invalid extras argument: %s" % extras)
256 
257  # Temporary transition code so that both "elements" and "extras"
258  # are supported as keyword arguments.
259  if extras and elements:
260  extras = dict(extras)
261  extras.update(elements)
262  elif elements:
263  extras = elements
264  self._extras = extras

266  # This solution is non-ideal as "stuff" is a global instance.
267  stuff.set_elements(extras)
268  stuff.set_actions(actions)
269 
270  element = self._parser.parse(spec)
271  if not element:
272  self._log.error("Invalid compound spec: %r" % spec)
273  raise SyntaxError("Invalid compound spec: %r" % spec)

283  def value(self, node):
284  if self._value_func is not None:
285  # Prepare *extras* dict for passing to value_func().
286  extras = {"_node": node}
287  for name, element in self._extras.iteritems():
