22 Compound element classes
23 ============================================================================
24 
25 The following special *element* classes exist as convenient ways of
26 constructing basic element types from string specifications:
27 
28  * :class:`Compound` --
29  a special element which parses a string spec to create a hierarchy of
30  basic elements.
31 
32  * :class:`Choice` --
33  a special element taking a :code:`choice` dictionary argument,
34  interpreting keys as :class:`Compound` string specifications and values
35  for what to return when compound specs are successfully decoded during
36  the recognition process.
37 
38  The :code:`choice` argument may also be a list or tuple of strings, in
39  which case the strings are also interpreted as :class:`Compound` strings
40  specifications. However, the values returned when compound specs are
41  successfully decoded during the recognition process are the recognized
42  words. **Note**: these values will be matching part(s) of the compound
43  specs.

63 class Compound(Alternative):
64  """
65  Element which parses a string spec to create a hierarchy of basic
66  elements.
67 
68  Constructor arguments:
69  - *spec* (*str*) -- compound specification
70  - *extras* (sequence) -- extras elements referenced from the
71  compound spec
72  - *actions* (*dict*) -- this argument is currently unused
73  - *name* (*str*) -- the name of this element
74  - *value* (*object*, default: *None*) --
75  value to be returned when this element is successfully decoded
76  If *None*, then the value(s) of child nodes are used instead
77  - *value_func* (*callable*, default: *None*) --
78  function to be called for the node value when this element is
79  successfully decoded. If *None*, then the value(s) of child nodes
80  are used. This argument takes precedence over the *value*
81  argument if it is present
82  - *elements* (sequence) -- same as the extras argument
83  - *default* (default: *None*) -- the default value of this element
84 
85  Example:
86 
87  .. code:: python
88 
89  # Define a command to type the sum of two spoken integers between
90  # 1 and 50 using a Compound element.
91  mapping = {
92  "type <XAndY>": Text("%(XAndY)d"),
93  }
94  extras = [
95  Compound(
96  # Pass the compound spec and element name.
97  spec="<x> and <y>",
98  name="XAndY",
99 
100  # Pass the internal IntegerRef extras.
101  extras=[IntegerRef("x", 1, 50), IntegerRef("y", 1, 50)],
102 
103  # Pass a value function that adds the two spoken integers
104  # together.
105  value_func=lambda node, extras: extras["x"] + extras["y"])
106  ]
107 
108  """

128  if isinstance(extras, (tuple, list)):
129  mapping = {}
130  for element in extras:
131  if not isinstance(element, ElementBase):
132  self._log.error("Invalid extras item: %s", element)
133  raise TypeError("Invalid extras item: %s" % element)
134  if not element.name:

136  raise TypeError("Extras item does not have a name: %s" % element)

176  def value(self, node):
177  if self._value_func is not None:
178  # Prepare *extras* dict for passing to value_func().
179  extras = {"_node": node}
180  for name, element in self._extras.items():
181  extra_node = node.get_child_by_name(name, shallow=True)
182  if extra_node:
183  extras[name] = extra_node.value()
184  elif element.has_default():
185  extras[name] = element.default
186  try:
187  value = self._value_func(node, extras)
188  except Exception as e:

190  raise

192  elif self._value is not None:

201 class Choice(Alternative):
202  """
203  Element allowing a dictionary of phrases (compound specs) to be
204  recognized to be mapped to objects to be used in an action.
205 
206  A list or tuple of phrases to be recognized may also be used. In
207  this case the strings are also interpreted as :class:`Compound`
208  string specifications. However, the values returned when compound
209  specs are successfully decoded during the recognition process are
210  the recognized words. **Note**: these values will be matching
211  part(s) of the compound specs.
212 
213  Constructor arguments:
214  - *name* (*str*) -- the name of this element
215  - *choices* (*dict*, *list* or *tuple*) -- dictionary mapping
216  recognized phrases to returned values **or** a list/tuple of
217  recognized phrases
218  - *extras* (*list*, default: *None*) -- a list of included
219  extras
220  - *default* (default: *None*) -- the default value of this
221  element
222 
223  Example using a dictionary:
224 
225  .. code:: python
226 
227  # Tab switching command, e.g. 'third tab'.
228  mapping = {
229  "<nth> tab": Key("c-%(nth)s"),
230  }
231  extras = [
232  Choice("nth", {
233  "first" : "1",
234  "second" : "2",
235  "third" : "3",
236  "fourth" : "4",
237  "fifth" : "5",
238  "sixth" : "6",
239  "seventh" : "7",
240  "eighth" : "8",
241  "(last | ninth)": "9",
242  "next" : "pgdown",
243  "previous" : "pgup",
244  }),
245  ]
246 
247  Example using a list:
248 
249  .. code:: python
250 
251  # Command for recognizing and typing nth words, e.g.
252  # 'type third'.
253  mapping = {
254  "type <nth>": Text("%(nth)s"),
255  }
256  extras = [
257  Choice("nth", [
258  "first",
259  "second",
260  "third",
261  "fourth",
262  "fifth",
263  "sixth",
264  "seventh",
265  "eighth",
266 
267  # Note that the decoded value for a compound spec like
268  # this one, when used in a list/tuple of choices,
269  # rather than a dictionary, is the recognized word:
270  # "last" or "ninth".
271  "(last | ninth)",
272 
273  "next",
274  "previous",
275  ]),
276  ]
277  """
278  def __init__(self, name, choices, extras=None, default=None):
279  # Argument type checking.
280  assert isinstance(name, string_types) or name is None
281  assert isinstance(choices, (dict, list, tuple))
282  choices_is_sequence = isinstance(choices, (list, tuple))
283  if choices_is_sequence:
284  choices = {k: None for k in choices}
285 
286  for k, v in choices.items():
287  assert isinstance(k, string_types)
288 
289  # Construct children from the given choice keys and values.
290  self._choices = choices
291  self._extras = extras
292  children = []
293  for k, v in choices.items():
294  child = Compound(spec=k, value=v, extras=extras)
295  children.append(child)
