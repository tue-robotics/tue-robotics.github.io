21 """
22 Fundamental element classes
23 ============================================================================
24 
25 Dragonfly grammars are built up out of a small set of fundamental building
26 blocks. These building blocks are implemented by the following *element*
27 classes:
28 
29  - :class:`ElementBase` --
30  the base class from which all other element classes are derived
31  - :class:`Sequence` --
32  sequence of child elements which must all match in the order given
33  - :class:`Alternative` --
34  list of possibilities of which only one will be matched
35  - :class:`Optional` --
36  wrapper around a child element which makes the child element optional
37  - :class:`Repetition` --
38  repetition of a child element
39  - :class:`Literal` --
40  literal word which must be said exactly by the speaker as given
41  - :class:`RuleRef` --
42  reference to a :class:`dragonfly.grammar.rule_base.Rule` object;
43  this element allows a rule to include (i.e. reference) another rule
44  - :class:`ListRef` --
45  reference to a :class:`dragonfly.grammar.list.List` object
46 
47 The following *element* classes are built up out of the fundamental
48 classes listed above:
49 
50  - :class:`Dictation` --
51  free-form dictation; this element matches any words the speaker says,
52  and includes facilities for formatting the spoken words with correct
53  spacing and capitalization
54  - :class:`DictListRef` --
55  reference to a :class:`dragonfly.all.DictList` object; this element is
56  similar to the :class:`dragonfly.all.ListRef` element, except that it
57  returns the value associated with the spoken words instead of the
58  spoken words themselves
59 
60 """

116  def element_tree_string(self):
117  """
118  Returns a formatted multi-line string representing this
119  element and its children.
120 
121  """
122  indent = " "
123  tree = []
124  stack = [(self, 0)]
125  while stack:
126  element, index = stack.pop()
127  if index == 0:
128  tree.append((element, len(stack)))
129  if len(element.children) > index:
130  stack.append((element, index + 1))
131  stack.append((element.children[index], 0))
132  lines = (indent*depth + str(element) for element, depth in tree)
133  return "\n".join(lines)

238 class Sequence(ElementBase):
239  """
240  Element class representing a sequence of child elements
241  which must all match a recognition in the correct order.
242 
243  Constructor arguments:
244  - *children* (iterable, default: *()*) --
245  the child elements of this element
246  - *name* (*str*, default: *None*) --
247  the name of this element
248 
249  For a recognition to match, all child elements must match
250  the recognition in the order that they were given in the
251  *children* constructor argument.
252 
253  Example usage:
254  >>> from dragonfly.test import ElementTester
255  >>> seq = Sequence([Literal("hello"), Literal("world")])
256  >>> test_seq = ElementTester(seq)
257  >>> test_seq.recognize("hello world")
258  ['hello', 'world']
259  >>> test_seq.recognize("hello universe")
260  RecognitionFailure
261 
262  """
263 
264  def __init__(self, children=(), name=None, default=None):
265  ElementBase.__init__(self, name=name, default=default)
266  self._children = self._copy_sequence(children,
267  "children", ElementBase)

294  # Methods for runtime recognition processing.
295 
296  def decode(self, state):
297  state.decode_attempt(self)
298 
299  # Special case for an empty sequence.
300  if len(self._children) == 0:
301  state.decode_success(self)
302  yield state
303  state.decode_retry(self)
304  state.decode_failure(self)
305  return
306 
307  # Attempt to walk a path through the entire sequence of children
308  # so that each one decodes successfully.
309  path = [self._children[0].decode(state)]
310  while path:
311  # Allow the last child to attempt decoding.
312  try: path[-1].next()
313  except StopIteration:
314  # Last child failed to decode, remove from path to
315  # allowed the one-before-last child to reattempt.
316  path.pop()
317  else:
318  # Last child successfully decoded.
319  if len(path) < len(self._children):
320  # Sequence not yet complete, append the next child.
321  path.append(self._children[len(path)].decode(state))
322  else:
323  # Sequence complete, all children decoded successfully.
324  state.decode_success(self)
325  yield state
326  state.decode_retry(self)
327 
328  # Sequence of children could not all decode successfully: failure.
329  state.decode_failure(self)
330  return

333  """
334  The *value* of a :class:`Sequence` is a list containing
335  the values of each of its children.
336 
337  """
338  return [child.value() for child in node.children]

388  #-----------------------------------------------------------------------
389  # Methods for runtime recognition processing.
390 
391  def decode(self, state):
392  state.decode_attempt(self)
393 
394  # If in greedy mode, allow the child to decode before.
395  if self._greedy:
396  for result in self._child.decode(state):
397  state.decode_success(self)
398  yield state
399  state.decode_retry(self)
400 
401  # Rollback decoding so that the null-decode can be yielded.
402  state.decode_rollback(self)
403 
404  # Yield the null-decode possibility.
405  state.decode_success(self)
406  yield state
407  state.decode_retry(self)
408 
409  # If not in greedy mode, allow the child to decode after.
410  if not self._greedy:
411  for result in self._child.decode(state):
412  state.decode_success(self)
413  yield state
414  state.decode_retry(self)
415 
416  # No more decoding possibilities available, failure.
417  state.decode_failure(self)
418  return

490  state.decode_success(self)

525 class Repetition(Sequence):
526  """
527  Element class representing a repetition of one child element.
528 
529  Constructor arguments:
530  - *child* (*ElementBase*) --
531  the child element of this element
532  - *min* (*int*, default: *1*) --
533  the minimum number of times that the child element must
534  be recognized; may be 0
535  - *max* (*int*, default: *None*) --
536  the maximum number of times that the child element must
537  be recognized; if *None*, the child element must be recognized
538  exactly *min* times (i.e. *max = min + 1*)
539  - *name* (*str*, default: *None*) --
540  the name of this element
541 
542  For a recognition to match, at least one of the child elements
543  must match the recognition. The first matching child is
544  used. Child elements are searched in the order they are given
545  in the *children* constructor argument.
546 
547  """
548 
549  def __init__(self, child, min=1, max=None, name=None, default=None):
550  if not isinstance(child, ElementBase):
551  raise TypeError("Child of %s object must be an"
552  " ElementBase instance." % self)
553  assert isinstance(min, int)
554  assert max is None or isinstance(max, int)
555  assert max is None or min < max
556 
557  self._child = child
558  self._min = min
559  if max is None: self._max = min + 1
560  else: self._max = max
561 
562  optional_length = self._max - self._min - 1
563  if optional_length > 0:
564  element = Optional(child)
565  for index in xrange(optional_length):
566  element = Optional(Sequence([child, element]))
567 
568  if self._min >= 1:
569  children = [child] * self._min + [element]
570  else:
571  children = [element]
572  elif self._min > 0:
573  children = [child] * self._min
574  else:
575  raise ValueError("Repetition not allowed to be empty.")
576 
577  Sequence.__init__(self, children, name=name, default=default)
578 
579  def dependencies(self, memo):
580  if self in memo:
581  return []
582  memo.append(self)
583  return self._child.dependencies(memo)

586  # Methods for runtime recognition processing.

588  def get_repetitions(self, node):
589  """
590  Returns a list containing the nodes associated with
591  each repetition of this element's child element.
592 
593  Argument:
594  - *node* (*Node*) --
595  the parse tree node associated with this repetition element;
596  necessary for searching for child elements within the parse
597  tree
598 
599  """
600  repetitions = []
601  for index in xrange(self._min):
602  element = node.children[index]
603  if element.actor != self._child:
604  raise TypeError("Invalid child of %s: %s" \
605  % (self, element.actor))
606  repetitions.append(element)
607 
608  if self._max - self._min > 1:
609  optional = node.children[-1]
610  while optional.children:
611  child = optional.children[0]
612  if isinstance(child.actor, Sequence):
613  assert len(child.children) == 2
614  element, optional = child.children
615  if element.actor != self._child:
616  raise TypeError("Invalid child of %s: %s" \
617  % (self, element.actor))
618  repetitions.append(element)
619  elif child.actor == self._child:
620  repetitions.append(child)
621  break
622  else:
623  raise TypeError("Invalid child of %s: %s" \
624  % (self, child.actor))
625 
626  return repetitions

630  The *value* of a :class:`Repetition` is a list containing

632 
633  The length of this list is equal to the number of times
634  that the child element was recognized.
635 
636  """
637  repetitions = self.get_repetitions(node)

801 
802  # If the next word(s) is/are in the list, success.
803  delta = 0
