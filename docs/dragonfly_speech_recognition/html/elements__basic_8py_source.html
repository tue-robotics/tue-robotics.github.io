21 """
22 Fundamental element classes
23 ============================================================================
24 
25 Dragonfly grammars are built up out of a small set of fundamental building
26 blocks. These building blocks are implemented by the following *element*
27 classes:
28 
29  - :class:`ElementBase` --
30  the base class from which all other element classes are derived
31  - :class:`Sequence` --
32  sequence of child elements which must all match in the order given
33  - :class:`Alternative` --
34  list of possibilities of which only one will be matched
35  - :class:`Optional` --
36  wrapper around a child element which makes the child element optional
37  - :class:`Repetition` --
38  repetition of a child element
39  - :class:`Literal` --
40  literal word which must be said exactly by the speaker as given
41  - :class:`RuleRef` --
42  reference to a :class:`dragonfly.grammar.rule_base.Rule` object;
43  this element allows a rule to include (i.e. reference) another rule
44  - :class:`ListRef` --
45  reference to a :class:`dragonfly.grammar.list.List` object
46  - :class:`Impossible` --
47  a special element that cannot be recognized
48  - :class:`Empty` --
49  a special element that is always recognized
50 
51 The following *element* classes are built up out of the fundamental
52 classes listed above:
53 
54  - :class:`Dictation` --
55  free-form dictation; this element matches any words the speaker says,
56  and includes facilities for formatting the spoken words with correct
57  spacing and capitalization
58  - :class:`Modifier` --
59  modifies the output of another element by applying a function to it
60  following recognition
61  - :class:`DictListRef` --
62  reference to a :class:`dragonfly.DictList` object; this element is
63  similar to the :class:`dragonfly.ListRef` element, except that it returns
64  the value associated with the spoken words instead of the spoken words
65  themselves
66  - :class:`RuleWrap` --
67  an element class used to wrap a Dragonfly element into a new private
68  rule to be referenced by the same element or other :class:`RuleRef`
69  elements
70 
71 See the following documentation sections for additional information and
72 usage examples:
73 
74  * :ref:`Elements (Object model section) <RefObjectModelElements>`
75  * :ref:`RefElementBasicDocTests`
76 
77 """

144  def element_tree_string(self):
145  """
146  Returns a formatted multi-line string representing this
147  element and its children.
148 
149  """
150  indent = " "
151  tree = []
152  stack = [(self, 0)]
153  while stack:
154  element, index = stack.pop()
155  if index == 0:
156  tree.append((element, len(stack)))
157  if len(element.children) > index:
158  stack.append((element, index + 1))
159  stack.append((element.children[index], 0))
160  lines = (indent*depth + str(element) for element, depth in tree)
161  return "\n".join(lines)

262 class Sequence(ElementBase):
263  """
264  Element class representing a sequence of child elements
265  which must all match a recognition in the correct order.
266 
267  Constructor arguments:
268  - *children* (iterable, default: *()*) --
269  the child elements of this element
270  - *name* (*str*, default: *None*) --
271  the name of this element
272  - *default* (*object*, default: *None*) --
273  the default value used if this element is optional and wasn't
274  spoken
275 
276  For a recognition to match, all child elements must match
277  the recognition in the order that they were given in the
278  *children* constructor argument.
279 
280  Example usage::
281 
282  >>> from dragonfly.test import ElementTester
283  >>> seq = Sequence([Literal("hello"), Literal("world")])
284  >>> test_seq = ElementTester(seq)
285  >>> test_seq.recognize("hello world")
286  ['hello', 'world']
287  >>> test_seq.recognize("hello universe")
288  RecognitionFailure
289 
290  """
291 
292  def __init__(self, children=(), name=None, default=None):
293  ElementBase.__init__(self, name=name, default=default)
294  self._children = self._copy_sequence(children,
295  "children", ElementBase)

321  #-----------------------------------------------------------------------
322  # Methods for runtime recognition processing.
323 
324  def decode(self, state):
325  state.decode_attempt(self)
326 
327  # Special case for an empty sequence.
328  if len(self._children) == 0:
329  state.decode_success(self)
330  yield state
331  state.decode_retry(self)
332  state.decode_failure(self)
333  return
334 
335  # Attempt to walk a path through the entire sequence of children
336  # so that each one decodes successfully.
337  path = [self._children[0].decode(state)]
338  while path:
339  # Allow the last child to attempt decoding.
340  try: next(path[-1])
341  except StopIteration:
342  # Last child failed to decode, remove from path to
343  # allowed the one-before-last child to reattempt.
344  path.pop()
345  else:
346  # Last child successfully decoded.
347  if len(path) < len(self._children):
348  # Sequence not yet complete, append the next child.
349  path.append(self._children[len(path)].decode(state))
350  else:
351  # Sequence complete, all children decoded successfully.
352  state.decode_success(self)
353  yield state
354  state.decode_retry(self)
355 
356  # Sequence of children could not all decode successfully: failure.
357  state.decode_failure(self)
358  return
359 
360  def value(self, node):
361  """
362  The *value* of a :class:`Sequence` is a list containing
363  the values of each of its children.
364 
365  """
366  return [child.value() for child in node.children]

371 class Optional(ElementBase):
372  """
373  Element class representing an optional child element.
374 
375  Constructor arguments:
376  - *child* (*ElementBase*) --
377  the child element of this element
378  - *name* (*str*, default: *None*) --
379  the name of this element
380  - *default* (*object*, default: *None*) --
381  the default value used if this element is optional and wasn't
382  spoken
383 
384  Recognitions always match this element. If the child element
385  does match the recognition, then that result is used.
386  Otherwise, this element itself does match but the child
387  is not processed.
388 
389  """
390 
391  def __init__(self, child, name=None, default=None):
392  ElementBase.__init__(self, name, default=default)
393 
394  if not isinstance(child, ElementBase):
395  raise TypeError("Child of %s object must be an"
396  " ElementBase instance." % self)
397  self._child = child
398  self._greedy = True
399 
400  #-----------------------------------------------------------------------
401  # Methods for runtime introspection.
402 
403  def _get_children(self):
404  """ Returns the optional child element. """
405  return (self._child, )
406 
407  #-----------------------------------------------------------------------
408  # Methods for load-time setup.
409 
410  def dependencies(self, memo):
411  if self._id in memo:
412  return []
413  memo.add(self._id)
414  return self._child.dependencies(memo)
415 
416  def gstring(self):
417  return "[" + self._child.gstring() + "]"
418 
419  #-----------------------------------------------------------------------
420  # Methods for runtime recognition processing.
421 
422  def decode(self, state):
423  # pylint: disable=unused-variable
424  state.decode_attempt(self)
425 
426  # If in greedy mode, allow the child to decode before.
427  if self._greedy:
428  for result in self._child.decode(state):
429  state.decode_success(self)
430  yield state
431  state.decode_retry(self)
432 
433  # Rollback decoding so that the null-decode can be yielded.
434  state.decode_rollback(self)
435 
436  # Yield the null-decode possibility.
437  state.decode_success(self)
438  yield state
439  state.decode_retry(self)
440 
441  # If not in greedy mode, allow the child to decode after.
442  if not self._greedy:
443  for result in self._child.decode(state):
444  state.decode_success(self)
445  yield state
446  state.decode_retry(self)
447 
448  # No more decoding possibilities available, failure.
449  state.decode_failure(self)
450 
451  def value(self, node):
452  """
453  The *value* of a :class:`Optional` is the value of its child,
454  if the child did match the recognition. Otherwise the
455  *value* is *None*.
456 
457  """
458  if node.children:
459  return node.children[0].value()
460  else:
461  return None

519  def decode(self, state):
520  state.decode_attempt(self)
521 
522  # Special case for an empty list of alternatives.
523  if len(self._children) == 0:
524  state.decode_success(self)
525  yield state
526  state.decode_retry(self)
527  state.decode_failure(self)
528  return
529 
530  # Iterate through the children.
531  for child in self._children:
532 
533  # Iterate through this child's possible decoding states.
534  # pylint: disable=unused-variable
535  for result in child.decode(state):
536  state.decode_success(self)
537  yield state
538  state.decode_retry(self)
539

541  # next child starts decoding without interference from the
542  # previous child.
543  state.decode_rollback(self)
544 
545  # None of the children could decode successfully: failure.
546  state.decode_failure(self)
547  return

560 class Repetition(Sequence):
561  """
562  Element class representing a repetition of one child element.
563 
564  Constructor arguments:
565  - *child* (*ElementBase*) --
566  the child element of this element
567  - *min* (*int*, default: *1*) --
568  the minimum number of times that the child element must
569  be recognized; may be 0
570  - *max* (*int*, default: *None*) --
571  the maximum number of times that the child element must
572  be recognized; if *None*, the child element must be recognized
573  exactly *min* times (i.e. *max = min + 1*)
574  - *name* (*str*, default: *None*) --
575  the name of this element
576  - *default* (*object*, default: *None*) --
577  the default value used if this element is optional and wasn't
578  spoken
579  - *optimize* (*bool*, default: *True*) --
580  whether the engine's compiler should compile the element
581  optimally
582 
583  For a recognition to match, at least one of the child elements
584  must match the recognition. The first matching child is
585  used. Child elements are searched in the order they are given
586  in the *children* constructor argument.
587 
588  If the *optimize* argument is set to *True*, the compiler will
589  ignore the *min* and *max* limits to reduce grammar complexity. If
590  the number of repetitions recognized is more than the *max* value,
591  the rule will fail to match.
592 
593  """
594 
595  # pylint: disable=redefined-builtin,unused-variable
596 
597  def __init__(self, child, min=1, max=None, name=None, default=None,
598  optimize=True):
599  if not isinstance(child, ElementBase):
600  raise TypeError("Child of %s object must be an"
601  " ElementBase instance." % self)
602  assert isinstance(min, integer_types)
603  assert max is None or isinstance(max, integer_types)
604  assert max is None or min < max, "min must be less than max"
605 
606  self._child = child
607  self._min = min
608  if max is None: self._max = min + 1
609  else: self._max = max
610  self._optimize = optimize
611 
612  optional_length = self._max - self._min - 1
613  if optional_length > 0:
614  element = Optional(child)
615  for index in range(optional_length-1):
616  element = Optional(Sequence([child, element]))
617 
618  if self._min >= 1:
619  children = [child] * self._min + [element]
620  else:
621  children = [element]
622  elif self._min > 0:
623  children = [child] * self._min
624  else:
625  raise ValueError("Repetition not allowed to be empty.")
626 
627  Sequence.__init__(self, children, name=name, default=default)
628 
629  min = property(
630  lambda self: self._min,
631  doc="The minimum number of times that the child element must be "
632  "recognized; may be 0. (Read-only)"
633  )
634 
635  max = property(
636  lambda self: self._max,
637  doc="The maximum number of times that the child element must be "
638  "recognized; if *None*, the child element must be "
639  "recognized exactly *min* times (i.e. *max = min + 1*). "
640  "(Read-only)"
641  )
642 
643  optimize = property(
644  lambda self: self._optimize,
645  doc="Whether the engine's compiler should compile the element "
646  "optimally. (Read-only)"
647  )
648 
649  def dependencies(self, memo):
650  if self._id in memo:
651  return []
652  memo.add(self._id)
653  return self._child.dependencies(memo)

658  def get_repetitions(self, node):
659  """
660  Returns a list containing the nodes associated with
661  each repetition of this element's child element.
662 
663  Argument:
664  - *node* (*Node*) --
665  the parse tree node associated with this repetition element;
666  necessary for searching for child elements within the parse
667  tree
668 
669  """
670  repetitions = []
671  for index in range(self._min):
672  element = node.children[index]
673  if element.actor != self._child:
674  raise TypeError("Invalid child of %s: %s" \
675  % (self, element.actor))
676  repetitions.append(element)
677 
678  if self._max - self._min > 1:
679  optional = node.children[-1]
680  while optional.children:
681  child = optional.children[0]
682  if isinstance(child.actor, Sequence):
683  assert len(child.children) == 2
684  element, optional = child.children
685  if element.actor != self._child:
686  raise TypeError("Invalid child of %s: %s" \
687  % (self, element.actor))
688  repetitions.append(element)
689  elif child.actor == self._child:
690  repetitions.append(child)
691  break
692  else:
693  raise TypeError("Invalid child of %s: %s" \
694  % (self, child.actor))
695 
696  return repetitions

698  def value(self, node):
699  """
700  The *value* of a :class:`Repetition` is a list containing
701  the values of its child.
702 
703  The length of this list is equal to the number of times
704  that the child element was recognized.
705 
706  """
707  repetitions = self.get_repetitions(node)
708  return [r.value() for r in repetitions]

713 class Literal(ElementBase):
714  """
715  Element class representing one or more literal words which must be
716  said exactly by the speaker as given.
717 
718  Quoted words can be used to potentially improve accuracy. This
719  currently only has an effect if using the Natlink SR engine
720  back-end.
721 
722  Constructor arguments:
723  - *text* (*str*) --
724  the words to be said by the speaker
725  - *name* (*str*, default: *None*) --
726  the name of this element
727  - *value* (*object*, default: *the recognized words*) --
728  value returned when this element is successfully decoded
729  - *default* (*object*, default: *None*) --
730  the default value used if this element is optional and wasn't
731  spoken
732  - *quote_start_str* (*str*, default: \") --
733  the string used for specifying the start of quoted words.
734  - *quote_end_str* (*str*, default: \") --
735  the string used for specifying the end of quoted words.
736  - *strip_quote_strs* (*bool*, default: *True*) --
737  whether the start and end quote strings should be stripped from
738  this element's word lists.
739 
740  """
741 
742  def __init__(self, text, name=None, value=None, default=None,
743  quote_start_str='"', quote_end_str='"',
744  strip_quote_strs=True):
745  ElementBase.__init__(self, name, default=default)
746  self._value = value
747 
748  if not isinstance(text, string_types):
749  raise TypeError("Text of %s object must be a"
750  " string." % self)
751 
752  # Construct the words and extended words lists. The latter includes
753  # quoted words as single items.
754  words = []
755  words_ext = []
756  current_quoted_sequence = []
757  for word in text.split():
758  # Begin quoted words.
759  if word.startswith(quote_start_str):
760  current_quoted_sequence.append(word)
761 
762  # End quoted words.
763  elif current_quoted_sequence and word.endswith(quote_end_str):
764  current_quoted_sequence.append(word)
765  quoted_words = " ".join(current_quoted_sequence)
766 
767  # Strip quote start and end strings if specified.
768  if strip_quote_strs:
769  quoted_words = quoted_words[len(quote_start_str):
770  len(quote_end_str) * -1]
771 
772  # Add the words to both lists.
773  words.extend(quoted_words.split())
774  words_ext.append(quoted_words)
775 
776  # Clear current sequence list.
777  del current_quoted_sequence[:]
778 
779  # Continuing quoted words.
780  elif current_quoted_sequence:
781  current_quoted_sequence.append(word)
782 
783  # Unquoted words.
784  else:
785  words.append(word)
786  words_ext.append(word)
787 
788  # Handle unfinished quoted words sequence by treating it as normal.
789  if current_quoted_sequence:
790  words.extend(current_quoted_sequence)
