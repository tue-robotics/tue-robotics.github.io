57 def map_word(word, encoding=getpreferredencoding(do_setlocale=False)):
58  """
59  Wraps output from Dragon.
60 
61  This wrapper ensures text output from the engine is Unicode. It assumes the
62  encoding of byte streams is the current locale's preferred encoding by default.
63  """
64  if isinstance(word, text_type):
65  return word
66  elif isinstance(word, binary_type):
67  return word.decode(encoding)
68  return word

202  def _load_grammar(self, grammar):
203  """ Load the given *grammar* into natlink. """
204  self._log.debug("Engine %s: loading grammar %s."
205  % (self, grammar.name))
206 
207  grammar_object = self.natlink.GramObj()
208  wrapper = GrammarWrapper(grammar, grammar_object, self,
209  self._recognition_observer_manager)
210  grammar_object.setBeginCallback(wrapper.begin_callback)
211  grammar_object.setResultsCallback(wrapper.results_callback)
212  grammar_object.setHypothesisCallback(None)
213 
214  c = NatlinkCompiler()
215  (compiled_grammar, rule_names) = c.compile_grammar(grammar)
216  wrapper.rule_names = rule_names
217 
218  all_results = (hasattr(grammar, "process_recognition_other")
219  or hasattr(grammar, "process_recognition_failure"))
220  hypothesis = False
221 
222  attempt_connect = False
223  try:
224  grammar_object.load(compiled_grammar, all_results, hypothesis)
225  except self.natlink.NatError as e:
226  # If loading failed because we're not connected yet,
227  # attempt to connect to natlink and reload the grammar.
228  if (str(e) == "Calling GramObj.load is not allowed before"
229  " calling natConnect"):
230  attempt_connect = True
231  else:
232  self._log.exception("Failed to load grammar %s: %s."
233  % (grammar, e))
234  raise EngineError("Failed to load grammar %s: %s."
235  % (grammar, e))
236  if attempt_connect:
237  self.connect()
238  try:
239  grammar_object.load(compiled_grammar, all_results, hypothesis)
240  except self.natlink.NatError as e:
241  self._log.exception("Failed to load grammar %s: %s."
242  % (grammar, e))
243  raise EngineError("Failed to load grammar %s: %s."
244  % (grammar, e))
245 
246  # Apply the threading fix if it hasn't been applied yet.
247  self.apply_threading_fix()
248 
249  # Return the grammar wrapper.
250  return wrapper

315  def mimic(self, words):
316  """
317  Mimic a recognition of the given *words*.
318 
319  .. note:: This method has a few quirks to be aware of:
320 
321  #. Mimic is not limited to one element per word as seen with
322  proper nouns from DNS. For example, "Buffalo Bills" can be
323  passed as one word.
324  #. Mimic can handle by the extra formatting by DNS built-in
325  commands.
326  #. Mimic is case sensitive.
327 
328  """
329  if isinstance(words, string_types):
330  words = words.split()
331 
332  try:
333  prepared_words = []
334  if PY2:
335  encoding = getpreferredencoding()
336  for word in words:
337  if isinstance(word, text_type):
338  word = word.encode(encoding)
339  prepared_words.append(word)
340  else:
341  for word in words:
342  prepared_words.append(word)
343  if len(prepared_words) == 0:
344  raise TypeError("empty list or string")
345  except Exception as e:
346  raise MimicFailure("Invalid mimic input %r: %s."
347  % (words, e))
348  try:
349  self.natlink.recognitionMimic(prepared_words)
350  except self.natlink.MimicFailed:
351  raise MimicFailure("No matching rule found for words %r."
352  % (prepared_words,))

370  def set_retain_directory(self, retain_dir):
371  """
372  Set the directory where audio data is saved.
373 
374  Retaining audio data may be useful for acoustic model training. This
375  is disabled by default.
376 
377  If a relative path is used and the code is running via natspeak.exe,
378  then the path will be made relative to the Natlink user directory or
379  base directory (e.g. ``MacroSystem``).
380 
381  :param retain_dir: retain directory path
382  :type retain_dir: string|None
383  """
384  is_string = isinstance(retain_dir, string_types)
385  if not (retain_dir is None or is_string):
386  raise EngineError("Invalid retain_dir: %r" % retain_dir)
387 
388  if is_string:
389  # Handle relative paths by using the Natlink user directory or
390  # base directory. Only do this if running via natspeak.exe.
391  try:
392  import natlinkstatus
393  except ImportError:
394  natlinkstatus = None
395  running_via_natspeak = (
396  sys.executable.endswith("natspeak.exe") and
397  natlinkstatus is not None
398  )
399  if not os.path.isabs(retain_dir) and running_via_natspeak:
400  status = natlinkstatus.NatlinkStatus()
401  user_dir = status.getUserDirectory()
402  retain_dir = os.path.join(
403  # Use the base dir if user dir isn't set.
404  user_dir if user_dir else status.BaseDirectory,
405  retain_dir
406  )
407 
408  # Set the retain directory.
409  self._retain_dir = retain_dir

430  def _decode_grammar_rules(self, state, words, results, *args):
431  # Iterate through this grammar's rules, attempting to decode each.
432  # If successful, call that rule's method for processing the
433  # recognition and return.
434  for rule in self.grammar.rules:
435  if not (rule.active and rule.exported): continue
436  state.initialize_decoding()
437  for _ in rule.decode(state):
438  if state.finished():
439  self._retain_audio(words, results, rule.name)
440  root = state.build_parse_tree()
441 
442  # Notify observers using the manager *before*
443  # processing.
444  # TODO Use words="other" instead, with a special
445  # recobs grammar wrapper at index 0.
446  notify_args = (words, rule, root, results)
447  self.recobs_manager.notify_recognition(
448  *notify_args
449  )
450  try:
451  rule.process_recognition(root)
452  except Exception as e:
453  self._log.exception("Failed to process rule "
454  "'%s': %s" % (rule.name, e))
455  self.recobs_manager.notify_post_recognition(
456  *notify_args
457  )
458  return True
459  return False

461  def results_callback(self, words, results):
462  self._log.debug("Grammar %s: received recognition %r."
463  % (self.grammar.name, words))
464 
465  if words == "other":
466  result_words = tuple(map_word(w) for w in results.getWords(0))
467  self.process_special_results(words, result_words, results)
468  return
469  elif words == "reject":
470  self.process_special_results(words, None, results)
471  return
472 
473  # If the words argument was not "other" or "reject", then
474  # it is a sequence of (word, rule_id) 2-tuples. Convert this
475  # into a tuple of unicode objects.
476  words_rules = tuple((map_word(w), r) for w, r in words)
477  words = tuple(w for w, r in words_rules)
478 
479  # Process this recognition.
480  if self.process_results(words_rules, self.rule_names, results):
481  return
482 
483  # Failed to decode recognition.
484  self._log.error("Grammar %s: failed to decode recognition %r."
485  % (self.grammar._name, words))

487  def _retain_audio(self, words, results, rule_name):
488  # Only write audio data and metadata if the directory exists.
489  retain_dir = self.engine._retain_dir
490  if retain_dir and not os.path.isdir(retain_dir):
491  self.engine._log.warning(
492  "Audio was not retained because '%s' was not a "
493  "directory" % retain_dir
494  )
495  elif retain_dir:
496  try:
497  audio = results.getWave()
498  # Make sure we have audio data
499  if len(audio) > 0:
500  # Write audio data.
501  now = datetime.now()
502  filename = ("retain_%s.wav"
503  % now.strftime("%Y-%m-%d_%H-%M-%S_%f"))
504  wav_path = os.path.join(retain_dir, filename)
505  with open(wav_path, "wb") as f:
506  f.write(audio)
507 
508  # Write metadata, assuming 11025Hz 16bit mono audio
509  text = ' '.join(words)
510  audio_length = float(len(audio) / 2) / 11025
511  tsv_path = os.path.join(retain_dir, "retain.tsv")
512  with open(tsv_path, "a") as tsv_file:
513  tsv_file.write('\t'.join([
514  filename, text_type(audio_length),
515  self.grammar.name, rule_name, text
516  ]) + '\n')

518  self.engine._log.exception("Exception retaining audio")
