76  def _load_grammar(self, grammar):
77  """ Load the given *grammar* into natlink. """
78  self._log.debug("Engine %s: loading grammar %s."
79  % (self, grammar.name))
80 
81  grammar.engine = self
82  grammar_object = self.natlink.GramObj()
83  wrapper = GrammarWrapper(grammar, grammar_object, self)
84  grammar_object.setBeginCallback(wrapper.begin_callback)
85  grammar_object.setResultsCallback(wrapper.results_callback)
86  grammar_object.setHypothesisCallback(None)
87 
88  # Dependency checking.
89  memo = []
90  for r in grammar._rules:
91  for d in r.dependencies(memo):
92  grammar.add_dependency(d)
93 
94  c = NatlinkCompiler()
95  (compiled_grammar, rule_names) = c.compile_grammar(grammar)
96  grammar._rule_names = rule_names
97 
98  if (hasattr(grammar, "process_recognition_other")
99  or hasattr(grammar, "process_recognition_failure")):
100  all_results = True
101  else:
102  all_results = False
103  hypothesis = False
104 
105  attempt_connect = False
106  try:
107  grammar_object.load(compiled_grammar, all_results, hypothesis)
108  except self.natlink.NatError, e:
109  # If loading failed because we're not connected yet,
110  # attempt to connect to natlink and reload the grammar.
111  if (str(e) == "Calling GramObj.load is not allowed before"
112  " calling natConnect"):
113  attempt_connect = True
114  else:
115  self._log.exception("Failed to load grammar %s: %s."
116  % (grammar, e))
117  raise EngineError("Failed to load grammar %s: %s."
118  % (grammar, e))
119  if attempt_connect:
120  self.connect()
121  try:
122  grammar_object.load(compiled_grammar, all_results, hypothesis)
123  except self.natlink.NatError, e:
124  self._log.exception("Failed to load grammar %s: %s."
125  % (grammar, e))
126  raise EngineError("Failed to load grammar %s: %s."
127  % (grammar, e))
128 
129  return wrapper

194  try:
195  prepared_words = []

256  def results_callback(self, words, results):
257  NatlinkEngine._log.debug("Grammar %s: received recognition %r."
258  % (self.grammar._name, words))
259 
260  if words == "other":
261  func = getattr(self.grammar, "process_recognition_other", None)
262  if func:
263  words = tuple(unicode(w, "windows-1252") for w in results.getWords(0))
264  func(words)
265  return
266  elif words == "reject":
267  func = getattr(self.grammar, "process_recognition_failure", None)
268  if func:
269  func()
270  return
271 
272  # If the words argument was not "other" or "reject", then
273  # it is a sequence of (word, rule_id) 2-tuples. Convert this
274  # into a tuple of unicode objects.
275  words_rules = tuple((unicode(w, "windows-1252"), r) for w, r in words)
276  words = tuple(w for w, r in words_rules)
277 
278  # Call the grammar's general process_recognition method, if present.
279  func = getattr(self.grammar, "process_recognition", None)
280  if func:
281  if not func(words):
282  return
283 
284  # Iterates through this grammar's rules, attempting
285  # to decode each. If successful, call that rule's
286  # method for processing the recognition and return.
287  s = state_.State(words_rules, self.grammar._rule_names, self.engine)
288  for r in self.grammar._rules:
289  if not r.active: continue
290  s.initialize_decoding()
291  for result in r.decode(s):
292  if s.finished():
293  root = s.build_parse_tree()
294  r.process_recognition(root)
295  return
296 
297  NatlinkEngine._log.warning("Grammar %s: failed to decode"
298  " recognition %r."
299  % (self.grammar._name, words))
