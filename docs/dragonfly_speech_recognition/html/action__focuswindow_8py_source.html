33 class FocusWindow(ActionBase):
34  """
35  Bring a window to the foreground action.
36 
37  Constructor arguments:
38  - *executable* (*str*) -- part of the filename of the
39  application's executable to which the target window belongs;
40  not case sensitive.
41  - *title* (*str*) -- part of the title of the target window;
42  not case sensitive.
43  - *index* (*str* or *int*) -- zero-based index of the target
44  window, for multiple matching windows; can be a string (for
45  substitution) but must be convertible to an integer.
46  - *filter_func* (*callable*) -- called with a single argument
47  (the window object), and should return ``True`` for your
48  target windows; example:
49  ``lambda window: window.get_position().dy > 100``.
50  - *focus_only* (*bool*, default *False*) -- if *True*, then
51  attempt to focus the window without raising it by using the
52  *Window.set_focus()* method instead of *set_foreground()*.
53  This argument may do nothing depending on the platform.
54 
55  This action searches all visible windows for a window which
56  matches the given parameters.
57 
58  """
59 
60  def __init__(self, executable=None, title=None, index=None,
61  filter_func=None, focus_only=False):

63  else: self.executable = None
64  if title: self.title = title.lower()

74  if index: arguments.append("index=%r" % index)

79  def _execute(self, data=None):
80  executable = self.executable
81  title = self.title
82  index = self.index
83  if data and isinstance(data, dict):
84  if executable: executable = (executable % data).lower()
85  if title: title = (title % data).lower()
86  if index: index = (index % data).lower()
87 
88  index = int(index) if index else 0
89 
90  # Get the first matching window and bring it to the foreground.
91  windows = Window.get_matching_windows(executable, title)
92  if self.filter_func:
93  windows = [window for window in windows
94  if self.filter_func(window)]
95  if windows and (index < len(windows)):
96  window = windows[index]
97  if self.focus_only:
98  window.set_focus()
99  else:
100  window.set_foreground()
101  else:
102  raise ActionError("Failed to find window (%s)." % self._str)
