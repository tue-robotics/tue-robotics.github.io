146 def run_pytest_suite(engine_name, pytest_options):
147  # Get test file paths.
148  paths = []
149  for name in engine_tests_dict[engine_name]:
150  if name.startswith("test_"):
151  # Use full module paths so pytest can import the files
152  # correctly.
153  name = "dragonfly/test/" + name + ".py"
154 
155  paths.append(name)
156 
157  # Initialize the engine and call connect().
158  engine_params = engine_params_dict.get(engine_name, {})
159  engine = get_engine(engine_name, **engine_params)
160  engine.connect()
161 
162  # Prevent the engine from running timers on its own. This lets us
163  # avoid race conditions.
164  engine._timer_manager.disable()
165  try:
166  # Run doctests through pytest.main() now that the engine is set up.
167  # Pass any specified pytest options, followed by doctest options for
168  # compatibility with both Python 2.7 and 3.
169  args = pytest_options + [
170  '-o',
171  'doctest_optionflags=ALLOW_UNICODE IGNORE_EXCEPTION_DETAIL'
172  ] + paths
173  return pytest.main(args)
174  finally:
175  # Disconnect after the tests.
176  engine.disconnect()
177 
178  # Check that the dragonfly engine was not changed during the tests.
179  assert engine is get_engine(), \
180  "The registered engine changed during the test suite."
