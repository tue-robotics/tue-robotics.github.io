102  continue

175  def _get_window_module(self):
176  # Get this window's process ID.
177  pid = self._get_window_pid()
178 
179  # Get the process handle of this window's process ID.
180  # Access permission flags:
181  # 0x0410 = PROCESS_QUERY_INFORMATION | PROCESS_VM_READ
182  handle = windll.kernel32.OpenProcess(0x0410, 0, pid)
183 
184  # Retrieve and return the process's executable path.
185  try:
186  # Try to use the QueryForProcessImageNameW function
187  # available since Windows Vista.
188  buffer_len = c_ulong(256)
189  buffer = (c_wchar * buffer_len.value)()
190  windll.kernel32.QueryFullProcessImageNameW(handle, 0,
191  pointer(buffer),
192  pointer(buffer_len))
193  buffer = buffer[:]
194  buffer = buffer[:buffer.index("\0")]
195  except Exception:
196  # If the function above failed, fall back to the older
197  # GetModuleFileNameEx function, available since windows XP.
198  # Note that this fallback function seems to fail when
199  # this process is 32 bit Python and handle refers to a
200  # 64-bit process.
201  buffer_len = 256
202  buffer = (c_wchar * buffer_len)()
203  windll.psapi.GetModuleFileNameExW(handle, 0, pointer(buffer),
204  buffer_len)
205  buffer = buffer[:]
206  buffer = buffer[:buffer.index("\0")]
207  finally:
208  windll.kernel32.CloseHandle(handle)
209 
210  return str(buffer)
