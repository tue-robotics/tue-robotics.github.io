121 def normalize_string(text):
122  """
123  Normalize a multi-line string.
124 
125  :param text: Multi-line text string.
126  :type text: str
127 
128  :return: Text string containing normalized lines.
129  :rtype: str
130 
131  Normalized lines means:
132  - Lines have been shifted to the left margin as far as possible.
133  - Lines have no trailing whitespace.
134  - Text has no leading or trailing empty lines.
135  """
136  lead_white_pat = re.compile('^( +)[^ ]')
137  lines = [line.rstrip() for line in text.split('\n')]
138 
139  # Find trailing empty lines.
140  last = len(lines) - 1
141  while last >= 0 and lines[last] == '':
142  last = last - 1
143  if last < 0:
144  return ""
145 
146  # Find leading empty lines.
147  first = 0
148  while lines[first] == '':
149  first = first + 1
150 
151  lines = lines[first:last + 1]
152 
153  # Strip common whitespace.
154  common_length = None
155  for line in lines:
156  m = lead_white_pat.match(line)
157  if m:
158  lead_length = len(m.group(1))
159  if common_length is None or lead_length < common_length:
160  common_length = lead_length
161 
162  if common_length > 0:
163  for i, line in enumerate(lines):
164  if line != '':
165  lines[i] = line[common_length:]
166 
167  return "\n".join(lines)
