21 """
22 Context classes
23 ============================================================================
24 
25 Dragonfly uses context classes to define when grammars and
26 rules should be active. A context is an object with a
27 :meth:`Context.matches` method which returns *True* if the
28 system is currently within that context, and *False* if it
29 is not.
30 
31 The following context classes are available:
32 
33  - :class:`Context` --
34  the base class from which all other context classes are derived
35  - :class:`AppContext` --
36  class which based on the application context, i.e. foreground window
37  executable, title, and handle
38 
39 
40 Logical operations
41 ----------------------------------------------------------------------------
42 
43 It is possible to modify and combine the behavior of contexts using the
44 Python's standard logical operators:
45 
46 :logical AND: ``context1 & context2`` -- *all* contexts must match
47 :logical OR: ``context1 | context2`` --
48  *one or more* of the contexts must match
49 :logical NOT: ``~context1`` -- *inversion* of when the context matches
50 
51 For example, to create a context which will match when
52 Firefox is in the foreground, but only if Google Reader is
53 *not* being viewed::
54 
55  firefox_context = AppContext(executable="firefox")
56  reader_context = AppContext(executable="firefox", title="Google Reader")
57  firefox_but_not_reader_context = firefox_context & ~reader_context
58

71 class Context(object):
72  """
73  Base class for other context classes.
74 
75  This base class implements some basic
76  infrastructure, including what's required for
77  logical operations on context objects. Derived
78  classes should at least do the following things:
79 
80  - During initialization, set *self._str* to some descriptive,
81  human readable value. This attribute is used by the
82  ``__str__()`` method.
83  - Overload the :meth:`Context.matches` method to implement
84  the logic to determine when to be active.
85 
86  The *self._log* logger objects should be used in methods of
87  derived classes for logging purposes. It is a standard logger
88  object from the *logger* module in the Python standard library.
89 
90  """
91 
92  _log = logging.getLogger("context.match")
93  _log_match = _log

147 class LogicAndContext(Context):

185 class AppContext(Context):
186  """
187  Context class using foreground application details.
188 
189  This class determines whether the foreground window meets
190  certain requirements. Which requirements must be met for this
191  context to match are determined by the constructor arguments.
192 
193  Constructor arguments:
194  - *executable* (*str*) --
195  (part of) the path name of the foreground application's
196  executable; case insensitive
197  - *title* (*str*) --
198  (part of) the title of the foreground window; case insensitive
199 
200  """
201 
202  #-----------------------------------------------------------------------
203  # Initialization methods.
204 
205  def __init__(self, executable=None, title=None, exclude=False):
206  Context.__init__(self)
207 
208  if isinstance(executable, str):
209  self._executable = executable.lower()
210  elif executable is None:
211  self._executable = None
212  else:
213  raise TypeError("executable argument must be a string or None;"
214  " received %r" % executable)
215 
216  if isinstance(title, str):
217  self._title = title.lower()
218  elif title is None:
219  self._title = None
220  else:
221  raise TypeError("title argument must be a string or None;"
222  " received %r" % title)
223 
224  self._exclude = bool(exclude)
225 
226  self._str = "%s, %s, %s" % (self._executable, self._title,
227  self._exclude)
228 
229  #-----------------------------------------------------------------------
230  # Matching methods.
231 
232  def matches(self, executable, title, handle):
233  executable = executable.lower()
234  title = title.lower()
235 
236  if self._executable:
237  found = (executable.find(self._executable) != -1)
238  if self._exclude == found:
239  self._log_match.debug("%s:"
240  " No match, executable doesn't match." % (self))
241  return False
242  if self._title:
243  found = (title.find(self._title) != -1)
244  if self._exclude == found:
245  self._log_match.debug("%s:"
246  " No match, title doesn't match." % (self))
247  return False
248 
249  if self._log_match: self._log_match.debug("%s: Match." % (self))
250  return True
