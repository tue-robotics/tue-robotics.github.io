21 """
22 Context classes
23 ============================================================================
24 
25 Dragonfly uses context classes to define when grammars and
26 rules should be active. A context is an object with a
27 :meth:`Context.matches` method which returns *True* if the
28 system is currently within that context, and *False* if it
29 is not.
30 
31 The following context classes are available:
32 
33  - :class:`Context` --
34  the base class from which all other context classes are derived
35  - :class:`AppContext` --
36  class which is based on the application context, i.e. foreground
37  window executable, title, and handle
38  - :class:`FuncContext` --
39  class that evaluates a given function/lambda/callable, whose return
40  value is interpreted as a *bool*, determining whether the context is
41  active
42 
43 
44 Logical operations
45 ----------------------------------------------------------------------------
46 
47 It is possible to modify and combine the behavior of contexts using the
48 Python's standard logical operators:
49 
50 :logical AND: ``context1 & context2`` -- *all* contexts must match
51 :logical OR: ``context1 | context2`` --
52  *one or more* of the contexts must match
53 :logical NOT: ``~context1`` -- *inversion* of when the context matches
54 
55 For example, to create a context which will match when
56 Firefox is in the foreground, but only if Google Reader is
57 *not* being viewed::
58 
59  firefox_context = AppContext(executable="firefox")
60  reader_context = AppContext(executable="firefox", title="Google Reader")
61  firefox_but_not_reader_context = firefox_context & ~reader_context
62 
63 
64 Matching other window attributes
65 ----------------------------------------------------------------------------
66 
67 The :class:`AppContext` class can be used to match window attributes and
68 properties other than the title and executable. To do this, pass extra
69 keyword arguments to the constructor::
70 
71  # Context for a maximized Firefox window.
72  maximized_firefox = AppContext(executable="firefox", is_maximized=True)
73 
74  # Context for a browser in fullscreen mode.
75  # 'role' and 'is_fullscreen' are X11 only.
76  fullscreen_browser = AppContext(role="browser", is_fullscreen=True)
77 
78  # Context for Android Studio or PyCharm using the X11 'cls' property.
79  AppContext(cls=["jetbrains-studio", "jetbrains-pycharm-ce"])
80

97 class Context(object):
98  """
99  Base class for other context classes.
100 
101  This base class implements some basic
102  infrastructure, including what's required for
103  logical operations on context objects. Derived
104  classes should at least do the following things:
105 
106  - During initialization, set *self._str* to some descriptive,
107  human readable value. This attribute is used by the
108  ``__str__()`` method.
109  - Overload the :meth:`Context.matches` method to implement
110  the logic to determine when to be active.
111 
112  The *self._log* logger objects should be used in methods of
113  derived classes for logging purposes. It is a standard logger
114  object from the *logger* module in the Python standard library.
115 
116  """
117 
118  _log = logging.getLogger("context.match")
119  _log_match = _log

215 class AppContext(Context):
216  """
217  Context class using foreground application details.
218 
219  This class determines whether the foreground window meets
220  certain requirements. Which requirements must be met for this
221  context to match are determined by the constructor arguments.
222 
223  If multiple strings are passed in a list, True will be returned if
224  the foreground window matches one or more of them. This applies to
225  the *executable* and *title* arguments and key word arguments for
226  most window attributes.
227 
228  Constructor arguments:
229  - *executable* (*str* or *list*) --
230  (part of) the path name of the foreground application's
231  executable; case insensitive
232  - *title* (*str* or *list*) --
233  (part of) the title of the foreground window; case insensitive
234  - *key word arguments* --
235  optional window attributes/properties and expected values;
236  case insensitive
237 
238  """
239 
240  # ----------------------------------------------------------------------
241  # Initialization methods.
242 
243  def __init__(self, executable=None, title=None, exclude=False,
244  **kwargs):
245  # pylint: disable=too-many-branches
246  # Suppress warnings about too many if-else branches.
247  Context.__init__(self)
248 
249  # Allow Unicode or strings to be used for executables and titles.
250  if isinstance(executable, string_types):
251  self._executable = [executable.lower()]
252  elif isinstance(executable, (list, tuple)):
253  self._executable = [e.lower() for e in executable]
254  elif executable is None:
255  self._executable = None
256  else:
257  raise TypeError("executable argument must be a string or None;"
258  " received %r" % executable)
259 
260  if isinstance(title, string_types):
261  self._title = [title.lower()]
262  elif isinstance(title, (list, tuple)):
263  self._title = [t.lower() for t in title]
264  elif title is None:
265  self._title = None
266  else:
267  raise TypeError("title argument must be a string or None;"
268  " received %r" % title)
269 
270  # Handle keyword arguments.
271  new_kwargs = {}
272  for key, value in kwargs.items():
273  if isinstance(value, string_types):
274  values = [value.lower()]
275  elif isinstance(value, list):
276  values = [str(v).lower() for v in value]
277  elif value is None:
278  values = None
279  else:
280  values = [value]
281  new_kwargs[key] = values
282 
283  self._exclude = bool(exclude)
284  self._str = "%s, %s, %s" % (self._executable, self._title,
285  self._exclude)
286  self._kwargs = new_kwargs
287  if self._kwargs:
288  self._str += ", %s" % self._kwargs
289 
290  # ----------------------------------------------------------------------
291  # Matching methods.
292 
293  def matches(self, executable, title, handle):
294  # pylint: disable=too-many-branches
295  # Suppress warnings about too many if-else branches.
296  if isinstance(executable, string_types):
297  executable = executable.lower()
298  if isinstance(title, string_types):
299  title = title.lower()
300 
301  if self._executable:
302  found = False
303  if isinstance(executable, string_types):
304  for match in self._executable:
305  if executable.find(match) != -1:
306  found = True
307  break
308  if self._exclude == found:
309  self._log_match.debug("%s: No match, executable doesn't "
310  "match.", self)
311  return False
312 
313  if self._title:
314  found = False
315  if isinstance(title, string_types):
316  for match in self._title:
317  if title.find(match) != -1:
318  found = True
319  break
320  if self._exclude == found:
321  self._log_match.debug("%s: No match, title doesn't match.",
322  self)
323  return False
324 
325  if self._kwargs:
326  # Import locally to avoid import cycles.
327  from dragonfly.windows import Window
328  window = Window.get_window(handle)
329 
330  found = False
331  for attr, expected_values in self._kwargs.items():
332  try:
333  # Get the window attribute.
334  attr_value = getattr(window, attr)
335  except AttributeError:
336  self._log_match.warning("%s: Skipped missing window"
337  " attribute '%s'",
338  self, attr)
339  continue
340 
341  # Check if the window attribute matched.
342  for match in expected_values:
343  is_string = isinstance(attr_value, string_types)
344  found = (
345  not is_string and match == attr_value or
346  is_string and attr_value.lower().find(match) != -1
347  )
348  if found:
349  break
350 
351  if self._exclude == found:
352  self._log_match.debug("%s: No match, not all extra"
353  " attributes match.", self)
354  return False
355 
356  if self._log_match:
357  self._log_match.debug("%s: Match.", self)
358  return True

363 class FuncContext(Context):
364  """
365  Context class that evaluates a given function, whose return
366  value is interpreted as a *bool*, determining whether the
367  context is active.
368 
369  The foreground application details are optionally passed to the
370  function as arguments named *executable*, *title*, and/or
371  *handle*, if any/each matches a so-named keyword argument of the
372  function. Default arguments may also be passed to the function,
373  through this class's constructor.
374 
375  """
376 
377  def __init__(self, function, **defaults):
378  """
379  Constructor arguments:
380  - *function* (callable) --
381  the function to call when this context is evaluated
382  - defaults --
383  optional default keyword-values for the arguments with
384  which the function will be called
385 
386  """
387 
388  Context.__init__(self)
389  self._function = function
390  self._defaults = defaults
391  self._str = "%s, defaults: %s" % (self._function, self._defaults)
392 
393  (args, _, varkw, defaults) = inspect.getargspec(self._function)
394  if varkw: self._filter_keywords = False
395  else: self._filter_keywords = True
396  self._valid_keywords = set(args)
397 
398  def matches(self, executable, title, handle):
399  arguments = dict(self._defaults)
400  arguments.update(executable=executable, title=title, handle=handle)
401 
402  if self._filter_keywords:
403  invalid_keywords = set(arguments.keys()) - self._valid_keywords
404  for key in invalid_keywords:
405  del arguments[key]
406 
407  try:
408  match = bool(self._function(**arguments))
409  if match:
410  self._log_match.debug("%s: Match.", self)
411  else:
412  self._log_match.debug("%s: No match, function"
413  " returned false.", self)
414  return match
415  except:
416  self._log.exception("Exception from function %s:",
417  self._function.__name__)
418  # Fallback to matching
419  return True
