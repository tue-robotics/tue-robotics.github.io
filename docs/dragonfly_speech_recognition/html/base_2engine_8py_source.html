280  def process_grammars_context(self, window=None):
281  """
282  Enable/disable grammars & rules based on their current contexts.
283 
284  This must be done preemptively for some SR engine back-ends,
285  such as WSR, that don't apply context changes upon/after the
286  utterance start has been detected. The WSR engine should call
287  this automatically whenever the foreground application (or its
288  title) changes. The user may want to call this manually to
289  update when custom contexts.

292  be passed in as an optimization if it has already been gathered.
293 
294  """

334  @classmethod
335  def _get_language_tag(cls, language_id):
336  # Get a language tag from the Windows language identifier.
337  tags = cls._language_tags.get(language_id)
338  if tags:
339  return tags[0]
340 
341  # The _language_tags dictionary didn't contain the language, so
342  # get the best match by using the primary language identifier.
343  # This allows us to match unlisted language variants.
344  primary_id = language_id & 0x00ff
345  for language_id2, (tag, _) in cls._language_tags.items():
346  if primary_id == language_id2 & 0x00ff: # Match found.
347  return tag
348 
349  # Speaker language wasn't found.
350  cls._log.error("Unknown speaker language: 0x%04x" % language_id)
351  raise EngineError("Unknown speaker language: 0x%04x" % language_id)
