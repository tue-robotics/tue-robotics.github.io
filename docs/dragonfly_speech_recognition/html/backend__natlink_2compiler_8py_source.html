44  def compile_grammar(self, grammar):

46

53 # print compiler.debug_state_string()

56  def _compile_rule(self, rule, compiler):
57  self._log.debug("%s: Compiling rule %s." % (self, rule.name))
58  if rule.imported:
59  return
60  compiler.start_rule_definition(rule.name, exported=rule.exported)
61  self.compile_element(rule.element, compiler)
62  compiler.end_rule_definition()

67  def _compile_sequence(self, element, compiler):
68  children = element.children
69  if len(children) > 1:
70  # Compile Sequence and Repetition elements differently.
71  is_rep = isinstance(element, elements_.Repetition)
72  if is_rep and element.optimize:
73  compiler.start_repetition()
74  self.compile_element(children[0], compiler)
75  compiler.end_repetition()
76  else:
77  compiler.start_sequence()
78  for c in children:
79  self.compile_element(c, compiler)
80  compiler.end_sequence()
81  elif len(children) == 1:
82  self.compile_element(children[0], compiler)
83 
84  def _compile_alternative(self, element, compiler):
85  children = element.children
86  if len(children) > 1:
87  compiler.start_alternative()
88  for c in children:
89  self.compile_element(c, compiler)
90  compiler.end_alternative()
91  elif len(children) == 1:
92  self.compile_element(children[0], compiler)
93 
94  def _compile_optional(self, element, compiler):
95  compiler.start_optional()
96  self.compile_element(element.children[0], compiler)
97  compiler.end_optional()

99  def _compile_literal(self, element, compiler):
100  # Use the element's words extension property to enable use of
101  # quoted words. DNS will receive quoted words as single strings.
102  # This can help with accuracy in some cases.
103  words = element.words_ext
104  if len(words) == 1:
105  compiler.add_word(words[0])
106  elif len(words) > 1:
107  compiler.start_sequence()
108  for w in words:
109  compiler.add_word(w)

151  def start_rule_definition(self, name, exported=False):
152  """start defining a rule."""
153 
154  # Make sure no rule is being defined at the moment.
155  if self._current_rule_name:
156  raise CompilerError("Cannot start defining a rule while" \
157  "a different rule is already being defined.")
158 
159  assert isinstance(name, string_types), ("The rule name must be a"
160  " string, received %r." % name)
161  self._current_rule_name = name
162  self._current_rule_export = exported
163  self._current_rule_definition = []
164 
165  def end_rule_definition(self):
166  """End defining a rule."""
167 
168  # Make sure a rule is being defined at the moment.
169  if not self._current_rule_name:
170  raise CompilerError("Cannot end defining a rule when" \
171  " no rule is being defined.")
172 
173  # Make sure that no other rule has been defined with this name.
174  if self._current_rule_name in self._rule_definitions:
175  raise CompilerError("Rule '%s' defined more than once." % \
176  self._current_rule_name)
177 
178  # If this rule has not been used before, register it.
179  if self._current_rule_name not in self._rules:
180  self._rules.append(self._current_rule_name)
181  if self._current_rule_export:
182  self._export_rules.append(self._current_rule_name)
183  self._rule_definitions[self._current_rule_name] = \
184  self._current_rule_definition
185 
186  self._current_rule_name = None
187  self._current_rule_export = None
188  self._current_rule_definition = None

333  def compile(self):
334  """Compile a binary grammar of this compiler's current state."""
335 
336  # Make sure no rule is being defined at the moment.
337  if self._current_rule_name:
338  raise CompilerError("Cannot compile grammar while a rule" \
339  " is being defined.")
340 
341  # Grammar header:
342  # - dwType; use 0.
343  # - dwFlags; use 0.
344  output = [struct.pack("LL", 0, 0)]
345 
346  # Lists of the names and IDs of exports, imports, lists, and words.
347  output.append(self._compile_id_chunk(4, self._export_rules, self._rules))
348  output.append(self._compile_id_chunk(5, self._import_rules, self._rules))
349  output.append(self._compile_id_chunk(6, self._lists, self._lists))
350  output.append(self._compile_id_chunk(2, self._words, self._words))
351 
352  # List of rule definitions.
353  output.append(self._compile_rule_chunk(3))
354 
355  # Return a concatenation of the header and chunks.
356  return b"".join(output)

358  def _compile_id_chunk(self, chunk_id, subset, ordered_superset):
359  # Loop through the elements of the superset, and if also present in
360  # the subset create a data entry of its ID and name. The IDs start
361  # counting at 1.
362  elements = []
363  for name, id in zip(ordered_superset,
364  range(1, len(ordered_superset) + 1)):
365 
366  # Skip names not included in the subset.
367  if name not in subset: continue
368 
369  # Chunk element:
370  # - dwSize; size of this element in bytes including this header.
371  # - dwNum; the element's ID.
372  # - szName; the element's name terminated by at least one 0.
373  # The element's name must be followed by one or more 0
374  # characters, so that its size in bytes is a multiple of 4.
375  if isinstance(name, text_type):
376  name = name.encode(getpreferredencoding())
377 
378  padded_len = (len(name) + 4) & (~3)
379  element = struct.pack(("LL%ds" % padded_len),
380  padded_len + 8, id, name)
381  elements.append(element)
382 
383  # Concatenate all the elements.
384  element_data = b"".join(elements)
385 
386  # Chunk header:
387  # - dwChunkId; words:2, rule definitions:3,
388  # exports:4, imports:5, lists:6
389  # - dwChunkSize; size of this chunk in bytes excluding this header.
390  header = struct.pack("LL", chunk_id, len(element_data))
391 
392  # Return the header and the elements.
393  return header + element_data

395  def _compile_rule_chunk(self, chunk_id):
396  # Loop through all known rule names, and if they are defined within
397  # this grammar create a rule definition entry.
398  definitions = []
399  for name, id in zip(self._rules,
400  range(1, len(self._rules) + 1)):
401 
402  # Skip imported rules.
403  if name in self._import_rules:
404  if name in self._rule_definitions:
405  raise CompilerError("Rule '%s' cannot be both imported" \
406  " and defined in a grammar" % name)
407  continue
408 
409  # Make sure that a definition has been given.
410  if name not in self._rule_definitions:
411  raise CompilerError("Rule '%s' is neither imported" \
412  " nor defined" % name)
413 
414  # Build the definition sequence for this rule.
415  elements = []
416  for t, v in self._rule_definitions[name]:
417  # Definition element:
418  # - wType; start:1, end:2, word:3, rule:4, list:6
419  # - wProb; probability rating, use 0.
420  # - dwValue; depends on wType as follows:
421  # - if wType is start or end, then dwValue is one of
422  # sequence:1, alternative:2, repetition:3, optional:4
423  # - if wType is word, rule, or list, then dwValue is
424  # the ID of the corresponding element.
425  element = struct.pack("HHL", t, 0, v)
426  elements.append(element)
427 
428  # Definition header:
429  # - dwSize; size of this rule definition in bytes including
430  # this header.
431  # - dwNum; the ID of this rule.
432  definition_size = 8 + sum([len(s) for s in elements])
433  definition = struct.pack("LL", definition_size, id)
434  definition += b"".join(elements)
435 
436  definitions.append(definition)
437 
438  # Concatenate all the rule definitions.
439  definition_data = b"".join(definitions)
440 
441  # Rule definition chunk header:
442  # - dwChunkId; rule definitions:3
443  # - dwChunkSize; size of this chunk in bytes excluding this header.
444  header = struct.pack("LL", chunk_id, len(definition_data))
445 
446  # Return the header and the rule definitions.
447  return header + definition_data
