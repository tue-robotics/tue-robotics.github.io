32 # Force the current process to be DPI aware. This will ensure that the
33 # resolution is reported properly even when DPI scaling is on. This appears to
34 # be set to PROCESS_SYSTEM_DPI_AWARE when running embedded in Dragon, which
35 # provides awareness of the DPI when the application starts, but not when it
36 # changes. Unfortunately, this cannot be changed (presumably it is declared in a
37 # manifest). When running from the command line, however, the default is
38 # PROCESS_DPI_UNAWARE and can be adjusted here.
39 try:
40  log = logging.getLogger("monitor.init")
41  value = 2 # PROCESS_PER_MONITOR_DPI_AWARE
42  hresult = ctypes.windll.shcore.SetProcessDpiAwareness(value)
43  if hresult == -2147024809: # E_INVALIDARG
44  log.warning("DPI awareness could not be set; "
45  "SetProcessDpiAwareness() received an invalid "
46  "argument: %d", value)
47  elif hresult == -2147024891: # E_ACCESSDENIED
48  log.warning("DPI awareness could not be set; it has been set "
49  "already.")
50 except OSError:
51  # Do nothing if SetProcessDpiAwareness() could not be called.
52  pass

66  @classmethod
67  def get_all_monitors(cls):
68  # Get an updated list of monitors.
69  monitors = []
70  for py_handle, _, rectangle in win32api.EnumDisplayMonitors():
71  # Get the monitor handle.
72  handle = py_handle.handle
73 
74  # Create a rectangle object representing the monitor's
75  # dimensions and relative position.
76  top_left_x, top_left_y, bottom_right_x, bottom_right_y = rectangle
77  dx = bottom_right_x - top_left_x
78  dy = bottom_right_y - top_left_y
79  rectangle = Rectangle(top_left_x, top_left_y, dx, dy)
80 
81  # Get a new or updated monitor object and add it to the list.
82  # Ensure that the primary monitor is first on the list.
83  monitor = cls.get_monitor(handle, rectangle)
84  if monitor.is_primary:
85  monitors.insert(0, monitor)
86  else:
87  monitors.append(monitor)
88 
89  return monitors
