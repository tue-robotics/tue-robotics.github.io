21 """
22 CompoundRule class
23 ============================================================================
24 
25 The CompoundRule class is designed to make it very easy to create a rule
26 based on a single compound spec.
27 
28 This rule class has the following parameters to customize its behavior:
29 
30  - *spec* -- compound specification for the rule's root element
31  - *extras* -- extras elements referenced from the compound spec
32  - *defaults* -- default values for the extras
33  - *exported* -- whether the rule is exported
34  - *context* -- context in which the rule will be active
35 
36 Each of these parameters can be passed as a (keyword) arguments to the
37 constructor, or defined as a class attribute in a derived class.
38 
39 
40 Example usage
41 ............................................................................
42 
43 The CompoundRule class can be used to define a voice-command as follows::
44 
45  class ExampleRule(CompoundRule):
46 
47  spec = "I want to eat <food>"
48  extras = [Choice("food", {
49  "(an | a juicy) apple": "good",
50  "a [greasy] hamburger": "bad",
51  }
52  )
53  ]
54 
55  def _process_recognition(self, node, extras):
56  good_or_bad = extras["food"]
57  print "That is a %s idea!" % good_or_bad
58 
59  rule = ExampleRule()
60  grammar.add_rule(rule)

136  def process_recognition(self, node):
137  """
138  Process a recognition of this rule.
139 
140  This method is called by the containing Grammar when this
141  rule is recognized. This method collects information about
142  the recognition and then calls *self._process_recognition*.
143 
144  - *node* -- The root node of the recognition parse tree.
145  """
146  # Prepare *extras* dict for passing to _process_recognition().
147  extras = {
148  "_grammar": self.grammar,
149  "_rule": self,
150  "_node": node,
151  }
152  extras.update(self._defaults)
153  for name, element in self._extras.items():
154  extra_node = node.get_child_by_name(name, shallow=True)
155  if extra_node:
156  extras[name] = extra_node.value()
157  elif element.has_default():
158  extras[name] = element.default
159 
160  # Call the method to do the actual processing.
161  self._process_recognition(node, extras)
