29 def send_rpc_request(url, method, params, id=0):
30  """
31  Utility function to send a JSON-RPC request to a server.
32 
33  This will raise an error if the request fails with an error or if
34  the server is unreachable.
35 
36  :param url: the URL to send the JSON-RPC request to.
37  :param method: name of the RPC method to call.
38  :param params: parameters of the RPC method to call.
39  :param id: ID of the JSON-RPC request (default: 0).
40  :type url: str
41  :type method: str
42  :type params: list
43  :type id: int
44  :returns: JSON-RPC response
45  :rtype: dict
46  :raises: RuntimeError
47  """
48  headers = {'content-type': 'application/json'}
49  payload = {
50  "method": method,
51  "params": params,
52  "jsonrpc": "2.0",
53  "id": id,
54  }
55  response = requests.post(
56  url, data=json.dumps(payload), headers=headers
57  ).json()
58 
59  # Handle an error in the response.
60  error_info = response.get("error")
61  if error_info:
62  raise RuntimeError(
63  "error calling RPC server method %s: %d, %s" %
64  (method, error_info["code"], error_info["message"])
65  )
66 
67  # Otherwise return the response.
68  return response
