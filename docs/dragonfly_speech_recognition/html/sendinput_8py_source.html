48  # From https://docs.microsoft.com/en-us/windows/desktop/inputdev/about-keyboard-input#extended-key-flag
49  # The extended keys consist of the ALT and CTRL keys
50  # on the right-hand side of the keyboard; the INS, DEL, HOME,
51  # END, PAGE UP, PAGE DOWN, and arrow keys in the clusters to
52  # the left of the numeric keypad; the NUM LOCK key; the BREAK
53  # (CTRL+PAUSE) key; the PRINT SCRN key; and the divide (/) and
54  # ENTER keys in the numeric keypad.
55  #
56  # It's unclear if the Windows keys are also "extended", so they
57  # have been included for historical reasons.
58  extended_keys = {
59  win32con.VK_UP,

83  def __init__(self, virtual_keycode, down, scancode=None, layout=None):
84  """Initialize structure based on key type."""
85  flags = 0
86  if virtual_keycode == 0:
87  flags |= 4 # KEYEVENTF_UNICODE
88  else:
89  # Translate *virtual_keycode* into a scan code, if necessary.
90  # Assume that *layout* is a keyboard layout (HKL).
91  if scancode is None:
92  user32 = windll.user32
93  if sys.getwindowsversion().major < 6:
94  map_type = 0 # MAPVK_VK_TO_VSC
95  else:
96  map_type = 4 # MAPVK_VK_TO_VSC_EX
97  if layout is None:
98  scancode = user32.MapVirtualKeyW(virtual_keycode,
99  map_type)
100  else:
101  scancode = user32.MapVirtualKeyExW(virtual_keycode,
102  map_type,
103  layout)
104 
105  # Add the KEYEVENTF_SCANCODE flag if the Win32 MapVirtualKey(Ex)
106  # function returned a translation. If not, then fallback on
107  # the specified keycode. This is also done for "soft" keys.
108  if scancode and virtual_keycode not in self.soft_keys:
109  flags |= 8 # KEYEVENTF_SCANCODE
110 
111  # Add the KEYEVENTF_EXTENDEDKEY flag, if necessary.
112  # Starting with Windows Vista, extended scan codes are
113  # specified with the high byte containing either 0xe0 or
114  # 0xe1.
115  if scancode >> 8 in (0xe0, 0xe1):
116  flags |= win32con.KEYEVENTF_EXTENDEDKEY
117 
118  # Always add the KEYEVENTF_EXTENDEDKEY flag for certain
119  # virtual-keys (see above).
120  if virtual_keycode in self.extended_keys:
121  flags |= win32con.KEYEVENTF_EXTENDEDKEY
122 
123  if not down:
124  flags |= win32con.KEYEVENTF_KEYUP
125 
126  extra = pointer(c_ulong(0))
127  Structure.__init__(self, virtual_keycode, scancode, flags, 0, extra)

153  def __init__(self, element):

174 def send_input_array(input_array):
175  length = len(input_array)
176  assert length >= 0
177  size = sizeof(input_array[0])
178  ptr = pointer(input_array)
179 
180  count_inserted = windll.user32.SendInput(length, ptr, size)
181 
182  if count_inserted != length:
183  last_error = win32api.GetLastError()
184  message = win32api.FormatMessage(last_error)
185  raise ValueError("windll.user32.SendInput(): %s" % (message))
