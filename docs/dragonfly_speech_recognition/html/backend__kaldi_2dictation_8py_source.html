66 class _UserDictationSequenceRule(CompoundRule):
67  spec = "<__kaldi_user_dictation_sequence>"
68  extras = [
69  Repetition(Alternative([
70  BaseDictation(format=False),
71  ListRef('__kaldi_user_dictation_listref', user_dictation_list),
72  DictListRef('__kaldi_user_dictation_dictlistref', user_dictation_dictlist),
73  ]), min=1, max=16, name="__kaldi_user_dictation_sequence"),
74  ]
75  exported = False
76  def value(self, node):
77  # This method returns the value of the root Repetition: a list of values of
78  # Alternatives, each (dictation or listref or dictlistref) being a string.
79  chunks = node.children[0].value()
80  # Make sure each chunk is a list (of strings), then concat them together.
81  return list(itertools.chain.from_iterable([chunk] if not isinstance(chunk, list) else chunk for chunk in chunks))
