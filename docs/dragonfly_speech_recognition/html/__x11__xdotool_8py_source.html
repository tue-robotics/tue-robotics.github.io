28 class XdotoolKeyboard(BaseX11Keyboard):
29  """Static class for typing keys with xdotool."""

31  _log = logging.getLogger("keyboard")
32 
33  # xdotool command
34  xdotool = "xdotool"
35 
36  @classmethod
37  def send_keyboard_events(cls, events):
38  """
39  Send a sequence of keyboard events.
40 
41  Positional arguments:
42  events -- a sequence of tuples of the form
43  (keycode, down, timeout), where
44  keycode (str): key symbol.
45  down (boolean): True means the key will be pressed down,
46  False means the key will be released.
47  timeout (float): number of seconds to sleep after
48  the keyboard event.
49 
50  """
51  cls._log.debug("Keyboard.send_keyboard_events %r", events)
52 
53  # Return early if there are no events (e.g. for Key("")).
54  if not events:
55  return
56 
57  arguments = []
58  for event in events:
59  (key, down, timeout) = event
60  key = KEY_TRANSLATION.get(key, key)
61 
62  # Add arguments for pressing keys.
63  if down:
64  arguments += ['keydown', '--delay', '0', key]
65  else:
66  arguments += ['keyup', '--delay', '0', key]
67 
68  # Instruct xdotool to sleep after the keyboard event if
69  # necessary.
70  if timeout:
71  arguments += ['sleep', '%.3f' % timeout]
72 
73  # Press/release the keys using xdotool, catching any errors.
74  command = [cls.xdotool] + arguments
75  readable_command = ' '.join(command)
76  cls._log.debug(readable_command)
77  try:
78  # Fork the process with setsid() if on a POSIX system such as
79  # Linux.
80  kwargs = {}
81  if os.name == 'posix':
82  kwargs.update(dict(preexec_fn=os.setsid))
83 
84  # Execute the xdotool child process.
85  p = subprocess.Popen(command, **kwargs)
86  if p.wait() > 0:
87  raise RuntimeError("xdotool command exited with non-zero "
88  "return code %d" % p.returncode)
89  except Exception as e:
90  cls._log.exception("Failed to execute xdotool command '%s': "
91  "%s", readable_command, e)
