107 def _callback(
108  hMonitor, # Handle to display monitor
109  hdcMonitor, # Handle to monitor DC
110  lprcMonitor, # Intersection rectangle of monitor
111  dwData # Data
112  ):
113  # Retrieves monitor info.
114  info = _monitor_info_t()
115  info.cbSize = ctypes.sizeof(_monitor_info_t)
116  info.rcMonitor = _rect_t()
117  info.rcWork = _rect_t()
118  res = ctypes.windll.user32.GetMonitorInfoA(hMonitor, ctypes.byref(info))
119 
120  # Store monitor info.
121  handle = long(hMonitor)
122  r = info.rcWork
123  rectangle = Rectangle(r.left, r.top, r.right - r.left, r.bottom - r.top)
124  monitor = Monitor(handle, rectangle)
125  Monitor._log.debug("Found monitor %s with geometry %s."
126  % (monitor, rectangle))
127  monitors.append(monitor)
128 
129  # Continue enumerating monitors.
