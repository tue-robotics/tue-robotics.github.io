58 class ContainerBase(object):

62  def __init__(self, **kwargs):
63  for name, default in self._attributes:
64  if name in kwargs: value = kwargs.pop(name)
65  elif callable(default): value = default()
66  else: value = default
67  setattr(self, name, value)
68  if kwargs:
69  names = sorted(kwargs.keys())
