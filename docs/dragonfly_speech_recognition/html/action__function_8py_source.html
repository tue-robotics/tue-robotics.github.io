22 Function action
23 ============================================================================
24 
25 The :class:`Function` action wraps a callable, optionally with some
26 default keyword argument values. On execution, the execution data
27 (commonly containing the recognition extras) are combined with the
28 default argument values (if present) to form the arguments with which
29 the callable will be called.
30 
31 Simple usage::
32 
33  >>> def func(count):
34  ... print("count: %d" % count)
35  ...
36  >>> action = Function(func)
37  >>> action.execute({"count": 2})
38  count: 2
39  True
40  >>> # Additional keyword arguments are ignored:
41  >>> action.execute({"count": 2, "flavor": "vanilla"})
42  count: 2
43  True
44 
45 Usage with default arguments::
46 
47  >>> def func(count, flavor):
48  ... print("count: %d" % count)
49  ... print("flavor: %s" % flavor)
50  ...
51  >>> # The Function object can be given default argument values:
52  >>> action = Function(func, flavor="spearmint")
53  >>> action.execute({"count": 2})
54  count: 2
55  flavor: spearmint
56  True
57  >>> # Arguments given at the execution-time to override default values:
58  >>> action.execute({"count": 2, "flavor": "vanilla"})
59  count: 2
60  flavor: vanilla
61  True
62 
63 Usage with the ``remap_data`` argument::
64 
65  >>> def func(x, y, z):
66  ... print("x: %d" % x)
67  ... print("y: %d" % y)
68  ... print("z: %d" % z)
69  ...
70  >>> # The Function object can optionally be given a second dictionary
71  >>> # argument to use extras with different names. It should be
72  >>> # compatible with the 'defaults' parameter:
73  >>> action = Function(func, dict(n="x", m="y"), z=4)
74  >>> action.execute({"n": 2, "m": 3})
75  x: 2
76  y: 3
77  z: 4
78  True

128  def _execute(self, data=None):
129  arguments = dict(self._defaults)
130  if isinstance(data, dict):
131  arguments.update(data)
132 
133  # Remap specified names.
134  if arguments and self._remap_data:
135  for old_name, new_name in self._remap_data.items():
136  if old_name in data:
137  arguments[new_name] = arguments.pop(old_name)
138

140  invalid_keywords = set(arguments.keys()) - self._valid_keywords
141  for key in invalid_keywords:
142  del arguments[key]
143

146  except Exception as e:

148  self._function.__name__)
