88  def _compile_rule(self, rule, grammar, grammar_handle):
89  self._log.debug("%s: Compiling rule %s." % (self, rule.name))
90 
91  # Determine whether this rule has already been compiled.
92  rule_handle = grammar_handle.Rules.FindRule(rule.name)
93  if rule_handle:
94  self._log.debug("%s: Already compiled rule %s." % (self, rule.name))
95  return
96 
97  # Determine the flags to set when adding this rule.
98  flags = 0
99  if rule.exported:
100  flags |= constants.SRATopLevel
101 
102  # Add this rule, and compile its root element.
103  rule_id = self._get_next_rule_id(grammar)
104  rule_handle = grammar_handle.Rules.Add(rule.name, flags, rule_id)
105  self._log.debug("%s: Compiling rule %r (id %r)." % (self, rule_handle.Name, rule_handle.Id))
106  self.compile_element(rule.element, rule_handle.InitialState, None, grammar, grammar_handle)
107 
108  return
