21 RPC methods
22 ----------------------------------------------------------------------------
23 
24 For RPC methods to work they must be added to the server with
25 :meth:`add_method`. For example::
26 
27  from dragonfly.engines import get_engine
28  from dragonfly.rpc import RPCServer
29 
30  # Initialise and start the server.
31  server = RPCServer()
32  server.start()
33 
34  # Add the RPC method via decoration.
35  @server.add_method
36  def get_engine_language():
37  return get_engine().language
38 
39  # add_method() can also be used normally.
40  server.add_method(get_engine_language)
41 
42 
43 Sending requests
44 ============================================================================
45 Requests can be sent to the server using, the :meth:`send_rpc_request`
46 function from Python::
47 
48  send_rpc_request(
49  server.url, method="get_engine_language",
50  params=[server.security_token], id=0
51  )
52 
53 Other tools such as `curl <https://curl.haxx.se/>`_ can also be used.
54 
55 .. code:: shell
56 
57  Using positional arguments:
58  $ curl --data-binary '{"jsonrpc":"2.0","id": "0","method": "speak","params": ["hello world", "<security-token>"]}' -H 'content-type:text/json;' http://127.0.0.1:50051

86 @_add_method
87 def list_grammars():
88  """
89  Get a list of grammars loaded into the current engine.
90 
91  This includes grammar rules and attributes.
92  """
93  # Send rule and grammar data back to the client.
94  grammars = []
95  engine = get_engine()
96  for grammar in engine.grammars:
97  rules = []
98  for rule in grammar.rules:
99  if isinstance(rule, CompoundRule):
100  specs = [rule.spec]
101  elif isinstance(rule, MappingRule):
102  specs = sorted(rule.specs)
103  else:
104  specs = [rule.element.gstring()]
105  rules.append({
106  "name": rule.name, "specs": specs,
107  "exported": rule.exported, "active": rule.active
108  })
109 
110  # Add grammar attributes to the grammars list.
111  is_active = any([r.active for r in grammar.rules])
112  grammars.append({"name": grammar.name, "rules": rules,
113  "enabled": grammar.enabled, "active": is_active})
114 
115  return grammars

179 @_add_method
180 def register_history(length=10, record_failures=False):
181  """
182  Register an internal recognition observer.
183 
184  :param length: length to initialize the ``RecognitionHistory`` instance
185  with (default ``10``).
186  :param record_failures: whether to record recognition failures
187  (default ``False``).
188  :type record_failures: bool
189  :type length: int
190  """

192  raise RuntimeError("history has already been registered")
