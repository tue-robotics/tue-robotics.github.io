35  The monitor class used on X11 (Linux).
36 
37  This implementation parses monitor information from
38  ``xrandr``.

59  @classmethod
60  def get_all_monitors(cls):
61  # pylint: disable=R0912,R0914,R0915
62  # Suppress warnings about too many local variables.
63 
64  # Get updated monitor information from xrandr.
65  try:
66  p = Popen(["xrandr"], stdout=PIPE)
67  stdout, _ = p.communicate()
68 
69  # Decode output if it is binary.
70  encoding = locale.getpreferredencoding()
71  if isinstance(stdout, binary_type):
72  stdout = stdout.decode(encoding)
73 
74  # Handle non-zero return codes.
75  if p.wait() > 0:
76  print(stdout)
77  raise RuntimeError("xrandr exited with non-zero return "
78  "code %d" % p.returncode)
79  except Exception as e:
80  cls._log.error("Failed to execute command: %s. Is "
81  "xrandr installed?", e)
82  raise e
83 
84  # Parse output from xrandr.
85  monitors = []
86  for line in stdout.split("\n")[1:]: # skip the first line
87  # Skip disconnected monitors, output modes and empty lines.
88  if not line or "disconnected" in line or line.startswith(" "):
89  continue
90 
91  # Get the information we need.
92  parts = line.split()
93  name = parts.pop(0)
94  primary = "primary" in parts
95  if primary: # this monitor will be first in the list
96  parts.remove("primary")
97 
98  # Find and remove info we don't need (e.g. 'connected' and
99  # orientation info).
100  i = 1
101  for i, part in enumerate(parts):
102  if "(" in part: # e.g. '(normal'
103  break
104  parts = parts[1:i]
105 
106  # Skip disabled monitors (no reported mode info).
107  if not parts:
108  continue
109 
110  # Get resolution and position info.
111  res_info = parts[0].split("+")
112  dx, dy = res_info.pop(0).split("x")
113  top_left_x, top_left_y = res_info
114 
115  # Convert values to integers and create a rectangle object
116  # representing the monitor's dimensions and relative position.
117  dx, dy = int(dx), int(dy)
118  top_left_x, top_left_y = int(top_left_x), int(top_left_y)
119  rectangle = Rectangle(top_left_x, top_left_y, dx, dy)
120 
121  # Get a new or updated monitor object and add it to the list.
122  monitor = cls.get_monitor(name, rectangle)
123  monitor.is_primary = primary
124 
125  # Ensure that the top_left monitor is the first in the list.
126  if top_left_x == 0:
127  monitors.insert(0, monitor)
128  else:
129  monitors.append(monitor)
130 
131  # Return the list of monitors.
132  return monitors
