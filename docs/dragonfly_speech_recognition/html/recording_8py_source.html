108  def _record(self):
109  # Start recording audio on the current thread until stop() is
110  # called.
111  p = pyaudio.PyAudio()
112  channels, rate = self.config.CHANNELS, self.config.RATE
113  frames_per_buffer = self.config.FRAMES_PER_BUFFER
114  pa_format = pyaudio.get_format_from_width(self.config.SAMPLE_WIDTH)
115  stream = p.open(input=True, format=pa_format, channels=channels,
116  rate=rate, frames_per_buffer=frames_per_buffer)
117 
118  # Start recognising in a loop
119  stream.start_stream()
120  while self._recording:
121  with self._condition:
122  self._buffers.append(stream.read(frames_per_buffer))
123 
124  # Notify waiting threads (if any).
125  self._condition.notifyAll()
126 
127  # This improves the performance; we don't need to process as
128  # much audio as the device can read.
129  time.sleep(self.read_interval)
