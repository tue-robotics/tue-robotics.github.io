21 """
22 Dragonfly's logging infrastructure is defined in the ``dragonfly.log``
23 module. It defines sane defaults for the various loggers used in the library
24 as well as functions for setting up logging and tracing.
25 
26 Adjusting logger levels
27 ----------------------------------------------------------------------------
28 
29 Dragonfly's logger levels can be adjusted much like Python logger levels:
30 
31 .. code::
32 
33  import logging
34  logging.getLogger("engine").setLevel(logging.DEBUG)
35 
36 The one caveat is that this must be done *after* the :meth:`setup_log`
37 function is called, otherwise the levels you set will be overridden.
38 By default, the function is only called near the top of the module loader
39 scripts (e.g. *dragonfly/examples/dfly-loader-wsr.py*), not within dragonfly
40 itself.
41 
42 It is not necessary to call :meth:`setup_log` at all. Standard Python
43 logging functions such as :meth:`basicConfig` can be used at the top of
44 module loader scripts instead.
45 
46 If you are not using dragonfly with a module loader, you will need to set up
47 a logging handler to avoid messages like tho following::
48 
49  No handlers could be found for logger "typeables"

188 def setup_log(use_stderr=True, use_file=True, use_stdout=False):
189  """
190  Setup Dragonfly's logging infrastructure with sane defaults.
191 
192  :param use_stderr: whether to output log messages to stderr
193  (default: True).
194  :param use_file: whether to output log messages to the
195  *~/.dragonfly.log* log file (default: True).
196  :param use_stdout: this parameter does nothing andhas been left in for
197  backwards-compatibility (default: False).
198  :type use_stderr: bool
199  :type use_file: bool
200  :type use_stdout: bool
201 
202  """
203  global _dispatching_handlers
204  global _stderr_handler, _file_handler
205  global _stderr_filters, _file_filters
206 
207  # Remove any previously created dispatching handlers.
208  for name, handler in _dispatching_handlers.items():
209  logger = logging.getLogger(name)
210  logger.removeHandler(handler)
211 
212  # Setup default handlers.
213  if use_stderr:
214  _stderr_handler = _setup_stderr_handler()
215  if use_file:
216  _file_handler = _setup_file_handler()
217 
218  # Create and register default filters.
219  for name, levels in default_levels.items():
220  stderr_level, file_level = levels
221  handler = DispatchingHandler()
222  _dispatching_handlers[name] = handler
223  if use_stderr:
224  stderr_filter = NameLevelFilter(name, stderr_level)
225  handler.add_handler_filter_pair(_stderr_handler, stderr_filter)
226  _stderr_filters[name] = stderr_filter
227  if use_file:
228  file_filter = NameLevelFilter(name, file_level)
229  handler.add_handler_filter_pair(_file_handler, file_filter)
230  _file_filters[name] = file_filter
231  logger = logging.getLogger(name)
232  logger.addHandler(handler)
233  logger.setLevel(min(stderr_level, file_level))
234  logger.propagate = False

240 def setup_tracing(output, limit=None):
241  """
242  Setup call tracing for low-level debugging.
243 
244  :param output: the file to write tracing messages to.
245  :type output: file
246  :param limit: the recursive depth limit for tracing (default: None).
247  :type limit: int|None
248  """
249  from pkg_resources import resource_filename
250  library_prefix = os.path.dirname(resource_filename(__name__, "setup.py"))
251  print("prefix:", library_prefix)
252  exclude_filenames = ("parser.py",)
253 
254  def _tracing_callback(frame, event, arg):
255  # Retrieve current function name, line number, etc.
256  code_object = frame.f_code
257  function_name = code_object.co_name
258  line_number = frame.f_lineno
259  filename = code_object.co_filename
260  if not filename.startswith(library_prefix):
261  return
262  else:
263  filename = filename[len(library_prefix)+1:]
264  if os.path.basename(filename) in exclude_filenames:
265  return
266 
267  depth = 0
268  if limit is not None:
269  # Determine call depth of current frame.
270  parent_frame = frame
271  while parent_frame:
272  parent_frame = parent_frame.f_back
273  depth += 1
274  del parent_frame
275  if depth > limit:
276  return
277 
278  # Write message to output.
279  indented_function_name = (" " * depth) + function_name
280  output.write("%2d %-40s %5s %-40s\n" % (depth, indented_function_name,
281  line_number, filename))
282  output.flush()
283 
284  sys.settrace(_tracing_callback)
