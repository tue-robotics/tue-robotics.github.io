107  def get_test_function(self):
108  """
109  Create and return a test function used for testing whether
110  key phrases or rules are processed correctly, insofar as they reach
111  the correct processing method/function.
112 
113  Note that returned test functions accept variable arguments.
114  :return: callable
115  """
116  def func(*_):
117  # Function was reached
118  try:
119  self.test_map[id(func)] += 1
120  except KeyError:
121  # Ignore any key errors; if this function's id is not in
122  # test_map, then it's not relevant to the currently running
123  # test method.
124  pass
125 
126  self.test_map[id(func)] = 0
127  return func
128 
129  def assert_test_function_called(self, func, n):
130  """
131  Assert that a test function was called n times.
132  :type func: callable
133  :param n: number of times the test function should have been called
134  """
135  x = self.test_map[id(func)]
136  self.assertEqual(x, n, "wrapped test function was called %d time(s) "
137  "instead of %d time(s)" % (x, n))
