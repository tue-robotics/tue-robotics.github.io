21 """
22 BringApp and StartApp actions
23 ============================================================================
24 
25 The :class:`StartApp` and :class:`BringApp` action classes are used to
26 start an application and bring it to the foreground. :class:`StartApp`
27 starts an application by running an executable file, while
28 :class:`BringApp` first checks whether the application is already running
29 and if so brings it to the foreground, otherwise starts it by running the
30 executable file.
31 
32 
33 Example usage
34 ----------------------------------------------------------------------------
35 
36 The following example brings Notepad to the foreground if it is already
37 open, otherwise it starts Notepad::
38 
39  BringApp(r"C:\\Windows\\system32\\notepad.exe").execute()
40 
41 Note that the path to *notepad.exe* given above might not be correct for
42 your computer, since it depends on the operating system and its
43 configuration.
44 
45 In some cases an application might be accessible simply through the file
46 name of its executable, without specifying the directory. This depends on
47 the operating system's path configuration. For example, on the author's
48 computer the following command successfully starts Notepad::
49 
50  BringApp("notepad").execute()
51 
52 Applications on MacOS are started and switched to using the application
53 name::
54 
55  BringApp("System Preferences").execute()
56

79 class StartApp(ActionBase):
80  """
81  Start an application.
82 
83  When this action is executed, it runs a file (executable),
84  optionally with commandline arguments.
85 
86  """
87 
88  def __init__(self, *args, **kwargs):
89  """
90  Constructor arguments:
91  - *args* (variable argument list of *str*'s) --
92  these strings are passed to subprocess.Popen()
93  to start the application as a child process
94  - *cwd* (*str*, default *None*) --
95  if not *None*, then start the application in this
96  directory
97  - *focus_after_start* (*bool*, default *False*) --
98  if *True*, then attempt to bring the window to the foreground
99  after starting the application.
100 
101  A single *list* or *tuple* argument can be used instead of
102  variable arguments.
103 
104  """
105  ActionBase.__init__(self)
106  if len(args) == 1 and isinstance(args[0], (tuple, list)):
107  args = args[0] # use the sub-list instead
108 
109  self._args = args
110 
111  self._cwd = kwargs.pop("cwd", None)
112  self._focus_after_start = kwargs.pop("focus_after_start", False)
113  if kwargs:
114  raise ActionError("Invalid keyword arguments: %r" % kwargs)
115 
116  # Expand any variables within path names.
117  self._args = [self._interpret(a) for a in self._args]
118  if self._cwd:
119  self._cwd = self._interpret(self._cwd)
120 
121  self._str = str(", ".join(repr(a) for a in self._args))
122 
123  @classmethod
124  def _interpret(cls, path):
125  if not isinstance(path, six.string_types):
126  raise TypeError("expected string argument for path, but got "
127  "%s" % path)
128 
129  # Encode text strings if this is Python 2.
130  if six.PY2 and isinstance(path, six.text_type):
131  encoding = locale.getpreferredencoding()
132  path = path.encode(encoding)
133 
134  return os.path.expanduser(os.path.expandvars(path))
135 
136  def _start_app_process(self):
137  try:
138  return Popen(self._args, cwd=self._cwd)
139  except Exception as e:
140  raise ActionError("Failed to start app %s: %s" % (self._str, e))
141 
142  def _focus_window_after_starting(self, pid):
143  timeout = 1.0
144  exe = self._args[0]
145 
146  # Wait for the window to appear for N seconds.
147  action = WaitWindow(executable=exe, timeout=timeout)
148  if action.execute():
149  # Bring the window to the foreground.
150  Window.get_foreground().set_foreground()
151 
152  # The application window wasn't focused, so try to focus it.
153  else:
154  target = pid
155  start = time.time()
156  while time.time() - start < timeout:
157  found = False
158  for window in Window.get_matching_windows(exe):
159  if target is None or window.pid == target:
160  window.set_foreground()
161  found = True
162  break
163  if found:
164  break

166  def _darwin_start_app(self):
167  # Try to use the macOS 'open' command-line program to start a new
168  # instance of the specified application. Return False if this is
169  # either not applicable or doesn't work.
170  try_using_open = (
171  sys.platform == "darwin" and len(self._args) == 1 and
172  not os.path.isabs(self._args[0])
173  )
174  if try_using_open:
175  process = Popen(['open', '-n', '-a', self._args[0]])
176  return process.wait() == 0
177  return False
178 
179  def _execute(self, data=None):
180  self._log.debug("Starting app: %r", self._args)
181  if self._darwin_start_app():
182  pid = None
183  else:
184  # This either isn't macOS or the 'open' program didn't work, so
185  # start the application as normal and get the process ID.
186  pid = self._start_app_process().pid
187 
188  # If specified, focus the application after starting it.
189  if self._focus_after_start:
190  self._focus_window_after_starting(pid)

195 class BringApp(StartApp):
196  """
197  Bring an application to the foreground, starting it if it is not
198  yet running.
199 
200  When this action is executed, it looks for an existing window of
201  the application specified in the constructor arguments. If an
202  existing window is found, that window is brought to the
203  foreground. On the other hand, if no window is found the
204  application is started.
205 
206  Note that the constructor arguments are identical to those used by
207  the :class:`StartApp` action class.
208 
209  """
210 
211  def __init__(self, *args, **kwargs):
212  """
213  Constructor arguments:
214  - *args* (variable argument list of *str*'s) --
215  these strings are passed to :meth:`subprocess.Popen`
216  to start the application as a child process
217  - *cwd* (*str*, default *None*) --
218  if not *None*, then start the application in this
219  directory
220  - *title* (*str*, default *None*) --
221  if not *None*, then match existing windows using this
222  title.
223  - *index* (*str* or *int*) -- zero-based index of the target
224  window, for multiple matching windows; can be a string (for
225  substitution) but must be convertible to an integer.
226  - *filter_func* (*callable*) -- called with a single argument
227  (the window object), and should return ``True`` for your
228  target windows; example:
229  ``lambda window: window.get_position().dy > 100``.
230  - *focus_after_start* (*bool*, default *False*) --
231  if *True*, then attempt to bring the window to the foreground
232  after starting the application. Does nothing if the
233  application is already running.
234  - *focus_only* (*bool*, default *False*) -- if *True*, then
235  attempt to focus a matching window without raising it by
236  using the *set_focus()* method instead of *set_foreground()*.
237  This argument may do nothing depending on the platform.
238 
239  """
240  self._title = kwargs.pop("title", None)
241  self._index = kwargs.pop("index", None)
242  self._filter_func = kwargs.pop("filter_func", None)
243  self._focus_only = kwargs.pop("focus_only", False)
244  StartApp.__init__(self, *args, **kwargs)
245 
246  def _execute(self, data=None):
247  self._log.debug("Bringing app: %r", self._args)
248  target = self._args[0].lower()
249  title = self._title
250  index = self._index
251  filter_func = self._filter_func
252  focus_only = self._focus_only
253  focus_action = FocusWindow(executable=target, title=title,
254  index=index, filter_func=filter_func,
255  focus_only=focus_only)
256  # Attempt to focus on an existing window.
257  if not focus_action.execute():
258  # Failed to focus on an existing window, so start
259  # the application.
260  StartApp._execute(self, data)
