46  @classmethod
47  def get_monitor(cls, handle, rectangle):
48  """
49  Get a :class:`Monitor` object given a monitor handle.
50 
51  Given the same handle, this method will return the same object.
52 
53  :param handle: monitor handle
54  :type handle: int
55  :param rectangle: monitor rectangle
56  :type rectangle: Rectangle
57  :rtype: Monitor
58  :returns: monitor
59  """
60  if handle in cls._monitors_by_handle:
61  monitor = cls._monitors_by_handle[handle]
62  monitor.rectangle = rectangle
63  else:
64  monitor = cls(handle, rectangle)
65  cls._monitors_by_handle[handle] = monitor
66  return monitor
