117  def _calculate_center(self, size):
118  w, h = size
119  desktop = win32gui.GetDesktopWindow()
120  al, at, ar, ab = win32gui.GetWindowRect(desktop)
121  cx = (ar-al)/2; cy = (ab-at)/2
122  cx, cy = win32gui.ClientToScreen(desktop, (cx, cy))
123  l = cx - w/2; t = cy - h/2

181  def _dialog_build_message_map(self):
182  # Collect all controls which expect callbacks.
183  map = {}
184  for control in self._dialog_controls:
185  for message, callback in control.message_callbacks.items():
186  map.setdefault(message, {})[control.id] = callback
187 
188  # Create dispatchers for each type of message.
189  for message, control_callbacks in map.items():
190  def dispatcher(hwnd, msg, wparam, lparam):
191  id = win32api.LOWORD(wparam)
192  if id in control_callbacks:
193  control_callbacks[id](hwnd, msg, wparam, lparam)
194  map[message] = dispatcher
195 
196  # Add the top-level callbacks handled by the window itself.
197  map.update({
198  win32con.WM_SIZE: self.on_size,
199  win32con.WM_INITDIALOG: self.on_init_dialog,
200  win32con.WM_GETMINMAXINFO: self.on_getminmaxinfo,
201  win32con.WM_CLOSE: self.on_close,
202  win32con.WM_DESTROY: self.on_destroy,
203  })

210  def _get_edge_sizes(self):
211  wl, wt, wr, wb = win32gui.GetWindowRect(self._hwnd)
212  cl, ct, cr, cb = win32gui.GetClientRect(self._hwnd)
213  il, it = win32gui.ClientToScreen(self._hwnd, (cl, ct))
214  ir, ib = win32gui.ClientToScreen(self._hwnd, (cr, cb))
215  el = il - wl
216  et = it - wt
217  er = wr - ir
218  eb = wb - ib
219  return el, et, er, eb
