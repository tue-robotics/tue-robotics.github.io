109  def copy_from_system(self, formats=None, clear=False):
110  """
111  Copy the Windows system clipboard contents into this instance.
112 
113  Arguments:
114  - *formats* (iterable, default: None) -- if not None, only the
115  given content formats will be retrieved. If None, all
116  available formats will be retrieved.
117  - *clear* (boolean, default: False) -- if true, the Windows
118  system clipboard will be cleared after its contents have been
119  retrieved.
120 
121  """
122  win32clipboard.OpenClipboard()
123  try:
124  # Determine which formats to retrieve.
125  if not formats:
126  format = 0
127  formats = []
128  while 1:
129  format = win32clipboard.EnumClipboardFormats(format)
130  if not format:
131  break
132  formats.append(format)
133  elif isinstance(formats, int):
134  formats = (formats,)
135 
136  # Verify that the given formats are valid.
137  try:
138  for format in formats:
139  if not isinstance(format, int):
140  raise TypeError("Invalid clipboard format: %r"
141  % format)
142  except Exception, e:
143  raise
144 
145  # Retrieve Windows system clipboard content.
146  contents = {}
147  for format in formats:
148  content = win32clipboard.GetClipboardData(format)
149  contents[format] = content
150  self._contents = contents
151 
152  # Clear the system clipboard, if requested, and close it.
153  if clear:
154  win32clipboard.EmptyClipboard()
155 
156  finally:
157  win32clipboard.CloseClipboard()
