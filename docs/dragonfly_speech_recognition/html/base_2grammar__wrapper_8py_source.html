85  def _decode_grammar_rules(self, state, words, results, *args):
86  # Iterate through this grammar's rules, attempting to decode each.
87  # If successful, call that rule's method for processing the
88  # recognition and return.
89  for rule in self.grammar.rules:
90  if not (rule.active and rule.exported): continue
91  state.initialize_decoding()
92  for _ in rule.decode(state):
93  if state.finished():
94  # TODO Use words="other" instead, with a special
95  # recobs grammar wrapper at index 0.
96  # Notify observers using the manager *before*
97  # processing.
98  root = state.build_parse_tree()
99  notify_args = (words, rule, root, results)
100  self.recobs_manager.notify_recognition(
101  *notify_args
102  )
103  try:
104  rule.process_recognition(root)
105  except Exception as e:
106  self._log.exception("Failed to process rule "
107  "'%s': %s" % (rule.name, e))
108  self.recobs_manager.notify_post_recognition(
109  *notify_args

114  def process_results(self, words, rule_names, results, *args):
115  # Disabled grammars cannot process recognitions.
116  if not self.grammar.enabled:
117  return False
118 
119  # Special result words *must* be handled separately by each

125  words_rules = tuple(words)
