112  def value(self, node):
113  month = node.get_child_by_name("month").value()
114  day = node.get_child_by_name("day").value()
115  year_node = node.get_child_by_name("year")
116  if year_node is None:
117  today = date.today()
118  year = today.year
119  if month - today.month > 6:
120  # More than six months in the future, use last year.
121  year -= 1
122  elif month - today.month < -6:
123  # More than six months in the past, use next year.
124  year += 1
125  else:
126  year = year_node.value()
127  return date(year, month, day)
