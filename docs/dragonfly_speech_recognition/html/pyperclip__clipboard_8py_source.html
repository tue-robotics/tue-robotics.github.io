37 class PyperclipClipboard(BaseClipboard):
38  """
39  Class for interacting with the system clipboard via the
40  `pyperclip <https://pyperclip.readthedocs.io/en/latest/>`__ Python
41  package.
42 
43  This is Dragonfly's default clipboard class on platforms other than
44  Windows.
45 
46  .. note::
47 
48  This class does work on Windows, however the Windows
49  :class:`dragonfly.windows.win32_clipboard.Win32Clipboard` class
50  should be used instead because this class doesn't support as many
51  clipboard formats.

61  @classmethod

63  # If *content* is None, clear the clipboard and return early.
64  if content is None:
65  cls.clear_clipboard()
66  return
67 
68  # Otherwise, convert *content* and set the system clipboard data.
69  content = cls.convert_format_content(cls.format_unicode, content)
70  pyperclip.copy(content)

78  def copy_from_system(self, formats=None, clear=False):
79  # Determine the supported formats. Copied file paths may be
80  # available to us on X11.
81  supported_formats = [self.format_text, self.format_unicode]
82  if os.environ.get("DISPLAY"):
83  supported_formats.append(self.format_hdrop)
84 
85  # Determine which formats to retrieve.
86  caller_specified_formats = bool(formats)
87  if not formats:
88  formats = supported_formats
89  elif isinstance(formats, integer_types):
90  formats = (formats,)
91 
92  # Verify that the given formats are valid.
93  for format in formats:
94  if not isinstance(format, integer_types):
95  raise TypeError("Invalid clipboard format: %r"
96  % format)
97 
98  # Retrieve the system clipboard content.
99  # This class only handles text formats.
100  text = self.get_system_text()
101  contents = {}
102 
103  # Populate the clipboard contents dictionary and raise errors for
104  # unavailable formats.
105  for format in formats:
106  try:
107  err = False
108  content = self.convert_format_content(format, text)
109  contents[format] = content
110  except (ValueError, TypeError):
111  err = True
112 
113  # Do not raise an error for CF_HDROP content if the caller did
114  # not specify it.
115  if format == self.format_hdrop and not caller_specified_formats:
116  err = False
117 
118  # Always raise an error if the format is not supported.
119  if format not in supported_formats:
120  err = True
121 
122  # Raise an error if a specified clipboard format was not
123  # available.
124  if err:
125  format_repr = self.format_names.get(format, format)
126  message = ("Specified clipboard format %r is not "
127  "available." % format_repr)
128  raise TypeError(message)
129  self._contents = contents
130 
131  # Then clear the system clipboard, if requested.
132  if clear:
133  self.clear_clipboard()
134 
135  def copy_to_system(self, clear=True):
136  # Clear the system clipboard, if requested.
137  if clear:
138  self.clear_clipboard()
139 
140  # Transfer text content to system clipboard, if necessary.
141  text = self.get_text()
142  if text is not None:
143  self.set_system_text(text)
