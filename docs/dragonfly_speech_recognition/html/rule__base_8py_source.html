35 class Rule(object):
36  """
37  Rule class for implementing complete or partial voice-commands.
38 
39  This rule class represents a voice-command or part of a voice-
40  command. It contains a root element, which defines the language
41  construct of this rule.
42 
43  Constructor arguments:
44  - *name* (*str*) -- name of this rule. If *None*, a unique
45  name will automatically be generated.
46  - *element* (*Element*) --
47  root element for this rule
48  - *context* (*Context*, default: *None*) --
49  context within which to be active. If *None*, the rule will
50  always be active when its grammar is active.
51  - *imported* (boolean, default: *False*) --
52  if true, this rule is imported from outside its grammar
53  - *exported* (boolean, default: *True*) --
54  if true, this rule is a complete top-level rule which can be
55  spoken by the user. This should be *True* for voice-commands
56  that the user can speak.
57 
58  The *self._log* logger objects should be used in methods of
59  derived classes for logging purposes. It is a standard logger
60  object from the *logger* module in the Python standard library.

187  def process_begin(self, executable, title, handle):
188  """
189  Start of phrase callback.
190 
191  This method is called when the speech recognition
192  engine detects that the user has begun to speak a
193  phrase. It is called by the rule's containing grammar
194  if the grammar and this rule are active.
195 
196  The default implementation of this method checks
197  whether this rule's context matches, and if it does
198  this method calls
199  :meth:`._process_begin`.
200 
201  Arguments:
202  - *executable* -- the full path to the module whose
203  window is currently in the foreground
204  - *title* -- window title of the foreground window
205  - *handle* -- window handle to the foreground window
206 
207  """
208  if not self.grammar:
209  raise GrammarError("A Dragonfly rule cannot be processed "
210  "before it is bound to a grammar.")
211  if not self._enabled:
212  if self._active:
213  self.deactivate()
214  return
215  if self._context:
216  if self._context.matches(executable, title, handle):
217  if not self._active:
218  self.activate()
219  self._process_begin()
220  else:
221  if self._active:
222  self.deactivate()
223  else:
224  if not self._active:
225  self.activate()
226  self._process_begin()

228  def activate(self, force=False):
229  if not self._grammar:
230  raise GrammarError("A Dragonfly rule cannot be activated "
231  "before it is bound to a grammar.")
232  if not self._enabled:
233  if self._active:
234  self.deactivate()
235  return
236  if not self._active or force:
237  self._grammar.activate_rule(self)
238  self._active = True

240  def deactivate(self):
241  if not self._grammar:
242  raise GrammarError("A Dragonfly rule cannot be deactivated "
243  "before it is bound to a grammar.")
244  if self._active:
245  try:
246  self._grammar.deactivate_rule(self)
247  except Exception as e:
248  self._log.warning("Failed to deactivate rule: %s (%s)",
249  self, e)
250  self._active = False
