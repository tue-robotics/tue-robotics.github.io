36 class CallbackRecognitionObserver(RecognitionObserver):
37  """
38  Observer class for calling recognition state change callbacks.
39 
40  This class is used by the ``register_*_callback`` functions and is
41  registered with the current engine on initialization.
42 
43  Constructor arguments:
44  - *event* (*str*) -- the name of the recognition event to register for
45  (e.g. ``"on_begin"``).
46  - *function* (*callable*) -- function to call on the specified
47  recognition event.
48 
49  """
50  def __init__(self, event, function):
51  RecognitionObserver.__init__(self)
52  if not hasattr(self, event):
53  raise ValueError("'%s' is not a valid recognition event"
54  % event)
55  if not callable(function):
56  raise TypeError("callback function for event '%s' is not "
57  "callable" % event)
58  self._function = function
59  self._event = event
60  self.register()
61 
62  def _process_recognition_event(self, cb_name, required_names, **kwargs):
63  if not (self._event == cb_name and callable(self._function)):
64  return
65 
66  # If the callback function takes keyword arguments, only send those
67  # that it accepts. Always pass required names.
68  func_kwargs = kwargs
69  if func_kwargs:
70  argspec = getargspec(self._function)
71  arg_names, kwargs_names = argspec[0], argspec[2]
72  if not kwargs_names:
73  func_kwargs = {k: v for (k, v) in func_kwargs.items()
74  if k in arg_names or k in required_names}
75 
76  # Call the callback function.
77  self._function(**func_kwargs)
