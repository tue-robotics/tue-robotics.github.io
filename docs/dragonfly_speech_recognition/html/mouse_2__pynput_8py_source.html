115 class ButtonEvent(BaseButtonEvent):
116 
117  def execute(self, window):
118  # Initialize the pynput mouse controller, if necessary.
119  _init_controller()
120 
121  # Process button events.
122  for ((button, event_type), flag) in self._flags:
123  # Check if the button is unknown.
124  if button is None:
125  event_type_s = "button" if event_type == 0 else "scroll"
126  raise ValueError("Unsupported %s event" % event_type_s)
127 
128  if event_type == 0: # Button press event
129  if flag:
130  _controller.press(button)
131  else:
132  _controller.release(button)
133  elif event_type == 1: # Scroll event
134  _controller.click(button, flag)
