348  def _do_recognition(self, timeout=None, single=False, audio_iter=None):
349  """
350  Loops performing recognition, by default forever, or for *timeout* seconds, or for a single recognition if *single=True*.
351  Returns ``False`` if timeout occurred without a recognition.
352  """
353  self._log.debug("do_recognition: timeout %s" % timeout)
354  if not self._decoder:
355  raise EngineError("Cannot recognize before connect()")
356  if audio_iter is None and self._audio is None:
357  raise EngineError("No audio input")
358  self._doing_recognition = True
359  self._in_phrase = False
360  self._ignore_current_phrase = False
361  in_complex = False
362  end_time = None
363  timed_out = False
364 
365  try:
366  self.prepare_for_recognition() # Try to get compilation out of the way before starting audio
367 
368  if timeout != None:
369  end_time = time.time() + timeout
370  timed_out = True
371 
372  if audio_iter == None:
373  self._audio.start()
374  audio_iter = self._audio_iter
375  self._log.info("Listening...")
376  next(audio_iter) # Prime the audio iterator
377 
378  # Loop until timeout (if set) or until disconnect() is called.
379  while (not self._deferred_disconnect) and ((not end_time) or (time.time() < end_time)):
380  block = audio_iter.send(in_complex)
381 
382  if block is False:
383  # No audio block available
384  time.sleep(0.001)
385 
386  elif block is not None:
387  if not self._in_phrase:
388  # Start of phrase
389  self._recognition_observer_manager.notify_begin()
390  with debug_timer(self._log.debug, "computing activity"):
391  kaldi_rules_activity = self._compute_kaldi_rules_activity()
392  self._in_phrase = True
393  self._ignore_current_phrase = False
394  self._grammar_wrappers_copy = self._grammar_wrappers.copy() # Keep a copy of valid grammar wrappers as of the start of utterance
395 
396  else:
397  # Ongoing phrase
398  kaldi_rules_activity = None
399  self._decoder.decode(block, False, kaldi_rules_activity)
400  if self.audio_store:
401  self.audio_store.add_block(block)
402  output, info = self._decoder.get_output()
403  self._log.log(5, "Partial phrase: %r [in_complex=%s]", output, in_complex)
404  kaldi_rule, words, words_are_dictation_mask, in_dictation = self._compiler.parse_partial_output(output)
405  in_complex = bool(in_dictation or (kaldi_rule and kaldi_rule.is_complex))
406 
407  else:
408  # End of phrase
409  self._decoder.decode(b'', True)
410  output, info = self._decoder.get_output()
411  if not self._ignore_current_phrase:
412  expected_error_rate = info.get('expected_error_rate', nan)
413  confidence = info.get('confidence', nan)
414  # output = self._compiler.untranslate_output(output)
415  recognition = self._parse_recognition(output)
416  is_acceptable_recognition = recognition.kaldi_rule and (recognition.has_dictation or not (
417  self._options['expected_error_rate_threshold'] and (expected_error_rate > self._options['expected_error_rate_threshold'])
418  ))
419  if is_acceptable_recognition:
420  recognition.process(expected_error_rate=expected_error_rate, confidence=confidence)
421  else:
422  recognition.fail(expected_error_rate=expected_error_rate, confidence=confidence)
423 
424  kaldi_rule, parsed_output = recognition.kaldi_rule, recognition.parsed_output
425  self._log.log(15, "End of phrase: eer=%.2f conf=%.2f%s, rule %s, %r",
426  expected_error_rate, confidence, (" [BAD]" if not is_acceptable_recognition else ""), kaldi_rule, parsed_output)
427  if self._saving_adaptation_state and is_acceptable_recognition: # Don't save adaptation state for bad recognitions
428  self._decoder.save_adaptation_state()
429  if self.audio_store:
430  if kaldi_rule and is_acceptable_recognition: # Don't store audio/metadata for bad recognitions
431  self.audio_store.finalize(parsed_output,
432  kaldi_rule.parent_grammar.name, kaldi_rule.parent_rule.name,
433  likelihood=expected_error_rate, has_dictation=recognition.has_dictation)
434  else:
435  self.audio_store.cancel()
436 
437  self._in_phrase = False
438  self._ignore_current_phrase = False
439  in_complex = False
440  timed_out = False
441  if single:
442  break
443  self.prepare_for_recognition() # Do any of this leftover, now that phrase is done
444 
445  self.call_timer_callback()
446 
447  except StopIteration:
448  if audio_iter == self._audio_iter:
449  self._log.warning("audio iterator stopped unexpectedly")
450 
451  finally:
452  self._doing_recognition = False
453  if (audio_iter == self._audio_iter) and self._audio:
454  try:
455  self._audio.stop() # We started the audio above, so we should stop it
456  except Exception:
457  self._log.exception("Error stopping audio")
458  if self._deferred_disconnect:
459  self.disconnect()
460 
461  return not timed_out

570  def _parse_recognition(self, output, mimic=False):
571  if mimic or self._compiler.parsing_framework == 'text':
572  with debug_timer(self._log.debug, "kaldi_rule parse time"):
573  detect_ambiguity = False
574  results = []
575  for kaldi_rule in sorted(self._active_kaldi_rules, key=lambda kr: 100 if kr.has_dictation else 0):

578  if words is None:
579  continue
580  # self._log.debug("success %d", kaldi_rule_id)
581  # Pass (kaldi_rule, words) to below.
582  results.append((kaldi_rule, words))
583  if not detect_ambiguity:
584  break
585 
586  if not results:
587  if not mimic:
588  # We should never receive an unparsable recognition from kaldi, only from mimic
589  self._log.error("unable to parse recognition: %r" % output)

591 
592  if len(results) > 1:

594  # FIXME: improve sorting criterion
595  results.sort(key=lambda result: 100 if result[0].has_dictation else 0)

599  words_are_dictation_mask = [True] * len(words)

603  dictation_info_func=lambda: (self.audio_store.current_audio_data, self._decoder.get_word_align(output)))
604  if kaldi_rule is None:
605  if words:
606  # We should never receive an unparsable recognition from kaldi, unless it's empty (from noise)
607  self._log.error("unable to parse recognition: %r" % output)
608  return Recognition.construct_empty(self)
609 
610  if self._log.isEnabledFor(12):
611  try:
612  self._log.log(12, "Alignment (word,time,length): %s", self._decoder.get_word_align(output))
613  except KaldiError:
614  self._log.warning("Exception logging word alignment")
615 
616  else:
617  raise EngineError("Invalid _compiler.parsing_framework")
618 
619  if not words:
620  # Empty recognition, so bail before calling into dragonfly parsing/processing
621  return Recognition.construct_empty(self)
622
