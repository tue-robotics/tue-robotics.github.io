106  def _move_relative(self, rectangle):
107  if self.from_left: horizontal = rectangle.x1
108  else: horizontal = rectangle.x2
109  if isinstance(self.horizontal, float):
110  distance = self.horizontal * rectangle.dx
111  else:
112  distance = self.horizontal
113  horizontal += distance
114 
115  if self.from_top: vertical = rectangle.y1
116  else: vertical = rectangle.y2
117  if isinstance(self.vertical, float):
118  distance = self.vertical * rectangle.dy
119  else:
120  distance = self.vertical
121  vertical += distance
122 
123  self._move_mouse(horizontal, vertical)
