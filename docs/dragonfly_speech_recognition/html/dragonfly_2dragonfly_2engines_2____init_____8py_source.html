42 def get_engine(name=None, **kwargs):
43  """
44  Get the engine implementation.
45 
46  This function will initialize an engine instance using the
47  ``get_engine`` and ``is_engine_available`` functions in the engine
48  packages and return an instance of the first available engine. If
49  one has already been initialized, it will be returned instead.
50 
51  If no specific engine is requested and no engine has already been
52  initialized, this function will initialize and return an instance of
53  the first available engine in the following order:
54 
55  ======================= =========================================
56  SR engine back-end Engine name string(s)
57  ======================= =========================================
58  1. Dragon/Natlink ``"natlink"``
59  2. Kaldi ``"kaldi"``
60  3. WSR/SAPI 5 ``"sapi5", "sapi5inproc", "sapi5shared"``
61  4. CMU Pocket Sphinx ``"sphinx"``
62  ======================= =========================================
63 
64  The :ref:`Text-input engine <RefTextEngine>` can be initialized by
65  specifying ``"text"`` as the engine name. This back-end will
66  **not** be initialized if no specific engine is requested because
67  the back-end is not a real SR engine and is used mostly for testing.
68 
69  **Arguments**:
70 
71  :param name: optional human-readable name of the engine to return.
72  :type name: str
73  :param \\**kwargs: optional keyword arguments passed through to the
74  engine for engine-specific configuration.
75  :rtype: EngineBase
76  :returns: engine instance
77  :raises: EngineError
78  """
79  # pylint: disable=too-many-statements,too-many-branches
80  global _default_engine, _engines_by_name
81  log = logging.getLogger("engine")
82 
83  if name and name in _engines_by_name:
84  # If the requested engine has already been initialized, return it.
85  engine = _engines_by_name[name]
86  elif not name and _default_engine:
87  # If no specific engine is requested and an engine has already
88  # been initialized, return it.
89  engine = _default_engine
90  else:
91  # No engine has been initialized yet.
92  engine = None
93 
94  # Check if there is an already initialized engine *and* custom engine
95  # initialization arguments. This is not allowed.
96  if engine and kwargs is not None and len(kwargs) > 0:
97  message = ("Error: Passing get_engine arguments to an engine "
98  "that has already been initialized, hence these "
99  "arguments are ignored.")
100  log.error(message)
101  raise EngineError(message)
102 
103  # If there is a relevant initialized engine already, then return it.
104  if engine:
105  return engine
106 
107  # Check if we're on Windows. If we're not on Windows, then we don't
108  # evaluate Windows-only engines like natlink.
109  windows = os.name == 'nt'
110 
111  if not engine and windows and name in (None, "natlink"):
112  # Attempt to retrieve the natlink back-end.
113  try:
114  from .backend_natlink import is_engine_available
115  from .backend_natlink import get_engine as get_specific_engine
116  if is_engine_available(**kwargs):
117  engine = get_specific_engine(**kwargs)
118  except Exception as e:
119  message = ("Exception while initializing natlink engine:"
120  " %s" % (e,))
121  log.warning(message)
122  if name:
123  raise EngineError(message)
124 
125  if not engine and name in (None, "kaldi"):
126  # Attempt to retrieve the Kaldi back-end.
127  try:
128  from .backend_kaldi import is_engine_available
129  from .backend_kaldi import get_engine as get_specific_engine
130  if is_engine_available(**kwargs):
131  engine = get_specific_engine(**kwargs)
132  except Exception as e:
133  message = ("Exception while initializing kaldi engine:"
134  " %s" % (e,))
135  log.warning(message)
136  if name:
137  raise EngineError(message)
138 
139  if not engine and windows and name in (None,) + _sapi5_names:
140  # Attempt to retrieve the sapi5 back-end.
141  try:
142  from .backend_sapi5 import is_engine_available
143  from .backend_sapi5 import get_engine as get_specific_engine
144  if is_engine_available(name, **kwargs):
145  engine = get_specific_engine(name, **kwargs)
146  except Exception as e:
147  message = ("Exception while initializing sapi5 engine:"
148  " %s" % (e,))
149  log.warning(message)
150  if name:
151  raise EngineError(message)
152 
153  if not engine and name in (None, "sphinx"):
154  # Attempt to retrieve the CMU Sphinx back-end.
155  try:
156  from .backend_sphinx import is_engine_available
157  from .backend_sphinx import get_engine as get_specific_engine
158  if is_engine_available(**kwargs):
159  engine = get_specific_engine(**kwargs)
160  except Exception as e:
161  message = ("Exception while initializing sphinx engine:"
162  " %s" % (e,))
163  log.warning(message)
164  if name:
165  raise EngineError(message)
166 
167  # Only retrieve the text input engine if explicitly specified; it is not
168  # an actual SR engine implementation and is mostly intended to be used
169  # for testing.
170  if not engine and name == "text":
171  # Attempt to retrieve the TextInput engine instance.
172  try:
173  from .backend_text import is_engine_available
174  from .backend_text import get_engine as get_specific_engine
175  if is_engine_available(**kwargs):
176  engine = get_specific_engine(**kwargs)
177  except Exception as e:
178  message = ("Exception while initializing text-input engine:"
179  " %s" % (e,))
180  log.warning(message)
181  if name:
182  raise EngineError(message)
183 
184  # Return the engine instance, if one has been initialized. Log a
185  # message about which SR engine back-end was used.
186  if engine:
187  message = "Initialized %r SR engine: %r." % (engine.name, engine)
188  log.info(message)
189  return engine
190  elif not name:
191  raise EngineError("No usable engines found.")
192  else:
193  if name not in _valid_engine_names:
194  raise EngineError("Requested engine %r is not a valid engine "
195  "name." % (name,))
196  else:
197  raise EngineError("Requested engine %r not available."
198  % (name,))

249 def get_speaker(name=None):
250  """
251  Get the text-to-speech (speaker) implementation.
252 
253  This function will initialize and return a speaker instance instance
254  of the available speaker back-end. If one has already been
255  initialized, it will be returned instead.
256 
257  If no specific speaker back-end is requested and no speaker has
258  already been initialized, this function will initialize and return
259  an instance of the first available back-end in the following order:
260 
261  ======================= =========================================
262  TTS speaker back-end Speaker name string(s)
263  ======================= =========================================
264  1. SAPI 5 ``"sapi5"``
265  2. Dragon/Natlink ``"natlink"``
266  3. eSpeak ``"espeak"``
267  4. CMU Flite ``"flite"``
268  5. Text (stdout) ``"text"``
269  ======================= =========================================
270 
271  The first two speaker back-ends are only available on Microsoft
272  Windows. The second requires that Dragon NaturallySpeaking and
273  Natlink are installed on the system.
274 
275  The third and fourth back-ends, eSpeak and CMU Flite, may be used on
276  most platforms. These require that the appropriate command-line
277  programs are installed on the system.
278 
279  The last back-end (text) is used as a fallback when no real speaker
280  implementation is available. This back-end writes input text to
281  stdout, i.e., prints text to the console.
282 
283  **Arguments**:
284 
285  :param name: optional human-readable name of the speaker to return.
286  :type name: str
287  :rtype: SpeakerBase
288  :returns: speaker instance
289  :raises: EngineError
290  """
291  global _default_speaker, _speakers_by_name
292  log = logging.getLogger("speaker")
293 
294  if name and name in _speakers_by_name:
295  speaker = _speakers_by_name[name]
296  elif not name and _default_speaker:
297  speaker = _default_speaker
298  else:
299  speaker = None
300  if speaker:
301  return speaker
302 
303  windows = os.name == 'nt'
304  if not speaker and windows and name in (None,) + _sapi5_names:
305  # Check if the sapi5 back-end is available.
306  try:
307  from .backend_sapi5 import is_engine_available
308  from .backend_sapi5.speaker import Sapi5Speaker
309  if is_engine_available(name):
310  speaker = Sapi5Speaker()
311  except Exception as e:
312  message = ("Exception while initializing sapi5 speaker:"
313  " %s" % (e,))
314  log.warning(message)
315  if name:
316  raise EngineError(message)
317 
318  if not speaker and windows and name in (None, "natlink"):
319  # Check if the natlink back-end is available.
320  try:
321  from .backend_natlink import is_engine_available
322  from .backend_natlink.speaker import NatlinkSpeaker
323  if is_engine_available():
324  speaker = NatlinkSpeaker()
325  except Exception as e:
326  message = ("Exception while initializing natlink speaker:"
327  " %s" % (e,))
328  log.warning(message)
329  if name:
330  raise EngineError(message)
331 
332  if not speaker and name in (None, "espeak"):
333  # Check if eSpeak is available.
334  try:
335  from .base.speaker_stdin import EspeakSpeaker
336  if EspeakSpeaker.is_available():
337  speaker = EspeakSpeaker()
338  except Exception as e:
339  message = ("Exception while initializing eSpeak speaker:"
340  " %s" % (e,))
341  log.warning(message)
342  if name:
343  raise EngineError(message)
344 
345  if not speaker and name in (None, "flite"):
346  # Check if CMU Flite is available.
347  try:
348  from .base.speaker_stdin import FliteSpeaker
349  if FliteSpeaker.is_available():
350  speaker = FliteSpeaker()
351  except Exception as e:
352  message = ("Exception while initializing Flite speaker:"
353  " %s" % (e,))
354  log.warning(message)
355  if name:
356  raise EngineError(message)
357 
358  if not speaker and name in (None, "text"):
359  # Check if the text back-end is available.
360  try:
361  from .backend_text import is_engine_available
362  from .backend_text.speaker import TextSpeaker
363  if is_engine_available():
364  speaker = TextSpeaker()
365  except Exception as e:
366  message = ("Exception while initializing text speaker:"
367  " %s" % (e,))
368  log.warning(message)
369  if name:
370  raise EngineError(message)
371 
372  # Return the speaker instance, if one has been initialized. Log a
373  # message about which SR speaker back-end was used.
374  if speaker:
375  message = "Initialized %r speaker: %r." % (speaker.name, speaker)
376  log.info(message)
377  return speaker
378  elif not name:
379  raise EngineError("No usable speakers found.")
380  else:
381  if name not in _valid_speaker_names:
382  raise EngineError("Requested speaker %r is not a valid speaker "
383  "name." % (name,))
384  else:
385  raise EngineError("Requested speaker %r not available."
386  % (name,))
