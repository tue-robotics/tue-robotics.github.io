55 def _map_to_str(text, encoding=locale.getpreferredencoding()):
56  # Translate unicode/bytes to whatever str is in this version of
57  # Python. Decoder methods seem to require str objects.
58  if not isinstance(text, string_types):
59  text = str(text)
60  if PY2 and isinstance(text, text_type):
61  text = text.encode(encoding)
62  elif not PY2 and isinstance(text, binary_type):
63  text = text.decode(encoding)
64  return text

322  def _validate_words(self, words, search_type):
323  unknown_words = []
324 
325  # Use 'set' to de-duplicate the 'words' list.
326  for word in set(words):
327  if not self.check_valid_word(word):
328  unknown_words.append(word)

330  if unknown_words:

345  def _set_grammar(self, wrapper, activate, partial=False):
346  if not wrapper:
347  return

352  def activate_search_if_necessary():
353  if activate:
354  self._decoder.end_utterance()
355  self._decoder.active_search = wrapper.search_name
356 
357  # Check if the wrapper's search name is valid.
358  # Set the search (again) if necessary.
359  valid_search = wrapper.search_name in self._valid_searches
360  if valid_search and not wrapper.set_search:
361  # wrapper.search_name is a valid search, so return.
362  activate_search_if_necessary()
363  return
364 
365  # Return early if 'partial' is True as an optimisation to avoid
366  # recompiling grammars for every rule activation/deactivation.
367  # Also return if the search doesn't need to be set.
368  if partial or not wrapper.set_search:
369  return
370 
371  # Compile and set the jsgf search.
372  compiled = wrapper.compile_jsgf()
373  self._log.debug(compiled)
374 
375  # Nothing further to do; no public rules.
376  if "public <root> = " not in compiled:
377  wrapper.set_search = False
378  return
379 
380  # Set the JSGF search.
381  self._decoder.end_utterance()
382  self._decoder.set_jsgf_string(wrapper.search_name,
383  _map_to_str(compiled))
384  activate_search_if_necessary()

387  wrapper.set_search = False

410  def set_keyphrase(self, keyphrase, threshold, func):
411  """
412  Add a keyphrase to listen for.
413 
414  Key phrases take precedence over grammars as they are processed first.
415  They cannot be set for specific contexts (yet).
416 
417  :param keyphrase: keyphrase to add.
418  :param threshold: keyphrase threshold value to use.
419  :param func: function or method to call when the keyphrase is heard.
420  :type keyphrase: str
421  :type threshold: float
422  :type func: callable
423  :raises: UnknownWordError
424  """
425  # Check that all words in the keyphrase are in the pronunciation dictionary.
426  # This can raise an UnknownWordError.
427  self._validate_words(keyphrase.split(), "keyphrase")
428 
429  # Check that the threshold is a float.
430  if not isinstance(threshold, float):
431  raise TypeError("threshold must be a float, not %s" % threshold)
432 
433  # Add parameters to the relevant dictionaries.
434  self._keyphrase_thresholds[keyphrase] = threshold
435  self._keyphrase_functions[keyphrase] = func
436 
437  # Set the keyphrase search (again)
438  self._decoder.end_utterance()
439  self._decoder.set_kws_list("_key_phrases", self._keyphrase_thresholds)

589  def _get_best_hypothesis(self, hypotheses):
590  """
591  Take a list of speech hypotheses and return the most likely one.
592 
593  :type hypotheses: iterable
594  :return: str | None
595  """
596  # Get all distinct, non-null hypotheses.
597  distinct = tuple([h for h in set(hypotheses) if bool(h)])
598  if not distinct:
599  return None
600  elif len(distinct) == 1:
601  return distinct[0] # only one choice
602 
603  # Decide between non-null hypotheses using a Pocket Sphinx search with
604  # each hypothesis as a grammar rule.
605  grammar = RootGrammar()
606  grammar.language_name = self.language
607  for i, hypothesis in enumerate(distinct):
608  grammar.add_rule(PublicRule("rule%d" % i, Literal(hypothesis)))
609 
610  compiled = grammar.compile_grammar()
611  name = "_temp"
612 
613  # Store the current search name.
614  original = self._decoder.active_search
615 
616  # Note that there is no need to validate words in this case because
617  # each literal in the _temp grammar came from a Pocket Sphinx
618  # hypothesis.
619  self._decoder.end_utterance()
620  self._decoder.set_jsgf_string(name, _map_to_str(compiled))
621  self._decoder.active_search = name
622 
623  # Do the processing.
624  hyp = self._decoder.batch_process(
625  self._audio_buffers,
626  use_callbacks=False
627  )
628  result = hyp.hypstr if hyp else None
629 
630  # Switch back to the previous search.
631  self._decoder.end_utterance() # just in case
632  self._decoder.active_search = original
633  self._decoder.unset_search("_temp")
634  return result

636  def _speech_start_callback(self, mimicking):
637  # Get context info.
638  fg_window = Window.get_foreground()
639  window_info = {
640  "executable": fg_window.executable,
641  "title": fg_window.title,
642  "handle": fg_window.handle,
643  }
644 
645  # Call process_begin for all grammars so that any out of context
646  # grammar will not be used.
647  for wrapper in self._grammar_wrappers.copy().values():
648  wrapper.process_begin(**window_info)
649 
650  if not mimicking:
651  # Trim excess audio buffers from the start of the list. Keep a maximum 1
652  # second of silence before speech start was detected. This should help
653  # increase the performance of batch reprocessing later.
654  chunk = self.config.FRAMES_PER_BUFFER
655  rate = self.config.RATE
656  seconds = 1
657  n_buffers = int(rate / chunk * seconds)
658  self._audio_buffers = self._audio_buffers[-1 * n_buffers:]
659 
660  # Notify observers
661  self._recognition_observer_manager.notify_begin()

663  def _hypothesis_callback(self, speech, mimicking):
664  """
665  Internal Pocket Sphinx hypothesis callback method. Calls _process_hypothesis
666  and does post-processing afterwards.
667  :param speech: speech hypothesis
668  :type speech: str | None
669  :param mimicking: whether to treat speech as mimicked speech.
670  :rtype: bool
671  """
672  # Clear any recorded audio buffers.
673  self._recorder.clear_buffers()
674 
675  # Process speech. We should get back a boolean for whether processing
676  # occurred as well as the final speech hypothesis.
677  processing_occurred, final_speech = self._process_hypotheses(
678  speech, mimicking
679  )
680 
681  # Notify observers of failure.
682  results_obj = None # TODO Use PS results object once implemented
683  if not processing_occurred:
684  self._recognition_observer_manager.notify_failure(results_obj)
685 
686  # Write the training data files if necessary.
687  data_dir = self.config.TRAINING_DATA_DIR
688  if not mimicking and data_dir and os.path.isdir(data_dir):
689  # Use the default search's hypothesis if final_speech was nil.
690  if not final_speech:
691  final_speech = speech
692  try:
693  write_training_data(self.config, self._audio_buffers,
694  final_speech)
695  except Exception as e:
696  self._log.exception("Failed to write training data: %s" % e)
697 
698  # Clear audio buffer list because utterance processing has finished.
699  self._audio_buffers = []
700 
701  # Ensure that the correct search is used.
702  self._set_default_search()
703 
704  # Return whether processing occurred in case this method was called
705  # by mimic.
706  return processing_occurred

708  def _process_key_phrases(self, speech, mimicking):
709  """
710  Processing key phrase searches and return the matched keyphrase
711  (if any).
712 
713  :type speech: str
714  :param mimicking: whether to treat speech as mimicked speech.
715  :rtype: str
716  """
717  # Return if speech is empty/null or if there are no key phrases set.
718  if not (speech and self._keyphrase_thresholds):
719  return "" # no matches
720 
721  if not mimicking:
722  # Reprocess using the key phrases search
723  self._decoder.end_utterance()
724  self._decoder.active_search = "_key_phrases"
725  hyp = self._decoder.batch_process(self._audio_buffers,
726  use_callbacks=False)
727 
728  # Get the hypothesis string.
729  speech = hyp.hypstr if hyp else ""
730 
731  # Restore search to the default search.
732  self._set_default_search()
733 
734  # Return if no key phrase matched.
735  if not speech:
736  return ""
737 
738  # Handle multiple matching key phrases. This appears to be a
739  # quirk of how Pocket Sphinx 'kws' searches work. Get the best
740  # match instead if this is the case.
741  recognised_phrases = speech.split(" ")
742  if len(recognised_phrases) > 1:
743  # Remove trailing space from the last phrase.
744  recognised_phrases[len(recognised_phrases) - 1].rstrip()
745  speech = self._get_best_hypothesis(recognised_phrases)
746  else:
747  speech = speech.rstrip() # remove trailing whitespace.
748 
749  # Notify observers if a keyphrase was matched.
750  results_obj = None # TODO Use PS results object once implemented
751  result = speech if speech in self._keyphrase_functions else ""
752  words = tuple(result.split())
753  if words:
754  self._recognition_observer_manager.notify_recognition(
755  words, None, None, results_obj
756  )
757 
758  # Call the registered function if there was a match and the function
759  # is callable.
760  func = self._keyphrase_functions.get(speech, None)
761  if callable(func):
762  try:
763  func()
764  except Exception as e:
765  self._log.exception(
766  "Exception caught when executing the function for "
767  "keyphrase '%s': %s" % (speech, e)
768  )
769 
770  # Notify observers after calling the keyphrase function.
771  if words:
772  self._recognition_observer_manager.notify_post_recognition(
773  words, None, None, results_obj
774  )

778  def _process_hypotheses(self, speech, mimicking):
779  """
780  Internal method to process speech hypotheses. This should only be called
781  from 'SphinxEngine._hypothesis_callback' because that method does important
782  post processing.
783 
784  :param speech: speech
785  :param mimicking: whether to treat speech as mimicked speech.
786  :rtype: tuple
787  """
788  # Check key phrases search first.
789  keyphrase = self._process_key_phrases(speech, mimicking)
790  if keyphrase:
791  # Keyphrase search matched.
792  return True, keyphrase
793 
794  # Otherwise do grammar processing.
795  hypotheses = {}
796  wrappers = self._grammar_wrappers.copy().values()
797 
798  # Save the LM hypothesis for later.
799  lm_hypothesis = speech
800 
801  # Filter out inactive grammars.
802  wrappers = [w for w in wrappers if w.grammar_is_active]
803 
804  # Include only exclusive grammars if at least one is active.
805  exclusive_count = 0
806  for wrapper in wrappers:
807  if wrapper.exclusive: exclusive_count += 1
808  if exclusive_count > 0:
809  wrappers = [w for w in wrappers if w.exclusive]
810 
811  # No grammar has been loaded.
812  if not wrappers:
813  return False, speech
814 
815  # Batch process audio buffers for each active grammar. Store each
816  # hypothesis.
817  for wrapper in wrappers:
818  if mimicking:
819  # Just use 'speech' for everything if mimicking.
820  hyp = speech
821  else:
822  # Switch to the search for this grammar and re-process the
823  # audio.
824  self._set_grammar(wrapper, True)
825  hyp = self._decoder.batch_process(
826  self._audio_buffers,
827  use_callbacks=False
828  )
829  if hyp:
830  hyp = hyp.hypstr
831 
832  # Set the hypothesis in the dictionary.
833  hypotheses[wrapper.search_name] = hyp
834 
835  # Get the best hypothesis.
836  speech = self._get_best_hypothesis(list(hypotheses.values()))
837 
838  # If we have an hypothesis, filter out irrelevant grammars and
839  # attempt to process it with the resulting subset. Stop on the
840  # first grammar that processes the hypothesis.
841  result = False
842  decoder_results = None # FIXME Expose P.S. decoder results
843  if speech:
844  wrappers_subset = [wrapper for wrapper in wrappers
845  if hypotheses[wrapper.search_name] == speech]
846  words_rules = self._get_words_rules(speech.split(), 0)
847  for wrapper in wrappers_subset:
848  result = wrapper.process_results(words_rules,
849  wrapper.grammar.rule_names,
850  decoder_results)
851  if result: break
852 
853  # If no processing has occurred by this point, try to process a
854  # grammar using the LM hypothesis instead, if there is one.
855  if not result and lm_hypothesis:
856  dictation_words = lm_hypothesis.split()
857  words_rules = self._get_words_rules(dictation_words, 1000000)
858  for wrapper in wrappers:
859  result = wrapper.process_results(words_rules,
860  wrapper.grammar.rule_names,
861  decoder_results)
862  if result: break
863  speech = lm_hypothesis
864 
865  # Return whether processing occurred, plus the final speech
866  # hypothesis for post-processing.
867  return result, speech

874  buffers. It will do nothing if :meth:`connect` hasn't been called.

890  self._audio_buffers.append(buf)

893  self.call_timer_callback()

902  def process_wave_file(self, path):
903  """
904  Recognise speech from a wave file and return the recognition results.
905 
906  This method checks that the wave file is valid. It raises an error
907  if the file doesn't exist, if it can't be read or if the WAV header
908  values do not match those in the engine configuration.
909 
910  If recognition is paused (sleep mode), this method will call
911  :meth:`resume_recognition`.
912 
913  The wave file must use the same sample width, sample rate and number
914  of channels that the acoustic model uses.
915 
916  If the file is valid, :meth:`process_buffer` is then used to process
917  the audio.
918 
919  Multiple utterances are supported.
920 
921  :param path: wave file path
922  :raises: IOError | OSError | ValueError
923  :returns: recognition results
924  :rtype: generator
925  """
926  if not self._decoder:
927  self.connect()
928 
929  # This method's implementation has been adapted from the PyAudio
930  # play wave example:
931  # http://people.csail.mit.edu/hubert/pyaudio/#play-wave-example
932 
933  # Check that path is a valid file.
934  if not os.path.isfile(path):
935  raise IOError("'%s' is not a file. Please use a different file path.")
936 
937  # Get required audio configuration from the engine config.
938  channels, sample_width, rate, chunk = (
939  self.config.CHANNELS,
940  self.config.SAMPLE_WIDTH,
941  self.config.RATE,
942  self.config.FRAMES_PER_BUFFER
943  )
944 
945  # Make sure recognition is not paused.
946  if self.recognition_paused:
947  self.resume_recognition(notify=False)
948 
949  # Open the wave file. Use contextlib to make sure that the file is
950  # closed whether errors are raised or not.
951  # Also register a custom recognition observer for the duration.
952  obs = WaveRecognitionObserver(self)
953  with contextlib.closing(wave.open(path, "rb")) as wf, obs as obs:
954  # Validate the wave file's header.
955  if wf.getnchannels() != channels:
956  message = ("WAV file '%s' should use %d channel(s), not %d!"
957  % (path, channels, wf.getnchannels()))
958  elif wf.getsampwidth() != sample_width:
959  message = ("WAV file '%s' should use sample width %d, not "
960  "%d!" % (path, sample_width, wf.getsampwidth()))
961  elif wf.getframerate() != rate:
962  message = ("WAV file '%s' should use sample rate %d, not "
963  "%d!" % (path, rate, wf.getframerate()))
964  else:
965  message = None
966 
967  if message:
968  raise ValueError(message)
969 
970  # Use process_buffer to process each buffer.
971  for _ in range(0, int(wf.getnframes() / chunk) + 1):
972  data = wf.readframes(chunk)
973  if not data:
974  break
975 
976  self.process_buffer(data)
977 
978  # Get the results from the observer.
979  if obs.words:
980  yield obs.words
981  obs.words = ""
982 
983  # Log warnings if speech start or end weren't detected.
984  if not obs.complete:
985  self._log.warning("Speech start/end wasn't detected in the wave "
986  "file!")
987  self._log.warning("Perhaps the Sphinx '-vad_prespeech' value "
988  "should be higher?")
989  self._log.warning("Or maybe '-vad_startspeech' or "
990  "'-vad_postspeech' should be lower?")

994  Start recognising from the default recording device until

996 
997  Recognition can be paused and resumed using either the sleep/wake

999  :meth:`resume_recognition`.
