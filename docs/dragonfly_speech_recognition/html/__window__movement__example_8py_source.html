47 class WindowMovementBase:
48  """
49  Base class for moving a target window.
50  """
51 
52  def __init__(self):
53  self.target_window = None
54 
55  def target_foreground_window(self):
56  """
57  Set the target window to the current foreground window.
58  """
59  self.target_window = Window.get_foreground()
60 
61  def move_target_window(self, window_animation):
62  """
63  Move the window, animating the movement of the window if
64  specified.
65  """
66  target_window = self.target_window
67  src_pos = target_window.get_position()
68  dst_pos = self.calculate_target_window_destination(src_pos)
69  target_window.move(dst_pos, window_animation)

72  """
73  Calculate and return the target window destination.
74 
75  This method should be overridden to return a Rectangle
76  object representing where the window should be moved.
77  """
78  raise NotImplementedError()

81 class MoveWindowHere(WindowMovementBase):
82  """
83  Class for moving a target window to the current cursor position.
84  """
85 
86  def calculate_target_window_destination(self, src_position):
87  # The destination of the target window is the source position with
88  # its top left-hand corner at the current cursor position.
89  x_mouse, y_mouse = get_cursor_position()
90  src_position.x1 = x_mouse
91  src_position.y1 = y_mouse
92  return src_position

95 class CenterWindowHere(WindowMovementBase):
96  """
97  Class for centering a target window on the current cursor position.
98  """
99 
100  def calculate_target_window_destination(self, src_position):
101  # The destination of the target window is the source position
102  # centered on the current cursor position.
103  x_mouse, y_mouse = get_cursor_position()
104  relative_x_center = src_position.x_center - src_position.x1
105  relative_y_center = src_position.y_center - src_position.y1
106  src_position.x1 = x_mouse - relative_x_center
107  src_position.y1 = y_mouse - relative_y_center
108  return src_position

154 class MoveNotepadWindowRule(MoveWindowRule):
155  """
156  Class for notepad window movement rules.
157  """
158 
159  def _process_recognition(self, node, extras):
160  # Find a notepad window, set it as the target window and move it.
161  # Display a warning if no notepad window was found.
162  notepad_windows = Window.get_matching_windows(executable="notepad",
163  title="notepad")
164  if not notepad_windows:
165  print("Could not find a notepad window to move.")
166  return
167 
168  self.window_movement.target_window = notepad_windows[0]
