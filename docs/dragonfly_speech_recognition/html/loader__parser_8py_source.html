286 class _ActionGroup(object):
287  def __init__(self, items, factor=1):
288  self.items = items
289  self.factor = factor
290  def build_tree(self):
291  if len(self.items) == 1 and self.factor == 1:
292  if isinstance(self.items[0], self.__class__):
293  return self.items[0].build_tree()
294  else:
295  return self.items[0]
296  branches = []
297  for item in self.items:
298  if isinstance(item, self.__class__):
299  branch = item.build_tree()
300  if isinstance(branch, tuple) and branch[1] == 1:
301  branches.extend(branch[0])
302  else:
303  branches.append(branch)
304  else:
305  branches.append(item)
306  return (branches, self.factor)
