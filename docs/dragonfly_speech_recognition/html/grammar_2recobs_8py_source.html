134 class RecognitionHistory(list, RecognitionObserver):
135  """
136  Storage class for recent recognitions.
137 
138  Instances of this class monitor recognitions and store them
139  internally. This class derives from the built in *list* type
140  and can be accessed as if it were a normal *list* containing
141  recent recognitions. Note that an instance's contents are
142  updated automatically as recognitions are received.
143 
144  """

147  list.__init__(self)

183 class PlaybackHistory(RecognitionHistory):
184  """
185  Storage class for playing back recent recognitions via the
186  :class:`Playback` action.
187 
188  Instances of this class monitor recognitions and store them
189  internally. This class derives from the built in *list* type
190  and can be accessed as if it were a normal *list* containing
191  :class:`Playback` actions for recent recognitions. Note that an
192  instance's contents are updated automatically as recognitions are
193  received.
194 
195  """
196 
197  def __init__(self, length=10):
198  RecognitionHistory.__init__(self, length)
199 
200  def _recognition_to_item(self, words):
201  return (words, time.time())
202 
203  def __getitem__(self, key):
204  result = RecognitionHistory.__getitem__(self, key)
205  if len(result) == 1:
206  return Playback([(result[0][0], 0)])
207  elif len(result) == 2 and isinstance(result[1], float):
208  return Playback([(result[0], 0)])
209  elif len(result) == 0:
210  return Playback(())
211  else:
212  pairs = [(w1, t2 - t1)
213  for ((w1, t1), (w2, t2))
214  in zip(result[:-1], result[1:])]
215  pairs.append((result[-1][0], 0))
216  return Playback(pairs)
