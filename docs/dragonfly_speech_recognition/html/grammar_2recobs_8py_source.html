68 class RecognitionHistory(list, RecognitionObserver):
69  """
70  Storage class for recent recognitions.
71 
72  Instances of this class monitor recognitions and store them
73  internally. This class derives from the built in *list* type
74  and can be accessed as if it were a normal *list* containing
75  recent recognitions. Note that an instance's contents are
76  updated automatically as recognitions are received.
77 
78  """
79 
80  def __init__(self, length=10):
81  list.__init__(self)
82  RecognitionObserver.__init__(self)
83  self._complete = True
84 
85  if (length is None or (isinstance(length, int) and length >= 1)):
86  self._length = length
87  else:
88  raise ValueError("length must be a positive int or None,"
89  " received %r." % length)
90 
91  @property

93  """ *False* if phrase-start detected but no recognition yet. """

99  def on_recognition(self, words):
100  self._complete = True
