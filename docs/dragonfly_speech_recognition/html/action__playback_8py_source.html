51 class Playback(ActionBase):
52  """ Playback a series of recognitions. """
53 
54  def __init__(self, series, speed=1):
55  """
56  Constructor arguments:
57  - *series* (sequence of 2-tuples) --
58  the recognitions to playback. Each element must be a
59  2-tuple of the form *(["words", "two", "mimic"], interval)*,
60  where *interval* is a float giving the number of seconds to
61  pause after the given words are mimicked.
62  - *speed* (*float*) --
63  the factor by which to speed up playback. The intervals
64  after each mimic are divided by this number.
65 
66  """
67  ActionBase.__init__(self)
68  self._series = tuple(series)
69  self._speed = float(speed)
70  self._str = str([w for w, i in self._series])
71

73  return self._speed
74  def _set_speed(self, speed):
75  self._speed = float(speed)
76  speed = property(fget=_get_speed, fset=_set_speed,
77  doc="Factor to speed up playback.")
78 
79  def _execute(self, data=None):
80  engine = get_engine()
81 
82  # Mimic the series of recognitions.
83  for words, interval in self._series:
84  self._log.debug("Mimicking recognition: %r" % (words,))
85  try:
86  engine.mimic(words)
87  if interval and self._speed:
88  sleep(interval / self._speed)
89  except Exception, e:
90  raise ActionError("Playback failed: %s" % e)
