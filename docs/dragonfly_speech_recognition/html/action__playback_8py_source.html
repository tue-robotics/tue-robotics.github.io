22 Playback action
23 ============================================================================
24 
25 The :class:`Playback` action mimics a sequence of recognitions. This is
26 for example useful for repeating a series of prerecorded or predefined
27 voice-commands.
28 
29 This class could for example be used to reload with one single action::
30 
31  action = Playback([
32  (["focus", "Natlink"], 1.0),
33  (["File"], 0.5),
34  (["Reload"], 0.0),
35  ])
36  action.execute()
37 
38 
39 Mimic quirks
40 ----------------------------------------------------------------------------
41 
42 Some SR engine back-ends have confusing :meth:`engine.mimic` method
43 behavior. See the engine-specific mimic method documentation in sections
44 under :ref:`RefEngines` for more information.
45 
46 
47 Class reference
48 ----------------------------------------------------------------------------

50 """

60 class Playback(ActionBase):
61  """ Playback a series of recognitions. """
62 
63  def __init__(self, series, speed=1):
64  """
65  Constructor arguments:
66  - *series* (sequence of 2-tuples) --
67  the recognitions to playback. Each element must be a
68  2-tuple of the form *(["words", "two", "mimic"], interval)*,
69  where *interval* is a float giving the number of seconds to
70  pause after the given words are mimicked.
71  - *speed* (*float*) --
72  the factor by which to speed up playback. The intervals
73  after each mimic are divided by this number.
74 
75  """
76  ActionBase.__init__(self)
77  self._series = tuple(series)
78  self._speed = float(speed)
79  self._str = str([w for w, i in self._series])
80 
81  def _get_speed(self):
82  return self._speed
83  def _set_speed(self, speed):
84  self._speed = float(speed)
85  speed = property(fget=_get_speed, fset=_set_speed,
86  doc="Factor to speed up playback.")
87 
88  def _execute(self, data=None):
89  engine = get_engine()
90 
91  # Mimic the series of recognitions.
92  for words, interval in self._series:
93  self._log.debug("Mimicking recognition: %r", words)
94  try:
95  engine.mimic(words)
96  if interval and self._speed:
97  sleep(interval / self._speed)
98  except Exception as e:
99  raise ActionError("Playback failed: %s" % e)
