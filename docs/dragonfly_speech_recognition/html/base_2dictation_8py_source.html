21 """
22 Dictation container base class
23 ============================================================================
24 
25 This class is used to store the recognized results of dictation elements
26 within voice-commands. It offers access to both the raw spoken-form words
27 and be formatted written-form text.
28 
29 The object can be expected to behave like a string,
30 responding as you would expect to string methods like :meth:`replace`.
31 The formatted text can be retrieved using
32 :meth:`~DictationContainerBase.format` or simply by calling ``str(...)``
33 on a dictation container object.
34 By default, formatting returns the words joined with
35 spaces, but custom formatting can be applied by calling
36 string methods on the :class:`Dictation` object.
37 A tuple of the raw spoken words can be
38 retrieved using :attr:`~DictationContainerBase.words`.
39 
40 
41 String Formatting Examples
42 ----------------------------------------------------------------------------
43 
44 The following examples demonstrate dictation input can be formatted by
45 calling string methods on :class:`Dictation` elements.
46 
47 Python example:
48 
49 .. code:: python
50 
51  mapping = {
52  # Define commands for writing Python methods, functions and classes.
53  "method [<under>] <snaketext>":
54  Text("def %(under)s%(snaketext)s(self):") + Key("left:2"),
55  "function <snaketext>":
56  Text("def %(snaketext)s():") + Key("left:2"),
57  "classy [<classtext>]":
58  Text("class %(classtext)s:") + Key("left"),
59 
60  # Define a command for accessing object members.
61  "selfie [<under>] [<snaketext>]":
62  Text("self.%(under)s%(snaketext)s"),
63  }
64 
65  extras = [
66  # Define a Dictation element that produces snake case text,
67  # e.g. hello_world.
68  Dictation("snaketext", default="").lower().replace(" ", "_"),
69 
70  # Define a Dictation element that produces text matching Python's
71  # class casing, e.g. DictationContainer.
72  Dictation("classtext", default="").title().replace(" ", ""),
73 
74  # Allow adding underscores before cased text.
75  Choice("under", {"under": "_"}, default=""),
76  ]
77 
78  rule = MappingRule(name="PythonExample", mapping=mapping, extras=extras)
79 
80 
81 Markdown example:
82 
83 .. code:: python
84 
85  mapping = {
86  # Define a command for typing Markdown headings 1 to 7 with optional
87  # capitalized text.
88  "heading [<num>] [<capitalised_text>]":
89  Text("#")*Repeat("num") + Text(" %(capitalised_text)s"),
90  }
91 
92  extras = [
93  Dictation("capitalised_text", default="").capitalize(),
94  IntegerRef("num", 1, 7, 1),
95  ]
96 
97  rule = MappingRule(name="MdExample", mapping=mapping, extras=extras)

100 Camel-case example using the :meth:`Dictation.camel` method:

102 .. code:: python

104  mapping = {
105  # Define a command for typing camel-case text, e.g. helloWorld.
106  "camel <camel_text>": Text(" %(camel_text)s"),

109  extras = [

115 
116 Example using the :meth:`Dictation.apply` method for random casing:
117 
118 .. code:: python
119 
120  from random import random
121 
122  def random_text(text):
123  # Randomize the case of each character.
124  result = ""
125  for c in text:
126  r = random()
127  if r < 0.5:
128  result += c.lower()
129  else:
130  result += c.upper()
131  return result
132 
133  mapping = {
134  "random <random_text>": Text("%(random_text)s"),
135  }
136 
137  extras = [
138  Dictation("random_text", default="").apply(random_text),
139  ]
140 
141  rule = MappingRule(name="RandomExample", mapping=mapping, extras=extras)

170  def __init__(self, words, methods=None):

173  of words as recognized by the backend SR engine.
174  Each word must be a Unicode string.
175

243  def apply_methods(self, joined_words):
244  """
245  Apply any string methods called on the :class:`Dictation` object to
246  a given string.
247 
248  Called during :meth:`format`.
249  """
250  result = joined_words
251  if result: # Do nothing for empty string
252  for method in self._methods:
253  if hasattr(result, method[0]):
254  function = getattr(result, method[0])
255  result = function(*method[1], **method[2])
256  elif hasattr(self, method[0]):
257  function = getattr(self, method[0])
258  result = function(result, *method[1], **method[2])
259  else:
260  raise AttributeError("'%s' is not a valid dictation or "
261  "string method" % method[0])
262  return result
