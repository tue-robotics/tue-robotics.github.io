41 class ContextAction(ActionBase):
42  '''
43  Action class to execute a different action depending on which context is
44  currently active.
45 
46  This is especially useful for allowing the same commands to work in
47  multiple applications without redefining them in other grammars. An
48  example of this is the redo shortcut. Some applications use
49  *Ctrl+Shift+Z*, while others might use *Ctrl+Y* instead.
50  ``ContextAction`` could be used to define *Ctrl+Y* as the default and
51  use *Ctrl+Shift+Z* for specific contexts::
52 
53  redo = ContextAction(default=Key('c-y'), actions=[
54  # Use cs-z for rstudio
55  (AppContext(executable="rstudio"), Key('cs-z')),
56  ])
57 
58  This class was originally written for the
59  `Aenea <https://github.com/dictation-toolbox/aenea>`__ project by Alex
60  Roper and has been modified to work without Aenea's functionality.
61  '''
62  def __init__(self, default=None, actions=None):
63  '''
64  Constructor arguments:
65  - *default* (action object, default *do nothing*) -- the
66  default action to execute if there was no matching context in
67  *actions*.
68  - *actions* (iterable, default *empty list*) -- an iterable
69  object containing context-action pairs. The action of the
70  first matching context will be executed.
71 
72  '''
73  if actions is None:
74  actions = []
75 
76  # Validate default action.
77  # Use a new ActionBase action (to do nothing) if default is None.
78  if default is None:
79  self._log_init.debug("Using default action for ContextAction")
80  default = ActionBase()
81 
82  # Otherwise check if default is an action.
83  elif not isinstance(default, ActionBase):
84  raise TypeError("Default action for ContextAction should be an "
85  "ActionBase or None, not %s." % default)
86 
87  # Set the default action.
88  self.default = default
89 
90  # Validate the actions list.
91  # Check if it can be converted to a dictionary. or convert it to a
92  # is one already (for consistent evaluation order).
93  if not isinstance(actions, dict):
94  dict(actions) # can raise a ValueError
95 
96  # Convert dictionaries to lists instead for consistent evaluation
97  # order.
98  else:
99  actions = list(actions.items())
100 
101  # Check the types of all keys and values.
102  for (context, action) in actions:
103  if not isinstance(context, Context):
104  raise TypeError("ContextAction actions list contains "
105  "unexpected object: %s, instead of a "
106  "Context." % context)
107  if not isinstance(action, ActionBase):
108  raise TypeError("ContextAction actions list contains "
109  "unexpected object: %s, instead of an "
110  "ActionBase." % action)
111 
112  # Set the actions list.
113  self.actions = actions
114  ActionBase.__init__(self)
115 
116  def add_context(self, context, action):
117  '''
118  Add a context-action pair to the *actions* list.
119 
120  :param context: dragonfly context
121  :param action: dragonfly action
122  :type context: Context
123  :type action: ActionBase
124  '''
125  self.actions.append((context, action))
126 
127  def _execute(self, data=None):
128  data = _ensure_execution_context(data)
129  win = data['_context']
130  try:
131  for (context, action) in self.actions:
132  if context.matches(win.executable, win.title, win.handle):
133  return action.execute(data)
134 
135  # Execute the default action and return the success.
136  return self.default.execute(data)
137  except Exception as e:
138  self._log.exception("Exception from matching context or "
139  "executing action %s: %s", self._str, e)
140  return False
