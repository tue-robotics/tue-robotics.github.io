69  caps_keys = [
70  'left',
71  'right',
72  'up',
73  'down',
74  'home',
75  'end',
76  'tab',
77  'insert',
78  'escape'
79  ]
80  caps_keys.extend('f%i' % i for i in range(1, 13))
81  for key in caps_keys:
82  translation[key] = key[0].upper() + key[1:]
83  for index in range(10):
84  translation['np%i' % index] = 'KP_%i' % index
85  letters = tuple(range(ord('a'), ord('z')))
86  digits = tuple(range(ord('0'), ord('9')))
87  for char in letters + digits:
88  translation[chr(char)] = chr(char)

99 class BaseX11Keyboard(BaseKeyboard):
100  """ Base Keyboard class for X11. """
101 
102  @classmethod
103  def get_typeable(cls, char, is_text=False):
104  """ Get a Typeable object. """
105  # Get a key translation if one exists. Otherwise use the character
106  # as the key name.
107  key = KEY_TRANSLATION.get(char, char)
108  if isinstance(key, binary_type):
109  key = key.decode(getpreferredencoding())
110 
111  # Convert single character keys to their Unicode code points.
112  # This allows typing any Unicode character with the Text and Key
113  # actions. It works with both X11 keyboard implementations.
114  if len(key) == 1:
115  # Get the Unicode code point for the character.
116  encoding = getpreferredencoding()
117  code_point = key.encode("unicode_escape")[2:].decode(encoding)
118 
119  # Handle ASCII keys by getting the hex code without '0x'.
120  if not code_point:
121  code_point = hex(ord(key))[2:]
122 
123  # Create a key name that xdotool will accept (e.g. U20AC).
124  key = 'U' + code_point.upper()
125  return X11Typeable(code=key, name=char, is_text=is_text)
