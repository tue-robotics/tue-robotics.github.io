36 class JSGFCompiler(CompilerBase):
37  """
38  Dragonfly compiler for JSpeech Grammar Format (JSGF)
39 
40  This class translates dragonfly elements, rules and grammars into
41  JSGF expansions, rules and grammars respectively.
42  """
43 
44  GrammarClass = jsgf.Grammar
45 
46  @staticmethod
47  def get_reference_name(o):
48  # Return a non-nil name string.
49  if hasattr(o, "name"):
50  if not o.name:
51  name = o.__class__.__name__
52  else:
53  name = o.name
54  else:
55  # Assume the object is a string.
56  name = o
57 
58  # JSGF and Pocket Sphinx don't allow spaces in names, but dragonfly
59  # does. Work around this by changing any spaces to underscores.
60  return name.replace(" ", "_")

65  def compile_grammar(self, grammar, *args, **kwargs):
66  """
67  Take a dragonfly grammar and translate it into a JSGF grammar object
68  with methods for compiling the grammar and matching speech.
69 
70  :param grammar:
71  :param args:
72  :param kwargs:
73  :return:
74  """
75  self._log.debug("%s: Compiling grammar %s." % (self, grammar.name))
76 
77  # Create a new JSGF Grammar object.
78  unknown_words = set()
79  result = self.GrammarClass(name=self.get_reference_name(grammar))
80 
81  # Compile each dragonfly rule and add it to the new grammar.
82  for rule in grammar.rules:
83  result.add_rule(self.compile_rule(rule, result, unknown_words))
84 
85  # Also compile and add any dragonfly Lists.
86  for lst in grammar.lists:
87  result.add_rule(self.compile_list(lst, result, unknown_words))
88 
89  # Log a warning about unknown words if necessary.
90  if unknown_words:
91  self._log.warning("Grammar '%s' used words not found in the "
92  "pronunciation dictionary: %s", result.name,
93  ", ".join(sorted(unknown_words)))
94 
95  # Return None for empty grammars.
96  if not result.rules:
97  return
98 
99  return result

101  def compile_rule(self, rule, *args, **kwargs):
102  return jsgf.Rule(
103  name=self.get_reference_name(rule),
104  visible=rule.exported,
105  expansion=self.compile_element(rule.element, *args, **kwargs)
106  )

108  # ----------------------------------------------------------------------
109  # Methods for compiling dragonfly lists and dictionary lists.
110  # These have no equivalent in JSGF, so hidden/private rules are used
111  # instead.
112 
113  def compile_list(self, lst, *args, **kwargs):
114  if isinstance(lst, List):
115  literal_list = [elements_.Literal(item) for item in lst]
116  elif isinstance(lst, DictList):
117  keys = list(lst.keys())
118  keys.sort()
119  literal_list = [elements_.Literal(key) for key in keys]
120  else:
121  raise CompilerError("Cannot compile dragonfly List %s"
122  % lst)
123 
124  return jsgf.HiddenRule(
125  self.get_reference_name(lst),
126  self.compile_element(elements_.Alternative(literal_list), *args,
127  **kwargs)
128  )
129 
130  def recompile_list(self, lst, jsgf_grammar):
131  # Used from the GrammarWrapper class to get an updated list and any
132  # unknown words.
133  unknown_words = set()
134  return (self.compile_list(lst, jsgf_grammar, unknown_words),
135  unknown_words)

140  def compile_element(self, element, *args, **kwargs):
141  # Look for a compiler method to handle the given element.
142  for element_type, compiler in self.element_compilers:
143  if isinstance(element, element_type):
144  return compiler(self, element, *args, **kwargs)
145 
146  # Didn't find a compiler method for this element type.
147  raise NotImplementedError("Compiler %s not implemented"
148  " for element type %s."
149  % (self, element))

151  def _compile_repetition(self, element, *args, **kwargs):
152  # Compile the first element only; pyjsgf doesn't support limits on
153  # repetition (yet).
154  children = element.children
155  if len(children) > 1:
156  self._log.debug("Ignoring limits of repetition element %s."
157  % element)
158  compiled_child = self.compile_element(children[0], *args, **kwargs)
159  return jsgf.Repeat(compiled_child)
160 
161  def _compile_sequence(self, element, *args, **kwargs):
162  # Compile Repetition elements separately.
163  if isinstance(element, elements_.Repetition):
164  return self._compile_repetition(element, *args, **kwargs)
165 
166  children = element.children
167  if len(children) > 1:
168  return jsgf.Sequence(*[
169  self.compile_element(c, *args, **kwargs) for c in children
170  ])
171  elif len(children) == 1:
172  # Skip redundant (1 child) sequences.
173  return self.compile_element(children[0], *args, **kwargs)
174  else:
175  # Compile an Empty element for empty sequences.
176  return self.compile_element(elements_.Empty(), *args, **kwargs)
177 
178  def _compile_alternative(self, element, *args, **kwargs):
179  children = element.children
180  if len(children) > 1:
181  return jsgf.AlternativeSet(*[
182  self.compile_element(c, *args, **kwargs) for c in children
183  ])
184  elif len(children) == 1:
185  # Skip redundant (1 child) alternatives.
186  return self.compile_element(children[0], *args, **kwargs)
187  else:
188  # Compile an Empty element for empty alternatives.
189  return self.compile_element(elements_.Empty(), *args, **kwargs)
190 
191  def _compile_optional(self, element, *args, **kwargs):
192  child = self.compile_element(element.children[0], *args, **kwargs)
193  return jsgf.OptionalGrouping(child)
194 
195  def _compile_literal(self, element, *args, **kwargs):
196  return jsgf.Literal(" ".join(element.words))
197 
198  def _compile_rule_ref(self, element, *args, **kwargs):
199  name = element.rule.name.replace(" ", "_")
200  return jsgf.NamedRuleRef(name)
201 
202  def _compile_list_ref(self, element, *args, **kwargs):
203  name = element.list.name.replace(" ", "_")
204  return jsgf.NamedRuleRef(name)
205 
206  def _compile_empty(self, element, *args, **kwargs):
207  return jsgf.NullRef()
208 
209  def _compile_impossible(self, element, *args, **kwargs):
210  return jsgf.VoidRef()

212  def _compile_dictation(self, element, *args, **kwargs):
213  # JSGF has no equivalent for dictation elements. Instead compile and
214  # return an Impossible element that allows dictation to be used,
215  # but not matched.
216  return self.compile_element(
217  elements_.Impossible(), *args, **kwargs

221 class PatchedRepeat(jsgf.Repeat):
222  """
223  Repeat class patched to compile JSGF repeats as
224  "expansion [expansion]*" to avoid a bug in Pocket Sphinx with the
225  repeat operator.
226  """
227  def compile(self, ignore_tags=False):
228  super(PatchedRepeat, self).compile()
229  compiled = self.child.compile(ignore_tags)
230  if self.tag and not ignore_tags:
231  return "(%s)[%s]*%s" % (compiled, compiled, self.tag)
232  else:
233  return "(%s)[%s]*" % (compiled, compiled)

236 class SphinxJSGFCompiler(JSGFCompiler):
237  """
238  JSGF compiler sub-class used by the CMU Pocket Sphinx backend.
239  """
240 
241  def __init__(self, engine):
242  JSGFCompiler.__init__(self)
243  self.engine = engine
244 
245  # Use a very unlikely phrase to replace unknown words. NullRefs are
246  # used instead if words aren't in the vocabulary.
247  self.impossible_literal = {
248  "en": "impossible " * 20,
249  }.get(engine.language, "")
250 
251  # ----------------------------------------------------------------------
252  # Methods for compiling elements.
253 
254  def _compile_repetition(self, element, *args, **kwargs):
255  # Compile the first element only; pyjsgf doesn't support limits on
256  # repetition (yet).
257  children = element.children
258  if len(children) > 1:
259  self._log.debug("Ignoring limits of repetition element %s."
260  % element)
261 
262  # Return a PatchedRepeat instead of a normal Repeat expansion.
263  compiled_child = self.compile_element(children[0], *args, **kwargs)
264  return PatchedRepeat(compiled_child)
265 
266  def _compile_literal(self, element, *args, **kwargs):
267  # Build literals as sequences and use <NULL> for unknown words.
268  children = []
269  for word in element.words:
270  if self.engine.check_valid_word(word):
271  children.append(jsgf.Literal(word))
272  else:
273  children.append(self.compile_element(
274  elements_.Impossible(), *args, **kwargs
275  ))
276 
277  # Save the unknown word.
278  args[1].add(word)
279 
280  return jsgf.Sequence(*children)
281 
282  def _compile_impossible(self, element, *args, **kwargs):
283  # Override this to avoid VoidRefs disabling entire rules/grammars.
284  # Use a special <_impossible> private rule instead. Only add the
285  # special rule if it isn't in the result grammar.
286  grammar = args[0]
287  if "_impossible" not in grammar.rule_names:
288  # Check that the impossible literal contains only valid words.
289  words = set(self.impossible_literal.split())
290  valid_literal = bool(words)
291  for word in words:
292  if not valid_literal:
293  break
294  if not self.engine.check_valid_word(word):
295  valid_literal = False
296 
297  if valid_literal:
298  expansion = jsgf.Literal(self.impossible_literal)
299  else:
300  # Fallback on a NullRef. There are some problems with using
301  # these, but they get the job done for simple rules.
302  expansion = jsgf.NullRef()
303  grammar.add_rule(jsgf.Rule(
304  name="_impossible", visible=False, expansion=expansion
305  ))
306 
307  return jsgf.NamedRuleRef("_impossible")

310  def _compile_dictation(self, element, *args, **kwargs):
311  return self.compile_element(
312  elements_.Impossible(), *args, **kwargs
