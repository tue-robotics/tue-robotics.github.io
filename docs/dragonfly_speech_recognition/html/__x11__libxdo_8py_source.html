39  def send_keyboard_events(cls, events):
40  """
41  Send a sequence of keyboard events.
42 
43  Positional arguments:
44  events -- a sequence of tuples of the form
45  (keycode, down, timeout), where
46  keycode (str): key symbol.
47  down (boolean): True means the key will be pressed down,
48  False means the key will be released.
49  timeout (int): number of seconds to sleep after
50  the keyboard event.
51 
52  """
53  cls._log.debug("Keyboard.send_keyboard_events %r", events)
54 
55  # TODO: We can distill this entire loop down to a single libxdo function
56  # call when we figure out how to properly user charcode_t entities from
57  # libxdo.
58  for event in events:
59  (key, down, timeout) = event
60  delay_micros = int(timeout * 1000.0)
61  key = KEY_TRANSLATION.get(key, key)
62 
63  # Press/release the key, catching any errors.
64  try:
65  if down:
66  cls.libxdo.send_keysequence_window_down(0, key, delay_micros)
67  else:
68  cls.libxdo.send_keysequence_window_up(0, key, delay_micros)
69  except Exception as e:
70  cls._log.exception("Failed to type key code %s: %s",
71  key, e)
72 
73  # Sleep after the keyboard event if necessary.
74  if timeout:
75  time.sleep(timeout)
