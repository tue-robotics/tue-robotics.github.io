83 def _phrase_to_regex(phrase):
84  # Treat whitespace between words as meaning anything other than alphanumeric
85  # characters.
86  pattern = r"[^\w--_]+".join(regex.escape(word) for word in phrase.split())
87  # Treat spaces at the beginning or end of the phrase as matching any
88  # whitespace character. This makes it easy to select stuff like non-breaking
89  # space, which occurs frequently in browsers.
90  # TODO Support newlines. Note that these are frequently implemented as
91  # separate text nodes in the accessibility tree, so the obvious
92  # implementation would not work well.
93  if phrase == " ":
94  pattern = r"\s"
95  else:
96  if phrase.startswith(" "):
97  pattern = r"\s" + pattern
98  if phrase.endswith(" "):
99  pattern = pattern + r"\s"
100  # Only match at boundaries of alphanumeric sequences if the phrase ends
101  # are alphanumeric.
102  if regex.search(r"^[\w--_]", phrase, regex.VERSION1 | regex.UNICODE):
103  pattern = r"(?<![\w--_])" + pattern
104  if regex.search(r"[\w--_]$", phrase, regex.VERSION1 | regex.UNICODE):
105  pattern = pattern + r"(?![\w--_])"
106  return pattern

109 def _find_text(query, expanded_text, cursor_offset):
110  # Convert query to regex with a single matching group. Note that this used
111  # to be simpler but buggy, and is now is complicated because lookbehind and
112  # lookahead regular expressions must be fixed-width, so we have to limit
113  # their usage and use a matching group instead, which requires matching up
114  # the parens surrounding the group.
115  pattern = r""
116 
117  # Add the start phrases, if present.
118  if query.start_phrase or query.start_relative_phrase:
119  if query.start_relative_phrase and query.start_relative_position == CursorPosition.AFTER:
120  pattern += _phrase_to_regex(query.start_relative_phrase)
121  if query.start_phrase:
122  pattern += r"[^\w--_]*"
123  pattern += r"(" + _phrase_to_regex(query.start_phrase)
124  if query.start_relative_phrase and query.start_relative_position == CursorPosition.BEFORE:
125  if query.start_phrase:
126  pattern += r"[^\w--_]*"
127  pattern += _phrase_to_regex(query.start_relative_phrase)
128  pattern += r".*?"
129 
130  # Add the end phrases.
131  if query.end_relative_phrase and query.end_relative_position == CursorPosition.AFTER:
132  pattern += _phrase_to_regex(query.end_relative_phrase)
133  if query.end_phrase:
134  pattern += r"[^\w--_]*"
135  # Add the initial "(" if we haven't already.
136  if not (query.start_phrase or query.start_relative_phrase):
137  pattern += r"("
138  pattern += _phrase_to_regex(query.end_phrase) + r")"
139  if query.end_relative_phrase and query.end_relative_position == CursorPosition.BEFORE:
140  if query.end_phrase:
141  pattern += r"[^\w--_]*"
142  pattern += _phrase_to_regex(query.end_relative_phrase)
143 
144  # Find all matches.
145  matches = regex.finditer(pattern,
146  expanded_text,
147  regex.IGNORECASE | regex.VERSION1 | regex.UNICODE,
148  overlapped=True)
149  ranges = [(match.start(1), match.end(1)) for match in matches]
150  if not ranges:
151  _log.warning("Not found: %s" % query)
152  return None
153  start_with_cursor = (query.through and not (query.start_phrase or query.start_relative_phrase))
154  if cursor_offset is None:
155  _log.warning("Cursor not found.")
156  if start_with_cursor:
157  _log.warning("Cannot perform cursor-relative query without cursor.")
158  return None
159  else:
160  # Pick arbitrary match (the first one).
161  return ranges[0]
162  else:
163  # Find nearest match.
164  range = min(ranges, key=lambda x: abs((x[0] + x[1]) / 2 - cursor_offset))
165  if start_with_cursor:
166  if range[0] < cursor_offset:
167  return (range[0], cursor_offset)
168  else:
169  return (cursor_offset, range[1])
170  else:
171  return range
