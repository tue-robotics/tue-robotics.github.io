227  # @trace_compile
228  def _compile_sequence(self, element, src_state, dst_state, grammar, kaldi_rule, fst):
229  src_state = self.add_weight_linkage(src_state, dst_state, self.get_weight(element), fst)
230  children = element.children
231  # Optimize for special lengths
232  if len(children) == 0:
233  fst.add_arc(src_state, dst_state, None)
234  return
235 
236  elif len(children) == 1:
237  return self.compile_element(children[0], src_state, dst_state, grammar, kaldi_rule, fst)
238 
239  else: # len(children) >= 2:
240  # Handle Repetition elements differently as a special case
241  is_repetition = isinstance(element, elements_.Repetition)
242  if is_repetition and element.optimize:
243  # Repetition...
244  # Insert new states, so back arc only affects child
245  s1 = fst.add_state()
246  s2 = fst.add_state()
247  fst.add_arc(src_state, s1, None)
248  # NOTE: to avoid creating an un-decodable epsilon loop, we must not allow an all-epsilon child here (compile_graph_agf should check this)
249  self.compile_element(children[0], s1, s2, grammar, kaldi_rule, fst)
250  if not fst.has_eps_path(s1, s2, self._eps_like_nonterms):
251  fst.add_arc(s2, s1, fst.eps_disambig, fst.eps) # Back arc, uses eps_disambig ('#0')
252  fst.add_arc(s2, dst_state, None)
253  return
254 
255  else:
256  # Cannot do optimize path, because of epsilon loop, so finish up with Sequence path
257  self._log.warning("%s: Cannot optimize Repetition element, because its child element can match empty string;"
258  " falling back to inefficient non-optimize path. (this is not that bad)" % self)
259  states = [src_state, s2] + [fst.add_state() for i in range(len(children)-2)] + [dst_state]
260  for i, child in enumerate(children[1:], start=1):
261  s1 = states[i]
262  s2 = states[i + 1]
263  self.compile_element(child, s1, s2, grammar, kaldi_rule, fst)
264  return
265 
266  else:
267  # Sequence, not Repetition...
268  # Insert new states for individual children elements
269  states = [src_state] + [fst.add_state() for i in range(len(children)-1)] + [dst_state]
270  for i, child in enumerate(children):
271  s1 = states[i]
272  s2 = states[i + 1]
273  self.compile_element(child, s1, s2, grammar, kaldi_rule, fst)
274  return

359  def get_weight(self, obj, name='weight'):
360  """ Gets the weight of given grammar or rule, checking for invalid values. """
361  weight = getattr(obj, name, 1)
362  if isinstance(obj, (elements_.Dictation, UserDictation)) and isinstance(weight, types.FunctionType):
363  # Ignore crazy string method handling on Dictation elements; use default value
364  weight = 1
365  try:
366  weight = float(weight)
367  except TypeError:
368  self._log.error("%s: Weight must be a numeric, but %s.%s is %s: %r" % (self, obj, name, type(weight), weight))
369  weight = 1
370  if weight <= 0:
371  self._log.error("%s: Weight cannot be negative or 0, but %s.%s is %s" % (self, obj, name, weight))
372  weight = 1e-9
373  return weight
