57  def _send_request(self, request_function):
58  # Send requests to the server using new threads to emulate requests
59  # from a different process.
60  request_thread = threading.Thread(target=request_function)
61  request_thread.start()
62 
63  # Manually process server requests by calling the timer manager's
64  # main callback directly.
65  while request_thread.is_alive():
66  get_engine()._timer_manager.main_callback()
67  time.sleep(0.05)
68 
69  # Handle errors and results from the request.
70  if isinstance(self.current_response, Exception):
71  raise self.current_response
72  else:
73  return self.current_response
