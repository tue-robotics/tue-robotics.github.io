120  def _do_recognition(self, delay=0):
121  """
122  Recognize words from standard input (stdin) in a loop until
123  interrupted or :meth:`disconnect` is called.
124 
125  :param delay: time in seconds to delay before mimicking each
126  command. This is useful for testing contexts.
127  """
128  self.connect()
129  try:
130  # Use iter to avoid a bug in Python 2.x:
131  # https://bugs.python.org/issue3907
132  for line in iter(sys.stdin.readline, ''):
133  line = line.strip()
134  if not line: # skip empty lines.
135  continue
136 
137  # Delay between mimic() calls if necessary.
138  if delay > 0:
139  time.sleep(delay)
140 
141  try:
142  self.mimic(line.split())
143  except MimicFailure:
144  self._log.error("Mimic failure for words: %s", line)
145 
146  # Finish early if disconnect() was called.
147  if self._connected is False:
148  break
149  except KeyboardInterrupt:
150  pass

152  def mimic(self, words, **kwargs):
153  """
154  Mimic a recognition of the given *words*.
155 
156  :param words: words to mimic
157  :type words: str|iter
158  :Keyword Arguments:
159 
160  optional *executable*, *title* and/or *handle* keyword arguments
161  may be used to simulate a specific foreground window context. The
162  current foreground window attributes will be used instead for any
163  keyword arguments not present.
164 
165  """
166  # The *words* argument should be a string or iterable.
167  # Words are put into lowercase for consistency.
168  if isinstance(words, string_types):
169  words = words.lower().split()
170  elif iter(words):
171  words = [w.lower() for w in words]
172  else:
173  raise TypeError("%r is not a string or other iterable object"
174  % words)
175 
176  # Fail on empty input.
177  if not words:
178  raise MimicFailure("Invalid mimic input %r" % words)
179 
180  # Notify observers that a recognition has begun.
181  self._recognition_observer_manager.notify_begin()
182 
183  # Get a sequence of (word, rule_id) 2-tuples.
184  words_rules = self._get_words_rules(words, 0)
185 
186  w = Window.get_foreground()
187  process_args = {
188  "executable": w.executable,
189  "title": w.title,
190  "handle": w.handle,
191  }
192  # Allows optional passing of window attributes to mimic
193  process_args.update(kwargs)
194 
195  # Call process_begin() for each grammar wrapper. Use a copy of
196  # _grammar_wrappers in case it changes.
197  # TODO Determine whether this should really be done for exclusive
198  # grammars.
199  for wrapper in self._grammar_wrappers.copy().values():
200  wrapper.process_begin(**process_args)
201 
202  # Take another copy of _grammar_wrappers to use for processing.
203  grammar_wrappers = self._grammar_wrappers.copy().values()
204 
205  # Filter out inactive grammars.
206  grammar_wrappers = [wrapper for wrapper in grammar_wrappers
207  if wrapper.grammar_is_active]
208 
209  # Include only exclusive grammars if at least one is active.
210  exclusive_count = 0
211  for wrapper in grammar_wrappers:
212  if wrapper.exclusive: exclusive_count += 1
213  if exclusive_count > 0:
214  grammar_wrappers = [wrapper for wrapper in grammar_wrappers
215  if wrapper.exclusive]
216 
217  # Attempt to process the mimicked words, stopping on the first
218  # grammar that processes them.
219  result = False
220  for wrapper in grammar_wrappers:
221  rule_names = wrapper.grammar.rule_names
222  result = wrapper.process_results(words_rules, rule_names, None)
223  if result: break
224 
225  # If no processing has occurred, then the mimic has failed.
226  if not result:
227  self._recognition_observer_manager.notify_failure(None)
228  raise MimicFailure("No matching rule found for words %r."
229  % (words,))
