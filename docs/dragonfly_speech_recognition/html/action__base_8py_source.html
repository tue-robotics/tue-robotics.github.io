250 class Repeat(object):
251  """
252  Action repeat factor.
253 
254  Integer Repeat factors ignore any supply data::
255 
256  >>> integer = Repeat(3)
257  >>> integer.factor()
258  3
259  >>> integer.factor({"foo": 4}) # Non-related data is ignored.
260  3
261 
262  Named Repeat factors retrieved their factor-value from the
263  supplied data::
264 
265  >>> named = Repeat(extra="foo")
266  >>> named.factor()
267  Traceback (most recent call last):
268  ...
269  ActionError: No extra repeat factor found for name 'foo' ('NoneType' object is unsubscriptable)
270  >>> named.factor({"foo": 4})
271  4
272 
273  Repeat factors with both integer count and named extra values set
274  combined (add) these together to determine their factor-value::
275 
276  >>> combined = Repeat(count=3, extra="foo")
277  >>> combined.factor()
278  Traceback (most recent call last):
279  ...
280  ActionError: No extra repeat factor found for name 'foo' ('NoneType' object is unsubscriptable)
281  >>> combined.factor({"foo": 4}) # Combined factors 3 + 4 = 7.
282  7
283 
284  """
285 
286  def __init__(self, count=None, extra=None):
287  if count is not None: self._count = count
288  else: self._count = 0
289  self._extra = extra
290 
291  def factor(self, data=None):
292  count = self._count
293  if self._extra:
294  try:
295  additional = int(data[self._extra])
296  except ValueError:
297  raise ActionError("Repeat factor %r has invalid value %r"
298  % (self._extra, data[self._extra]))
299  except KeyError:
300  raise ActionError("No extra repeat factor found for name %r"
301  % (self._extra,))
302  except Exception, e:
303  raise ActionError("No extra repeat factor found for name %r"
304  " (%s)" % (self._extra, e))
305  count += additional
306  return count
