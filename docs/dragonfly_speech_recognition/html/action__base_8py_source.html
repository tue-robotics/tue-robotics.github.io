317 class Repeat(object):
318  # pylint: disable=line-too-long
319 
320  """
321  Action repeat factor.
322 
323  Integer Repeat factors ignore any supply data::
324 
325  >>> integer = Repeat(count=3)
326  >>> integer.factor()
327  3
328  >>> integer.factor({"foo": 4}) # Non-related data is ignored.
329  3
330 
331  Integer Repeat factors can be specified with the ``*`` operator::
332 
333  >>> from dragonfly import Function
334  >>> def func():
335  ... print("executing 'func'")
336  ...
337  >>> action = Function(func) * 3
338  >>> action.execute()
339  executing 'func'
340  executing 'func'
341  executing 'func'
342 
343  Named Repeat factors retrieved their factor-value from the
344  supplied data::
345 
346  >>> named = Repeat("foo")
347  >>> named.factor()
348  Traceback (most recent call last):
349  ...
350  ActionError: No extra repeat factor found for name 'foo' ('NoneType' object is unsubscriptable)
351  >>> named.factor({"foo": 4})
352  4
353 
354  Repeat factors with both integer count and named extra values set
355  combined (add) these together to determine their factor-value::
356 
357  >>> combined = Repeat(extra="foo", count=3)
358  >>> combined.factor()
359  Traceback (most recent call last):
360  ...
361  ActionError: No extra repeat factor found for name 'foo' ('NoneType' object is unsubscriptable)
362  >>> combined.factor({"foo": 4}) # Combined factors 3 + 4 = 7.
363  7
364 
365  """
366 
367  def __init__(self, extra=None, count=None):
368  # Backward compatibility for swapped arguments
369  # (#103)
370  if isinstance(extra, integer_types):
371  self._count = extra
372  self._extra = count
373  else:
374  self._extra = extra
375  self._count = count if count is not None else 0
376 
377  def factor(self, data=None):
378  count = self._count
379  if self._extra:
380  try:
381  additional = int(data[self._extra])
382  except ValueError:
383  raise ActionError("Repeat factor %r has invalid value %r"
384  % (self._extra, data[self._extra]))
385  except KeyError:
386  raise ActionError("No extra repeat factor found for name %r"
387  % (self._extra,))
388  except Exception as e:
389  raise ActionError("No extra repeat factor found for name %r"
390  " (%s)" % (self._extra, e))
391  count += additional
392  return count
393 
394  def __str__(self):
395  return '{}'.format(self._extra if self._extra else self._count)
