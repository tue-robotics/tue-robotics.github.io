140 class CharChoice(RuleRef):
141 
142  def __init__(self, name=None):
143  letter_element = Sequence([
144  Optional(Compound(uppercase_prefix, name="upper")),
145  Choice("letter", letter_names),
146  ])
147  other_choices = {}
148  other_choices.update(digit_names)
149  other_choices.update(symbol_names)
150  other_element = Choice(name="other", choices=other_choices)
151 
152  root_element = Alternative([letter_element, other_element])
153  root_rule = Rule(element=root_element)
154  RuleRef.__init__(self, name=name, rule=root_rule)
155 
156  def value(self, node):
157  child = node.get_child_by_name("letter")
158  if child:
159  value = child.value()
160  if node.has_child_with_name("upper"):
161  value = value.upper()
162  return value
163  child = node.get_child_by_name("other")
164  return child.value()

179 def choice_series_wrap_class(choices):
180  class _ChoiceWrapClass(RuleRef):
181  _choices = choices
182  def __init__(self, name=None, min=1, max=8):
183  choice_element = Choice(name=None, choices=self._choices)
184  choice_rule = Rule(element=choice_element)
185  choice_ref = RuleRef(rule=choice_rule)
186  series_element = Repetition(child=choice_ref, min=min, max=max)
187  series_rule = Rule(element=series_element)
188  RuleRef.__init__(self, name=name, rule=series_rule)
189  def value(self, node):
