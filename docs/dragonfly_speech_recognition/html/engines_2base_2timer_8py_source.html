34 class Timer(object):
35  """
36  Timer class for calling a function every N seconds.
37 
38  Constructor arguments:
39 
40  - *function* (*callable*) -- the function to call every N seconds. Must
41  have no required arguments.
42  - *interval* (*float*) -- number of seconds between calls to the
43  function. Note that this is on a best-effort basis only.
44  - *manager* (:class:`TimerManagerBase`) -- engine timer manager instance.
45  - *repeating* (*bool*) -- whether to call the function every N seconds
46  or just once (default: True).
47 
48  Instances of this class are normally initialised from
49  :meth:`engine.create_timer`.
50  """
51 
52  _log = logging.getLogger("engine.timer")
53 
54  def __init__(self, function, interval, manager, repeating=True):
55  self.function = function
56  self.interval = interval
57  self.manager = manager
58  self.repeating = repeating
59  self.active = False
60  self.next_time = None
61  self.start()
62 
63  def start(self):
64  """
65  Start calling the timer's function on an interval.
66 
67  This method is called on initialisation.
68  """
69  if self.active:
70  return
71  self.manager.add_timer(self)
72  self.active = True
73  self.next_time = time.time() + self.interval
74 
75  def stop(self):
76  """ Stop calling the timer's function on an interval. """
77  if not self.active:
78  return
79  self.manager.remove_timer(self)
80  self.active = False
81  self.next_time = None
82 
83  def call(self):
84  """
85  Call the timer's function.
86 
87  This method is normally called by the timer manager.
88  """
89  self.next_time = time.time() + self.interval
90  try:
91  self.function()
92  except Exception as e:
93  self._log.exception("Exception during timer callback: %s" % (e,))
94  if not self.repeating:
95  self.stop()

98 class TimerManagerBase(object):
99  """ Base timer manager class. """
100 
101  _log = logging.getLogger("engine.timer")
102 
103  def __init__(self, interval, engine):
104  self.interval = interval
105  self.engine = engine
106  self.timers = []
107  self._lock = RLock()
108  self._enabled = True
109  self._active = False
110 
111  def add_timer(self, timer):
112  """ Add a timer and activate the main callback if required. """
113  with self._lock:
114  self.timers.append(timer)
115  if len(self.timers) == 1 and self._enabled:
116  self._activate_main_callback(self.main_callback,
117  self.interval)
118  self._active = True
119 
120  def remove_timer(self, timer):
121  """ Remove a timer and deactivate the main callback if required. """
122  try:
123  with self._lock:
124  self.timers.remove(timer)
125  except Exception as e:
126  self._log.exception("Failed to remove timer: %s" % e)
127  return
128  if len(self.timers) == 0 and self._enabled:
129  self._deactivate_main_callback()
130  self._active = False
131 
132  def enable(self):
133  """
134  Method to re-enable the main timer callback.
135 
136  The main timer callback is enabled by default. This method is only
137  useful if :meth:`disable` is called.
138  """
139  self._enabled = True
140 
141  def disable(self):
142  """
143  Method to disable execution of the main timer callback.
144 
145  This method is used for testing timer-related functionality without
146  race conditions.
147  """
148  self._enabled = False
149 
150  def main_callback(self):
151  """ Method to call each timer's function when required. """
152  # Deactivate the engine's main callback if necessary. We don't
153  # return early here because this method should work if called
154  # manually.
155  if not self._enabled and self._active:
156  self._deactivate_main_callback()
157  self._active = False
158 
159  now = time.time()
160  for c in tuple(self.timers):
161  if c.next_time < now:
162  try:
163  c.call()
164  except Exception as e:
165  self._log.exception("Exception occurred during"
166  " timer callback: %s" % (e,))

185 class ThreadedTimerManager(TimerManagerBase):
186  """
187  Timer manager class using a daemon thread.
188 
189  This class is used by the "text" engine. It is only suitable for engine
190  backends with no recognition loop to execute timer functions on.
191 
192  .. warning::
193 
194  The timer interface is **not** thread-safe. Use the :meth:`enable`
195  and :meth:`disable` methods if you need finer control over timer
196  function execution.
197  """
198 
199  def __init__(self, interval, engine):
200  TimerManagerBase.__init__(self, interval, engine)
201  self._running = False
202  self._thread = None
203 
204  def _activate_main_callback(self, callback, sec):
205  """"""
206  # Do nothing if the thread is already running.
207  if self._running:
208  return
209 
210  def run():
211  while self._running:
212  time.sleep(sec)
213  callback()
214 
215  self._running = True
216  self._thread = Thread(target=run)
217  self._thread.setDaemon(True)
218  self._thread.start()
219 
220  def _deactivate_main_callback(self):
221  """"""
222  # Stop the thread's main loop and wait until it finishes, timing out
223  # after 5 seconds.
224  should_join = (self._thread and self._thread.is_alive() and
225  self._thread is not current_thread())
226  self._running = False
227  if should_join:
228  timeout = 5
229  self._thread.join(timeout=timeout)
230  if self._thread.is_alive():
231  raise RuntimeError("failed to deactivate main callback "
232  "after %d seconds" % timeout)
