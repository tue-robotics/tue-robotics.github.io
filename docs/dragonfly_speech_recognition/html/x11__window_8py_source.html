70  @classmethod
71  def _run_command(cls, command, arguments):
72  """
73  Run a command with arguments and return the result.
74 
75  :param command: command to run
76  :type command: str
77  :param arguments: arguments to append
78  :type arguments: list
79  :returns: stdout, return_code
80  :rtype: tuple
81  """
82  arguments = [str(arg) for arg in arguments]
83  full_command = [command] + arguments
84  full_readable_command = ' '.join(full_command)
85  kwargs = dict(stdout=PIPE, stderr=PIPE)
86 
87  # Fork the process with setsid() if on a POSIX system.
88  if os.name == 'posix':
89  kwargs.update(dict(preexec_fn=os.setsid))
90 
91  # Execute the command with Popen, logging an error and re-raising
92  # the exception on failure.
93  cls._log.debug(full_readable_command)
94  try:
95  p = Popen(full_command, **kwargs)
96  stdout, stderr = p.communicate()
97  except OSError as exception:
98  cls._log.error("Failed to execute command '%s': %s.",
99  full_readable_command, exception)
100  raise exception
101 
102  # Decode output if it is binary.
103  encoding = locale.getpreferredencoding()
104  if isinstance(stdout, binary_type):
105  stdout = stdout.decode(encoding)
106  if isinstance(stderr, binary_type):
107  stderr = stderr.decode(encoding)
108 
109  # Print error messages to stderr. Filter BadWindow messages.
110  stderr = stderr.rstrip()
111  if stderr and "BadWindow" not in stderr:
112  print(stderr, file=sys.stderr)
113 
114  # Return the process output and return code.
115  return stdout.rstrip(), p.returncode

238  def _get_properties_from_xprop(self, *properties):
239  # This method retrieves windows properties by shelling out to xprop.
240  result = {}
241  args = ['-id', self.id] + list(properties)
242  stdout, return_code = self._run_xprop_command(args)
243  if return_code > 0:
244  return {}
245 
246  for line in stdout.split('\n'):
247  line = line.split(' =', 1)
248  if len(line) != 2:
249  continue
250 
251  raw_key, value = line
252  raw_key, value = raw_key.strip(), value.strip()
253  if 'STRING)' in raw_key: # This also handles (UTF8_STRING).
254  value = value[1:-1]
255 
256  # Use a list if there are multiple strings.
257  if '", "' in value:
258  value = [string for string in value.split('", "')]
259 
260  # Get the key without the property type.
261  key = raw_key[0:raw_key.find('(')]
262  result[key] = value
263 
264  # Split up class and class name.
265  if 'WM_CLASS' in result:
266  window_class_name, window_class = result.pop('WM_CLASS')
267  result['cls_name'] = window_class_name
268  result['cls'] = window_class
269 
270  # Return the requested properties.
271  return result
