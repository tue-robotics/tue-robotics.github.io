58 class DgnImported(RuleRef):
59  """
60  Base class that implements an imported rule and takes care of
61  decoding recognitions accordingly.

64 
65  def __init__(self, name=None, imported_name=None, default=None):
66  if not imported_name:
67  imported_name = name
68  self.imported_name = imported_name

70  RuleRef.__init__(self, self.imported_rule, name, default=default)
71 
72  def decode(self, state):
73  state.decode_attempt(self)
74 
75  # Check that at least one word has been dictated, otherwise fail.
76  if state.rule() != self.imported_name:
77  state.decode_failure(self)
78  return
79 
80  # Determine how many words have been dictated.
81  count = 1
82  while state.rule(count) == self.imported_name:
83  count += 1
84 
85  # Yield possible states where the number of dictated words
86  # gobbled is decreased by 1 between yields.
87  for i in range(count, 0, -1):
88  state.next(i)
89  state.decode_success(self)
90  yield state
91  state.decode_retry(self)
92  state.decode_rollback(self)
93 
94  # None of the possible states were accepted, failure.
95  state.decode_failure(self)
96  return
