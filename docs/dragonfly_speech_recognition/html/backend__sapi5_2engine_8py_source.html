203  def wait_for_recognition(self, timeout=None):
204  NULL = c_int(win32con.NULL)
205  if timeout != None:
206  begin_time = time.time()
207  timed_out = False
208  windll.user32.SetTimer(NULL, NULL, int(timeout * 1000), NULL)
209 
210  message = MSG()
211  message_pointer = pointer(message)
212 
213  while (not timeout) or (time.time() - begin_time < timeout):
214  self._log.error("loop")
215  if windll.user32.GetMessageW(message_pointer, NULL, 0, 0) == 0:
216  msg = str(WinError())
217  self._log.error("GetMessageW() failed: %s" % msg)
218  raise EngineError("GetMessageW() failed: %s" % msg)
219 
220  if message.message == win32con.WM_TIMER:
221  self._log.error("loop, timeout")
222  # A timer message means this loop has timed out.
223  timed_out = True
224  break
225  else:
226  self._log.error("loop, dispatch")
227  # Process other messages as normal.
228  windll.user32.TranslateMessage(message_pointer)
229  windll.user32.DispatchMessageW(message_pointer)
230 
231  return not timed_out

282  def select_audio_source(self, audio_source):
283  """
284  Configure the speech recognition engine to use the given
285  audio source.
286 
287  The audio source may be specified as follows:
288  - As an *int* specifying the index of the audio source to use
289  - As a *str* containing the description of the audio source
290  to use, or a substring thereof
291 
292  This class' method *get_audio_sources()* can be used to
293  retrieve the available sources together with their indices
294  and descriptions.
295 
296  """
297 
298  available_sources = self._recognizer.GetAudioInputs()
299 
300  if isinstance(audio_source, (int, long)):
301  # Parameter is the index of the source to use.
302  if 0 <= audio_source < available_sources.Count:
303  selected_source = available_sources.Item(audio_source)
304  else:
305  raise EngineError("Invalid audio source index: %r"
306  " (%s sources available, so index must be"
307  " in range 0 to %s)"
308  % (audio_source, available_sources.Count,
309  available_sources.Count - 1))
310 
311  elif isinstance(audio_source, basestring):
312  for item in collection_iter(available_sources):
313  if audio_source in item.GetDescription():
314  selected_source = item
315  break
316  else:
317  raise EngineError("Audio source not found: %r"
318  % (audio_source))
319 
320  else:
321  raise EngineError("Invalid audio source qualifier: %r"
322  % (audio_source))
323 
324  self._log.info("Selecting audio source: %r"
325  % (selected_source.GetDescription(),))
326  self._recognizer.AudioInput = selected_source

365  def recognition_callback(self, StreamNumber, StreamPosition, RecognitionType, Result):
366  try:
367  newResult = Dispatch(Result)
368  phrase_info = newResult.PhraseInfo
369  rule_name = phrase_info.Rule.Name
370 
371  #---------------------------------------------------------------
372  # Build a list of rule names for each element.
373 
374  # First populate it with the top level rule name.
375  element = phrase_info.Rule
376  name = element.Name
377  start = element.FirstElement
378  count = element.NumberOfElements
379  rule_names = [name] * count
380 
381  # Walk the tree of child rules and put their names in the list.
382  stack = [collection_iter(phrase_info.Rule.Children)]
383  while stack:
384  try: element = stack[-1].next()
385  except StopIteration: stack.pop(); continue
386  name = element.Name
387  start = element.FirstElement
388  count = element.NumberOfElements
389  rule_names[start:start + count] = [name] * count
390  if element.Children:
391  stack.append(collection_iter(element.Children))
392 
393  #---------------------------------------------------------------
394  # Prepare the words and rule names for the element parsers.
395 
396  replacements = [False] * len(rule_names)
397  if phrase_info.Replacements:
398  for replacement in collection_iter(phrase_info.Replacements):
399  begin = replacement.FirstElement
400  end = begin + replacement.NumberOfElements
401  replacements[begin] = replacement.Text
402  for index in range(begin + 1, end):
403  replacements[index] = True
404 
405  results = []
406  rule_set = list(set(rule_names))
407  elements = phrase_info.Elements
408  for index in range(len(rule_names)):
409  element = elements.Item(index)
410  rule_id = rule_set.index(rule_names[index])
411  replacement = replacements[index]
412  info = [element.LexicalForm, rule_id,
413  element.DisplayText, element.DisplayAttributes,
414  replacement]
415  results.append(info)
416 
417  #---------------------------------------------------------------
418  # Attempt to parse the recognition.
419 
420  func = getattr(self.grammar, "process_recognition", None)
421  if func:
422  words = [r[2] for r in results]
423  if not func(words):
424  return
425 
426  s = State(results, rule_set, self.engine)
427  for r in self.grammar._rules:
428  if r.name != rule_name:
429  continue
430  s.initialize_decoding()
431  for result in r.decode(s):
432  if s.finished():
433  root = s.build_parse_tree()
434  r.process_recognition(root)
435  return
436 
437  except Exception, e:
438  Sapi5Engine._log.error("Grammar %s: exception: %s"
439  % (self.grammar._name, e), exc_info=True)
440 
441  #-------------------------------------------------------------------
442  # If this point is reached, then the recognition was not
443  # processed successfully..
444 
445  self.engine._log.error("Grammar %s: failed to decode"
446  " recognition %r."
447  % (self.grammar._name,
448  [r[0] for r in results]))
