229  def mimic(self, words):
230  """
231  Mimic a recognition of the given *words*.
232 
233  .. note:: This method has a few quirks to be aware of:
234 
235  #. Mimic can fail to recognize a command if the relevant grammar
236  is not yet active.
237  #. Mimic does not work reliably with the shared recognizer unless
238  there are one or more exclusive grammars active.
239  #. Mimic can **crash the process** in some circumstances, e.g.
240  when mimicking non-ASCII characters.
241 
242  """
243  self._log.debug("SAPI5 mimic: %r" % (words,))
244  if isinstance(words, string_types):
245  phrase = words
246  else:
247  phrase = " ".join(words)
248 
249  # Fail on empty input.
250  if not phrase:
251  raise MimicFailure("Invalid mimic input %r" % phrase)
252 
253  # Register a recognition observer for checking the success of this
254  # mimic.
255  observer = MimicObserver()
256  observer.register()
257 
258  # Emulate recognition of the phrase and wait for recognition to
259  # finish, timing out after 2 seconds.
260  self._recognizer.EmulateRecognition(phrase)
261  timeout = 2
262  NULL = c_int(win32con.NULL)
263  if timeout != None:
264  begin_time = time.time()
265  windll.user32.SetTimer(NULL, NULL, int(timeout * 1000), NULL)
266 
267  message = MSG()
268  message_pointer = pointer(message)
269  while (not timeout) or (time.time() - begin_time < timeout):
270  if timeout:
271  self._log.debug("SAPI5 message loop: %s sec left"
272  % (timeout + begin_time - time.time()))
273  else:
274  self._log.debug("SAPI5 message loop: no timeout")
275 
276  if windll.user32.GetMessageW(message_pointer, NULL, 0, 0) == 0:
277  msg = str(WinError())
278  self._log.error("GetMessageW() failed: %s" % msg)
279  raise EngineError("GetMessageW() failed: %s" % msg)
280 
281  self._log.debug("SAPI5 message: %r" % (message.message,))
282  if message.message == win32con.WM_TIMER:
283  # A timer message means this loop has timed out.
284  self._log.debug("SAPI5 message loop timed out: %s sec left"
285  % (timeout + begin_time - time.time()))
286  break
287  else:
288  # Process other messages as normal.
289  self._log.debug("SAPI5 message translating and dispatching.")
290  windll.user32.TranslateMessage(message_pointer)
291  windll.user32.DispatchMessageW(message_pointer)
292 
293  if observer.status.startswith("recognition:"):
294  # The previous message was a recognition which matched.
295  self._log.debug("SAPI5 message caused recognition.")
296 
297  # Unregister the observer and check its status.
298  observer.unregister()
299  if observer.status == "failure":
300  raise MimicFailure("Mimic failed.")
301  elif observer.status == "none":
302  raise MimicFailure("Mimic failed, nothing happened.")

326  def _do_recognition(self):
327  """
328  Recognize speech in a loop.
329 
330  This will also call any scheduled timer functions and ensure
331  that the correct window context is used.
332 
333  """
334 
335  # Register for window change events to activate/deactivate grammars
336  # and rules on window changes, including window title changes. This
337  # is done here because the SAPI5 'OnPhraseStart' grammar callback is
338  # called after grammar state changes are allowed.
339  WinEventProcType = WINFUNCTYPE(None, HANDLE, DWORD, HWND, LONG,
340  LONG, DWORD, DWORD)
341 
342  self._last_foreground_window = None
343  self._last_foreground_window_title = None
344 
345  def callback(hWinEventHook, event, hwnd, idObject, idChild,
346  dwEventThread, dwmsEventTime):
347  window = Window.get_foreground()
348  # Note: hwnd doesn't always match window.handle, even when
349  # foreground window changed (and sometimes it didn't change)
350  window_changed = (
351  window != self._last_foreground_window or
352  window == self._last_foreground_window and
353  window.title != self._last_foreground_window_title
354  )
355  if window_changed:
356  self.process_grammars_context(window)
357  self._last_foreground_window = window
358  self._last_foreground_window_title = window.title
359 
360  def set_hook(win_event_proc, event_type):
361  return windll.user32.SetWinEventHook(
362  event_type, event_type, 0, win_event_proc, 0, 0,
363  win32con.WINEVENT_OUTOFCONTEXT)
364 
365  win_event_proc = WinEventProcType(callback)
366  windll.user32.SetWinEventHook.restype = HANDLE
367 
368  events = {win32con.EVENT_SYSTEM_FOREGROUND,
369  win32con.EVENT_OBJECT_NAMECHANGE}
370  hook_ids = [set_hook(win_event_proc, event) for event in events]
371 
372  # Recognize speech, call timer functions and handle window change
373  # events in a loop. Stop on disconnect().
374  self.speak('beginning loop!')
375  try:
376  while self._recognizer is not None:
377  pythoncom.PumpWaitingMessages()
378  self.call_timer_callback()
379  time.sleep(0.005)
380  finally:
381  # Unregister event hooks.
382  for hook_id in hook_ids:
383  windll.user32.UnhookWinEvent(hook_id)

561  def recognition_callback(self, StreamNumber, StreamPosition,
562  RecognitionType, Result):
563  try:
564  newResult = Dispatch(Result)
565  phrase_info = newResult.PhraseInfo
566  rule_name = phrase_info.Rule.Name
567 
568  #---------------------------------------------------------------
569  # Build a list of rule names for each element.
570 
571  # First populate it with the top level rule name.
572  element = phrase_info.Rule
573  name = element.Name
574  start = element.FirstElement
575  count = element.NumberOfElements
576  rule_names = [name] * count
577 
578  # Walk the tree of child rules and put their names in the list.
579  stack = [collection_iter(phrase_info.Rule.Children)]
580  while stack:
581  try: element = next(stack[-1])
582  except StopIteration: stack.pop(); continue
583  name = element.Name
584  start = element.FirstElement
585  count = element.NumberOfElements
586  rule_names[start:start + count] = [name] * count
587  if element.Children:
588  stack.append(collection_iter(element.Children))
589 
590  #---------------------------------------------------------------
591  # Prepare the words and rule names for the element parsers.
592 
593  replacements = [False] * len(rule_names)
594  if phrase_info.Replacements:
595  for replacement in collection_iter(phrase_info.Replacements):
596  begin = replacement.FirstElement
597  end = begin + replacement.NumberOfElements
598  replacements[begin] = replacement.Text
599  for index in range(begin + 1, end):
600  replacements[index] = True
601 
602  results = []
603  rule_set = list(set(rule_names))
604 
605  elements = phrase_info.Elements
606  for index in range(len(rule_names)):
607  element = elements.Item(index)
608  rule_id = rule_set.index(rule_names[index])
609 
610  # Map dictation rule IDs to 1M so that dragonfly recognizes
611  # the words as dictation.
612  if rule_names[index] == "dgndictation":
613  rule_id = 1000000
614 
615  replacement = replacements[index]
616  info = [element.LexicalForm, rule_id,
617  element.DisplayText, element.DisplayAttributes,
618  replacement]
619  results.append(info)
620 
621  #---------------------------------------------------------------
622  # Retain audio
623 
624  self._retain_audio(results, newResult, rule_name)
625 
626  #---------------------------------------------------------------
627  # Attempt to parse the recognition.
628 
629  if self.process_results(results, rule_set, newResult): return
630 
631  except Exception as e:
632  Sapi5Engine._log.error("Grammar %s: exception: %s"
633  % (self.grammar._name, e), exc_info=True)
634 
635  #-------------------------------------------------------------------
636  # If this point is reached, then the recognition was not
637  # processed successfully..
638 
639  self._log.error("Grammar %s: failed to decode recognition %r."
640  % (self.grammar._name, [r[0] for r in results]))
