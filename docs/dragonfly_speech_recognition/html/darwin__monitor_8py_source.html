33 class DarwinMonitor(BaseMonitor):
34  """
35  The monitor class used on Mac OS (Darwin).
36 
37  Monitors for Mac OS are **not** automatically updated because ``AppKit``
38  doesn't update monitor information after it is retrieved. A simple
39  workaround for this is to restart the Python interpreter after a monitor
40  change.
41 
42  """
43 
44  #-----------------------------------------------------------------------
45  # Class methods to create new Monitor objects.
46 
47  @classmethod
48  def get_all_monitors(cls):
49  # Get monitor information from AppKit.
50  monitors = []
51  for screen in NSScreen.screens():
52  # Get the monitor handle.
53  handle = screen.deviceDescription()['NSScreenNumber']
54 
55  # Create a rectangle object representing the monitor's
56  # dimensions and relative position.
57  frame = screen.frame()
58  dx = frame.size.width - frame.origin.x
59  dy = frame.size.height - frame.origin.y
60  rectangle = Rectangle(frame.origin.x, frame.origin.y, dx, dy)
61 
62  # Get a new or updated monitor object and add it to the list.
63  monitors.append(cls.get_monitor(handle, rectangle))
64 
65  return monitors
66 
67  #-----------------------------------------------------------------------
68  # Methods that control attribute access.
69 
70  @property
71  def name(self):
72  """ The name of this monitor. """
73  handle = self.handle
74  for screen in NSScreen.screens():
75  if screen.deviceDescription()['NSScreenNumber'] == handle:
76  return screen.localizedName()
77 
78  return u"macOS monitor"
