117  def position_string(self, width=20, width_before=6):
118  mark, continuation = ">>", "."
119  width -= 2
120 
121  # Locate edges.
122  i1 = self._index - width_before
123  i2 = self._index - width_before + width - len(mark)
124 
125  # Account for edge effects.
126  if i1 < 0:
127  i2 = min(i2 - i1, len(self._data))
128  i1 = 0
129  elif i2 > len(self._data):
130  width_before += i2 - len(self._data)
131  i1 = max(i1 - i2 + len(self._data), 0)
132  i2 = len(self._data)
133 
134  # Collect character representations.
135  before = [repr(c)[1:-1] for c in self._data[i1:self._index]]
136  after = [repr(c)[1:-1] for c in self._data[self._index:i2]]
137 
138  # Build before string.
139  if i1 == 0 and len("".join(before)) <= width_before:
140  before = '"' + "".join(before)
141  else:
142  while before and len("".join(before)) > width_before - 2:
143  before.pop(0)
144  before = "".join(before)
145  before = '%s"%s' % (continuation * (width_before - len(before)), before)
146 
147  # Build after string.
148  width_after = width - len(before) - len(mark)
149  if i2 == len(self._data) and len("".join(after)) <= width_after:
150  after = "".join(after) + '"'
151  else:
152  while after and len("".join(after)) > width_after - 2:
153  after.pop()
154  after = "".join(after)
155  after = '%s"%s' % (after, continuation * (width_after - len(after)))
156 
157  # Build and return result.
158  return '%s%s%s' % (before, mark, after)

163  def remaining(self):

169  def peek(self, length):

173  length = len(self._data) - self._index

177  if self._index == len(self._data):
178  return None

180  length = len(self._data) - self._index
181  self._index += length
182  return self._data[self._index - length : self._index]

212  def decode_attempt(self, element):
213  # assert isinstance(element, ParserElementBase)
214  self._depth += 1
215  self._stack.append(State._Frame(self._depth, element, self._index))
216  if self._log_debug:
217  self._log_step(element, "attempt")

219  def decode_retry(self, element):
220  # assert isinstance(element, ParserElementBase)
221  frame = self._get_frame_from_actor(element)
222  self._depth = frame.depth
223  if self._log_debug:
224  self._log_step(element, "retry")
225 
226  def decode_rollback(self, element):
227  # assert isinstance(element, ParserElementBase)
228  frame = self._get_frame_from_depth()
229  if not frame or frame.actor != element:
230  raise ParserError("Parser decoding stack broken")
231  if frame is self._stack[-1]:
232  # Last parser on the stack, rollback.
233  self._index = frame.begin
234  else:
235  raise ParserError("Parser decoding stack broken")
236  if self._log_debug:
237  self._log_step(element, "rollback")
238 
239  def decode_success(self, element, value=None):
240  # assert isinstance(element, ParserElementBase)
241  if self._log_debug:
242  self._log_step(element, "success")
243  frame = self._get_frame_from_depth()
244  if not frame or frame.actor != element:
245  raise ParserError("Parser decoding stack broken.")
246  frame.end = self._index
247  frame.value = value
248  self._depth -= 1
249 
250  def decode_failure(self, element):
251  # assert isinstance(element, ParserElementBase)
252  frame = self._stack.pop()
253  self._index = frame.begin
254  self._depth = frame.depth
255  if self._log_debug:
256  self._log_step(element, "failure")
257  self._depth -= 1
258 
259  def _get_frame_from_depth(self):
260  for frame in reversed(self._stack):
261  if frame.depth == self._depth:
262  return frame
263  return None
264 
265  def _get_frame_from_actor(self, actor):
266  for frame in reversed(self._stack):
267  if frame.actor is actor:
268  return frame
269  return None

271  def _log_step(self, parser, message):
272  indent = " " * self._depth
273  output = "%s%s: %s" % (indent, message, parser)
274  self._log.debug(output)
275  if self._index != self._previous_index:
276  self._previous_index = self._index
277  output = "%s -- Decoding State: '%s'" % (indent, str(self))
278  self._log.debug(output)

411  def parse(self, state):
412  state.decode_attempt(self)
413 
414  # Special case for an empty sequence.
415  if not self._children:
416  state.decode_success(self)
417  yield state
418  state.decode_retry(self)

421 
422  # Attempt to walk a path through the entire sequence of children
423  # so that each one decodes successfully.

426  # Allow the last child to attempt decoding.
427  try:
428  next(path[-1])

432  path.pop()

436  # Sequence not yet complete, append the next child.
437  path.append(self._children[len(path)].parse(state))
438  else:

718 
719  def char_matches(self, c):
720  if self._set:
721  return c in self._set
722  elif self._pattern:
723  return bool(self._pattern.match(c))
724  else:
725  return False
726 
727  def parse(self, state):
728  state.decode_attempt(self)
729 
730  # Gobble as many valid characters as possible.
731  count = 0
732  if self._exclude:
733  while (not state.finished() and not
734  self.char_matches(state.peek(1))):
735  state.next(1)
736  count += 1
737  else:
738  while (not state.finished() and
739  self.char_matches(state.peek(1))):
740  state.next(1)
741  count += 1
742 
743  if self._optional or count > 0:
744  state.decode_success(self)
745  yield state
746  state.decode_retry(self)
747 
748  state.decode_failure(self)

832 class QuotedStringContent(ParserElementBase):
833 
834  escape_char = "\\"
835  valid_chars = string.printable
836  escaped_chars = {
837  "n": "\n",
838  "t": "\t",
839  }
840  valid_char_pattern = re.compile(r"\w", re.UNICODE)
841 
842  def __init__(self, delimiter_string, name=None):
843  self.delimiter_string = delimiter_string
844  ParserElementBase.__init__(self, name)
845 
846  def valid_char(self, c):
847  """
848  Check whether a character is a valid. Valid characters include all
849  Unicode alphanumeric characters and any characters in
850  self.valid_chars.
851  :type c: str
852  :rtype: bool
853  """
854  if c in self.valid_chars:
855  return True
856  else:
857  return bool(self.valid_char_pattern.match(c))
858 
859  def parse(self, state):
860  state.decode_attempt(self)
861 
862  # Gobble as many valid characters as possible.
863  characters = []
864  while not state.finished():
865  next_char = state.peek(1)
866 
867  # If next_char escapes, look forward past it and
868  # handle the escaped character.
869  if next_char == self.escape_char:
870  # If there's nothing after the escaped char, then
871  # we can't gobble the escape character itself.
872  # So we're done.
873  if state.remaining() < 2:
874  break
875  escaped_char = state.peek(2)[1]
876  if escaped_char in self.escaped_chars:
877  # The escaped character has a special transformation.
878  next_char = self.escaped_chars[escaped_char]
879  else:
880  # The escaped character is not special,
881  # so simply unescape it.
882  next_char = escaped_char
883  if not self.valid_char(next_char):
884  break
885  state.next(1) # Gobble self.escape_char.
886 
887  # If self.delimiter_string is next, then we're done.
888  elif state.peek(len(self.delimiter_string)) == self.delimiter_string:
889  break
890 
891  # If next_char is not acceptable, don't gobble it.
892  elif not self.valid_char(next_char):
893  break
894 
895  state.next(1) # Gobble next_char.
896  characters.append(next_char)
897 
898  value = "".join(characters)
899  state.decode_success(self, value)
900  yield state
901  state.decode_retry(self)
902  state.decode_failure(self)
903  return
