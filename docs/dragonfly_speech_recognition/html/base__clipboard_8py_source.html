89  @classmethod
90  def _clipboard_formats_changed(cls, formats, clipboard1, clipboard2):
91  # Check if the content of any specified format has changed.
92  # Use all formats, if specified.
93  if formats == "all":
94  return clipboard1 != clipboard2
95  result = False
96  for format in formats:
97  format_available = clipboard1.has_format(format)
98  result = (
99  format_available != clipboard2.has_format(format) or
100  format_available and clipboard1.get_format(format)
101  != clipboard2.get_format(format))
102  if result:
103  break
104  return result

106  @classmethod
107  def wait_for_change(cls, timeout, step=0.001, formats=None,
108  initial_clipboard=None):
109  """
110  Wait (poll) for the system clipboard to change.
111 
112  This is a blocking method which returns whether or not the
113  system clipboard changed within a specified timeout period.
114 
115  Arguments:
116  - *timeout* (float) -- timeout in seconds.
117  - *step* (float, default: 0.001) -- number of seconds between
118  each check.
119  - *formats* (iterable, default: None) -- if not None, only
120  changes to the given content formats will register. If None,
121  all formats will be observed.
122  - *initial_clipboard* (Clipboard, default: None) -- if a
123  clipboard is given, the method will wait until the system
124  clipboard differs from the instance's contents.
125 
126  """
127  # By default, this method retrieves the system clipboard every
128  # *step* seconds until the contents change. This method should be
129  # overridden by the sub-class if there is a more efficient way for
130  # the platform.
131  if not initial_clipboard:
132  initial_clipboard = cls(from_system=True)
133  clipboard2 = cls()
134  timeout = time.time() + float(timeout)
135  step = float(step)
136  if isinstance(formats, integer_types):
137  formats = (formats,)
138  elif formats:
139  for format in formats:
140  if not isinstance(format, integer_types):
141  raise TypeError("Invalid clipboard format: %r"
142  % format)
143  result = False
144  while time.time() < timeout:
145  # Check if the content of any relevant format has changed.
146  clipboard2.copy_from_system()
147  formats_to_compare = formats if formats else "all"
148  result = cls._clipboard_formats_changed(formats_to_compare,
149  initial_clipboard,
150  clipboard2)
151 
152  if result:
153  break
154 
155  # Failure. Try again after *step* seconds.
156  time.sleep(step)
157  return result

160  @contextlib.contextmanager
161  def synchronized_changes(cls, timeout, step=0.001, formats=None,
162  initial_clipboard=None):
163  """
164  Context manager for synchronizing local and system clipboard
165  changes. This takes the same arguments as the
166  :meth:`wait_for_change` method.
167 
168  Arguments:
169  - *timeout* (float) -- timeout in seconds.
170  - *step* (float, default: 0.001) -- number of seconds between
171  each check.
172  - *formats* (iterable, default: None) -- if not None, only
173  changes to the given content formats will register. If None,
174  all formats will be observed.
175  - *initial_clipboard* (Clipboard, default: None) -- if a
176  clipboard is given, the method will wait until the system
177  clipboard differs from the instance's contents.
178 
179  Use with a Python 'with' block::
180 
181  from dragonfly import Clipboard, Key
182 
183  # Copy the selected text with Ctrl+C and wait until a system
184  # clipboard change is detected.
185  timeout = 3
186  with Clipboard.synchronized_changes(timeout):
187  Key("c-c", use_hardware=True).execute()
188 
189  # Retrieve the system text.
190  text = Clipboard.get_system_text()
191 
192  """
193  # Save the current clipboard contents, if necessary.
194  if initial_clipboard:
195  initial_clipboard = cls(from_system=True)
196  try:
197  # Yield for clipboard operations.
198  yield
199  finally:
200  # Wait for the system clipboard to change.
201  cls.wait_for_change(timeout, step, formats, initial_clipboard)

205  @classmethod
206  def _convert_format_text(cls, content):
207  format_name = cls.format_names[cls.format_text]
208  if not isinstance(content, (text_type, binary_type)):
209  raise TypeError("Invalid content for format %s: (%r)"
210  % (format_name, content,))
211 
212  # Use a binary string for CF_TEXT content.
213  if isinstance(content, text_type):
214  encoding = locale.getpreferredencoding()
215  content = content.encode(encoding)
216 
217  return content

219  @classmethod
220  def _convert_format_unicode(cls, content):
221  format_name = cls.format_names[cls.format_unicode]
222  if not isinstance(content, (text_type, binary_type)):
223  raise TypeError("Invalid content for format %s: (%r)"
224  % (format_name, content,))
225 
226  # Use a text string for CF_UNICODETEXT content.
227  if isinstance(content, binary_type):
228  encoding = locale.getpreferredencoding()
229  content = content.decode(encoding)
230 
231  return content

233  @classmethod
234  def _convert_format_hdrop(cls, content):
235  # The CF_HDROP clipboard format is for copied file paths.
236  format_name = cls.format_names[cls.format_hdrop]
237  result = content
238 
239  # Convert string content into a list of file paths. String content
240  # must be a list of file paths separated by new lines and/or null
241  # characters. The following example string is acceptable:
242  # "c:\\temp1.txt\0c:\\temp2.txt\0\0".
243  if isinstance(content, (text_type, binary_type)):
244  delimiter = u"[\0\r\n]"
245  if isinstance(content, binary_type):
246  delimiter = delimiter.encode()
247 
248  # Convert string content into a list and remove empty
249  # strings.
250  result = re.split(delimiter, content)
251  result = [string for string in result if string]
252 
253  # Verify that the list/tuple items are non-empty strings.
254  elif isinstance(content, (list, tuple)):
255  string_items = all([
256  isinstance(item, (text_type, binary_type)) and item
257  for item in content
258  ])
259  if not string_items:
260  raise TypeError("Invalid content type for format %s: (%r)"
261  % (format_name, content))
262 
263  # Only strings, tuples and lists are accepted for CF_HDROP
264  # content.
265  else:
266  raise TypeError("Invalid content type for format %s: (%r)"
267  % (format_name, content))
268 
269  # Remove the file scheme from strings if it is present.
270  file_paths = []
271  for string in result:
272  file_scheme = (u"file://" if isinstance(string, text_type)
273  else b"file://")
274  if string.startswith(file_scheme):
275  string = string[7:]
276  file_paths.append(string)
277  result = file_paths
278 
279  # Empty lists/tuples are not acceptable.
280  if not result:
281  raise ValueError("Invalid content value for format %s: (%r)"
282  % (format_name, content))
283 
284  # Verify that strings in the content list are existing, absolute
285  # file paths. Relative paths are not accepted because they do not
286  # make sense without the context of a working directory. The system
287  # clipboard does not typically have such a context.
288  file_paths_exist = all([
289  os.path.isabs(item) and os.path.exists(item)
290  for item in result
291  ])
292  if not (file_paths_exist and result):
293  raise ValueError("Invalid content value for format %s: (%r)"
294  % (format_name, content))
295 
296  # Use a tuple for CF_HDROP content.
297  if isinstance(result, list):
298  result = tuple(result)
299 
300  return result

302  @classmethod
303  def convert_format_content(cls, format, content):
304  """
305  Convert content for the given *format*, if necessary, and return
306  it.
307 
308  This method operates on the following formats:
309  - *text* -- encodes content to a binary string, if necessary
310  and if possible.
311  - *unicode* -- decodes content to a text string, if necessary
312  and if possible.
313  - *hdrop* -- converts content into a tuple of file paths, if
314  necessary and if possible.
315 
316  String content must be a list of existing, absolute file
317  paths separated by new lines and/or null characters. All
318  specified file paths must be absolute paths referring to
319  existing files on the system.
320 
321  If the content cannot be converted for the given *format*, an
322  error is raised.
323 
324  Arguments:
325  - *format* (int) -- the clipboard format to convert.
326  - *content* (string) -- the clipboard contents to convert.
327 
328  """
329  convert_method = {
330  cls.format_text: cls._convert_format_text,
331  cls.format_unicode: cls._convert_format_unicode,
332  cls.format_hdrop: cls._convert_format_hdrop,
333  }.get(format)
334  if convert_method is not None:
335  content = convert_method(content)
336 
337  # Return the content.
338  return content
