161  def update(self, hardware_events_required):
162  # Nothing to do for virtual keys.
163  if self._char is None:
164  return True
165 
166  # Get updated key code and modifiers for this Typeable, falling back
167  # on the key codes in CHAR_VK_MAP where necessary.
168  try:
169  self._is_text = False
170  code, modifiers = Win32Keyboard.get_keycode_and_modifiers(
171  self._char, char_vk_fallback=hardware_events_required)
172  except ValueError:
173  if hardware_events_required:
174  # This Typeable cannot be typed in the current context.
175  return False
176  else:
177  # Fallback on Unicode events.
178  code, modifiers = self._char, ()
179  self._is_text = True
180 
181  # Set key code and modifiers.
182  self._code, self._modifiers = code, modifiers
183  return True

185  def _unicode_events(self, down, timeout):
186  character = self._code
187  if not isinstance(character, text_type):
188  raise TypeError("cannot create Unicode events for character: %r"
189  % character)
190 
191  # Construct the UTF-16 (LE) events that sendinput expects.
192  events = []
193  byte_stream = character.encode("utf-16-le")
194  format_string = "<" + str(len(byte_stream) // 2) + "H"
195  for short in unpack(format_string, byte_stream):
196  if down is True or down is False:
197  events.extend([(short, down, timeout, True)])
198  else: # both for events()
199  events.extend([(short, True, timeout, True),
200  (short, False, timeout, True)])
201  return events
202 
203  def on_events(self, timeout=0):
204  """Return events for pressing this key down."""
205  if self._is_text:
206  events = self._unicode_events(True, timeout)
207  else:
208  events = [(m, True, 0) for m in self._modifiers]
209  events.append((self._code, True, timeout))
210  return events
211 
212  def off_events(self, timeout=0):
213  """Return events for releasing this key."""
214  if self._is_text:
215  events = self._unicode_events(False, timeout)
216  else:
217  events = [(m, False, 0) for m in self._modifiers]
218  events.append((self._code, False, timeout))
219  events.reverse()
220  return events
221 
222  def events(self, timeout=0):
223  """Return events for pressing and then releasing this key."""
224  if self._is_text:
225  events = self._unicode_events("both", timeout)
226  else:
227  events = [(self._code, True, 0), (self._code, False, timeout)]
228  for m in self._modifiers[-1::-1]:
229  events.insert(0, (m, True, 0))
230  events.append((m, False, 0))
231  return events

249  @classmethod
250  def send_keyboard_events(cls, events):
251  """
252  Send a sequence of keyboard events.
253 
254  Positional arguments:
255  events -- a sequence of tuples of the form
256  (keycode, down, timeout), where
257  keycode (int): Win32 Virtual Key code.
258  down (boolean): True means the key will be pressed down,
259  False means the key will be released.
260  timeout (int): number of seconds to sleep after
261  the keyboard event.
262  or
263  (character, down, timeout, is_text), where
264  character (str): Unicode character.
265  down (boolean): True means the key will be pressed down,
266  False means the key will be released.
267  timeout (int): number of seconds to sleep after
268  the keyboard event.
269  is_text (boolean): True means that the keypress is targeted
270  at a window or control that accepts Unicode text.
271  """
272  layout = cls.get_current_layout()
273 
274  # Process and send keyboard events.
275  items = []
276  for event in events:
277  if len(event) == 3:
278  # Pass scancode=0 to press keys via virtual-key codes
279  # instead of scancodes.
280  keycode, down, timeout = event
281  input_structure = KeyboardInput(keycode, down,
282  # scancode=0,
283  layout=layout)
284  elif len(event) == 4 and event[3]:
285  character, down, timeout = event[:3]
286  input_structure = KeyboardInput(0, down, scancode=character)
287  else:
288  raise ValueError("invalid keyboard event tuple: %r" % event)
289  items.append(input_structure)
290  if timeout:
291  array = make_input_array(items)
292  items = []
293  send_input_array(array)
294  time.sleep(timeout)
295  if items:
296  array = make_input_array(items)
297  send_input_array(array)
298  if timeout: time.sleep(timeout)

300  @classmethod
301  def _get_initial_keycode(cls, char, char_vk_fallback=True):
302  # Get the code for this character.
303  layout = cls.get_current_layout()
304  try:
305  code = win32api.VkKeyScanEx(char, layout)
306  except TypeError:
307  code = -1
308 
309  # Fallback on the keycode in CHAR_VK_MAP, if applicable.
310  # Note: char_vk_fallback=False is used when typing these characters
311  # with the Unicode keyboard.
312  if (char_vk_fallback and code == -1 and
313  char in Win32KeySymbols.CHAR_VK_MAP):
314  code = Win32KeySymbols.CHAR_VK_MAP[char]
315 
316  if code == -1:
317  raise ValueError("Unknown char: %r" % char)
318 
319  return code

321  @classmethod
322  def xget_virtual_keycode(cls, char, char_vk_fallback=True):
323  code = cls._get_initial_keycode(char, char_vk_fallback)
324 
325  # Construct a list of the virtual key code and modifiers.
326  codes = [code & 0x00ff]
327  if code & 0x0100: codes.append(cls.shift_code)
328  if code & 0x0200: codes.append(cls.ctrl_code)
329  if code & 0x0400: codes.append(cls.alt_code)
330  return codes

332  @classmethod

334  code = cls._get_initial_keycode(char, char_vk_fallback)
335 
336  # Construct a list of the virtual key code and modifiers.
337  modifiers = []
338  if code & 0x0100: modifiers.append(cls.shift_code)
339  if code & 0x0200: modifiers.append(cls.ctrl_code)
340  if code & 0x0400: modifiers.append(cls.alt_code)
341  code &= 0x00ff
