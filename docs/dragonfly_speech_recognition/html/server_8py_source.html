20 """
21 RPC server
22 ----------------------------------------------------------------------------
23 
24 Dragonfly's RPC server handles requests by processing each method through
25 the current engine's :ref:`multiplexing timer interface <RefEngineTimers>`.
26 This allows engines to handle requests safely and keeps engine-specific
27 implementation details out of the *dragonfly.rpc* sub-package.
28 
29 Security tokens
30 ============================================================================
31 
32 The RPC server uses mandatory security tokens to authenticate client
33 requests. This avoids some security issues where, for example, malicious web
34 pages could send POST requests to the server, even if running on localhost.
35 
36 If sending requests over an open network, please ensure that the connection
37 is secure by using TLS or SSH port forwarding.
38 
39 If the server's :code:`security_token` constructor parameter is not
40 specified, a new token will be generated and printed to the console. Clients
41 must specify the security token either as the last positional argument or as
42 the :code:`security_token` keyword argument.
43 
44 Errors will be raised if clients send no security token or a token that
45 doesn't match the server's.

80 # RPC method decorator used by the server internally.

82 @decorator
83 def _rpc_method(method, *args, **kwargs):
84  # Create a thread condition for waiting for the method's result.
85  condition = threading.Condition()
86 
87  closure = []
88  def timer_func():
89  try:
90  closure.append(method(*args, **kwargs))
91  except Exception as e:
92  # Log any exceptions.
93  _log.exception("Exception occurred during RPC method '%s': %s"
94  % (method.__name__, e))
95  closure.append(e)
96 
97  # Notify the waiting thread that the result is ready.
98  with condition:
99  condition.notify()
100 
101  # Start a non-repeating timer to execute timer_func().
102  get_engine().create_timer(timer_func, 0, repeating=False)
103 
104  # Wait for the result.
105  with condition:
106  condition.wait()
107 
108  # Raise an error if it's an exception (json-rpc handles this),
109  # otherwise return it.
110  result = closure[0]
111  if isinstance(result, Exception):
112  raise result
113  else:
114  return result

136  def checked_rpc_method(*args, **kwargs):
137  assert not (args and kwargs)

140  # there is no token.

148  "'security_token' argument")

153  if not compare_security_token(self.server.security_token,
154  rpc_security_token):

156  " the server\'s")
157  _log.error(message)
158  raise PermissionDeniedError(message)
159

161  return f(*args, **kwargs)

171 class RPCServer(object):
172  """
173  RPC server class.
174 
175  This class will run a local web server on port 50051 by default. The
176  server expects requests using
177  `JSON-RPC 2.0 <https://www.jsonrpc.org/specification>`__.
178 
179  Constructor arguments:
180 
181  - *address* -- address to use (*str*, default: "127.0.0.1")
182  - *port* -- port to use (*int*, default: 50051)
183  - *ssl_context* -- SSL context object to pass to
184  *werkzeug.serving.run_simple* (*SSLContext*, default: None).
185  - *threaded* -- whether to use a separate thread to process each request
186  (*bool*, default: True).
187  - *security_token* -- security token for authenticating clients
188  (*str*, default: None). A new token will be generated and printed if
189  this parameter is unspecified.
190 
191  The *ssl_context* parameter is explained in more detail in Werkzeug's
192  `SSL serving documentation
193  <http://werkzeug.pocoo.org/docs/0.14/serving/#ssl>`__.
194 
195  Secure connections can also be set up using OpenSSH port forwarding with
196  a command such as::
197 
198  $ ssh -NTf -L 50051:127.0.0.1:50051 <system-with-rpc-server>
199 
200  *Minor note*: using an IP address instead of a hostname for the
201  *address* parameter should increase performance somewhat, e.g.
202  "127.0.0.1" instead of "localhost".
203 
204  .. warning::
205 
206  Do **not** send requests to the server from the main engine thread;
207  thread deadlocks *will* occur if you do this because the main thread
208  cannot call timer functions and wait for a response at the same time.
209  The RPC framework was designed to be used from *remote* processes.
210 
211  Requests will not be processed if the engine is not connected and
212  processing speech.
213 
214 
215  """

264  def send_request(self, method, params, id=0):
265  """
266  Utility method to send a JSON-RPC request to the server. This will
267  block the current thread until a response is received.
268 
269  This method is mostly used for testing. If called from the engine's
270  main thread, a deadlock *will* occur.
271 
272  This will raise an error if the request fails with an error or if
273  the server is unreachable.
274 
275  The server's security token will automatically be added to the
276  ``params`` list/dictionary.
277 
278  :param method: name of the RPC method to call.
279  :param params: parameters of the RPC method to call.
280  :param id: ID of the JSON-RPC request (default: 0).
281  :type method: str
282  :type params: list | dict
283  :type id: int
284  :returns: JSON-RPC response
285  :rtype: dict
286  :raises: RuntimeError
287  """
288  # Add the server's security token to params.
289  security_token = self.security_token
290  if isinstance(params, dict):
291  params["security_token"] = security_token
292  else:
293  params.append(security_token)
294 
295  return send_rpc_request(self.url, method, params, id)

297  def start(self):
298  """
299  Start the server.
300 
301  This method is non-blocking, the RPC server will run on a separate
302  daemon thread. This way it is not necessary to call :meth:`stop`
303  before the main thread terminates.
304  """
305  # Handle a previous Thread.join() timeout.
306  if self._thread and not self._thread.is_alive():
307  self._thread = None
308 
309  # Return if the server thread is already running.
310  elif self._thread:
311  return
312 
313  def run():
314  # Note: 'threaded' means that each request is handled using a
315  # separate thread. This introduces some overhead costs.
316  try:
317  return run_simple(self._address, self._port,
318  self._application,
319  ssl_context=self._ssl_context,
320  threaded=self._threaded)
321  except Exception as e:
322  self._log.exception("Exception caught on RPC server thread:"
323  " %s" % e)
324 
325  self._thread = threading.Thread(target=run)
326  self._thread.setDaemon(True)
327  self._thread.start()
328 
329  # Wait a few milliseconds to allow the server to start properly.
330  time.sleep(0.1)
331 
332  def stop(self):
333  """
334  Stop the server if it is running.
335  """
336  # Handle a previous Thread.join() timeout.
337  if self._thread and not self._thread.is_alive():
338  self._thread = None
339 
340  # Return if the server isn't currently running.
341  if not self._thread:
342  return
343 
344  # werkzeug is normally stopped through a request to the server.
345  self.send_request("shutdown_rpc_server", [])
346  timeout = 5
347  self._thread.join(timeout)
348  if self._thread.is_alive():
349  self._log.warning("RPC server thread failed to stop after %d "
350  "seconds" % timeout)
351  else:
352  self._thread = None
