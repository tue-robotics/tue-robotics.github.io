18 class CheckIfWeHaveItAll(State):
19  def __init__(self, robot, max_iterations=20):
20  State.__init__(self, outcomes=["we_have_it_all", "keep_going"], input_keys=["item_picked"])
21  self.robot = robot
22  self.iteration = 0
23  self.items_picked = []
24  self.max_iterations = max_iterations
25 
26  def execute(self, userdata):
27  self.iteration += 1
28  if self.iteration > self.max_iterations:
29  return "we_have_it_all"
30 
31  item_picked = userdata["item_picked"]
32 
33  if item_picked and item_picked not in self.items_picked:
34  self.items_picked.append(item_picked)
35  else:
36  rospy.logwarn("Invalid item picked: %s", item_picked)
37 
38  missing_items = [item for item in ITEMS if item not in self.items_picked]
39 
40  if missing_items:
41  self.robot.speech.speak(f"So far we have: {' '.join(self.items_picked)}", block=False)
42  self.robot.speech.speak(f"Still missing the {' '.join(missing_items)}", block=False)
43  else:
44  self.robot.speech.speak("We have everything now", block=False)
45 
46  return "keep_going" if missing_items else "we_have_it_all"
