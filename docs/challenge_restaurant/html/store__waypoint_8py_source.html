16 def _get_area(convex_hull):
17  pts = [[c.x(), c.y()] for c in convex_hull]
18  lines = np.hstack([pts, np.roll(pts, -1, axis=0)])
19  area = 0.5 * abs(sum(x1 * y2 - x2 * y1 for x1, y1, x2, y2 in lines))
20  return area
